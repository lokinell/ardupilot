51 // This interrupt triggers on a negative transiution of the PPM-SIM pin
52 void FLYMAPLERCInput::_timer_capt_cb(void)
53 {
54  _last_input_interrupt_time = hal.scheduler->millis();
55 
56  static uint16 previous_count;
57  static uint8 channel_ctr;
58 
59  // Read the CCR register, where the time count since the last input pin transition will be
60  timer_dev *tdev = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_device;
61  uint8 timer_channel = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_channel;
62  uint16 current_count = timer_get_compare(tdev, timer_channel);
63  uint32 sr = (tdev->regs).gen->SR;
64  uint32 overcapture_mask = (1 << (TIMER_SR_CC1OF_BIT + timer_channel - 1));
65 
66  if (sr & overcapture_mask)
67  {
68  // Hmmm, lost an interrupt somewhere? Ignore this sample
69  (tdev->regs).gen->SR &= ~overcapture_mask; // Clear overcapture flag
70  return;
71  }
72 
73  uint16_t pulse_width;
74  if (current_count < previous_count) {
75  pulse_width = current_count + FLYMAPLE_TIMER_RELOAD - previous_count;
76  } else {
77  pulse_width = current_count - previous_count;
78  }
79 
80  // Pulse sequence repetition rate is about 22ms.
81  // Longest servo pulse is about 1.8ms
82  // Shortest servo pulse is about 0.5ms
83  // Shortest possible PPM sync pulse with 10 channels is about 4ms = 22 - (10 channels * 1.8)
84  if (pulse_width > 8000) { // 4ms
85  // sync pulse detected. Pass through values if at least a minimum number of channels received
86  if( channel_ctr >= FLYMAPLE_RC_INPUT_MIN_CHANNELS ) {
87  _valid_channels = channel_ctr;
88  // Clear any remaining channels, in case they were corrupted during a connect or something
89  while (channel_ctr < FLYMAPLE_RC_INPUT_NUM_CHANNELS)
90  _pulse_capt[channel_ctr++] = 0;
91  }
92  channel_ctr = 0;
93  } else {
94 
95 // if (channel_ctr == 0)
96 // hal.uartA->printf("ch 0 %d\n", pulse_width);
97 
98  if (channel_ctr < FLYMAPLE_RC_INPUT_NUM_CHANNELS) {
99  _pulse_capt[channel_ctr] = pulse_width;
100  channel_ctr++;
101  if (channel_ctr == FLYMAPLE_RC_INPUT_NUM_CHANNELS) {
102  _valid_channels = FLYMAPLE_RC_INPUT_NUM_CHANNELS;
103  }
104  }
105  }
106  previous_count = current_count;
107 }

109 void FLYMAPLERCInput::init(void* machtnichts)
110 {
111  /* initialize overrides */
112  clear_overrides();
113 
114  // Configure pin 6 input to timer 1 CH1 bRin Input Capture mode
115  pinMode(FLYMAPLE_RC_INPUT_PIN, INPUT_PULLDOWN);
116 
117  timer_dev *tdev = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_device;
118  uint8 timer_channel = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_channel;
119  timer_pause(tdev); // disabled
120  timer_set_prescaler(tdev, (CYCLES_PER_MICROSECOND/2) - 1); // 2MHz = 0.5us timer ticks
121  timer_set_reload(tdev, FLYMAPLE_TIMER_RELOAD-1);
122  // Without a filter, can get triggering on the wrong edge and other problems.
123  (tdev->regs).gen->CCMR1 = TIMER_CCMR1_CC1S_INPUT_TI1 | (3 << 4); // no prescaler, input from T1, filter internal clock, N=8
124  (tdev->regs).gen->CCER = TIMER_CCER_CC1P | TIMER_CCER_CC1E; // falling edge, enable capture
125  timer_attach_interrupt(tdev, timer_channel, _timer_capt_cb);
126  timer_generate_update(tdev);
127  timer_resume(tdev); // reenabled
128 }

147 uint16_t FLYMAPLERCInput::read(uint8_t ch) {
148  timer_dev *tdev = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_device;
149  uint8 timer_channel = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_channel;
150 
151  /* constrain ch */
152  if (ch >= FLYMAPLE_RC_INPUT_NUM_CHANNELS)
153  return 0;
154  /* grab channel from isr's memory in critical section*/
155  timer_disable_irq(tdev, timer_channel);
156  uint16_t capt = _pulse_capt[ch];
157  timer_enable_irq(tdev, timer_channel);
158  /* scale _pulse_capt from 0.5us units to 1us units. */
159  uint16_t pulse = constrain_pulse(capt >> 1);
160  /* Check for override */
161  uint16_t over = _override[ch];
162  return (over == 0) ? pulse : over;
163 }

165 uint8_t FLYMAPLERCInput::read(uint16_t* periods, uint8_t len) {
166  timer_dev *tdev = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_device;
167  uint8 timer_channel = PIN_MAP[FLYMAPLE_RC_INPUT_PIN].timer_channel;
168 
169  /* constrain len */
170  if (len > FLYMAPLE_RC_INPUT_NUM_CHANNELS)
171  len = FLYMAPLE_RC_INPUT_NUM_CHANNELS;
172  /* grab channels from isr's memory in critical section */
173  timer_disable_irq(tdev, timer_channel);
174  for (uint8_t i = 0; i < len; i++) {
175  periods[i] = _pulse_capt[i];
176  }
177  timer_enable_irq(tdev, timer_channel);
178  /* Outside of critical section, do the math (in place) to scale and
179  * constrain the pulse. */
180  for (uint8_t i = 0; i < len; i++) {
181  /* scale _pulse_capt from 0.5us units to 1us units. */
182  periods[i] = constrain_pulse(periods[i] >> 1);
183  /* check for override */
184  if (_override[i] != 0) {
185  periods[i] = _override[i];
186  }
187  }
188  return _valid_channels;
189 }

199 bool FLYMAPLERCInput::set_override(uint8_t channel, int16_t override) {
200  if (override < 0) return false; /* -1: no change. */
201  if (channel < FLYMAPLE_RC_INPUT_NUM_CHANNELS) {
