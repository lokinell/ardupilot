83 /*
84  update distance_cm
85  */
86 void AP_RangeFinder_analog::update(void)
87 {
88  update_voltage();
89  float v = state.voltage_mv * 0.001f;
90  float dist_m = 0;
91  float scaling = ranger._scaling[state.instance];
92  float offset = ranger._offset[state.instance];
93  RangeFinder::RangeFinder_Function function = (RangeFinder::RangeFinder_Function)ranger._function[state.instance].get();
94  int16_t max_distance_cm = ranger._max_distance_cm[state.instance];
95 
96  switch (function) {
97  case RangeFinder::FUNCTION_LINEAR:
98  dist_m = (v - offset) * scaling;
99  break;
100 
101  case RangeFinder::FUNCTION_INVERTED:
102  dist_m = (offset - v) * scaling;
103  break;
104 
105  case RangeFinder::FUNCTION_HYPERBOLA:
106  if (v <= offset) {
107  dist_m = 0;
108  }
109  dist_m = scaling / (v - offset);
110  if (isinf(dist_m) || dist_m > max_distance_cm * 0.01f) {
111  dist_m = max_distance_cm * 0.01f;
112  }
113  break;
114  }
115  if (dist_m < 0) {
116  dist_m = 0;
117  }
118  state.distance_cm = dist_m * 100.0f;
119 
120  // update range_valid state based on distance measured
121  update_status();
122 }
