171  system, and to allow initial sensor read loops to run

180  perf_counter_t perf_loop = perf_alloc(PC_ELAPSED, "APM_loop");
181  perf_counter_t perf_overrun = perf_alloc(PC_COUNT, "APM_overrun");
182  struct hrt_call loop_overtime_call;
183 
184  thread_running = true;
185 
186  /*
187  switch to high priority for main loop
188  */
189  set_priority(APM_MAIN_PRIORITY);
190 
191  while (!_vrbrain_thread_should_exit) {
192  perf_begin(perf_loop);
193 
194  /*
195  this ensures a tight loop waiting on a lower priority driver
196  will eventually give up some time for the driver to run. It
197  will only ever be called if a loop() call runs for more than
198  0.1 second
199  */
200  hrt_call_after(&loop_overtime_call, 100000, (hrt_callout)loop_overtime, NULL);
201 
202  loop();
203 
204  if (vrbrain_ran_overtime) {
205  /*
206  we ran over 1s in loop(), and our priority was lowered
207  to let a driver run. Set it back to high priority now.
208  */
209  set_priority(APM_MAIN_PRIORITY);
210  perf_count(perf_overrun);
211  vrbrain_ran_overtime = false;
212  }
213 
214  perf_end(perf_loop);
215 
216  /*
217  give up 500 microseconds of time, to ensure drivers get a
218  chance to run. This relies on the accurate semaphore wait
219  using hrt in semaphore.cpp
220  */
221  hal.scheduler->delay_microseconds(500);
222  }
223  thread_running = false;
224  return 0;
