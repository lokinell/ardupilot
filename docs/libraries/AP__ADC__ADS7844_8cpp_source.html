18  * AP_ADC_ADS7844.cpp - ADC ADS7844 Library for Ardupilot Mega
19  * Code by Jordi Muï¿½oz and Jose Julio. DIYDrones.com
20  *
21  * Modified by John Ihlein 6 / 19 / 2010 to:
22  * 1)Prevent overflow of adc_counter when more than 8 samples collected between reads. Probably
23  * only an issue on initial read of ADC at program start.
24  * 2)Reorder analog read order as follows:
25  * p, q, r, ax, ay, az
26  * External ADC ADS7844 is connected via Serial port 2 (in SPI mode)
27  * TXD2 = MOSI = pin PH1
28  * RXD2 = MISO = pin PH0
29  * XCK2 = SCK = pin PH2
30  * Chip Select pin is PC4 (33) [PH6 (9)]
31  * We are using the 16 clocks per conversion timming to increase efficiency (fast)
32  *
33  * The sampling frequency is 1kHz (Timer2 overflow interrupt)
34  *
35  * So if our loop is at 50Hz, our needed sampling freq should be 100Hz, so
36  * we have an 10x oversampling and averaging.
37  *
38  * Methods:
39  * Init() : Initialization of interrupts an Timers (Timer2 overflow interrupt)
40  * Ch(ch_num) : Return the ADC channel value
41  *
42  * // HJI - Input definitions. USB connector assumed to be on the left, Rx and servo
43  * // connector pins to the rear. IMU shield components facing up. These are board
44  * // referenced sensor inputs, not device referenced.
45  * On Ardupilot Mega Hardware, oriented as described above:
46  * Chennel 0 : yaw rate, r
47  * Channel 1 : roll rate, p
48  * Channel 2 : pitch rate, q
49  * Channel 3 : x / y gyro temperature
50  * Channel 4 : x acceleration, aX
51  * Channel 5 : y acceleration, aY
52  * Channel 6 : z acceleration, aZ
53  * Channel 7 : Differential pressure sensor port
54  *
55  */

67 static const unsigned char adc_cmd[17] =

83 void AP_ADC_ADS7844::read(void)
84 {
85  static int semfail_ctr = 0;
86  uint8_t ch;
87 
89  bool got = _spi_sem->take_nonblocking();
90  if (!got) {
91  semfail_ctr++;
92  if (semfail_ctr > 100) {
93  hal.scheduler->panic("PANIC: failed to take _spi_sem "
94  "100 times in AP_ADC_ADS7844::read");
95  }
96  return;
97  } else {
98  semfail_ctr = 0;
99  }
100 
101  uint8_t rx[17];
102  _spi->transaction(adc_cmd, rx, 17);
103 
104  for (ch = 0; ch < 8; ch++) {
105  uint16_t v = (rx[2*ch+1] << 8) | rx[2*ch+2];
106  if (v & 0x8007) {
107  // this is a 12-bit ADC, shifted by 3 bits.
108  // if we get other bits set then the value is
109  // bogus and should be ignored
110  continue;
111  }
112  if (++_count[ch] == 0) {
113  // overflow ... shouldn't happen too often
114  // unless we're just not using the
115  // channel. Notice that we overflow the count
116  // to 1 here, not zero, as otherwise the
117  // reader below could get a division by zero
118  _sum[ch] = 0;
119  _count[ch] = 1;
120  }
121  _sum[ch] += (v >> 3);
122  }
123 
124  _spi_sem->give();
125 
126  // record time of this sample
127  _ch6_last_sample_time_micros = hal.scheduler->micros();
128 }

135 void AP_ADC_ADS7844::Init()
136 {
137  hal.scheduler->suspend_timer_procs();
138  _spi = hal.spi->device(AP_HAL::SPIDevice_ADS7844);
139  if (_spi == NULL) {
140  hal.scheduler->panic("PANIC: AP_ADC_ADS7844 missing SPI device "
141  "driver\n");
142  }
143 
144  _spi_sem = _spi->get_semaphore();
145 
146  if (_spi_sem == NULL) {
147  hal.scheduler->panic("PANIC: AP_ADC_ADS7844 missing SPI device "
148  "semaphore");
149  }
150 
151  if (!_spi_sem->take(0)) {
152  hal.scheduler->panic("PANIC: failed to take _spi_sem in"
153  "AP_ADC_ADS7844::Init");
154  }
155 
156  _spi->cs_assert();
157  // get an initial value for each channel. This ensures
158  // _count[] is never zero
159  for (uint8_t i=0; i<8; i++) {
160  uint16_t adc_tmp;
161  adc_tmp = _spi->transfer(0) << 8;
162  adc_tmp |= _spi->transfer(adc_cmd[i + 1]);
163  _count[i] = 1;
164  _sum[i] = adc_tmp;
165  }
166 
167  _spi->cs_release();
168 
169  _spi_sem->give();
170 
171  _ch6_last_sample_time_micros = hal.scheduler->micros();
172 
173  hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_ADC_ADS7844::read, void));
174  hal.scheduler->resume_timer_procs();
175 
176 }

178 // Read one channel value
179 float AP_ADC_ADS7844::Ch(uint8_t ch_num)
180 {
181  uint16_t count;
182  uint32_t sum;
183 
184  // ensure we have at least one value
185  while (_count[ch_num] == 0) /* noop */;
186 
187  // grab the value with timer procs disabled, and clear the count
188  hal.scheduler->suspend_timer_procs();
189  count = _count[ch_num];
190  sum = _sum[ch_num];
191  _count[ch_num] = 0;
192  _sum[ch_num] = 0;
193  hal.scheduler->resume_timer_procs();
194 
195  return ((float)sum)/count;
196 }

212 // Read 6 channel values
213 // this assumes that the counts for all of the 6 channels are
214 // equal. This will only be true if we always consistently access a
215 // sensor by either Ch6() or Ch() and never mix them. If you mix them
216 // then you will get very strange results
217 uint32_t AP_ADC_ADS7844::Ch6(const uint8_t *channel_numbers, float *result)
218 {
219  uint16_t count[6];
220  uint32_t sum[6];
221  uint8_t i;
222 
223  // ensure we have at least one value
224  for (i=0; i<6; i++) {
225  while (_count[channel_numbers[i]] == 0) /* noop */;
226  }
227 
228  // grab the values with timer procs disabled, and clear the counts
229  hal.scheduler->suspend_timer_procs();
230  for (i=0; i<6; i++) {
231  count[i] = _count[channel_numbers[i]];
232  sum[i] = _sum[channel_numbers[i]];
233  _count[channel_numbers[i]] = 0;
234  _sum[channel_numbers[i]] = 0;
235  }
236 
237  // calculate the delta time.
238  // we do this before re-enabling interrupts because another sensor read could fire immediately and change the _last_sensor_time value
239  uint32_t ret = _ch6_last_sample_time_micros - _ch6_delta_time_start_micros;
240  _ch6_delta_time_start_micros = _ch6_last_sample_time_micros;
241 
242  hal.scheduler->resume_timer_procs();
243 
244  // calculate averages. We keep this out of the cli region
245  // to prevent us stalling the ISR while doing the
246  // division. That costs us 36 bytes of stack, but I think its
247  // worth it.
248  for (i = 0; i < 6; i++) {
249  result[i] = sum[i] / (float)count[i];
250  }
251 
252  // return number of microseconds since last call
253  return ret;
254 }
