47 void AP_Mount_Backend::control(int32_t pitch_or_lat, int32_t roll_or_lon, int32_t yaw_or_alt, MAV_MOUNT_MODE mount_mode)
48 {
49  _frontend.set_mode(_instance, mount_mode);
50 
51  // interpret message fields based on mode
52  switch (_frontend.get_mode(_instance)) {
53  case MAV_MOUNT_MODE_RETRACT:
54  case MAV_MOUNT_MODE_NEUTRAL:
55  // do nothing with request if mount is retracted or in neutral position
56  break;
57 
58  // set earth frame target angles from mavlink message
59  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
60  set_angle_targets(roll_or_lon*0.01f, pitch_or_lat*0.01f, yaw_or_alt*0.01f);
61  break;
62 
63  // Load neutral position and start RC Roll,Pitch,Yaw control with stabilization
64  case MAV_MOUNT_MODE_RC_TARGETING:
65  // do nothing if pilot is controlling the roll, pitch and yaw
66  break;
67 
68  // set lat, lon, alt position targets from mavlink message
69  case MAV_MOUNT_MODE_GPS_POINT:
70  Location target_location;
71  memset(&target_location, 0, sizeof(target_location));
72  target_location.lat = pitch_or_lat;
73  target_location.lng = roll_or_lon;
74  target_location.alt = yaw_or_alt;
75  target_location.flags.relative_alt = true;
76  set_roi_target(target_location);
77  break;
78 
79  default:
80  // do nothing
81  break;
82  }
83 }

85 // update_targets_from_rc - updates angle targets using input from receiver
86 void AP_Mount_Backend::update_targets_from_rc()
87 {
88 #define rc_ch(i) RC_Channel::rc_channel(i-1)
89 
90  uint8_t roll_rc_in = _state._roll_rc_in;
91  uint8_t tilt_rc_in = _state._tilt_rc_in;
92  uint8_t pan_rc_in = _state._pan_rc_in;
93 
94  // if joystick_speed is defined then pilot input defines a rate of change of the angle
95  if (_frontend._joystick_speed) {
96  // allow pilot speed position input to come directly from an RC_Channel
97  if (roll_rc_in && rc_ch(roll_rc_in)) {
98  _angle_ef_target_rad.x += rc_ch(roll_rc_in)->norm_input_dz() * 0.0001f * _frontend._joystick_speed;
99  _angle_ef_target_rad.x = constrain_float(_angle_ef_target_rad.x, radians(_state._roll_angle_min*0.01f), radians(_state._roll_angle_max*0.01f));
100  }
101  if (tilt_rc_in && (rc_ch(tilt_rc_in))) {
102  _angle_ef_target_rad.y += rc_ch(tilt_rc_in)->norm_input_dz() * 0.0001f * _frontend._joystick_speed;
103  _angle_ef_target_rad.y = constrain_float(_angle_ef_target_rad.y, radians(_state._tilt_angle_min*0.01f), radians(_state._tilt_angle_max*0.01f));
104  }
105  if (pan_rc_in && (rc_ch(pan_rc_in))) {
106  _angle_ef_target_rad.z += rc_ch(pan_rc_in)->norm_input_dz() * 0.0001f * _frontend._joystick_speed;
107  _angle_ef_target_rad.z = constrain_float(_angle_ef_target_rad.z, radians(_state._pan_angle_min*0.01f), radians(_state._pan_angle_max*0.01f));
108  }
109  } else {
110  // allow pilot position input to come directly from an RC_Channel
111  if (roll_rc_in && (rc_ch(roll_rc_in))) {
112  _angle_ef_target_rad.x = angle_input_rad(rc_ch(roll_rc_in), _state._roll_angle_min, _state._roll_angle_max);
113  }
114  if (tilt_rc_in && (rc_ch(tilt_rc_in))) {
115  _angle_ef_target_rad.y = angle_input_rad(rc_ch(tilt_rc_in), _state._tilt_angle_min, _state._tilt_angle_max);
116  }
117  if (pan_rc_in && (rc_ch(pan_rc_in))) {
118  _angle_ef_target_rad.z = angle_input_rad(rc_ch(pan_rc_in), _state._pan_angle_min, _state._pan_angle_max);
119  }

135 // calc_angle_to_location - calculates the earth-frame roll, tilt and pan angles (and radians) to point at the given target
136 void AP_Mount_Backend::calc_angle_to_location(const struct Location &target, Vector3f& angles_to_target_rad, bool calc_tilt, bool calc_pan)
137 {
138  float GPS_vector_x = (target.lng-_frontend._current_loc.lng)*cosf(ToRad((_frontend._current_loc.lat+target.lat)*0.00000005f))*0.01113195f;
139  float GPS_vector_y = (target.lat-_frontend._current_loc.lat)*0.01113195f;
140  float GPS_vector_z = (target.alt-_frontend._current_loc.alt); // baro altitude(IN CM) should be adjusted to known home elevation before take off (Set altimeter).
141  float target_distance = 100.0f*pythagorous2(GPS_vector_x, GPS_vector_y); // Careful , centimeters here locally. Baro/alt is in cm, lat/lon is in meters.
142 
143  // initialise all angles to zero
144  angles_to_target_rad.zero();
145 
146  // tilt calcs
147  if (calc_tilt) {
148  angles_to_target_rad.y = atan2f(GPS_vector_z, target_distance);
149  }
150 
151  // pan calcs
152  if (calc_pan) {
153  // calc absolute heading and then onvert to vehicle relative yaw
154  angles_to_target_rad.z = wrap_PI(atan2f(GPS_vector_x, GPS_vector_y) - _frontend._ahrs.yaw);
155  }
156 }
