74 // update - updates device according to timed_updated. Should be
75 // called at 50Hz
76 void OreoLED_PX4::update()
77 {
78  static uint8_t counter = 0; // counter to reduce rate from 50hz to 10hz
79  static uint8_t step = 0; // step to control pattern
80  static uint8_t last_stage = 0; // unique id of the last messages sent to the LED, used to reduce resends which disrupt some patterns
81  static uint8_t initialization_done = 0; // Keep track if initialization has begun. There is a period when the driver
82  // is running but initialization has not yet begun -- this prevents post-initialization
83  // LED patterns from displaying before initialization has completed.
84 
85  // return immediately if not healthy
86  if (!_overall_health) {
87  return;
88  }
89 
90  // handle firmware update event
91  if (AP_Notify::flags.firmware_update) {
92  // Force a syncronisation before setting the free-running colour cycle macro
93  send_sync();
94  set_macro(OREOLED_INSTANCE_ALL, OREOLED_PARAM_MACRO_COLOUR_CYCLE);
95  return;
96  }
97 
98  // return immediately if custom pattern has been sent
99  if (OreoLED_PX4::_pattern_override != 0) {
100  // reset stage so patterns will be resent once override clears
101  last_stage = 0;
102  return;
103  }
104 
105  // slow rate from 50Hz to 10hz
106  counter++;
107  if (counter < 5) {
108  return;
109  }
110  counter = 0;
111 
112  // move forward one step
113  step++;
114  if (step >= 10) {
115  step = 0;
116  }
117 
118  // Pre-initialization pattern is all solid green
119  if (!initialization_done) {
120  set_rgb(OREOLED_ALL_INSTANCES, 0, OREOLED_BRIGHT, 0);
121  }
122 
123  // initialising pattern
124  if (AP_Notify::flags.initialising) {
125  initialization_done = 1; // Record initialization has begun
126  last_stage = 1; // record stage
127 
128  // exit so no other status modify this pattern
129  return;
130  }
131 
132  // save trim and esc calibration pattern
133  if (AP_Notify::flags.save_trim || AP_Notify::flags.esc_calibration) {
134  switch(step) {
135  case 0:
136  case 3:
137  case 6:
138  // red
139  set_rgb(OREOLED_INSTANCE_ALL, OREOLED_BRIGHT, 0, 0);
140  break;
141 
142  case 1:
143  case 4:
144  case 7:
145  // blue
146  set_rgb(OREOLED_INSTANCE_ALL, 0, 0, OREOLED_BRIGHT);
147  break;
148 
149  case 2:
150  case 5:
151  case 8:
152  // green on
153  set_rgb(OREOLED_INSTANCE_ALL, 0, OREOLED_BRIGHT, 0);
154  break;
155 
156  case 9:
157  // all off
158  set_rgb(OREOLED_INSTANCE_ALL, 0, 0, 0);
159  break;
160  }
161  // record stage
162  last_stage = 2;
163  // exit so no other status modify this pattern
164  return;
165  }
166 
167  // radio failsafe pattern: Alternate between front red/rear black and front black/rear red
168  if (AP_Notify::flags.failsafe_radio) {
169  switch(step) {
170  case 0:
171  case 1:
172  case 2:
173  case 3:
174  case 4:
175  // Front red/rear black
176  set_rgb(OREOLED_FRONTLEFT, OREOLED_BRIGHT, 0, 0);
177  set_rgb(OREOLED_FRONTRIGHT, OREOLED_BRIGHT, 0, 0);
178  set_rgb(OREOLED_BACKLEFT, 0, 0, 0);
179  set_rgb(OREOLED_BACKRIGHT, 0, 0, 0);
180  break;
181  case 5:
182  case 6:
183  case 7:
184  case 8:
185  case 9:
186  // Front black/rear red
187  set_rgb(OREOLED_FRONTLEFT, 0, 0, 0);
188  set_rgb(OREOLED_FRONTRIGHT, 0, 0, 0);
189  set_rgb(OREOLED_BACKLEFT, OREOLED_BRIGHT, 0, 0);
190  set_rgb(OREOLED_BACKRIGHT, OREOLED_BRIGHT, 0, 0);
191  break;
192  }
193  // record stage
194  last_stage = 3;
195  // exit so no other status modify this pattern
196  return;
197  }
198 
199  // send colours (later we will set macro if required)
200  if (last_stage < 10 && initialization_done) {
201  set_macro(OREOLED_INSTANCE_ALL, OREOLED_PARAM_MACRO_AUTOMOBILE);
202  last_stage = 10;
203  } else if (last_stage >= 10) {
204  static uint8_t previous_autopilot_mode = -1;
205  if (previous_autopilot_mode != AP_Notify::flags.autopilot_mode) {
206 
207  if (AP_Notify::flags.autopilot_mode) {
208  // autopilot flight modes start breathing macro
209  set_macro(OREOLED_INSTANCE_ALL, OREOLED_PARAM_MACRO_AUTOMOBILE);
210  set_macro(OREOLED_INSTANCE_ALL, OREOLED_PARAM_MACRO_BREATH);
211  } else {
212  // manual flight modes stop breathing -- solid color
213  set_macro(OREOLED_INSTANCE_ALL, OREOLED_PARAM_MACRO_AUTOMOBILE);
214  }
215 
216  // record we have processed this change
217  previous_autopilot_mode = AP_Notify::flags.autopilot_mode;
218  }
219  last_stage = 11;
220  }
221 }

223 // set_rgb - set color as a combination of red, green and blue values for one or all LEDs
224 void OreoLED_PX4::set_rgb(uint8_t instance, uint8_t red, uint8_t green, uint8_t blue)
225 {
226  // return immediately if no healty leds
227  if (!_overall_health) {
228  return;
229  }
230 
231  // get semaphore
232  _state_desired_semaphore = true;
233 
234  // check for all instances
235  if (instance == OREOLED_INSTANCE_ALL) {
236  // store desired rgb for all LEDs
237  for (uint8_t i=0; i<OREOLED_NUM_LEDS; i++) {
238  if (_state_desired[i].mode != OREOLED_MODE_RGB || _state_desired[i].red != red || _state_desired[i].green != green || _state_desired[i].blue != blue) {
239  _state_desired[i].mode = OREOLED_MODE_RGB;
240  _state_desired[i].red = red;
241  _state_desired[i].green = green;
242  _state_desired[i].blue = blue;
243  _send_required = true;
244  }
245  }
246  } else if (instance < OREOLED_NUM_LEDS) {
247  // store desired rgb for one LED
248  if (_state_desired[instance].mode != OREOLED_MODE_RGB || _state_desired[instance].red != red || _state_desired[instance].green != green || _state_desired[instance].blue != blue) {
249  _state_desired[instance].mode = OREOLED_MODE_RGB;
250  _state_desired[instance].red = red;
251  _state_desired[instance].green = green;
252  _state_desired[instance].blue = blue;
253  _send_required = true;
254  }
255  }
256 
257  // release semaphore
258  _state_desired_semaphore = false;
259 }

317 // update_timer - called by scheduler and updates PX4 driver with commands
318 void OreoLED_PX4::update_timer(void)
319 {
320  // exit immediately if unhealthy
321  if (!_overall_health) {
322  return;
323  }
324 
325  // exit immediately if send not required, or state is being updated
326  if (!_send_required || _state_desired_semaphore) {
327  return;
328  }
329 
330  // for each LED
331  for (uint8_t i=0; i<OREOLED_NUM_LEDS; i++) {
332 
333  // check for state change
334  if (!(_state_desired[i] == _state_sent[i])) {
335  switch (_state_desired[i].mode) {
336  case OREOLED_MODE_PATTERN:
337  // not yet supported
338  break;
339  case OREOLED_MODE_MACRO:
340  {
341  oreoled_macrorun_t macro_run = {i, _state_desired[i].macro};
342  ioctl(_oreoled_fd, OREOLED_RUN_MACRO, (unsigned long)&macro_run);
343  }
344  break;
345  case OREOLED_MODE_RGB:
346  {
347  oreoled_rgbset_t rgb_set = {i, OREOLED_PATTERN_SOLID, _state_desired[i].red, _state_desired[i].green, _state_desired[i].blue};
348  ioctl(_oreoled_fd, OREOLED_SET_RGB, (unsigned long)&rgb_set);
349  }
350  break;
351  case OREOLED_MODE_SYNC:
352  {
353  ioctl(_oreoled_fd, OREOLED_FORCE_SYNC, 0);
354  }
355  break;
356  }
357  // save state change
358  _state_sent[i] = _state_desired[i];
359  }
360  }
361 
362  // flag updates sent
363  _send_required = false;
364 }

366 // handle a LED_CONTROL message
367 void OreoLED_PX4::handle_led_control(mavlink_message_t *msg)
368 {
369  // exit immediately if unhealthy
370  if (!_overall_health) {
371  return;
372  }
373 
374  // decode mavlink message
375  mavlink_led_control_t packet;
376  mavlink_msg_led_control_decode(msg, &packet);
377 
378  // exit immediately if instance is invalid
379  if (packet.instance >= OREOLED_NUM_LEDS && packet.instance != OREOLED_INSTANCE_ALL) {
380  return;
381  }
382 
383  // if pattern is OFF, we clear pattern override so normal lighting should resume
384  if (packet.pattern == LED_CONTROL_PATTERN_OFF) {
385  _pattern_override = 0;
386  return;
387  }
388 
389  // custom patterns not implemented
390  if (packet.pattern == LED_CONTROL_PATTERN_CUSTOM) {
391  return;
392  }
393 
394  // other patterns sent as macro
395  set_macro(packet.instance, (oreoled_macro)packet.pattern);
396  _pattern_override = packet.pattern;
397 }
