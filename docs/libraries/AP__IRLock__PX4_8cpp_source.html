49 // retrieve latest sensor data - returns true if new data is available
50 bool AP_IRLock_PX4::update()
51 {
52  // return immediately if not healthy
53  if (!_flags.healthy) {
54  return false;
55  }
56 
57  // read position of all objects
58  struct irlock_s report;
59  uint16_t count = 0;
60  while(::read(_fd, &report, sizeof(struct irlock_s)) == sizeof(struct irlock_s) && report.timestamp >_last_timestamp) {
61  _target_info[count].timestamp = report.timestamp;
62  _target_info[count].target_num = report.target_num;
63  _target_info[count].angle_x = report.angle_x;
64  _target_info[count].angle_y = report.angle_y;
65  _target_info[count].size_x = report.size_x;
66  _target_info[count].size_y = report.size_y;
67  count++;
68  _last_timestamp = report.timestamp;
69  _last_update = hal.scheduler->millis();
70  }
71 
72  // update num_blocks and implement timeout
73  if (count > 0) {
74  _num_targets = count;
75  } else if ((hal.scheduler->millis() - _last_update) > IRLOCK_TIMEOUT_MS) {
76  _num_targets = 0;
77  }
78 
79  // return true if new data found
80  return (_num_targets > 0);
81 }
