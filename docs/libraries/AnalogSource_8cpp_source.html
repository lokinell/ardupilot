105 /* read_average is called from the normal thread (not an interrupt). */
106 float FLYMAPLEAnalogSource::_read_average()
107 {
108  uint16_t sum;
109  uint8_t sum_count;
110 
111  if (_sum_count == 0) {
112  // avoid blocking waiting for new samples
113  return _last_average;
114  }
115 
116  /* Read and clear in a critical section */
117  noInterrupts();
118  sum = _sum;
119  sum_count = _sum_count;
120  _sum = 0;
121  _sum_count = 0;
122 
123  interrupts();
124  float avg = sum / (float) sum_count;
125 
126  _last_average = avg;
127  return avg;
128 }

130 void FLYMAPLEAnalogSource::setup_read() {
131  if (_stop_pin != ANALOG_INPUT_NONE) {
132  uint8_t digital_pin = hal.gpio->analogPinToDigitalPin(_stop_pin);
133  hal.gpio->pinMode(digital_pin, HAL_GPIO_OUTPUT);
134  hal.gpio->write(digital_pin, 1);
135  }
136  if (_settle_time_ms != 0) {
137  _read_start_time_ms = hal.scheduler->millis();
138  }
139  adc_reg_map *regs = ADC1->regs;
140  adc_set_reg_seqlen(ADC1, 1);
141  uint8 channel = 0;
142  if (_pin == ANALOG_INPUT_BOARD_VCC)
143  channel = PIN_MAP[FLYMAPLE_VCC_ANALOG_IN_PIN].adc_channel;
144  else if (_pin == ANALOG_INPUT_NONE)
145  ; // NOOP
146  else
147  channel = PIN_MAP[_pin].adc_channel;
148  regs->SQR3 = channel;
149 }

167 /* new_sample is called from an interrupt. It always has access to
168  * _sum and _sum_count. Lock out the interrupts briefly with
169  * noInterrupts()/interrupts() to read these variables from outside an interrupt. */
170 void FLYMAPLEAnalogSource::new_sample(uint16_t sample) {
171  _sum += sample;
172  _latest = sample;
173  if (_sum_count >= 15) { // Flymaple has a 12 bit ADC, so can only sum 16 in a uint16_t
174  _sum >>= 1;
175  _sum_count = 8;
176  } else {
177  _sum_count++;
178  }
179 }
