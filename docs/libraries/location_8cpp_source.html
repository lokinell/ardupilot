33 float longitude_scale(const struct Location &loc)
34 {
35 #if HAL_CPU_CLASS < HAL_CPU_CLASS_150
36  static int32_t last_lat;
37  static float scale = 1.0;
38  // don't optimise on faster CPUs. It causes some minor errors on Replay
39  if (labs(last_lat - loc.lat) < 100000) {
40  // we are within 0.01 degrees (about 1km) of the
41  // same latitude. We can avoid the cos() and return
42  // the same scale factor.
43  return scale;
44  }
45  scale = cosf(loc.lat * 1.0e-7f * DEG_TO_RAD);
46  scale = constrain_float(scale, 0.01f, 1.0f);
47  last_lat = loc.lat;
48  return scale;
49 #else
50  float scale = cosf(loc.lat * 1.0e-7f * DEG_TO_RAD);
51  return constrain_float(scale, 0.01f, 1.0f);
52 #endif
53 }

94 /*
95  return the proportion we are along the path from point1 to
96  point2, along a line parallel to point1<->point2.
97 
98  This will be less than >1 if we have passed point2
99  */
100 float location_path_proportion(const struct Location &location,
101  const struct Location &point1,
102  const struct Location &point2)
103 {
104  Vector2f vec1 = location_diff(point1, point2);
105  Vector2f vec2 = location_diff(point1, location);
106  float dsquared = sq(vec1.x) + sq(vec1.y);
107  if (dsquared < 0.001f) {
108  // the two points are very close together
109  return 1.0f;
110  }
111  return (vec1 * vec2) / dsquared;
112 }

114 /*
115  * extrapolate latitude/longitude given bearing and distance
116  * Note that this function is accurate to about 1mm at a distance of
117  * 100m. This function has the advantage that it works in relative
118  * positions, so it keeps the accuracy even when dealing with small

121 void location_update(struct Location &loc, float bearing, float distance)
122 {

249 void wgsllh2ecef(const Vector3d &llh, Vector3d &ecef) {
250  double d = WGS84_E * sin(llh[0]);
251  double N = WGS84_A / sqrt(1. - d*d);
252 
253  ecef[0] = (N + llh[2]) * cos(llh[0]) * cos(llh[1]);
254  ecef[1] = (N + llh[2]) * cos(llh[0]) * sin(llh[1]);
255  ecef[2] = ((1 - WGS84_E*WGS84_E)*N + llh[2]) * sin(llh[0]);
256 }

259 void wgsecef2llh(const Vector3d &ecef, Vector3d &llh) {
260  /* Distance from polar axis. */
261  const double p = sqrt(ecef[0]*ecef[0] + ecef[1]*ecef[1]);
262 
263  /* Compute longitude first, this can be done exactly. */
264  if (!is_zero(p))
265  llh[1] = atan2(ecef[1], ecef[0]);
266  else
267  llh[1] = 0;
268 
269  /* If we are close to the pole then convergence is very slow, treat this is a
270  * special case. */
271  if (p < WGS84_A*1e-16) {
272  llh[0] = copysign(M_PI_2, ecef[2]);
273  llh[2] = fabs(ecef[2]) - WGS84_B;
274  return;
275  }
276 
277  /* Caluclate some other constants as defined in the Fukushima paper. */
278  const double P = p / WGS84_A;
279  const double e_c = sqrt(1. - WGS84_E*WGS84_E);
280  const double Z = fabs(ecef[2]) * e_c / WGS84_A;
281 
282  /* Initial values for S and C correspond to a zero height solution. */
283  double S = Z;
284  double C = e_c * P;
285 
286  /* Neither S nor C can be negative on the first iteration so
287  * starting prev = -1 will not cause and early exit. */
288  double prev_C = -1;
289  double prev_S = -1;
290 
291  double A_n, B_n, D_n, F_n;
292 
293  /* Iterate a maximum of 10 times. This should be way more than enough for all
294  * sane inputs */
295  for (int i=0; i<10; i++)
296  {
297  /* Calculate some intermmediate variables used in the update step based on
298  * the current state. */
299  A_n = sqrt(S*S + C*C);
300  D_n = Z*A_n*A_n*A_n + WGS84_E*WGS84_E*S*S*S;
301  F_n = P*A_n*A_n*A_n - WGS84_E*WGS84_E*C*C*C;
302  B_n = 1.5*WGS84_E*S*C*C*(A_n*(P*S - Z*C) - WGS84_E*S*C);
303 
304  /* Update step. */
305  S = D_n*F_n - B_n*S;
306  C = F_n*F_n - B_n*C;
307 
308  /* The original algorithm as presented in the paper by Fukushima has a
309  * problem with numerical stability. S and C can grow very large or small
310  * and over or underflow a double. In the paper this is acknowledged and
311  * the proposed resolution is to non-dimensionalise the equations for S and
312  * C. However, this does not completely solve the problem. The author caps
313  * the solution to only a couple of iterations and in this period over or
314  * underflow is unlikely but as we require a bit more precision and hence
315  * more iterations so this is still a concern for us.
316  *
317  * As the only thing that is important is the ratio T = S/C, my solution is
318  * to divide both S and C by either S or C. The scaling is chosen such that
319  * one of S or C is scaled to unity whilst the other is scaled to a value
320  * less than one. By dividing by the larger of S or C we ensure that we do
321  * not divide by zero as only one of S or C should ever be zero.
322  *
323  * This incurs an extra division each iteration which the author was
324  * explicityl trying to avoid and it may be that this solution is just
325  * reverting back to the method of iterating on T directly, perhaps this
326  * bears more thought?
327  */
328 
329  if (S > C) {
330  C = C / S;
331  S = 1;
332  } else {
333  S = S / C;
334  C = 1;
335  }
336 
337  /* Check for convergence and exit early if we have converged. */
338  if (fabs(S - prev_S) < 1e-16 && fabs(C - prev_C) < 1e-16) {
339  break;
340  } else {
341  prev_S = S;
342  prev_C = C;
343  }
344  }
345 
346  A_n = sqrt(S*S + C*C);
347  llh[0] = copysign(1.0, ecef[2]) * atan(S / (e_c*C));
348  llh[2] = (p*e_c*C + fabs(ecef[2])*S - WGS84_A*e_c*A_n) / sqrt(e_c*e_c*C*C + S*S);
349 }
