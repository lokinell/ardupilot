32  convert airspeed in m/s to an airspeed sensor value
33  */
34 uint16_t SITL_State::_airspeed_sensor(float airspeed)
35 {
36  const float airspeed_ratio = 1.9936f;
37  const float airspeed_offset = 2013;
38  float airspeed_pressure, airspeed_raw;
39 
40  airspeed_pressure = (airspeed*airspeed) / airspeed_ratio;
41  airspeed_raw = airspeed_pressure + airspeed_offset;
42  if (airspeed_raw/4 > 0xFFFF) {
43  return 0xFFFF;
44  }
45  // add delay
46  uint32_t now = hal.scheduler->millis();
47  uint32_t best_time_delta_wind = 200; // initialise large time representing buffer entry closest to current time - delay.
48  uint8_t best_index_wind = 0; // initialise number representing the index of the entry in buffer closest to delay.
49 
50  // storing data from sensor to buffer
51  if (now - last_store_time_wind >= 10) { // store data every 10 ms.
52  last_store_time_wind = now;
53  if (store_index_wind > wind_buffer_length-1) { // reset buffer index if index greater than size of buffer
54  store_index_wind = 0;
55  }
56  buffer_wind[store_index_wind].data = airspeed_raw; // add data to current index
57  buffer_wind[store_index_wind].time = last_store_time_wind; // add time to current index
58  store_index_wind = store_index_wind + 1; // increment index
59  }
60 
61  // return delayed measurement
62  delayed_time_wind = now - _sitl->wind_delay; // get time corresponding to delay
63  // find data corresponding to delayed time in buffer
64  for (uint8_t i=0; i<=wind_buffer_length-1; i++) {
65  time_delta_wind = abs(delayed_time_wind - buffer_wind[i].time); // find difference between delayed time and time stamp in buffer
66  // if this difference is smaller than last delta, store this time
67  if (time_delta_wind < best_time_delta_wind) {
68  best_index_wind = i;
69  best_time_delta_wind = time_delta_wind;
70  }
71  }
72  if (best_time_delta_wind < 200) { // only output stored state if < 200 msec retrieval error
73  airspeed_raw = buffer_wind[best_index_wind].data;
74  }
75 
76  return airspeed_raw/4;
77 }

80 float SITL_State::_gyro_drift(void)
81 {
82  if (_sitl->drift_speed == 0.0f ||
83  _sitl->drift_time == 0.0f) {
84  return 0;
85  }
86  double period = _sitl->drift_time * 2;
87  double minutes = fmod(_scheduler->_micros64() / 60.0e6, period);
88  if (minutes < period/2) {
89  return minutes * ToRad(_sitl->drift_speed);
90  }
91  return (period - minutes) * ToRad(_sitl->drift_speed);

98 uint16_t SITL_State::_ground_sonar(void)
99 {
100  float altitude = height_agl();
101 
102  float voltage = 5.0f;
103  if (fabsf(_sitl->state.rollDeg) < 90 &&
104  fabsf(_sitl->state.pitchDeg) < 90) {
105  // adjust for apparent altitude with roll
106  altitude /= cosf(radians(_sitl->state.rollDeg)) * cosf(radians(_sitl->state.pitchDeg));
107 
108  altitude += _sitl->sonar_noise * _rand_float();
109 
110  // Altitude in in m, scaler in meters/volt
111  voltage = altitude / _sitl->sonar_scale;
112  voltage = constrain_float(voltage, 0, 5.0f);
113 
114  if (_sitl->sonar_glitch >= (_rand_float() + 1.0f)/2.0f) {
115  voltage = 5.0f;
116  }
117  }
118 
119  return 1023*(voltage / 5.0f);
120 }
