58  if (_init_blob_index < (sizeof(_initialisation_blob) / sizeof(_initialisation_blob[0]))) {

75 bool
76 AP_GPS_SBF::parse(uint8_t temp)
77 {
78  switch (sbf_msg.sbf_state)
79  {
80  default:
81  case sbf_msg_parser_t::PREAMBLE1:
82  if (temp == SBF_PREAMBLE1) {
83  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE2;
84  sbf_msg.read = 0;
85  }
86  break;
87  case sbf_msg_parser_t::PREAMBLE2:
88  if (temp == SBF_PREAMBLE2) {
89  sbf_msg.sbf_state = sbf_msg_parser_t::CRC1;
90  }
91  else
92  {
93  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
94  }
95  break;
96  case sbf_msg_parser_t::CRC1:
97  sbf_msg.crc = temp;
98  sbf_msg.sbf_state = sbf_msg_parser_t::CRC2;
99  break;
100  case sbf_msg_parser_t::CRC2:
101  sbf_msg.crc += (uint16_t)(temp << 8);
102  sbf_msg.sbf_state = sbf_msg_parser_t::BLOCKID1;
103  break;
104  case sbf_msg_parser_t::BLOCKID1:
105  sbf_msg.blockid = temp;
106  sbf_msg.sbf_state = sbf_msg_parser_t::BLOCKID2;
107  break;
108  case sbf_msg_parser_t::BLOCKID2:
109  sbf_msg.blockid += (uint16_t)(temp << 8);
110  sbf_msg.sbf_state = sbf_msg_parser_t::LENGTH1;
111  break;
112  case sbf_msg_parser_t::LENGTH1:
113  sbf_msg.length = temp;
114  sbf_msg.sbf_state = sbf_msg_parser_t::LENGTH2;
115  break;
116  case sbf_msg_parser_t::LENGTH2:
117  sbf_msg.length += (uint16_t)(temp << 8);
118  sbf_msg.sbf_state = sbf_msg_parser_t::DATA;
119  if (sbf_msg.length % 4 != 0) {
120  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
121  Debug("bad packet length=%u\n", (unsigned)sbf_msg.length);
122  }
123  break;
124  case sbf_msg_parser_t::DATA:
125  if (sbf_msg.read >= sizeof(sbf_msg.data)) {
126  Debug("parse overflow length=%u\n", (unsigned)sbf_msg.read);
127  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
128  break;
129  }
130  sbf_msg.data.bytes[sbf_msg.read] = temp;
131  sbf_msg.read++;
132  if (sbf_msg.read >= (sbf_msg.length - 8)) {
133  uint16_t crc = crc16_ccitt((uint8_t*)&sbf_msg.blockid, 2, 0);
134  crc = crc16_ccitt((uint8_t*)&sbf_msg.length, 2, crc);
135  crc = crc16_ccitt((uint8_t*)&sbf_msg.data, sbf_msg.length - 8, crc);
136 
137  sbf_msg.sbf_state = sbf_msg_parser_t::PREAMBLE1;
138 
139  if (sbf_msg.crc == crc) {
140  return process_message();
141  } else {
142  Debug("crc fail\n");
143  crc_error_counter++;
144  }
145  }
146  break;
147  }
148 
149  return false;
150 }

181 bool
182 AP_GPS_SBF::process_message(void)
183 {
184  uint16_t blockid = (sbf_msg.blockid & 4095u);
185 
186  Debug("BlockID %d", blockid);
187 
188  // ExtEventPVTGeodetic
189  if (blockid == 4038) {
190  log_ExtEventPVTGeodetic(sbf_msg.data.msg4007u);
191  }
192  // PVTGeodetic
193  if (blockid == 4007) {
194  const msg4007 &temp = sbf_msg.data.msg4007u;
195 
196  // Update time state
197  if (temp.WNc != 65535) {
198  state.time_week = temp.WNc;
199  state.time_week_ms = (uint32_t)(temp.TOW);
200  }
201 
202  state.last_gps_time_ms = hal.scheduler->millis();
203 
204  state.hdop = last_hdop;
205 
206  // Update velocity state (dont use 210^10)
207  if (temp.Vn > -200000) {
208  state.velocity.x = (float)(temp.Vn);
209  state.velocity.y = (float)(temp.Ve);
210  state.velocity.z = (float)(-temp.Vu);
211 
212  state.have_vertical_velocity = true;
213 
214  float ground_vector_sq = state.velocity[0] * state.velocity[0] + state.velocity[1] * state.velocity[1];
215  state.ground_speed = (float)safe_sqrt(ground_vector_sq);
216 
217  state.ground_course_cd = (int32_t)(100 * ToDeg(atan2f(state.velocity[1], state.velocity[0])));
218  state.ground_course_cd = wrap_360_cd(state.ground_course_cd);
219 
220  state.horizontal_accuracy = (float)temp.HAccuracy * 0.01f;
221  state.vertical_accuracy = (float)temp.VAccuracy * 0.01f;
222  state.have_horizontal_accuracy = true;
223  state.have_vertical_accuracy = true;
224  }
225 
226  // Update position state (dont use 210^10)
227  if (temp.Latitude > -200000) {
228  state.location.lat = (int32_t)(temp.Latitude * RAD_TO_DEG_DOUBLE * 1e7);
229  state.location.lng = (int32_t)(temp.Longitude * RAD_TO_DEG_DOUBLE * 1e7);
230  state.location.alt = (int32_t)((float)temp.Height * 1e2f);
231  }
232 
233  if (temp.NrSV != 255) {
234  state.num_sats = temp.NrSV;
235  }
236 
237  Debug("temp.Mode=0x%02x\n", (unsigned)temp.Mode);
238  switch (temp.Mode & 15) {
239  case 0: // no pvt
240  state.status = AP_GPS::NO_FIX;
241  break;
242  case 1: // standalone
243  state.status = AP_GPS::GPS_OK_FIX_3D;
244  break;
245  case 2: // dgps
246  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
247  break;
248  case 3: // fixed location
249  state.status = AP_GPS::GPS_OK_FIX_3D;
250  break;
251  case 4: // rtk fixed
252  state.status = AP_GPS::GPS_OK_FIX_3D_RTK;
253  break;
254  case 5: // rtk float
255  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
256  break;
257  case 6: // sbas
258  state.status = AP_GPS::GPS_OK_FIX_3D;
259  break;
260  case 7: // moving rtk fixed
261  state.status = AP_GPS::GPS_OK_FIX_3D_RTK;
262  break;
263  case 8: // moving rtk float
264  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
265  break;
266  }
267 
268  if ((temp.Mode & 64) > 0) // gps is in base mode
269  state.status = AP_GPS::NO_FIX;
270  if ((temp.Mode & 128) > 0) // gps only has 2d fix
271  state.status = AP_GPS::GPS_OK_FIX_2D;
272 
273  return true;
274  }
275  // DOP
276  if (blockid == 4001) {
277  const msg4001 &temp = sbf_msg.data.msg4001u;
278 
279  last_hdop = temp.HDOP;
280 
281  state.hdop = last_hdop;
282  }
283 
284  return false;
285 }
