97 /*
98  update state for MAV_CMD_DO_REPEAT_SERVO and MAV_CMD_DO_REPEAT_RELAY
99 */
100 void AP_ServoRelayEvents::update_events(void)
101 {
102  if (repeat == 0 || (hal.scheduler->millis() - start_time_ms) < delay_ms) {
103  return;
104  }
105 
106  start_time_ms = hal.scheduler->millis();
107 
108  switch (type) {
109  case EVENT_TYPE_SERVO:
110  hal.rcout->enable_ch(channel-1);
111  if (repeat & 1) {
112  hal.rcout->write(channel-1, RC_Channel::rc_channel(channel-1)->radio_trim);
113  } else {
114  hal.rcout->write(channel-1, servo_value);
115  }
116  break;
117 
118  case EVENT_TYPE_RELAY:
119  relay.toggle(channel);
120  break;
121  }
122 
123  if (repeat > 0) {
124  repeat--;
125  } else {
126  // toggle bottom bit so servos flip in value
127  repeat ^= 1;
128  }
129 }
