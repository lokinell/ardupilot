122 void AP_L1_Control::_prevent_indecision(float &Nu)
123 {
124  const float Nu_limit = 0.9f*M_PI_F;
125  if (fabsf(Nu) > Nu_limit &&
126  fabsf(_last_Nu) > Nu_limit &&
127  fabsf(wrap_180_cd(_target_bearing_cd - _ahrs.yaw_sensor)) > 12000 &&
128  Nu * _last_Nu < 0.0f) {
129  // we are moving away from the target waypoint and pointing
130  // away from the waypoint (not flying backwards). The sign
131  // of Nu has also changed, which means we are
132  // oscillating in our decision about which way to go
133  Nu = _last_Nu;
134  }

137 // update L1 control for waypoint navigation
138 void AP_L1_Control::update_waypoint(const struct Location &prev_WP, const struct Location &next_WP)
139 {
140 
141  struct Location _current_loc;
142  float Nu;
143  float xtrackVel;
144  float ltrackVel;
145 
146  // Calculate L1 gain required for specified damping
147  float K_L1 = 4.0f * _L1_damping * _L1_damping;
148 
149  // Get current position and velocity
150  _ahrs.get_position(_current_loc);
151 
152  Vector2f _groundspeed_vector = _ahrs.groundspeed_vector();
153 
154  // update _target_bearing_cd
155  _target_bearing_cd = get_bearing_cd(_current_loc, next_WP);
156 
157  //Calculate groundspeed
158  float groundSpeed = _groundspeed_vector.length();
159  if (groundSpeed < 0.1f) {
160  // use a small ground speed vector in the right direction,
161  // allowing us to use the compass heading at zero GPS velocity
162  groundSpeed = 0.1f;
163  _groundspeed_vector = Vector2f(cosf(_ahrs.yaw), sinf(_ahrs.yaw)) * groundSpeed;
164  }
165 
166  // Calculate time varying control parameters
167  // Calculate the L1 length required for specified period
168  // 0.3183099 = 1/1/pipi
169  _L1_dist = 0.3183099f * _L1_damping * _L1_period * groundSpeed;
170 
171  // Calculate the NE position of WP B relative to WP A
172  Vector2f AB = location_diff(prev_WP, next_WP);
173 
174  // Check for AB zero length and track directly to the destination
175  // if too small
176  if (AB.length() < 1.0e-6f) {
177  AB = location_diff(_current_loc, next_WP);
178  if (AB.length() < 1.0e-6f) {
179  AB = Vector2f(cosf(_ahrs.yaw), sinf(_ahrs.yaw));
180  }
181  }
182  AB.normalize();
183 
184  // Calculate the NE position of the aircraft relative to WP A
185  Vector2f A_air = location_diff(prev_WP, _current_loc);
186 
187  // calculate distance to target track, for reporting
188  _crosstrack_error = A_air % AB;
189 
190  //Determine if the aircraft is behind a +-135 degree degree arc centred on WP A
191  //and further than L1 distance from WP A. Then use WP A as the L1 reference point
192  //Otherwise do normal L1 guidance
193  float WP_A_dist = A_air.length();
194  float alongTrackDist = A_air * AB;
195  if (WP_A_dist > _L1_dist && alongTrackDist/max(WP_A_dist, 1.0f) < -0.7071f)
196  {
197  //Calc Nu to fly To WP A
198  Vector2f A_air_unit = (A_air).normalized(); // Unit vector from WP A to aircraft
199  xtrackVel = _groundspeed_vector % (-A_air_unit); // Velocity across line
200  ltrackVel = _groundspeed_vector * (-A_air_unit); // Velocity along line
201  Nu = atan2f(xtrackVel,ltrackVel);
202  _nav_bearing = atan2f(-A_air_unit.y , -A_air_unit.x); // bearing (radians) from AC to L1 point
203 
204  } else { //Calc Nu to fly along AB line
205 
206  //Calculate Nu2 angle (angle of velocity vector relative to line connecting waypoints)
207  xtrackVel = _groundspeed_vector % AB; // Velocity cross track
208  ltrackVel = _groundspeed_vector * AB; // Velocity along track
209  float Nu2 = atan2f(xtrackVel,ltrackVel);
210  //Calculate Nu1 angle (Angle to L1 reference point)
211  float sine_Nu1 = _crosstrack_error/max(_L1_dist, 0.1f);
212  //Limit sine of Nu1 to provide a controlled track capture angle of 45 deg
213  sine_Nu1 = constrain_float(sine_Nu1, -0.7071f, 0.7071f);
214  float Nu1 = asinf(sine_Nu1);
215 
216  // compute integral error component to converge to a crosstrack of zero when traveling
217  // straight but reset it when disabled or if it changes. That allows for much easier
218  // tuning by having it re-converge each time it changes.
219  if (_L1_xtrack_i_gain <= 0 || !is_equal(_L1_xtrack_i_gain, _L1_xtrack_i_gain_prev)) {
220  _L1_xtrack_i = 0;
221  _L1_xtrack_i_gain_prev = _L1_xtrack_i_gain;
222  } else if (fabsf(Nu1) < radians(5)) {
223 
224  const float dt = 0.1f; // 10Hz
225  _L1_xtrack_i += Nu1 * _L1_xtrack_i_gain * dt;
226 
227  // an AHRS_TRIM_X=0.1 will drift to about 0.08 so 0.1 is a good worst-case to clip at
228  _L1_xtrack_i = constrain_float(_L1_xtrack_i, -0.1f, 0.1f);
229  }
230 
231  // to converge to zero we must push Nu1 harder
232  Nu1 += _L1_xtrack_i;
233 
234  Nu = Nu1 + Nu2;
235  _nav_bearing = atan2f(AB.y, AB.x) + Nu1; // bearing (radians) from AC to L1 point
236  }
237 
238  _prevent_indecision(Nu);
239  _last_Nu = Nu;
240 
241  //Limit Nu to +-pi
242  Nu = constrain_float(Nu, -1.5708f, +1.5708f);
243  _latAccDem = K_L1 * groundSpeed * groundSpeed / _L1_dist * sinf(Nu);
244 
245  // Waypoint capture status is always false during waypoint following
246  _WPcircle = false;
247 
248  _bearing_error = Nu; // bearing error angle (radians), +ve to left of track
249 }

251 // update L1 control for loitering
252 void AP_L1_Control::update_loiter(const struct Location &center_WP, float radius, int8_t loiter_direction)
253 {
254  struct Location _current_loc;
255 
256  // scale loiter radius with square of EAS2TAS to allow us to stay
257  // stable at high altitude
258  radius *= sq(_ahrs.get_EAS2TAS());
259 
260  // Calculate guidance gains used by PD loop (used during circle tracking)
261  float omega = (6.2832f / _L1_period);
262  float Kx = omega * omega;
263  float Kv = 2.0f * _L1_damping * omega;
264 
265  // Calculate L1 gain required for specified damping (used during waypoint capture)
266  float K_L1 = 4.0f * _L1_damping * _L1_damping;
267 
268  //Get current position and velocity
269  _ahrs.get_position(_current_loc);
270 
271  Vector2f _groundspeed_vector = _ahrs.groundspeed_vector();
272 
273  //Calculate groundspeed
274  float groundSpeed = max(_groundspeed_vector.length() , 1.0f);
275 
276 
277  // update _target_bearing_cd
278  _target_bearing_cd = get_bearing_cd(_current_loc, center_WP);
279 
280 
281  // Calculate time varying control parameters
282  // Calculate the L1 length required for specified period
283  // 0.3183099 = 1/pi
284  _L1_dist = 0.3183099f * _L1_damping * _L1_period * groundSpeed;
285 
286  //Calculate the NE position of the aircraft relative to WP A
287  Vector2f A_air = location_diff(center_WP, _current_loc);
288 
289  // Calculate the unit vector from WP A to aircraft
290  // protect against being on the waypoint and having zero velocity
291  // if too close to the waypoint, use the velocity vector
292  // if the velocity vector is too small, use the heading vector
293  Vector2f A_air_unit;
294  if (A_air.length() > 0.1f) {
295  A_air_unit = A_air.normalized();
296  } else {
297  if (_groundspeed_vector.length() < 0.1f) {
298  A_air_unit = Vector2f(cosf(_ahrs.yaw), sinf(_ahrs.yaw));
299  } else {
300  A_air_unit = _groundspeed_vector.normalized();
301  }
302  }
303 
304  //Calculate Nu to capture center_WP
305  float xtrackVelCap = A_air_unit % _groundspeed_vector; // Velocity across line - perpendicular to radial inbound to WP
306  float ltrackVelCap = - (_groundspeed_vector * A_air_unit); // Velocity along line - radial inbound to WP
307  float Nu = atan2f(xtrackVelCap,ltrackVelCap);
308 
309  _prevent_indecision(Nu);
310  _last_Nu = Nu;
311 
312  Nu = constrain_float(Nu, -M_PI_2, M_PI_2); //Limit Nu to +- Pi/2
313 
314  //Calculate lat accln demand to capture center_WP (use L1 guidance law)
315  float latAccDemCap = K_L1 * groundSpeed * groundSpeed / _L1_dist * sinf(Nu);
316 
317  //Calculate radial position and velocity errors
318  float xtrackVelCirc = -ltrackVelCap; // Radial outbound velocity - reuse previous radial inbound velocity
319  float xtrackErrCirc = A_air.length() - radius; // Radial distance from the loiter circle
320 
321  // keep crosstrack error for reporting
322  _crosstrack_error = xtrackErrCirc;
323 
324  //Calculate PD control correction to circle waypoint_ahrs.roll
325  float latAccDemCircPD = (xtrackErrCirc * Kx + xtrackVelCirc * Kv);
326 
327  //Calculate tangential velocity
328  float velTangent = xtrackVelCap * float(loiter_direction);
329 
330  //Prevent PD demand from turning the wrong way by limiting the command when flying the wrong way
331  if (ltrackVelCap < 0.0f && velTangent < 0.0f) {
332  latAccDemCircPD = max(latAccDemCircPD, 0.0f);
333  }
334 
335  // Calculate centripetal acceleration demand
336  float latAccDemCircCtr = velTangent * velTangent / max((0.5f * radius), (radius + xtrackErrCirc));
337 
338  //Sum PD control and centripetal acceleration to calculate lateral manoeuvre demand
339  float latAccDemCirc = loiter_direction * (latAccDemCircPD + latAccDemCircCtr);
340 
341  // Perform switchover between 'capture' and 'circle' modes at the
342  // point where the commands cross over to achieve a seamless transfer
343  // Only fly 'capture' mode if outside the circle
344  if (xtrackErrCirc > 0.0f && loiter_direction * latAccDemCap < loiter_direction * latAccDemCirc) {
345  _latAccDem = latAccDemCap;
346  _WPcircle = false;
347  _bearing_error = Nu; // angle between demanded and achieved velocity vector, +ve to left of track
348  _nav_bearing = atan2f(-A_air_unit.y , -A_air_unit.x); // bearing (radians) from AC to L1 point
349  } else {
350  _latAccDem = latAccDemCirc;
351  _WPcircle = true;
352  _bearing_error = 0.0f; // bearing error (radians), +ve to left of track
353  _nav_bearing = atan2f(-A_air_unit.y , -A_air_unit.x); // bearing (radians)from AC to L1 point
354  }
355 }

358 // update L1 control for heading hold navigation
359 void AP_L1_Control::update_heading_hold(int32_t navigation_heading_cd)
360 {
361  // Calculate normalised frequency for tracking loop
362  const float omegaA = 4.4428f/_L1_period; // sqrt(2)*pi/period
363  // Calculate additional damping gain
364 
365  int32_t Nu_cd;
366  float Nu;
367 
368  // copy to _target_bearing_cd and _nav_bearing
369  _target_bearing_cd = wrap_180_cd(navigation_heading_cd);
370  _nav_bearing = radians(navigation_heading_cd * 0.01f);
371 
372  Nu_cd = _target_bearing_cd - wrap_180_cd(_ahrs.yaw_sensor);
373  Nu_cd = wrap_180_cd(Nu_cd);
374  Nu = radians(Nu_cd * 0.01f);
375 
376  Vector2f _groundspeed_vector = _ahrs.groundspeed_vector();
377 
378  //Calculate groundspeed
379  float groundSpeed = _groundspeed_vector.length();
380 
381  // Calculate time varying control parameters
382  _L1_dist = groundSpeed / omegaA; // L1 distance is adjusted to maintain a constant tracking loop frequency
383  float VomegaA = groundSpeed * omegaA;
384 
385  // Waypoint capture status is always false during heading hold
386  _WPcircle = false;
387 
388  _crosstrack_error = 0;
389 
390  _bearing_error = Nu; // bearing error angle (radians), +ve to left of track
391 
392  // Limit Nu to +-pi
393  Nu = constrain_float(Nu, -M_PI_2, M_PI_2);
394  _latAccDem = 2.0f*sinf(Nu)*VomegaA;
395 }
