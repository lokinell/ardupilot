201 void AP_MotorsHeli::output()
202 {

204  update_throttle_filter();
205 
206  if (_flags.armed) {

209  } else if (_flags.stabilizing) {

258 // output_disarmed - sends commands to the motors
259 void AP_MotorsHeli::output_disarmed()
260 {
261  if (_servo_test_cycle_counter > 0){
262  // perform boot-up servo test cycle if enabled
263  servo_test();
264  } else {
265  // manual override (i.e. when setting up swash)
266  switch (_servo_mode) {
267  case SERVO_CONTROL_MODE_MANUAL_PASSTHROUGH:
268  // pass pilot commands straight through to swash
269  _roll_control_input = _roll_radio_passthrough;
270  _pitch_control_input = _pitch_radio_passthrough;
271  _throttle_control_input = _throttle_radio_passthrough;
272  _yaw_control_input = _yaw_radio_passthrough;
273  break;
274  case SERVO_CONTROL_MODE_MANUAL_CENTER:
275  // fixate mid collective
276  _roll_control_input = 0;
277  _pitch_control_input = 0;
278  _throttle_control_input = _collective_mid_pwm;
279  _yaw_control_input = 0;
280  break;
281  case SERVO_CONTROL_MODE_MANUAL_MAX:
282  // fixate max collective
283  _roll_control_input = 0;
284  _pitch_control_input = 0;
285  _throttle_control_input = 1000;
286  _yaw_control_input = 4500;
287  break;
288  case SERVO_CONTROL_MODE_MANUAL_MIN:
289  // fixate min collective
290  _roll_control_input = 0;
291  _pitch_control_input = 0;
292  _throttle_control_input = 0;
293  _yaw_control_input = -4500;
294  break;
295  case SERVO_CONTROL_MODE_MANUAL_OSCILLATE:
296  // use servo_test function from child classes
297  servo_test();
298  break;
299  default:
300  // no manual override
301  break;
302  }
303  }
304 
305  // ensure swash servo endpoints haven't been moved
306  init_outputs();
307 
308  // continuously recalculate scalars to allow setup
309  calculate_scalars();
310 
311  // helicopters always run stabilizing flight controls
312  move_actuators(_roll_control_input, _pitch_control_input, _throttle_control_input, _yaw_control_input);
313 
314  update_motor_control(ROTOR_CONTROL_STOP);
315 }

317 // parameter_check - check if helicopter specific parameters are sensible
318 bool AP_MotorsHeli::parameter_check(bool display_msg) const
319 {
320  // returns false if _rsc_setpoint is not higher than _rsc_critical as this would not allow rotor_runup_complete to ever return true
321  if (_rsc_critical >= _rsc_setpoint) {
322  if (display_msg) {
323  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: H_RSC_CRITICAL too large");
324  }
325  return false;
326  }
327 
328  // returns false if RSC Mode is not set to a valid control mode
329  if (_rsc_mode <= (int8_t)ROTOR_CONTROL_MODE_DISABLED || _rsc_mode > (int8_t)ROTOR_CONTROL_MODE_CLOSED_LOOP_POWER_OUTPUT) {
330  if (display_msg) {
331  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: H_RSC_MODE invalid");
332  }
333  return false;
334  }
335 
336  // returns false if RSC Runup Time is less than Ramp time as this could cause undesired behaviour of rotor speed estimate
337  if (_rsc_runup_time <= _rsc_ramp_time){
338  if (display_msg) {
339  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: H_RUNUP_TIME too small");
340  }
341  return false;
342  }
343 
344  // returns false if idle output is higher than critical rotor speed as this could block runup_complete from going false
345  if ( _rsc_idle_output >= _rsc_critical){
346  if (display_msg) {
347  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: H_RSC_IDLE too large");
348  }
349  return false;
350  }
351 
352  // all other cases parameters are OK
353  return true;
354 }
