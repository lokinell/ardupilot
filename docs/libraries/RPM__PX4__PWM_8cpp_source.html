69 void AP_RPM_PX4_PWM::update(void)
70 {
71  if (_fd == -1) {
72  return;
73  }
74 
75  struct pwm_input_s pwm;
76  uint16_t count = 0;
77  const float scaling = ap_rpm._scaling[state.instance];
78  float sum = 0;
79 
80  while (::read(_fd, &pwm, sizeof(pwm)) == sizeof(pwm)) {
81  // the px4 pwm_input driver reports the period in microseconds
82  if (pwm.period == 0) {
83  continue;
84  }
85  float rpm = scaling * (1.0e6f * 60) / pwm.period;
86  float maximum = ap_rpm._maximum[state.instance];
87  if (maximum <= 0 || rpm <= maximum) {
88  sum += rpm;
89  count++;
90  }
91  }
92 
93  if (count != 0) {
94  state.rate_rpm = sum / count;
95  state.last_reading_ms = hal.scheduler->millis();
96  }
97 }
