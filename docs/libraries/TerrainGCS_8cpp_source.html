34 /*
35  request any missing 4x4 grids from a block, given a grid_cache
36  */
37 bool AP_Terrain::request_missing(mavlink_channel_t chan, struct grid_cache &gcache)
38 {
39  struct grid_block &grid = gcache.grid;
40 
41  if (grid.spacing != grid_spacing) {
42  // an invalid grid
43  return false;
44  }
45 
46  // see if we are waiting for disk read
47  if (gcache.state == GRID_CACHE_DISKWAIT) {
48  // don't request data from the GCS till we know its not on disk
49  return false;
50  }
51 
52  // see if it is fully populated
53  if ((grid.bitmap & bitmap_mask) == bitmap_mask) {
54  // it is fully populated, nothing to do
55  return false;
56  }
57 
58  if (comm_get_txspace(chan) < MAVLINK_NUM_NON_PAYLOAD_BYTES + MAVLINK_MSG_ID_TERRAIN_REQUEST_LEN) {
59  // not enough buffer space
60  return false;
61  }
62 
63  /*
64  ask the GCS to send a set of 4x4 grids
65  */
66  mavlink_msg_terrain_request_send(chan, grid.lat, grid.lon, grid_spacing, bitmap_mask & ~grid.bitmap);
67  last_request_time_ms[chan] = hal.scheduler->millis();
68 
69  return true;
70 }

82 /*
83  send any pending terrain request to the GCS
84  */
85 void AP_Terrain::send_request(mavlink_channel_t chan)
86 {
87  if (enable == 0 || !allocate()) {
88  // not enabled
89  return;
90  }
91 
92  // see if we need to schedule some disk IO
93  schedule_disk_io();
94 
95  Location loc;
96  if (!ahrs.get_position(loc)) {
97  // we don't know where we are
98  return;
99  }
100 
101  // always send a terrain report
102  send_terrain_report(chan, loc, true);
103 
104  // did we request recently?
105  if (hal.scheduler->millis() - last_request_time_ms[chan] < 2000) {
106  // too soon to request again
107  return;
108  }
109 
110  // request any missing 4x4 blocks in the current grid
111  struct grid_info info;
112  calculate_grid_info(loc, info);
113 
114  if (request_missing(chan, info)) {
115  return;
116  }
117 
118  // also request a larger set of up to 9 grids
119  for (int8_t x=-1; x<=1; x++) {
120  for (int8_t y=-1; y<=1; y++) {
121  Location loc2 = loc;
122  location_offset(loc2,
123  x*TERRAIN_GRID_BLOCK_SIZE_X*0.7f*grid_spacing,
124  y*TERRAIN_GRID_BLOCK_SIZE_Y*0.7f*grid_spacing);
125  struct grid_info info2;
126  calculate_grid_info(loc2, info2);
127  if (request_missing(chan, info2)) {
128  return;
129  }
130  }
131  }
132 
133  // check cache blocks that may have been setup by a TERRAIN_CHECK
134  for (uint16_t i=0; i<cache_size; i++) {
135  if (cache[i].state >= GRID_CACHE_VALID) {
136  if (request_missing(chan, cache[i])) {
137  return;
138  }
139  }
140  }
141 
142  // request the current loc last to ensure it has highest last
143  // access time
144  if (request_missing(chan, info)) {
145  return;
146  }
147 }

157 /*
158  get some statistics for TERRAIN_REPORT
159 */
160 void AP_Terrain::get_statistics(uint16_t &pending, uint16_t &loaded)
161 {
162  pending = 0;
163  loaded = 0;
164  for (uint16_t i=0; i<cache_size; i++) {
165  if (cache[i].grid.spacing != grid_spacing) {
166  continue;
167  }
168  if (cache[i].state == GRID_CACHE_INVALID) {
169  continue;
170  }
171  uint8_t maskbits = TERRAIN_GRID_BLOCK_MUL_X*TERRAIN_GRID_BLOCK_MUL_Y;
172  if (cache[i].state == GRID_CACHE_DISKWAIT) {
173  pending += maskbits;
174  continue;
175  }
176  if (cache[i].state == GRID_CACHE_DIRTY) {
177  // count dirty grids as a pending, so we know where there
178  // are disk writes pending
179  pending++;
180  }
181  uint8_t bitcount = bitcount64(cache[i].grid.bitmap);
182  pending += maskbits - bitcount;
183  loaded += bitcount;
184  }
185 }

201 /*
202  send a TERRAIN_REPORT for a location
203  */
204 void AP_Terrain::send_terrain_report(mavlink_channel_t chan, const Location &loc, bool extrapolate)
205 {
206  float terrain_height = 0;
207  float home_terrain_height = 0;
208  uint16_t spacing = 0;
209  Location current_loc;
210  if (ahrs.get_position(current_loc) &&
211  height_amsl(ahrs.get_home(), home_terrain_height) &&
212  height_amsl(loc, terrain_height)) {
213  // non-zero spacing indicates we have data
214  spacing = grid_spacing;
215  } else if (extrapolate && have_current_loc_height) {
216  // show the extrapolated height, so logs show what height is
217  // being used for navigation
218  terrain_height = last_current_loc_height;
219  }
220  uint16_t pending, loaded;
221  get_statistics(pending, loaded);
222 
223  float current_height;
224  if (spacing == 0 && !(extrapolate && have_current_loc_height)) {
225  current_height = 0;
226  } else {
227  if (current_loc.flags.relative_alt) {
228  current_height = current_loc.alt*0.01f;
229  } else {
230  current_height = (current_loc.alt - ahrs.get_home().alt)*0.01f;
231  }
232  }
233  current_height += home_terrain_height - terrain_height;
234 
235  if (comm_get_txspace(chan) >= MAVLINK_NUM_NON_PAYLOAD_BYTES + MAVLINK_MSG_ID_TERRAIN_REPORT_LEN) {
236  mavlink_msg_terrain_report_send(chan, loc.lat, loc.lng, spacing,
237  terrain_height, current_height,
238  pending, loaded);
239  }
240 }

255 /*
256  handle TERRAIN_DATA messages from GCS
257  */
258 void AP_Terrain::handle_terrain_data(mavlink_message_t *msg)
259 {
260  mavlink_terrain_data_t packet;
261  mavlink_msg_terrain_data_decode(msg, &packet);
262 
263  uint16_t i;
264  for (i=0; i<cache_size; i++) {
265  if (cache[i].grid.lat == packet.lat &&
266  cache[i].grid.lon == packet.lon &&
267  cache[i].grid.spacing == packet.grid_spacing &&
268  grid_spacing == packet.grid_spacing &&
269  packet.gridbit < 56) {
270  break;
271  }
272  }
273  if (i == cache_size) {
274  // we don't have that grid, ignore data
275  return;
276  }
277  struct grid_cache &gcache = cache[i];
278  struct grid_block &grid = gcache.grid;
279  uint8_t idx_x = (packet.gridbit / TERRAIN_GRID_BLOCK_MUL_Y) * TERRAIN_GRID_MAVLINK_SIZE;
280  uint8_t idx_y = (packet.gridbit % TERRAIN_GRID_BLOCK_MUL_Y) * TERRAIN_GRID_MAVLINK_SIZE;
281  ASSERT_RANGE(idx_x,0,(TERRAIN_GRID_BLOCK_MUL_X-1)*TERRAIN_GRID_MAVLINK_SIZE);
282  ASSERT_RANGE(idx_y,0,(TERRAIN_GRID_BLOCK_MUL_Y-1)*TERRAIN_GRID_MAVLINK_SIZE);
283  for (uint8_t x=0; x<TERRAIN_GRID_MAVLINK_SIZE; x++) {
284  for (uint8_t y=0; y<TERRAIN_GRID_MAVLINK_SIZE; y++) {
285  grid.height[idx_x+x][idx_y+y] = packet.data[x*TERRAIN_GRID_MAVLINK_SIZE+y];
286  }
287  }
288  gcache.grid.bitmap |= ((uint64_t)1) << packet.gridbit;
289 
290  // mark dirty for disk IO
291  gcache.state = GRID_CACHE_DIRTY;
292 
293 #if TERRAIN_DEBUG
294  hal.console->printf("Filled bit %u idx_x=%u idx_y=%u\n",
295  (unsigned)packet.gridbit, (unsigned)idx_x, (unsigned)idx_y);
296  if (gcache.grid.bitmap == bitmap_mask) {
297  hal.console->printf("--lat=%12.7f --lon=%12.7f %u\n",
298  grid.lat*1.0e-7f,
299  grid.lon*1.0e-7f,
300  grid.height[0][0]);
301  Location loc2;
302  loc2.lat = grid.lat;
303  loc2.lng = grid.lon;
304  location_offset(loc2, 28*grid_spacing, 32*grid_spacing);
305  hal.console->printf("--lat=%12.7f --lon=%12.7f %u\n",
306  loc2.lat*1.0e-7f,
307  loc2.lng*1.0e-7f,
308  grid.height[27][31]);
309  }
310 #endif
311 
312  // see if we need to schedule some disk IO
313  update();
314 }
