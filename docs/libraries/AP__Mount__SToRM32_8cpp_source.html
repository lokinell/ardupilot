18 // update mount position - should be called periodically
19 void AP_Mount_SToRM32::update()
20 {
21  // exit immediately if not initialised
22  if (!_initialised) {
23  find_gimbal();
24  return;
25  }
26 
27  // flag to trigger sending target angles to gimbal
28  bool resend_now = false;
29 
30  // update based on mount mode
31  switch(get_mode()) {
32  // move mount to a "retracted" position. To-Do: remove support and replace with a relaxed mode?
33  case MAV_MOUNT_MODE_RETRACT:
34  {
35  const Vector3f &target = _state._retract_angles.get();
36  _angle_ef_target_rad.x = ToRad(target.x);
37  _angle_ef_target_rad.y = ToRad(target.y);
38  _angle_ef_target_rad.z = ToRad(target.z);
39  }
40  break;
41 
42  // move mount to a neutral position, typically pointing forward
43  case MAV_MOUNT_MODE_NEUTRAL:
44  {
45  const Vector3f &target = _state._neutral_angles.get();
46  _angle_ef_target_rad.x = ToRad(target.x);
47  _angle_ef_target_rad.y = ToRad(target.y);
48  _angle_ef_target_rad.z = ToRad(target.z);
49  }
50  break;
51 
52  // point to the angles given by a mavlink message
53  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
54  // do nothing because earth-frame angle targets (i.e. _angle_ef_target_rad) should have already been set by a MOUNT_CONTROL message from GCS
55  break;
56 
57  // RC radio manual angle control, but with stabilization from the AHRS
58  case MAV_MOUNT_MODE_RC_TARGETING:
59  // update targets using pilot's rc inputs
60  update_targets_from_rc();
61  resend_now = true;
62  break;
63 
64  // point mount to a GPS point given by the mission planner
65  case MAV_MOUNT_MODE_GPS_POINT:
66  if(_frontend._ahrs.get_gps().status() >= AP_GPS::GPS_OK_FIX_2D) {
67  calc_angle_to_location(_state._roi_target, _angle_ef_target_rad, true, true);
68  resend_now = true;
69  }
70  break;
71 
72  default:
73  // we do not know this mode so do nothing
74  break;
75  }
76 
77  // resend target angles at least once per second
78  if (resend_now || ((hal.scheduler->millis() - _last_send) > AP_MOUNT_STORM32_RESEND_MS)) {
79  send_do_mount_control(ToDeg(_angle_ef_target_rad.y), ToDeg(_angle_ef_target_rad.x), ToDeg(_angle_ef_target_rad.z), MAV_MOUNT_MODE_MAVLINK_TARGETING);
80  }
81 }
