65 bool AP_Compass_PX4::init(void)
66 {
67  _mag_fd[0] = open(MAG_BASE_DEVICE_PATH"0", O_RDONLY);
68  _mag_fd[1] = open(MAG_BASE_DEVICE_PATH"1", O_RDONLY);
69  _mag_fd[2] = open(MAG_BASE_DEVICE_PATH"2", O_RDONLY);
70 
71  _num_sensors = 0;
72  for (uint8_t i=0; i<COMPASS_MAX_INSTANCES; i++) {
73  if (_mag_fd[i] >= 0) {
74  _num_sensors = i+1;
75  }
76  }
77  if (_num_sensors == 0) {
78  hal.console->printf("Unable to open " MAG_BASE_DEVICE_PATH"0" "\n");
79  return false;
80  }
81 
82  for (uint8_t i=0; i<_num_sensors; i++) {
83  _instance[i] = register_compass();
84 
85  // get device id
86  set_dev_id(_instance[i], ioctl(_mag_fd[i], DEVIOCGDEVICEID, 0));
87 
88  // average over up to 20 samples
89  if (ioctl(_mag_fd[i], SENSORIOCSQUEUEDEPTH, 20) != 0) {
90  hal.console->printf("Failed to setup compass queue\n");
91  return false;
92  }
93 
94  // remember if the compass is external
95  set_external(_instance[i], ioctl(_mag_fd[i], MAGIOCGEXTERNAL, 0) > 0);
96  _count[i] = 0;
97  _sum[i].zero();
98 
99  }
100 
101  // give the driver a chance to run, and gather one sample
102  hal.scheduler->delay(40);
103  accumulate();
104  if (_count[0] == 0) {
105  hal.console->printf("Failed initial compass accumulate\n");
106  }
107 
108  return true;
109 }

111 void AP_Compass_PX4::read(void)
112 {
113  // try to accumulate one more sample, so we have the latest data
114  accumulate();
115 
116  for (uint8_t i=0; i<_num_sensors; i++) {
117  uint8_t frontend_instance = _instance[i];
118  // avoid division by zero if we haven't received any mag reports
119  if (_count[i] == 0) continue;
120 
121  _sum[i] /= _count[i];
122 
123  publish_filtered_field(_sum[i], frontend_instance);
124 
125  _sum[i].zero();
126  _count[i] = 0;
127  }
128 }

130 void AP_Compass_PX4::accumulate(void)
131 {
132  struct mag_report mag_report;
133  for (uint8_t i=0; i<_num_sensors; i++) {
134  uint8_t frontend_instance = _instance[i];
135  while (::read(_mag_fd[i], &mag_report, sizeof(mag_report)) == sizeof(mag_report) &&
136  mag_report.timestamp != _last_timestamp[i]) {
137 
138  uint32_t time_us = (uint32_t)mag_report.timestamp;
139  // get raw_field - sensor frame, uncorrected
140  Vector3f raw_field = Vector3f(mag_report.x, mag_report.y, mag_report.z)*1.0e3f;
141 
142  // rotate raw_field from sensor frame to body frame
143  rotate_field(raw_field, frontend_instance);
144 
145  // publish raw_field (uncorrected point sample) for calibration use
146  publish_raw_field(raw_field, time_us, frontend_instance);
147 
148  // correct raw_field for known errors
149  correct_field(raw_field, frontend_instance);
150 
151  // publish raw_field (corrected point sample) for EKF use
152  publish_unfiltered_field(raw_field, time_us, frontend_instance);
153 
154  // accumulate into averaging filter
155  _sum[i] += raw_field;
156  _count[i]++;
157 
158  _last_timestamp[i] = mag_report.timestamp;
159  }
160  }
161 }
