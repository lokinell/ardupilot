599 /*
600  * Fuse compass measurements using explicit algebraic equations generated with Matlab symbolic toolbox.
601  * The script file used to generate these and other equations in this filter can be found here:
602  * https://github.com/priseborough/InertialNav/blob/master/derivations/RotationVectorAttitudeParameterisation/GenerateNavFilterEquations.m
603  * This fusion method only modifies the orientation, does not require use of the magnetic field states and is computatonally cheaper.
604  * It is suitable for use when the external magnetic field environment is disturbed (eg close to metal structures, on ground).
605  * It is not as robust to magneometer failures.
606 */
607 void NavEKF2_core::fuseCompass()
608 {
609  float q0 = stateStruct.quat[0];
610  float q1 = stateStruct.quat[1];
611  float q2 = stateStruct.quat[2];
612  float q3 = stateStruct.quat[3];
613 
614  float magX = magDataDelayed.mag.x;
615  float magY = magDataDelayed.mag.y;
616  float magZ = magDataDelayed.mag.z;
617 
618  // compass measurement error variance (rad^2)
619  const float R_MAG = 3e-2f;
620 
621  // Calculate observation Jacobian
622  float t2 = q0*q0;
623  float t3 = q1*q1;
624  float t4 = q2*q2;
625  float t5 = q3*q3;
626  float t6 = q0*q2*2.0f;
627  float t7 = q1*q3*2.0f;
628  float t8 = t6+t7;
629  float t9 = q0*q3*2.0f;
630  float t13 = q1*q2*2.0f;
631  float t10 = t9-t13;
632  float t11 = t2+t3-t4-t5;
633  float t12 = magX*t11;
634  float t14 = magZ*t8;
635  float t19 = magY*t10;
636  float t15 = t12+t14-t19;
637  float t16 = t2-t3+t4-t5;
638  float t17 = q0*q1*2.0f;
639  float t24 = q2*q3*2.0f;
640  float t18 = t17-t24;
641  float t20 = 1.0f/t15;
642  float t21 = magY*t16;
643  float t22 = t9+t13;
644  float t23 = magX*t22;
645  float t28 = magZ*t18;
646  float t25 = t21+t23-t28;
647  float t29 = t20*t25;
648  float t26 = tan(t29);
649  float t27 = 1.0f/(t15*t15);
650  float t30 = t26*t26;
651  float t31 = t30+1.0f;
652  float H_MAG[3];
653  H_MAG[0] = -t31*(t20*(magZ*t16+magY*t18)+t25*t27*(magY*t8+magZ*t10));
654  H_MAG[1] = t31*(t20*(magX*t18+magZ*t22)+t25*t27*(magX*t8-magZ*t11));
655  H_MAG[2] = t31*(t20*(magX*t16-magY*t22)+t25*t27*(magX*t10+magY*t11));
656 
657  // Calculate innovation variance and Kalman gains, taking advantage of the fact that only the first 3 elements in H are non zero
658  float PH[3];
659  float varInnov = R_MAG;
660  for (uint8_t rowIndex=0; rowIndex<=2; rowIndex++) {
661  PH[rowIndex] = 0.0f;
662  for (uint8_t colIndex=0; colIndex<=2; colIndex++) {
663  PH[rowIndex] += P[rowIndex][colIndex]*H_MAG[colIndex];
664  }
665  varInnov += H_MAG[rowIndex]*PH[rowIndex];
666  }
667  float varInnovInv;
668  if (varInnov >= R_MAG) {
669  varInnovInv = 1.0f / varInnov;
670  // All three magnetometer components are used in this measurement, so we output health status on three axes
671  faultStatus.bad_xmag = false;
672  faultStatus.bad_ymag = false;
673  faultStatus.bad_zmag = false;
674  } else {
675  // the calculation is badly conditioned, so we cannot perform fusion on this step
676  // we reset the covariance matrix and try again next measurement
677  CovarianceInit();
678  // All three magnetometer components are used in this measurement, so we output health status on three axes
679  faultStatus.bad_xmag = true;
680  faultStatus.bad_ymag = true;
681  faultStatus.bad_zmag = true;
682  return;
683  }
684  for (uint8_t rowIndex=0; rowIndex<=stateIndexLim; rowIndex++) {
685  Kfusion[rowIndex] = 0.0f;
686  for (uint8_t colIndex=0; colIndex<=2; colIndex++) {
687  Kfusion[rowIndex] += P[rowIndex][colIndex]*H_MAG[colIndex];
688  }
689  Kfusion[rowIndex] *= varInnovInv;
690  }
691 
692  // Calculate the innovation
693  float innovation = calcMagHeadingInnov();
694 
695  // Copy raw value to output variable used for data logging
696  innovYaw = innovation;
697 
698  // limit the innovation so that initial corrections are not too large
699  if (innovation > 0.5f) {
700  innovation = 0.5f;
701  } else if (innovation < -0.5f) {
702  innovation = -0.5f;
703  }
704 
705  // calculate the innovation test ratio
706  yawTestRatio = sq(innovation) / (sq(max(frontend->_magInnovGate,1)) * varInnov);
707 
708  // Declare the magnetometer unhealthy if the innovation test fails
709  if (yawTestRatio > 1.0f) {
710  magHealth = false;
711  // On the ground a large innovation could be due to large initial gyro bias or magnetic interference from nearby objects
712  // If we are flying, then it is more likely due to a magnetometer fault and we should not fuse the data
713  if (inFlight) {
714  return;
715  }
716  } else {
717  magHealth = true;
718  }
719 
720  // correct the state vector
721  stateStruct.angErr.zero();
722  for (uint8_t i=0; i<=stateIndexLim; i++) {
723  statesArray[i] -= Kfusion[i] * innovation;
724  }
725 
726  // the first 3 states represent the angular misalignment vector. This is
727  // is used to correct the estimated quaternion on the current time step
728  stateStruct.quat.rotate(stateStruct.angErr);
729 
730  // correct the covariance using P = P - K*H*P taking advantage of the fact that only the first 3 elements in H are non zero
731  float HP[24];
732  for (uint8_t colIndex=0; colIndex<=stateIndexLim; colIndex++) {
733  HP[colIndex] = 0.0f;
734  for (uint8_t rowIndex=0; rowIndex<=2; rowIndex++) {
735  HP[colIndex] += H_MAG[rowIndex]*P[rowIndex][colIndex];
736  }
737  }
738  for (uint8_t rowIndex=0; rowIndex<=stateIndexLim; rowIndex++) {
739  for (uint8_t colIndex=0; colIndex<=stateIndexLim; colIndex++) {
740  P[rowIndex][colIndex] -= Kfusion[rowIndex] * HP[colIndex];
741  }
742  }
743 
744  // force the covariance matrix to be symmetrical and limit the variances to prevent
745  // ill-condiioning.
746  ForceSymmetry();
747  ConstrainVariances();
748 }

756 */
757 void NavEKF2_core::FuseDeclination()
758 {
759  // declination error variance (rad^2)
760  const float R_DECL = 1e-2f;
761 
762  // copy required states to local variables
763  float magN = stateStruct.earth_magfield.x;
764  float magE = stateStruct.earth_magfield.y;
765 
766  // prevent bad earth field states from causing numerical errors or exceptions
767  if (magN < 1e-3f) {
768  return;
769  }
770 
771  // Calculate observation Jacobian and Kalman gains
772  float t2 = 1.0f/magN;
773  float t4 = magE*t2;
774  float t3 = tanf(t4);
775  float t5 = t3*t3;
776  float t6 = t5+1.0f;
777  float t7 = 1.0f/(magN*magN);
778  float t8 = P[17][17]*t2*t6;
779  float t15 = P[16][17]*magE*t6*t7;
780  float t9 = t8-t15;
781  float t10 = t2*t6*t9;
782  float t11 = P[17][16]*t2*t6;
783  float t16 = P[16][16]*magE*t6*t7;
784  float t12 = t11-t16;
785  float t17 = magE*t6*t7*t12;
786  float t13 = R_DECL+t10-t17;
787  float t14 = 1.0f/t13;
788  float t18 = magE;
789  float t19 = magN;
790  float t21 = 1.0f/t19;
791  float t22 = t18*t21;
792  float t20 = tanf(t22);
793  float t23 = t20*t20;
794  float t24 = t23+1.0f;
795 
796  float H_MAG[24];
797  H_MAG[16] = -t18*1.0f/(t19*t19)*t24;
798  H_MAG[17] = t21*t24;
799 
800  for (uint8_t i=0; i<=15; i++) {
801  Kfusion[i] = t14*(P[i][17]*t2*t6-P[i][16]*magE*t6*t7);
802  }
803  Kfusion[16] = -t14*(t16-P[16][17]*t2*t6);
804  Kfusion[17] = t14*(t8-P[17][16]*magE*t6*t7);
805  for (uint8_t i=17; i<=23; i++) {

807  }

810  float magDecAng = use_compass() ? _ahrs->get_compass()->get_declination() : 0;
811 
812  // Calculate the innovation
813  float innovation = atanf(t4) - magDecAng;
814 
815  // limit the innovation to protect against data errors
816  if (innovation > 0.5f) {
817  innovation = 0.5f;
818  } else if (innovation < -0.5f) {
819  innovation = -0.5f;
820  }
821 
822  // zero the attitude error state - by definition it is assumed to be zero before each observaton fusion
823  stateStruct.angErr.zero();
824 
825  // correct the state vector
826  for (uint8_t j= 0; j<=stateIndexLim; j++) {
827  statesArray[j] = statesArray[j] - Kfusion[j] * innovation;
828  }
829 
830  // the first 3 states represent the angular misalignment vector. This is
831  // is used to correct the estimated quaternion on the current time step
832  stateStruct.quat.rotate(stateStruct.angErr);
833 
834  // correct the covariance P = (I - K*H)*P
835  // take advantage of the empty columns in KH to reduce the
836  // number of operations
837  for (unsigned i = 0; i<=stateIndexLim; i++) {
838  for (unsigned j = 0; j<=15; j++) {
839  KH[i][j] = 0.0f;
840  }
841  KH[i][16] = Kfusion[i] * H_MAG[16];
842  KH[i][17] = Kfusion[i] * H_MAG[17];
843  for (unsigned j = 18; j<=23; j++) {
844  KH[i][j] = 0.0f;
845  }
846  }
847  for (unsigned j = 0; j<=stateIndexLim; j++) {
848  for (unsigned i = 0; i<=stateIndexLim; i++) {
849  KHP[i][j] = KH[i][16] * P[16][j] + KH[i][17] * P[17][j];
850  }
851  }
852  for (unsigned i = 0; i<=stateIndexLim; i++) {
853  for (unsigned j = 0; j<=stateIndexLim; j++) {
854  P[i][j] = P[i][j] - KHP[i][j];
855  }
856  }

859  // ill-condiioning.
860  ForceSymmetry();
861  ConstrainVariances();

865 // Calculate magnetic heading innovation
866 float NavEKF2_core::calcMagHeadingInnov()
867 {
868  // rotate predicted earth components into body axes and calculate
869  // predicted measurements
870  Matrix3f Tbn_temp;
871  stateStruct.quat.rotation_matrix(Tbn_temp);
872  Vector3f magMeasNED = Tbn_temp*magDataDelayed.mag;
873 
874  // calculate the innovation where the predicted measurement is the angle wrt magnetic north of the horizontal component of the measured field
875  float innovation = atan2f(magMeasNED.y,magMeasNED.x) - _ahrs->get_compass()->get_declination();
876 
877  // wrap the innovation so it sits on the range from +-pi
878  if (innovation > M_PI_F) {
879  innovation = innovation - 2*M_PI_F;
880  } else if (innovation < -M_PI_F) {
881  innovation = innovation + 2*M_PI_F;
882  }
883 
884  // Unwrap so that a large yaw gyro bias offset that causes the heading to wrap does not lead to continual uncontrolled heading drift
885  if (innovation - lastInnovation > M_PI_F) {
886  // Angle has wrapped in the positive direction to subtract an additional 2*Pi
887  innovationIncrement -= 2*M_PI_F;
888  } else if (innovation -innovationIncrement < -M_PI_F) {
889  // Angle has wrapped in the negative direction so add an additional 2*Pi
890  innovationIncrement += 2*M_PI_F;
891  }
892  lastInnovation = innovation;
893 
894  return innovation + innovationIncrement;
895 }
