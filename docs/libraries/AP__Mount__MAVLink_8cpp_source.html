26 // update mount position - should be called periodically
27 void AP_Mount_MAVLink::update()
28 {
29  // exit immediately if not initialised
30  if (!_initialised) {
31  return;
32  }
33 
34  // update based on mount mode
35  switch(get_mode()) {
36  // move mount to a "retracted" position. we do not implement a separate servo based retract mechanism
37  case MAV_MOUNT_MODE_RETRACT:
38  break;
39 
40  // move mount to a neutral position, typically pointing forward
41  case MAV_MOUNT_MODE_NEUTRAL:
42  break;
43 
44  // point to the angles given by a mavlink message
45  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
46  // do nothing because earth-frame angle targets (i.e. _angle_ef_target_rad) should have already been set by a MOUNT_CONTROL message from GCS
47  break;
48 
49  // RC radio manual angle control, but with stabilization from the AHRS
50  case MAV_MOUNT_MODE_RC_TARGETING:
51  // update targets using pilot's rc inputs
52  update_targets_from_rc();
53  break;
54 
55  // point mount to a GPS point given by the mission planner
56  case MAV_MOUNT_MODE_GPS_POINT:
57  if(_frontend._ahrs.get_gps().status() >= AP_GPS::GPS_OK_FIX_2D) {
58  calc_angle_to_location(_state._roi_target, _angle_ef_target_rad, true, true);
59  }
60  break;
61 
62  default:
63  // we do not know this mode so do nothing
64  break;
65  }
66 }
