129 bool AP_Arming::ins_checks(bool report)
130 {
131  if ((checks_to_perform & ARMING_CHECK_ALL) ||
132  (checks_to_perform & ARMING_CHECK_INS)) {
133  const AP_InertialSensor &ins = ahrs.get_ins();
134  if (!ins.get_gyro_health_all()) {
135  if (report) {
136  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: gyros not healthy");
137  }
138  return false;
139  }
140  if (!ins.gyro_calibrated_ok_all()) {
141  if (report) {
142  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: gyros not calibrated");
143  }
144  return false;
145  }
146  if (!ins.get_accel_health_all()) {
147  if (report) {
148  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: accels not healthy");
149  }
150  return false;
151  }
152  if (!ins.accel_calibrated_ok_all()) {
153  if (report) {
154  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: 3D accel cal needed");
155  }
156  return false;
157  }
158 
159  // check all accelerometers point in roughly same direction
160  if (ins.get_accel_count() > 1) {
161  const Vector3f &prime_accel_vec = ins.get_accel();
162  for(uint8_t i=0; i<ins.get_accel_count(); i++) {
163  // get next accel vector
164  const Vector3f &accel_vec = ins.get_accel(i);
165  Vector3f vec_diff = accel_vec - prime_accel_vec;
166  // allow for up to 0.75 m/s/s difference. Has to pass
167  // in last 10 seconds
168  float threshold = 0.75f;
169  if (i >= 2) {
170  /*
171  we allow for a higher threshold for IMU3 as it
172  runs at a different temperature to IMU1/IMU2,
173  and is not used for accel data in the EKF
174  */
175  threshold *= 3;
176  }
177  if (vec_diff.length() <= threshold) {
178  last_accel_pass_ms[i] = hal.scheduler->millis();
179  }
180  if (hal.scheduler->millis() - last_accel_pass_ms[i] > 10000) {
181  if (report) {
182  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: inconsistent Accelerometers");
183  }
184  return false;
185  }
186  }
187  }
188 
189  // check all gyros are giving consistent readings
190  if (ins.get_gyro_count() > 1) {
191  const Vector3f &prime_gyro_vec = ins.get_gyro();
192  for(uint8_t i=0; i<ins.get_gyro_count(); i++) {
193  // get next gyro vector
194  const Vector3f &gyro_vec = ins.get_gyro(i);
195  Vector3f vec_diff = gyro_vec - prime_gyro_vec;
196  // allow for up to 5 degrees/s difference. Pass if its
197  // been OK in last 10 seconds
198  if (vec_diff.length() <= radians(5)) {
199  last_gyro_pass_ms[i] = hal.scheduler->millis();
200  }
201  if (hal.scheduler->millis() - last_gyro_pass_ms[i] > 10000) {
202  if (report) {
203  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: inconsistent gyros");
204  }
205  return false;
206  }
207  }
208  }
209  }
210 
211  return true;
212 }

214 bool AP_Arming::compass_checks(bool report)
215 {
216  if ((checks_to_perform) & ARMING_CHECK_ALL ||
217  (checks_to_perform) & ARMING_CHECK_COMPASS) {
218 
219  if (!_compass.use_for_yaw()) {
220  // compass use is disabled
221  return true;
222  }
223 
224  if (!_compass.healthy()) {
225  if (report) {
226  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass not healthy");
227  }
228  return false;
229  }
230  // check compass learning is on or offsets have been set
231  if (!_compass.learn_offsets_enabled() && !_compass.configured()) {
232  if (report) {
233  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass not calibrated");
234  }
235  return false;
236  }
237 
238  //check if compass is calibrating
239  if (_compass.is_calibrating()) {
240  if (report) {
241  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "Arm: Compass calibration running");
242  }
243  return false;
244  }
245 
246  //check if compass has calibrated and requires reboot
247  if (_compass.compass_cal_requires_reboot()) {
248  if (report) {
249  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass calibrated requires reboot");
250  }
251  return false;
252  }
253 
254  // check for unreasonable compass offsets
255  Vector3f offsets = _compass.get_offsets();
256  if (offsets.length() > AP_ARMING_COMPASS_OFFSETS_MAX) {
257  if (report) {
258  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Compass offsets too high");
259  }
260  return false;
261  }
262 
263  // check for unreasonable mag field length
264  float mag_field = _compass.get_field().length();
265  if (mag_field > AP_ARMING_COMPASS_MAGFIELD_MAX || mag_field < AP_ARMING_COMPASS_MAGFIELD_MIN) {
266  if (report) {
267  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "PreArm: Check mag field");
268  }
269  return false;
270  }
271 
272  // check all compasses point in roughly same direction
273  if (!_compass.consistent()) {
274  if (report) {
275  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL,"PreArm: inconsistent compasses");
276  }
277  return false;
278  }
279  }
280 
281  return true;
282 }

368 bool AP_Arming::pre_arm_checks(bool report)
369 {
370  bool ret = true;
371  if (armed || require == NONE) {
372  // if we are already armed or don't need any arming checks
373  // then skip the checks
374  return true;
375  }
376 
377  ret &= hardware_safety_check(report);
378  ret &= barometer_checks(report);
379  ret &= ins_checks(report);
380  ret &= compass_checks(report);
381  ret &= gps_checks(report);
382  ret &= battery_checks(report);
383  ret &= logging_checks(report);
384  ret &= manual_transmitter_checks(report);
385  ret &= board_voltage_checks(report);
386 
387  return ret;
388 }
