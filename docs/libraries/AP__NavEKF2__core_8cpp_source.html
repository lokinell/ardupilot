421  * Update the quaternion, velocity and position states using delayed IMU measurements
422  * because the EKF is running on a delayed time horizon. Note that the quaternion is
423  * not used by the EKF equations, which instead estimate the error in the attitude of
424  * the vehicle when each observtion is fused. This attitude error is then used to correct
425  * the quaternion.
426 */
427 void NavEKF2_core::UpdateStrapdownEquationsNED()
428 {
429  // apply correction for earths rotation rate
430  // % * - and + operators have been overloaded
431  correctedDelAng = imuDataDelayed.delAng - prevTnb * earthRateNED*imuDataDelayed.delAngDT;
432 
433  // convert the rotation vector to its equivalent quaternion
434  correctedDelAngQuat.from_axis_angle(correctedDelAng);
435 
436  // update the quaternion states by rotating from the previous attitude through
437  // the delta angle rotation quaternion and normalise
438  stateStruct.quat *= correctedDelAngQuat;
439  stateStruct.quat.normalize();
440 
441  // calculate the body to nav cosine matrix
442  Matrix3f Tbn_temp;
443  stateStruct.quat.rotation_matrix(Tbn_temp);
444  prevTnb = Tbn_temp.transposed();
445 
446  // transform body delta velocities to delta velocities in the nav frame
447  // * and + operators have been overloaded
448  Vector3f delVelNav; // delta velocity vector in earth axes
449  delVelNav = Tbn_temp*imuDataDelayed.delVel;
450  delVelNav.z += GRAVITY_MSS*imuDataDelayed.delVelDT;
451 
452  // calculate the rate of change of velocity (used for launch detect and other functions)
453  velDotNED = delVelNav / imuDataDelayed.delVelDT;
454 
455  // apply a first order lowpass filter
456  velDotNEDfilt = velDotNED * 0.05f + velDotNEDfilt * 0.95f;
457

459  // variance estimation)
460  accNavMag = velDotNEDfilt.length();
461  accNavMagHoriz = pythagorous2(velDotNEDfilt.x , velDotNEDfilt.y);
462

464  Vector3f lastVelocity = stateStruct.velocity;
465 
466  // sum delta velocities to get velocity
467  stateStruct.velocity += delVelNav;
468 
469  // apply a trapezoidal integration to velocities to calculate position
470  stateStruct.position += (stateStruct.velocity + lastVelocity) * (imuDataDelayed.delVelDT*0.5f);
471

473  delAngBodyOF += imuDataDelayed.delAng - stateStruct.gyro_bias;
474  delTimeOF += imuDataDelayed.delAngDT;

480 /*
481  * Propagate PVA solution forward from the fusion time horizon to the current time horizon
482  * using simple observer which performs two functions:
483  * 1) Corrects for the delayed time horizon used by the EKF.
484  * 2) Applies a LPF to state corrections to prevent 'stepping' in states due to measurement
485  * fusion introducing unwanted noise into the control loops.
486  * The inspiration for using a complementary filter to correct for time delays in the EKF
487  * is based on the work by A Khosravian.
488  *
489  * Recursive Attitude Estimation in the Presence of Multi-rate and Multi-delay Vector Measurements
490  * A Khosravian, J Trumpf, R Mahony, T Hamel, Australian National University
491 */
492 void NavEKF2_core::calcOutputStatesFast() {
493 
494  // Calculate strapdown solution at the current time horizon
495 
496  // remove gyro scale factor errors
497  Vector3f delAng;
498  delAng.x = imuDataNew.delAng.x * stateStruct.gyro_scale.x;
499  delAng.y = imuDataNew.delAng.y * stateStruct.gyro_scale.y;
500  delAng.z = imuDataNew.delAng.z * stateStruct.gyro_scale.z;
501 
502  // remove sensor bias errors
503  delAng -= stateStruct.gyro_bias;
504  Vector3f delVel;
505  delVel = imuDataNew.delVel;
506  delVel.z -= stateStruct.accel_zbias;
507 
508  // apply corections to track EKF solution
509  delAng += delAngCorrection;
510 
511  // convert the rotation vector to its equivalent quaternion
512  Quaternion deltaQuat;
513  deltaQuat.from_axis_angle(delAng);
514 
515  // update the quaternion states by rotating from the previous attitude through
516  // the delta angle rotation quaternion and normalise
517  outputDataNew.quat *= deltaQuat;
518  outputDataNew.quat.normalize();
519 
520  // calculate the body to nav cosine matrix
521  Matrix3f Tbn_temp;
522  outputDataNew.quat.rotation_matrix(Tbn_temp);
523 
524  // transform body delta velocities to delta velocities in the nav frame
525  // Add the earth frame correction required to track the EKF states
526  // * and + operators have been overloaded
527  Vector3f delVelNav = Tbn_temp*delVel + delVelCorrection;
528  delVelNav.z += GRAVITY_MSS*imuDataNew.delVelDT;
529 
530  // save velocity for use in trapezoidal intergration for position calcuation
531  Vector3f lastVelocity = outputDataNew.velocity;
532 
533  // sum delta velocities to get velocity
534  outputDataNew.velocity += delVelNav;
535 
536  // apply a trapezoidal integration to velocities to calculate position, applying correction required to track EKF solution
537  outputDataNew.position += (outputDataNew.velocity + lastVelocity) * (imuDataNew.delVelDT*0.5f) + velCorrection * imuDataNew.delVelDT;
538 
539  // store the output in the FIFO buffer if this is a filter update step
540  if (runUpdates) {
541  StoreOutput();
542  }
543 
544  // extract data at the fusion time horizon from the FIFO buffer
545  RecallOutput();
546 
547  // compare quaternion data with EKF quaternion at the fusion time horizon and calculate correction
548 
549  // divide the demanded quaternion by the estimated to get the error
550  Quaternion quatErr = stateStruct.quat / outputDataDelayed.quat;
551 
552  // Convert to a delta rotation using a small angle approximation
553  quatErr.normalize();
554  Vector3f deltaAngErr;
555  float scaler;
556  if (quatErr[0] >= 0.0f) {
557  scaler = 2.0f;
558  } else {
559  scaler = -2.0f;
560  }
561  deltaAngErr.x = scaler * quatErr[1];
562  deltaAngErr.y = scaler * quatErr[2];
563  deltaAngErr.z = scaler * quatErr[3];
564 
565  // multiply the angle error vector by a gain to calculate the delta angle correction required to track the EKF solution
566  const float Kang = 1.0f;
567  delAngCorrection = deltaAngErr * imuDataNew.delAngDT * Kang;
568 
569  // multiply velocity error by a gain to calculate the delta velocity correction required to track the EKF solution
570  const float Kvel = 1.0f;
571  delVelCorrection = (stateStruct.velocity - outputDataDelayed.velocity) * imuDataNew.delVelDT * Kvel;
572 
573  // multiply position error by a gain to calculate the velocity correction required to track the EKF solution
574  const float Kpos = 1.0f;
575  velCorrection = (stateStruct.position - outputDataDelayed.position) * Kpos;
576 
577  // update vertical velocity and position states used to provide a vertical position derivative output
578  // using a simple complementary filter
579  float lastPosDownDerivative = posDownDerivative;
580  posDownDerivative = 2.0f * (outputDataNew.position.z - posDown);
581  posDown += (posDownDerivative + lastPosDownDerivative + 2.0f*delVelNav.z) * (imuDataNew.delVelDT*0.5f);
582 }
