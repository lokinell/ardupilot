144  perf_counter_t perf_loop = perf_alloc(PC_ELAPSED, "APM_loop");
145  perf_counter_t perf_overrun = perf_alloc(PC_COUNT, "APM_overrun");
146  struct hrt_call loop_overtime_call;
147 
148  thread_running = true;
149 
150  /*
151  switch to high priority for main loop
152  */
153  hal_px4_set_priority(APM_MAIN_PRIORITY);
154 
155  while (!_px4_thread_should_exit) {
156  perf_begin(perf_loop);
157 
158  /*
159  this ensures a tight loop waiting on a lower priority driver
160  will eventually give up some time for the driver to run. It
161  will only ever be called if a loop() call runs for more than
162  0.1 second
163  */
164  hrt_call_after(&loop_overtime_call, 100000, (hrt_callout)loop_overtime, NULL);
165 
166  g_callbacks->loop();
167 
168  if (px4_ran_overtime) {
169  /*
170  we ran over 1s in loop(), and our priority was lowered
171  to let a driver run. Set it back to high priority now.
172  */
173  hal_px4_set_priority(APM_MAIN_PRIORITY);
174  perf_count(perf_overrun);
175  px4_ran_overtime = false;
176  }
177 
178  perf_end(perf_loop);
179 
180  /*
181  give up 250 microseconds of time, to ensure drivers get a
182  chance to run. This relies on the accurate semaphore wait
183  using hrt in semaphore.cpp
184  */
185  hal.scheduler->delay_microseconds(250);
186  }
