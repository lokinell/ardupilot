169 // calibrate the airspeed. This must be called at least once before
170 // the get_airspeed() interface can be used
171 void AP_Airspeed::calibrate(bool in_startup)
172 {
173  float sum = 0;
174  uint8_t count = 0;
175  if (!_enable) {
176  return;
177  }
178  if (in_startup && _skip_cal) {
179  return;
180  }
181  // discard first reading
182  get_pressure();
183  for (uint8_t i = 0; i < 10; i++) {
184  hal.scheduler->delay(100);
185  float p = get_pressure();
186  if (_healthy) {
187  sum += p;
188  count++;
189  }
190  }
191  if (count == 0) {
192  // unhealthy sensor
193  hal.console->println("Airspeed sensor unhealthy");
194  _offset.set(0);
195  return;
196  }
197  float raw = sum/count;
198  _offset.set_and_save(raw);
199  _airspeed = 0;
200  _raw_airspeed = 0;
201 }

203 // read the airspeed sensor
204 void AP_Airspeed::read(void)
205 {
206  float airspeed_pressure;
207  if (!_enable) {
208  return;
209  }
210  airspeed_pressure = get_pressure() - _offset;
211 
212  // remember raw pressure for logging
213  _raw_pressure = airspeed_pressure;
214 
215  /*
216  we support different pitot tube setups so used can choose if
217  they want to be able to detect pressure on the static port
218  */
219  switch ((enum pitot_tube_order)_tube_order.get()) {
220  case PITOT_TUBE_ORDER_NEGATIVE:
221  airspeed_pressure = -airspeed_pressure;
222  // no break
223  case PITOT_TUBE_ORDER_POSITIVE:
224  if (airspeed_pressure < -32) {
225  // we're reading more than about -8m/s. The user probably has
226  // the ports the wrong way around
227  _healthy = false;
228  }
229  break;
230  case PITOT_TUBE_ORDER_AUTO:
231  default:
232  airspeed_pressure = fabsf(airspeed_pressure);
233  break;
234  }
235  airspeed_pressure = max(airspeed_pressure, 0);
236  _last_pressure = airspeed_pressure;
237  _raw_airspeed = sqrtf(airspeed_pressure * _ratio);
238  _airspeed = 0.7f * _airspeed + 0.3f * _raw_airspeed;
239  _last_update_ms = hal.scheduler->millis();
240 }
