16 /*
17  This is an INS driver for the combination L3G4200D gyro and ADXL345 accelerometer.
18  This combination is available as a cheap 10DOF sensor on ebay
19 
20  This sensor driver is an example only - it should not be used in any
21  serious autopilot as the latencies on I2C prevent good timing at
22  high sample rates. It is useful when doing an initial port of
23  ardupilot to a board where only I2C is available, and a cheap sensor
24  can be used.
25 
26 Datasheets:
27  ADXL345 Accelerometer http://www.analog.com/static/imported-files/data_sheets/ADXL345.pdf
28  L3G4200D gyro http://www.st.com/st-web-ui/static/active/en/resource/technical/document/datasheet/CD00265057.pdf
29 */

149 bool AP_InertialSensor_L3G4200D::_init_sensor(void)
150 {
151  // get pointer to i2c bus semaphore
152  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
153 
154  // take i2c bus sempahore
155  if (!i2c_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER))
156  return false;
157 
158  // Init the accelerometer
159  uint8_t data = 0;
160  hal.i2c->readRegister(ADXL345_ACCELEROMETER_ADDRESS, ADXL345_ACCELEROMETER_ADXLREG_DEVID, &data);
161  if (data != ADXL345_ACCELEROMETER_XL345_DEVID) {
162  hal.scheduler->panic("AP_InertialSensor_L3G4200D: could not find ADXL345 accelerometer sensor");
163  }
164  hal.i2c->writeRegister(ADXL345_ACCELEROMETER_ADDRESS, ADXL345_ACCELEROMETER_ADXLREG_POWER_CTL, 0x00);
165  hal.scheduler->delay(5);
166  hal.i2c->writeRegister(ADXL345_ACCELEROMETER_ADDRESS, ADXL345_ACCELEROMETER_ADXLREG_POWER_CTL, 0xff);
167  hal.scheduler->delay(5);
168  // Measure mode:
169  hal.i2c->writeRegister(ADXL345_ACCELEROMETER_ADDRESS, ADXL345_ACCELEROMETER_ADXLREG_POWER_CTL, 0x08);
170  hal.scheduler->delay(5);
171 
172  // Full resolution, 8g:
173  // Caution, this must agree with ADXL345_ACCELEROMETER_SCALE_1G
174  // In full resoution mode, the scale factor need not change
175  hal.i2c->writeRegister(ADXL345_ACCELEROMETER_ADDRESS, ADXL345_ACCELEROMETER_ADXLREG_DATA_FORMAT, 0x08);
176  hal.scheduler->delay(5);
177 
178  // Normal power, 800Hz Output Data Rate, 400Hz bandwidth:
179  hal.i2c->writeRegister(ADXL345_ACCELEROMETER_ADDRESS, ADXL345_ACCELEROMETER_ADXLREG_BW_RATE, 0x0d);
180  hal.scheduler->delay(5);
181 
182  // enable FIFO in stream mode. This will allow us to read the accelerometers much less frequently
183  hal.i2c->writeRegister(ADXL345_ACCELEROMETER_ADDRESS,
184  ADXL345_ACCELEROMETER_ADXLREG_FIFO_CTL,
185  ADXL345_ACCELEROMETER_ADXLREG_FIFO_CTL_STREAM);
186 
187  // Init the Gyro
188  // Expect to read the right 'WHO_AM_I' value
189  hal.i2c->readRegister(L3G4200D_I2C_ADDRESS, L3G4200D_REG_WHO_AM_I, &data);
190  if (data != L3G4200D_REG_WHO_AM_I_VALUE)
191  hal.scheduler->panic("AP_InertialSensor_L3G4200D: could not find L3G4200D gyro sensor");
192 
193  // setup for 800Hz sampling with 110Hz filter
194  hal.i2c->writeRegister(L3G4200D_I2C_ADDRESS,
195  L3G4200D_REG_CTRL_REG1,
196  L3G4200D_REG_CTRL_REG1_DRBW_800_110 |
197  L3G4200D_REG_CTRL_REG1_PD |
198  L3G4200D_REG_CTRL_REG1_XYZ_ENABLE);
199  hal.scheduler->delay(1);
200 
201  hal.i2c->writeRegister(L3G4200D_I2C_ADDRESS,
202  L3G4200D_REG_CTRL_REG1,
203  L3G4200D_REG_CTRL_REG1_DRBW_800_110 |
204  L3G4200D_REG_CTRL_REG1_PD |
205  L3G4200D_REG_CTRL_REG1_XYZ_ENABLE);
206  hal.scheduler->delay(1);
207 
208  hal.i2c->writeRegister(L3G4200D_I2C_ADDRESS,
209  L3G4200D_REG_CTRL_REG1,
210  L3G4200D_REG_CTRL_REG1_DRBW_800_110 |
211  L3G4200D_REG_CTRL_REG1_PD |
212  L3G4200D_REG_CTRL_REG1_XYZ_ENABLE);
213  hal.scheduler->delay(1);
214 
215  // setup for 2000 degrees/sec full range
216  hal.i2c->writeRegister(L3G4200D_I2C_ADDRESS,
217  L3G4200D_REG_CTRL_REG4,
218  L3G4200D_REG_CTRL_REG4_FS_2000);
219  hal.scheduler->delay(1);
220 
221  // enable FIFO
222  hal.i2c->writeRegister(L3G4200D_I2C_ADDRESS,
223  L3G4200D_REG_CTRL_REG5,
224  L3G4200D_REG_CTRL_REG5_FIFO_EN);
225  hal.scheduler->delay(1);
226 
227  // enable FIFO in stream mode. This will allow us to read the gyros much less frequently
228  hal.i2c->writeRegister(L3G4200D_I2C_ADDRESS,
229  L3G4200D_REG_FIFO_CTL,
230  L3G4200D_REG_FIFO_CTL_STREAM);
231  hal.scheduler->delay(1);
232 
233 
234  // Set up the filter desired
235  _set_filter_frequency(_accel_filter_cutoff());
236 
237  // give back i2c semaphore
238  i2c_sem->give();
239 
240  // start the timer process to read samples
241  hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_InertialSensor_L3G4200D::_accumulate, void));
242 
243  _gyro_instance = _imu.register_gyro();
244  _accel_instance = _imu.register_accel();
245 
246  _product_id = AP_PRODUCT_ID_L3G4200D;
247 
248  return true;
249 }

263 bool AP_InertialSensor_L3G4200D::update(void)
264 {
265  Vector3f accel, gyro;
266 
267  pthread_spin_lock(&_data_lock);
268  unsigned int idx = !_data_idx;
269  accel = _data[idx].accel_filtered;
270  gyro = _data[idx].gyro_filtered;
271  _have_sample = false;
272  pthread_spin_unlock(&_data_lock);
273 
274  _publish_accel(_accel_instance, accel);
275  _publish_gyro(_gyro_instance, gyro);
276 
277  if (_last_filter_hz != _accel_filter_cutoff()) {
278  _set_filter_frequency(_accel_filter_cutoff());
279  _last_filter_hz = _accel_filter_cutoff();
280  }
281 
282  return true;
283 }

285 // Accumulate values from accels and gyros
286 void AP_InertialSensor_L3G4200D::_accumulate(void)
287 {
288  // get pointer to i2c bus semaphore
289  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
290 
291  // take i2c bus sempahore
292  if (!i2c_sem->take_nonblocking())
293  return;
294 
295  uint8_t num_samples_available;
296  uint8_t fifo_status = 0;
297 
298  // Read gyro FIFO status
299  fifo_status = 0;
300  hal.i2c->readRegister(L3G4200D_I2C_ADDRESS,
301  L3G4200D_REG_FIFO_SRC,
302  &fifo_status);
303  if (fifo_status & L3G4200D_REG_FIFO_SRC_OVERRUN) {
304  // FIFO is full
305  num_samples_available = 32;
306  } else if (fifo_status & L3G4200D_REG_FIFO_SRC_EMPTY) {
307  // FIFO is empty
308  num_samples_available = 0;
309  } else {
310  // FIFO is partly full
311  num_samples_available = fifo_status & L3G4200D_REG_FIFO_SRC_ENTRIES_MASK;
312  }
313 
314  if (num_samples_available > 0) {
315  // read all the entries in one go, using AUTO_INCREMENT. This saves a lot of time on I2C setup
316  int16_t buffer[num_samples_available][3];
317  if (hal.i2c->readRegisters(L3G4200D_I2C_ADDRESS, L3G4200D_REG_XL | L3G4200D_REG_AUTO_INCREMENT,
318  sizeof(buffer), (uint8_t *)&buffer[0][0]) == 0) {
319  for (uint8_t i=0; i<num_samples_available; i++) {
320  Vector3f gyro = Vector3f(buffer[i][0], -buffer[i][1], -buffer[i][2]);
321  // Adjust for chip scaling to get radians/sec
322  gyro *= L3G4200D_GYRO_SCALE_R_S;
323  _rotate_and_correct_gyro(_gyro_instance, gyro);
324  _notify_new_gyro_raw_sample(_gyro_instance, gyro);
325  _data[_data_idx].gyro_filtered = _gyro_filter.apply(gyro);
326  _have_gyro_sample = true;
327  }
328  }
329  }
330 
331  // Read accelerometer FIFO to find out how many samples are available
332  hal.i2c->readRegister(ADXL345_ACCELEROMETER_ADDRESS,
333  ADXL345_ACCELEROMETER_ADXLREG_FIFO_STATUS,
334  &fifo_status);
335  num_samples_available = fifo_status & 0x3F;
336 
337  // read the samples and apply the filter
338  if (num_samples_available > 0) {
339  int16_t buffer[num_samples_available][3];
340  if (hal.i2c->readRegistersMultiple(ADXL345_ACCELEROMETER_ADDRESS,
341  ADXL345_ACCELEROMETER_ADXLREG_DATAX0,
342  sizeof(buffer[0]), num_samples_available,
343  (uint8_t *)&buffer[0][0]) == 0) {
344  for (uint8_t i=0; i<num_samples_available; i++) {
345  Vector3f accel = Vector3f(buffer[i][0], -buffer[i][1], -buffer[i][2]);
346  // Adjust for chip scaling to get m/s/s
347  accel *= ADXL345_ACCELEROMETER_SCALE_M_S;
348  _rotate_and_correct_accel(_accel_instance, accel);
349  _notify_new_accel_raw_sample(_accel_instance, accel);
350  _data[_data_idx].accel_filtered = _accel_filter.apply(accel);
351  _have_accel_sample = true;
352  }
353  }
354  }
355 
356  // give back i2c semaphore
357  i2c_sem->give();
358 
359  if (_have_accel_sample && _have_gyro_sample) {
360  _have_gyro_sample = false;
361  _have_accel_sample = false;
362 
363  pthread_spin_lock(&_data_lock);
364  _data_idx = !_data_idx;
365  _have_sample = true;
366  pthread_spin_unlock(&_data_lock);
367  }
368 }
