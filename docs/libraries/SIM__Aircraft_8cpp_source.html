151 /*
152  try to synchronise simulation time with wall clock time, taking
153  into account desired speedup
154  This tries to take account of possible granularity of
155  get_wall_time_us() so it works reasonably well on windows
156 */
157 void Aircraft::sync_frame_time(void)
158 {
159  frame_counter++;
160  uint64_t now = get_wall_time_us();
161  if (frame_counter >= 40 &&
162  now > last_wall_time_us) {
163  float rate = frame_counter * 1.0e6f/(now - last_wall_time_us);
164  achieved_rate_hz = (0.99f*achieved_rate_hz) + (0.01f*rate);
165  if (achieved_rate_hz < rate_hz * target_speedup) {
166  scaled_frame_time_us *= 0.999f;
167  } else {
168  scaled_frame_time_us /= 0.999f;
169  }
170 #if 0
171  ::printf("achieved_rate_hz=%.3f rate=%.2f rate_hz=%.3f sft=%.1f\n",
172  (double)achieved_rate_hz,
173  (double)rate,
174  (double)rate_hz,
175  (double)scaled_frame_time_us);
176 #endif
177  uint32_t sleep_time = scaled_frame_time_us*frame_counter;
178  if (sleep_time > min_sleep_time) {
179  usleep(sleep_time);
180  }
181  last_wall_time_us = now;
182  frame_counter = 0;
183  }
184 }

197 /*
198  normal distribution random numbers
199  See
200  http://en.literateprograms.org/index.php?title=Special:DownloadCode/Box-Muller_transform_%28C%29&oldid=7011
201 */
202 double Aircraft::rand_normal(double mean, double stddev)
203 {
204  static double n2 = 0.0;
205  static int n2_cached = 0;
206  if (!n2_cached)
207  {
208  double x, y, r;
209  do
210  {
211  x = 2.0*rand()/RAND_MAX - 1;
212  y = 2.0*rand()/RAND_MAX - 1;
213 
214  r = x*x + y*y;
215  }
216  while (r == 0.0 || r > 1.0);
217  {
218  double d = sqrt(-2.0*log(r)/r);
219  double n1 = x*d;
220  n2 = y*d;
221  double result = n1*stddev + mean;
222  n2_cached = 1;
223  return result;
224  }
225  }
226  else
227  {
228  n2_cached = 0;
229  return n2*stddev + mean;
230  }
231 }

264 uint64_t Aircraft::get_wall_time_us() const
265 {
266 #ifdef __CYGWIN__
267  static DWORD tPrev;
268  static uint64_t last_ret_us;
269  if (tPrev == 0) {
270  tPrev = timeGetTime();
271  return 0;
272  }
273  DWORD now = timeGetTime();
274  last_ret_us += (uint64_t)((now - tPrev)*1000UL);
275  tPrev = now;
276  return last_ret_us;
277 #else
