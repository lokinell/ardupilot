103 MultiCopter::MultiCopter(const char *home_str, const char *frame_str) :
104  Aircraft(home_str, frame_str),
105  frame(NULL),
106  hover_throttle(0.51),
107  terminal_velocity(15.0),
108  terminal_rotation_rate(4*radians(360.0))
109 {
110  for (uint8_t i=0; i < ARRAY_SIZE(supported_frames); i++) {
111  if (strcasecmp(frame_str, supported_frames[i].name) == 0) {
112  frame = &supported_frames[i];
113  }
114  }
115  if (frame == NULL) {
116  printf("Frame '%s' not found", frame_str);
117  exit(1);
118  }
119  /*
120  scaling from total motor power to Newtons. Allows the copter
121  to hover against gravity when each motor is at hover_throttle
122  */
123  mass = 1.5;
124  thrust_scale = (mass * GRAVITY_MSS) / (frame->num_motors * hover_throttle);
125 
126  frame_height = 0.1;
127 }

130  update the multicopter simulation by one time step
131  */
132 void MultiCopter::update(const struct sitl_input &input)
133 {
134  float motor_speed[frame->num_motors];
135 
136  for (uint8_t i=0; i<frame->num_motors; i++) {
137  uint16_t servo = input.servos[frame->motors[i].servo-1];
138  // assume 1000 to 2000 PWM range
139  if (servo <= 1000) {
140  motor_speed[i] = 0;
141  } else {
142  motor_speed[i] = (servo-1000) / 1000.0f;
143  }
144  }
145 
146  // how much time has passed?
147  float delta_time = frame_time_us * 1.0e-6f;
148 
149  // rotational acceleration, in rad/s/s, in body frame
150  Vector3f rot_accel;
151  float thrust = 0.0f; // newtons
152 
153  for (uint8_t i=0; i<frame->num_motors; i++) {
154  rot_accel.x += -radians(5000.0) * sinf(radians(frame->motors[i].angle)) * motor_speed[i];
155  rot_accel.y += radians(5000.0) * cosf(radians(frame->motors[i].angle)) * motor_speed[i];
156  if (frame->motors[i].clockwise) {
157  rot_accel.z -= motor_speed[i] * radians(400.0);
158  } else {
159  rot_accel.z += motor_speed[i] * radians(400.0);
160  }
161  thrust += motor_speed[i] * thrust_scale; // newtons
162  }
163 
164  // rotational air resistance
165  rot_accel.x -= gyro.x * radians(5000.0) / terminal_rotation_rate;
166  rot_accel.y -= gyro.y * radians(5000.0) / terminal_rotation_rate;
167  rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
168 
169  // update rotational rates in body frame
170  gyro += rot_accel * delta_time;
171 
172  // update attitude
173  dcm.rotate(gyro * delta_time);
174  dcm.normalize();
175 
176  // air resistance
177  Vector3f air_resistance = -velocity_ef * (GRAVITY_MSS/terminal_velocity);
178 
179  accel_body = Vector3f(0, 0, -thrust / mass);
180  Vector3f accel_earth = dcm * accel_body;
181 
182  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
183  accel_earth += air_resistance;
184 
185  // if we're on the ground, then our vertical acceleration is limited
186  // to zero. This effectively adds the force of the ground on the aircraft
187  if (on_ground(position) && accel_earth.z > 0) {
188  accel_earth.z = 0;
189  }
190 
191  // work out acceleration as seen by the accelerometers. It sees the kinematic
192  // acceleration (ie. real movement), plus gravity
193  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
194 
195  // add some noise
196  add_noise(thrust / (thrust_scale * frame->num_motors));
197 
198  // new velocity vector
199  velocity_ef += accel_earth * delta_time;
200 
201  // new position vector
202  Vector3f old_position = position;
203  position += velocity_ef * delta_time;
204 
205  // assume zero wind for now
206  airspeed = velocity_ef.length();
207 
208  // constrain height to the ground
209  if (on_ground(position)) {
210  if (!on_ground(old_position)) {
211  printf("Hit ground at %f m/s\n", velocity_ef.z);
212 
213  velocity_ef.zero();
214 
215  // zero roll/pitch, but keep yaw
216  float r, p, y;
217  dcm.to_euler(&r, &p, &y);
218  dcm.from_euler(0, 0, y);
219 
220  position.z = -(ground_level + frame_height - home.alt*0.01f);
221  }
222  }
223 
224  // update lat/lon/altitude
225  update_position();
226 }
