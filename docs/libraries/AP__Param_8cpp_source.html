53 // the 'GROUP_ID' of a element of a group is the 18 bit identifier
54 // used to distinguish between this element of the group and other
55 // elements of the same group. It is calculated using a bit shift per
56 // level of nesting, so the first level of nesting gets 6 bits the 2nd
57 // level gets the next 6 bits, and the 3rd level gets the last 6
58 // bits. This limits groups to having at most 64 elements.
59 #define GROUP_ID(grpinfo, base, i, shift) ((base)+(((uint16_t)PGM_UINT8(&grpinfo[i].idx))<<(shift)))

123 bool AP_Param::check_group_info(const struct AP_Param::GroupInfo * group_info,
124  uint16_t * total_size,
125  uint8_t group_shift,
126  uint8_t prefix_length)
127 {
128  uint8_t type;
129  int8_t max_idx = -1;
130  for (uint8_t i=0;
131  (type=PGM_UINT8(&group_info[i].type)) != AP_PARAM_NONE;
132  i++) {
133 #ifdef AP_NESTED_GROUPS_ENABLED
134  if (type == AP_PARAM_GROUP) {
135  // a nested group
136  const struct GroupInfo *ginfo = (const struct GroupInfo *)PGM_POINTER(&group_info[i].group_info);
137  if (group_shift + _group_level_shift >= _group_bits) {
138  Debug("double group nesting in %s", group_info[i].name);
139  return false;
140  }
141  if (ginfo == NULL ||
142  !check_group_info(ginfo, total_size, group_shift + _group_level_shift, prefix_length + strlen(group_info[i].name))) {
143  return false;
144  }
145  continue;
146  }
147 #endif // AP_NESTED_GROUPS_ENABLED
148  uint8_t idx = PGM_UINT8(&group_info[i].idx);
149  if (idx >= (1<<_group_level_shift)) {
150  Debug("idx too large (%u) in %s", idx, group_info[i].name);
151  return false;
152  }
153  if ((int8_t)idx <= max_idx) {
154  Debug("indexes must be in increasing order in %s", group_info[i].name);
155  return false;
156  }
157  max_idx = (int8_t)idx;
158  uint8_t size = type_size((enum ap_var_type)type);
159  if (size == 0) {
160  Debug("invalid type in %s", group_info[i].name);
161  return false;
162  }
163  if (prefix_length + strlen(group_info[i].name) > 16) {
164  Debug("suffix is too long in %s", group_info[i].name);
165  return false;
166  }
167  (*total_size) += size + sizeof(struct Param_header);
168  }
169  return true;

467 // return true if found, along with the offset in the EEPROM where
468 // the variable is stored
469 // if not found return the offset of the sentinal
470 // if the sentinal isn't found either, the offset is set to 0xFFFF
471 bool AP_Param::scan(const AP_Param::Param_header *target, uint16_t *pofs)
472 {
473  struct Param_header phdr;
474  uint16_t ofs = sizeof(AP_Param::EEPROM_header);
475  while (ofs < _storage.size()) {
476  _storage.read_block(&phdr, ofs, sizeof(phdr));
477  if (phdr.type == target->type &&
478  phdr.key == target->key &&
479  phdr.group_element == target->group_element) {
480  // found it
481  *pofs = ofs;
482  return true;
483  }
484  // note that this is an ||, not an &&, as this makes us more
485  // robust to power off while adding a variable to EEPROM
486  if (phdr.type == _sentinal_type ||

492  }

494  }

706 // Save the variable to EEPROM, if supported
707 //
708 bool AP_Param::save(bool force_save)
709 {
710  uint32_t group_element = 0;
711  const struct GroupInfo *ginfo;
712  uint8_t idx;
713  const struct AP_Param::Info *info = find_var_info(&group_element, &ginfo, &idx);
714  const AP_Param *ap;
715 
716  if (info == NULL) {
717  // we don't have any info on how to store it
718  return false;
719  }
720 
721  struct Param_header phdr;
722 
723  // create the header we will use to store the variable
724  if (ginfo != NULL) {
725  phdr.type = PGM_UINT8(&ginfo->type);
726  } else {
727  phdr.type = PGM_UINT8(&info->type);
728  }
729  phdr.key = PGM_UINT8(&info->key);
730  phdr.group_element = group_element;
731 
732  ap = this;
733  if (phdr.type != AP_PARAM_VECTOR3F && idx != 0) {
734  // only vector3f can have non-zero idx for now
735  return false;
736  }
737  if (idx != 0) {
738  ap = (const AP_Param *)((uintptr_t)ap) - (idx*sizeof(float));
739  }
740 
741  char name[AP_MAX_NAME_SIZE+1];
742  copy_name_info(info, ginfo, idx, name, sizeof(name), true);
743 
744  // scan EEPROM to find the right location
745  uint16_t ofs;
746  if (scan(&phdr, &ofs)) {
747  // found an existing copy of the variable
748  eeprom_write_check(ap, ofs+sizeof(phdr), type_size((enum ap_var_type)phdr.type));
749  send_parameter(name, (enum ap_var_type)phdr.type);
750  return true;
751  }
752  if (ofs == (uint16_t) ~0) {
753  return false;
754  }
755 
756  // if the value is the default value then don't save
757  if (phdr.type <= AP_PARAM_FLOAT) {
758  float v1 = cast_to_float((enum ap_var_type)phdr.type);
759  float v2;
760  if (ginfo != NULL) {
761  v2 = get_default_value(&ginfo->def_value);
762  } else {
763  v2 = get_default_value(&info->def_value);
764  }
765  if (is_equal(v1,v2) && !force_save) {
766  GCS_MAVLINK::send_parameter_value_all(name, (enum ap_var_type)info->type, v2);
767  return true;
768  }
769  if (phdr.type != AP_PARAM_INT32 &&
770  (fabsf(v1-v2) < 0.0001f*fabsf(v1))) {
771  // for other than 32 bit integers, we accept values within
772  // 0.01 percent of the current value as being the same
773  GCS_MAVLINK::send_parameter_value_all(name, (enum ap_var_type)info->type, v2);
774  return true;
775  }
776  }
777 
778  if (ofs+type_size((enum ap_var_type)phdr.type)+2*sizeof(phdr) >= _storage.size()) {
779  // we are out of room for saving variables
780  hal.console->println("EEPROM full");
781  return false;
782  }
783 
784  // write a new sentinal, then the data, then the header
785  write_sentinal(ofs + sizeof(phdr) + type_size((enum ap_var_type)phdr.type));
786  eeprom_write_check(ap, ofs+sizeof(phdr), type_size((enum ap_var_type)phdr.type));
787  eeprom_write_check(&phdr, ofs, sizeof(phdr));
788 
789  send_parameter(name, (enum ap_var_type)phdr.type);
790  return true;
791 }
