55  int r;
56 
57  pthread_attr_init(&attr);
58  /*
59  we need to run as root to get realtime scheduling. Allow it to

292 void *Scheduler::_timer_thread(void* arg)
293 {
294  Scheduler* sched = (Scheduler *)arg;
295 
296  while (sched->system_initializing()) {
297  poll(NULL, 0, 1);
298  }
299  /*
300  this aims to run at an average of 1kHz, so that it can be used
301  to drive 1kHz processes without drift
302  */
303  uint64_t next_run_usec = sched->micros64() + 1000;
304  while (true) {
305  uint64_t dt = next_run_usec - sched->micros64();
306  if (dt > 2000) {
307  // we've lost sync - restart
308  next_run_usec = sched->micros64();
309  } else {
310  sched->_microsleep(dt);
311  }
312  next_run_usec += 1000;
313  // run registered timers
314  sched->_run_timers(true);
315  }
316  return NULL;
317 }
