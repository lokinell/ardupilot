61  /* work out the new rate mask. The PX4IO board has 3 groups of servos.
62 
63  Group 0: channels 0 1
64  Group 1: channels 4 5 6 7
65  Group 2: channels 2 3
66 
67  Channels within a group must be set to the same rate.
68 
69  For the moment we never set the channels above 8 to more than
70  50Hz
71  */
72  if (freq_hz > 50) {

74  _rate_mask |= chmask & 0xFF;
75  if (_rate_mask & 0x07) {
76  _rate_mask |= 0x07;
77  }
78  if (_rate_mask & 0x38) {
79  _rate_mask |= 0x38;
80  }
81  if (_rate_mask & 0xC0) {
82  _rate_mask |= 0xC0;
83  }
84  } else {
85  // we are setting low rates on the given channels
86  if (chmask & 0x07) {
87  _rate_mask &= ~0x07;
88  }
