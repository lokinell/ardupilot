197 // returns just the PWM without the offset from radio_min
198 void
199 RC_Channel::calc_pwm(void)
200 {
201  if(_type == RC_CHANNEL_TYPE_RANGE) {
202  pwm_out = range_to_pwm();
203  radio_out = (_reverse >= 0) ? (radio_min + pwm_out) : (radio_max - pwm_out);
204 
205  }else if(_type == RC_CHANNEL_TYPE_ANGLE_RAW) {
206  pwm_out = (float)servo_out * 0.1f;
207  int16_t reverse_mul = (_reverse==-1?-1:1);
208  radio_out = (pwm_out * reverse_mul) + radio_trim;
209 
210  }else{ // RC_CHANNEL_TYPE_ANGLE
211  pwm_out = angle_to_pwm();
212  radio_out = pwm_out + radio_trim;
213  }
214 
215  radio_out = constrain_int16(radio_out, radio_min.get(), radio_max.get());
216 }

277 /*
278  return an "angle in centidegrees" (normally -4500 to 4500) from
279  the current radio_in value using the specified dead_zone
280  */
281 int16_t
282 RC_Channel::pwm_to_angle_dz(uint16_t dead_zone)
283 {
284  int16_t radio_trim_high = radio_trim + dead_zone;
285  int16_t radio_trim_low = radio_trim - dead_zone;
286 
287  // prevent div by 0
288  if ((radio_trim_low - radio_min) == 0 || (radio_max - radio_trim_high) == 0)
289  return 0;
290 
291  int16_t reverse_mul = (_reverse==-1?-1:1);
292  if(radio_in > radio_trim_high) {
293  return reverse_mul * ((int32_t)_high * (int32_t)(radio_in - radio_trim_high)) / (int32_t)(radio_max - radio_trim_high);
294  }else if(radio_in < radio_trim_low) {
295  return reverse_mul * ((int32_t)_high * (int32_t)(radio_in - radio_trim_low)) / (int32_t)(radio_trim_low - radio_min);
296  }else
297  return 0;
298 }

322 /*
323  convert a pulse width modulation value to a value in the configured
324  range, using the specified deadzone
325  */
326 int16_t
327 RC_Channel::pwm_to_range_dz(uint16_t dead_zone)
328 {
329  int16_t r_in = constrain_int16(radio_in, radio_min.get(), radio_max.get());
330 
331  if (_reverse == -1) {

334

336

367 float
368 RC_Channel::norm_input()
369 {
370  float ret;
371  int16_t reverse_mul = (_reverse==-1?-1:1);
372  if (radio_in < radio_trim) {
373  ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_trim - radio_min);
374  } else {
375  ret = reverse_mul * (float)(radio_in - radio_trim) / (float)(radio_max - radio_trim);
376  }
377  return constrain_float(ret, -1.0f, 1.0f);
378 }

382 {

385  float ret;

389  } else if (radio_in > dz_max && radio_max > dz_max) {

391  } else {
392  ret = 0;
393  }
394  return constrain_float(ret, -1.0f, 1.0f);

397 /*
398  get percentage input from 0 to 100. This ignores the trim value.
399  */
400 uint8_t
401 RC_Channel::percent_input()
402 {
403  if (radio_in <= radio_min) {

405  }

407  return _reverse==-1?0:100;
408  }

416 float
417 RC_Channel::norm_output()
418 {
419  int16_t mid = (radio_max + radio_min) / 2;
420  float ret;
421  if (mid <= radio_min) {
422  return 0;
423  }
424  if (radio_out < mid) {
425  ret = (float)(radio_out - mid) / (float)(mid - radio_min);
426  } else if (radio_out > mid) {
427  ret = (float)(radio_out - mid) / (float)(radio_max - mid);
428  } else {
429  ret = 0;
430  }
431  if (_reverse == -1) {
432  ret = -ret;
433  }
434  return ret;
435 }
