86 void PX4RCOutput::set_freq(uint32_t chmask, uint16_t freq_hz)
87 {
88  // we can't set this per channel yet
89  if (freq_hz > 50) {
90  // we're being asked to set the alt rate
91  if (ioctl(_pwm_fd, PWM_SERVO_SET_UPDATE_RATE, (unsigned long)freq_hz) != 0) {
92  hal.console->printf("RCOutput: Unable to set alt rate to %uHz\n", (unsigned)freq_hz);
93  return;
94  }
95  _freq_hz = freq_hz;
96  }
97 
98  /* work out the new rate mask. The PX4IO board has 3 groups of servos.
99 
100  Group 0: channels 0 1
101  Group 1: channels 4 5 6 7
102  Group 2: channels 2 3
103 
104  Channels within a group must be set to the same rate.
105 
106  For the moment we never set the channels above 8 to more than
107  50Hz
108  */
109  if (freq_hz > 50) {
110  // we are setting high rates on the given channels
111  _rate_mask |= chmask & 0xFF;
112  if (_rate_mask & 0x3) {
113  _rate_mask |= 0x3;
114  }
115  if (_rate_mask & 0xc) {
116  _rate_mask |= 0xc;
117  }
118  if (_rate_mask & 0xF0) {
119  _rate_mask |= 0xF0;
120  }
121  } else {
122  // we are setting low rates on the given channels
123  if (chmask & 0x3) {
124  _rate_mask &= ~0x3;
125  }
126  if (chmask & 0xc) {
127  _rate_mask &= ~0xc;
128  }
129  if (chmask & 0xf0) {
130  _rate_mask &= ~0xf0;
131  }
132  }
133 
134  if (ioctl(_pwm_fd, PWM_SERVO_SET_SELECT_UPDATE_RATE, _rate_mask) != 0) {
135  hal.console->printf("RCOutput: Unable to set alt rate mask to 0x%x\n", (unsigned)_rate_mask);
136  }

239 {
240  if (ch >= PX4_NUM_OUTPUT_CHANNELS) {

242  }

289 void PX4RCOutput::_publish_actuators(void)
290 {
291  struct actuator_direct_s actuators;
292 
293  if (_esc_pwm_min == 0 ||
294  _esc_pwm_max == 0) {
295  // not initialised yet
296  return;
297  }
298 
299  actuators.nvalues = _max_channel;
300  if (actuators.nvalues > actuators.NUM_ACTUATORS_DIRECT) {
301  actuators.nvalues = actuators.NUM_ACTUATORS_DIRECT;
302  }
303  // don't publish more than 8 actuators for now, as the uavcan ESC
304  // driver refuses to update any motors if you try to publish more
305  // than 8
306  if (actuators.nvalues > 8) {
307  actuators.nvalues = 8;
308  }
309  bool armed = hal.util->get_soft_armed();
310  actuators.timestamp = hrt_absolute_time();
311  for (uint8_t i=0; i<actuators.nvalues; i++) {
312  if (!armed) {
313  actuators.values[i] = 0;
314  } else {
315  actuators.values[i] = (_period[i] - _esc_pwm_min) / (float)(_esc_pwm_max - _esc_pwm_min);
316  }
317  // actuator values are from -1 to 1
318  actuators.values[i] = actuators.values[i]*2 - 1;
319  }
320 
321  if (_actuator_direct_pub == NULL) {
322  _actuator_direct_pub = orb_advertise(ORB_ID(actuator_direct), &actuators);
323  } else {
324  orb_publish(ORB_ID(actuator_direct), _actuator_direct_pub, &actuators);
325  }
326  if (hal.util->safety_switch_state() != AP_HAL::Util::SAFETY_DISARMED) {
327  _arm_actuators(true);
328  }

331 void PX4RCOutput::_timer_tick(void)
332 {
333  uint32_t now = hal.scheduler->micros();
334 
335  if ((_enabled_channels & ((1U<<_servo_count)-1)) == 0) {
336  // no channels enabled
337  _arm_actuators(false);
338  goto update_pwm;
339  }
340 
341  // always send at least at 20Hz, otherwise the IO board may think
342  // we are dead
343  if (now - _last_output > 50000) {
344  _need_update = true;
345  }
346 
347  if (_need_update && _pwm_fd != -1) {
348  _need_update = false;
349  perf_begin(_perf_rcout);
350  if (_max_channel <= _servo_count) {
351  ::write(_pwm_fd, _period, _max_channel*sizeof(_period[0]));
352  } else {
353  // we're using both sets of outputs
354  ::write(_pwm_fd, _period, _servo_count*sizeof(_period[0]));
355  if (_alt_fd != -1 && _alt_servo_count > 0) {
356  uint8_t n = _max_channel - _servo_count;
357  if (n > _alt_servo_count) {
358  n = _alt_servo_count;
359  }
360  ::write(_alt_fd, &_period[_servo_count], n*sizeof(_period[0]));
361  }
362  }
363 
364  // also publish to actuator_direct
365  _publish_actuators();
366 
367  perf_end(_perf_rcout);
368  _last_output = now;
369  }
370 
371 update_pwm:
372  for (uint8_t i=0; i<ORB_MULTI_MAX_INSTANCES; i++) {
373  bool rc_updated = false;
374  if (_outputs[i].pwm_sub >= 0 &&
375  orb_check(_outputs[i].pwm_sub, &rc_updated) == 0 &&
376  rc_updated) {
377  orb_copy(ORB_ID(actuator_outputs), _outputs[i].pwm_sub, &_outputs[i].outputs);
378  }
379  }
380 
381 }
