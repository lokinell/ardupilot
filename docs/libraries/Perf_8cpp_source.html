95  base->name = name;

124 void Util::perf_end(perf_counter_t perf)
125 {
126  struct perf_counter_elapsed_t *perf_elapsed = (struct perf_counter_elapsed_t *)perf;
127 
128  if (perf_elapsed == NULL) {
129  return;
130  }
131 
132  if (perf_elapsed->base.type != PC_ELAPSED) {
133  hal.console->printf("perf_end() called over a perf_counter_t(%s) "
134  "that is not of the PC_ELAPSED type.\n",
135  perf_elapsed->base.name);
136  return;
137  }
138  if (perf_elapsed->start == 0) {
139  hal.console->printf("perf_end() called before an perf_begin() on %s.\n",
140  perf_elapsed->base.name);
141  return;
142  }
143 
144  struct timespec ts;
145  clock_gettime(CLOCK_MONOTONIC, &ts);
146  const uint64_t elapsed = timespec_to_nsec(&ts) - perf_elapsed->start;
147 
148  perf_elapsed->count++;
149  perf_elapsed->total += elapsed;
150 
151  if (perf_elapsed->least > elapsed) {
152  perf_elapsed->least = elapsed;
153  }
154 
155  if (perf_elapsed->most < elapsed) {
156  perf_elapsed->most = elapsed;
157  }
158 
159  /*
160  * Maintain mean and variance of interval in nanoseconds
161  * Knuth/Welford recursive mean and variance of update intervals (via Wikipedia)
162  * Same implementation of PX4.
163  */
164  const double delta_intvl = elapsed - perf_elapsed->mean;
165  perf_elapsed->mean += (delta_intvl / perf_elapsed->count);
166  perf_elapsed->m2 += (delta_intvl * (elapsed - perf_elapsed->mean));
167 
168  perf_elapsed->start = 0;
169 }
