323 void AP_InertialSensor_PX4::_get_sample()
324 {
325  for (uint8_t i=0; i<max(_num_accel_instances,_num_gyro_instances);i++) {
326  struct accel_report accel_report;
327  struct gyro_report gyro_report;
328 
329  bool gyro_valid = _get_gyro_sample(i,gyro_report);
330  bool accel_valid = _get_accel_sample(i,accel_report);
331 
332  while(gyro_valid || accel_valid) {
333  // interleave accel and gyro samples by time - this will allow sculling corrections later
334  // check the next gyro measurement to see if it needs to be integrated first
335  if(gyro_valid && accel_valid && gyro_report.timestamp <= accel_report.timestamp) {
336  _new_gyro_sample(i,gyro_report);
337  gyro_valid = _get_gyro_sample(i,gyro_report);
338  continue;
339  }
340  // if not, try to integrate an accelerometer sample
341  if(accel_valid) {
342  _new_accel_sample(i,accel_report);
343  accel_valid = _get_accel_sample(i,accel_report);
344  continue;
345  }
346  // if not, we've only got gyro samples left in the buffer
347  if(gyro_valid) {
348  _new_gyro_sample(i,gyro_report);
349  gyro_valid = _get_gyro_sample(i,gyro_report);
350  }
351  }
352  }
353  _last_get_sample_timestamp = hal.scheduler->micros64();
354 }
