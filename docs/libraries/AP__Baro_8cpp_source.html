85 // calibrate the barometer. This must be called at least once before
86 // the altitude() or climb_rate() interfaces can be used
87 void AP_Baro::calibrate()
88 {
89  // reset the altitude offset when we calibrate. The altitude
90  // offset is supposed to be for within a flight
91  _alt_offset.set_and_save(0);
92 
93  // start by assuming all sensors are calibrated (for healthy() test)
94  for (uint8_t i=0; i<_num_sensors; i++) {
95  sensors[i].calibrated = true;
96  sensors[i].alt_ok = true;
97  }
98 
99  // let the barometer settle for a full second after startup
100  // the MS5611 reads quite a long way off for the first second,
101  // leading to about 1m of error if we don't wait
102  for (uint8_t i = 0; i < 10; i++) {
103  uint32_t tstart = hal.scheduler->millis();
104  do {
105  update();
106  if (hal.scheduler->millis() - tstart > 500) {
107  hal.scheduler->panic("PANIC: AP_Baro::read unsuccessful "
108  "for more than 500ms in AP_Baro::calibrate [2]\r\n");
109  }
110  hal.scheduler->delay(10);
111  } while (!healthy());
112  hal.scheduler->delay(100);
113  }
114 
115  // now average over 5 values for the ground pressure and
116  // temperature settings
117  float sum_pressure[BARO_MAX_INSTANCES] = {0};
118  float sum_temperature[BARO_MAX_INSTANCES] = {0};
119  uint8_t count[BARO_MAX_INSTANCES] = {0};
120  const uint8_t num_samples = 5;
121 
122  for (uint8_t c = 0; c < num_samples; c++) {
123  uint32_t tstart = hal.scheduler->millis();
124  do {
125  update();
126  if (hal.scheduler->millis() - tstart > 500) {
127  hal.scheduler->panic("PANIC: AP_Baro::read unsuccessful "
128  "for more than 500ms in AP_Baro::calibrate [3]\r\n");
129  }
130  } while (!healthy());
131  for (uint8_t i=0; i<_num_sensors; i++) {
132  if (healthy(i)) {
133  sum_pressure[i] += sensors[i].pressure;
134  sum_temperature[i] += sensors[i].temperature;
135  count[i] += 1;
136  }
137  }
138  hal.scheduler->delay(100);
139  }
140  for (uint8_t i=0; i<_num_sensors; i++) {
141  if (count[i] == 0) {
142  sensors[i].calibrated = false;
143  } else {
144  sensors[i].ground_pressure.set_and_save(sum_pressure[i] / count[i]);
145  sensors[i].ground_temperature.set_and_save(sum_temperature[i] / count[i]);
146  }
147  }
148 
149  // panic if all sensors are not calibrated
150  for (uint8_t i=0; i<_num_sensors; i++) {
151  if (sensors[i].calibrated) {
152  return;
153  }
154  }
155  hal.scheduler->panic("AP_Baro: all sensors uncalibrated");
156 }

158 /*
159  update the barometer calibration
160  this updates the baro ground calibration to the current values. It
161  can be used before arming to keep the baro well calibrated
162 */
163 void AP_Baro::update_calibration()
164 {
165  for (uint8_t i=0; i<_num_sensors; i++) {
166  if (healthy(i)) {
167  sensors[i].ground_pressure.set(get_pressure(i));
168  }
169  float last_temperature = sensors[i].ground_temperature;
170  sensors[i].ground_temperature.set(get_calibration_temperature(i));
171 
172  // don't notify the GCS too rapidly or we flood the link
173  uint32_t now = hal.scheduler->millis();
174  if (now - _last_notify_ms > 10000) {
175  sensors[i].ground_pressure.notify();
176  sensors[i].ground_temperature.notify();
177  _last_notify_ms = now;
178  }
179  if (fabsf(last_temperature - sensors[i].ground_temperature) > 3) {
180  // reset _EAS2TAS to force it to recalculate. This happens
181  // when a digital airspeed sensor comes online
182  _EAS2TAS = 0;
183  }
184  }
185 }

187 // return altitude difference in meters between current pressure and a
188 // given base_pressure in Pascal
189 float AP_Baro::get_altitude_difference(float base_pressure, float pressure) const
190 {
191  float ret;
192  float temp = get_ground_temperature() + 273.15f;
193  float scaling = pressure / base_pressure;
194 
195  // This is an exact calculation that is within +-2.5m of the standard
196  // atmosphere tables in the troposphere (up to 11,000 m amsl).
197  ret = 153.8462f * temp * (1.0f - expf(0.190259f * logf(scaling)));
198 
199  return ret;
200 }

203 // return current scale factor that converts from equivalent to true airspeed
204 // valid for altitudes up to 10km AMSL
205 // assumes standard atmosphere lapse rate
206 float AP_Baro::get_EAS2TAS(void)
207 {
208  float altitude = get_altitude();
209  if ((fabsf(altitude - _last_altitude_EAS2TAS) < 100.0f) && !is_zero(_EAS2TAS)) {
210  // not enough change to require re-calculating
211  return _EAS2TAS;
212  }
213 
214  float tempK = get_calibration_temperature() + 273.15f - 0.0065f * altitude;
215  _EAS2TAS = safe_sqrt(1.225f / ((float)get_pressure() / (287.26f * tempK)));
216  _last_altitude_EAS2TAS = altitude;
217  return _EAS2TAS;
218 }

220 // return air density / sea level density - decreases as altitude climbs
221 float AP_Baro::get_air_density_ratio(void)
222 {
223  float eas2tas = get_EAS2TAS();
224  if (eas2tas > 0.0f) {
225  return 1.0f/(sq(get_EAS2TAS()));
226  } else {
227  return 1.0f;
228  }

231 // return current climb_rate estimeate relative to time that calibrate()
232 // was called. Returns climb rate in meters/s, positive means up

251 /*
252  get the temperature in degrees C to be used for calibration purposes
253  */
254 float AP_Baro::get_calibration_temperature(uint8_t instance) const
255 {
256  // if we have a recent external temperature then use it
257  if (_last_external_temperature_ms != 0 && hal.scheduler->millis() - _last_external_temperature_ms < 10000) {
258  return _external_temperature;
259  }
260  // if we don't have an external temperature then use the minimum
261  // of the barometer temperature and 25 degrees C. The reason for
262  // not just using the baro temperature is it tends to read high,
263  // often 30 degrees above the actual temperature. That means the
264  // EAS2TAS tends to be off by quite a large margin
265  float ret = get_temperature(instance);
266  if (ret > 25) {
267  ret = 25;
268  }
269  return ret;
270 }

332 /*
333  call update on all drivers
334  */
335 void AP_Baro::update(void)
336 {
337  if (!_hil_mode) {
338  for (uint8_t i=0; i<_num_drivers; i++) {
339  drivers[i]->update();
340  }
341  }
342 
343  // consider a sensor as healthy if it has had an update in the
344  // last 0.5 seconds
345  uint32_t now = hal.scheduler->millis();
346  for (uint8_t i=0; i<_num_sensors; i++) {
347  sensors[i].healthy = (now - sensors[i].last_update_ms < 500) && !is_zero(sensors[i].pressure);
348  }
349 
350  for (uint8_t i=0; i<_num_sensors; i++) {
351  if (sensors[i].healthy) {
352  // update altitude calculation
353  if (is_zero(sensors[i].ground_pressure)) {
354  sensors[i].ground_pressure = sensors[i].pressure;
355  }
356  float altitude = get_altitude_difference(sensors[i].ground_pressure, sensors[i].pressure);
357  // sanity check altitude
358  sensors[i].alt_ok = !(isnan(altitude) || isinf(altitude));
359  if (sensors[i].alt_ok) {
360  sensors[i].altitude = altitude + _alt_offset;
361  }
362  }
363  }
364 
365  // ensure the climb rate filter is updated
366  if (healthy()) {
367  _climb_rate_filter.update(get_altitude(), get_last_update());
368  }
369 
370  // choose primary sensor
371  if (_primary_baro >= 0 && _primary_baro < _num_sensors && healthy(_primary_baro)) {
372  _primary = _primary_baro;
373  } else {
374  _primary = 0;
375  for (uint8_t i=0; i<_num_sensors; i++) {
376  if (healthy(i)) {
377  _primary = i;
378  break;
379  }
380  }
381  }
382 }
