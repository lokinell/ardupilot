34 /*
35  forward a MAVLink message to the right port. This also
36  automatically learns the route for the sender if it is not
37  already known.
38 
39  This returns true if the message should be processed locally
40 
41  Theory of MAVLink routing:
42 
43  When a flight controller receives a message it should process it
44  locally if any of these conditions hold:
45 
46  1a) the message has no target_system field
47 
48  1b) the message has a target_system of zero
49 
50  1c) the message has the flight controllers target system and has no
51  target_component field
52 
53  1d) the message has the flight controllers target system and has
54  the flight controllers target_component
55 
56  1e) the message has the flight controllers target system and the
57  flight controller has not seen any messages on any of its links
58  from a system that has the messages
59  target_system/target_component combination
60 
61  When a flight controller receives a message it should forward it
62  onto another different link if any of these conditions hold for that
63  link:
64 
65  2a) the message has no target_system field
66 
67  2b) the message has a target_system of zero
68 
69  2c) the message does not have the flight controllers target_system
70  and the flight controller has seen a message from the messages
71  target_system on the link
72 
73  2d) the message has the flight controllers target_system and has a
74  target_component field and the flight controllers has seen a
75  message from the target_system/target_component combination on
76  the link
77 
78 Note: This proposal assumes that ground stations will not send command
79 packets to a non-broadcast destination (sysid/compid combination)
80 until they have received at least one package from that destination
81 over the link. This is essential to prevent a flight controller from
82 acting on a message that is not meant for it. For example, a PARAM_SET
83 cannot be sent to a specific sysid/compid combination until the GCS
84 has seen a packet from that sysid/compid combination on the link.
85 
86 The GCS must also reset what sysid/compid combinations it has seen on
87 a link when it sees a SYSTEM_TIME message with a decrease in
88 time_boot_ms from a particular sysid/compid. That is essential to
89 detect a reset of the flight controller, which implies a reset of its
90 routing table.
91 
92 */
93 bool MAVLink_routing::check_and_forward(mavlink_channel_t in_channel, const mavlink_message_t* msg)
94 {
95  // handle the case of loopback of our own messages, due to
96  // incorrect serial configuration.
97  if (msg->sysid == mavlink_system.sysid &&
98  msg->compid == mavlink_system.compid) {
99  return true;
100  }
101 
102  // learn new routes
103  learn_route(in_channel, msg);
104 
105  if (msg->msgid == MAVLINK_MSG_ID_HEARTBEAT) {
106  // heartbeat needs special handling
107  handle_heartbeat(in_channel, msg);
108  return true;
109  }
110 
111  // extract the targets for this packet
112  int16_t target_system = -1;
113  int16_t target_component = -1;
114  get_targets(msg, target_system, target_component);
115 
116  bool broadcast_system = (target_system == 0 || target_system == -1);
117  bool broadcast_component = (target_component == 0 || target_component == -1);
118  bool match_system = broadcast_system || (target_system == mavlink_system.sysid);
119  bool match_component = match_system && (broadcast_component ||
120  (target_component == mavlink_system.compid));
121  bool process_locally = match_system && match_component;
122 
123  if (process_locally && !broadcast_system && !broadcast_component) {
124  // nothing more to do - it can only be for us
125  return true;
126  }
127 
128  // forward on any channels matching the targets
129  bool forwarded = false;
130  bool sent_to_chan[MAVLINK_COMM_NUM_BUFFERS];
131  memset(sent_to_chan, 0, sizeof(sent_to_chan));
132  for (uint8_t i=0; i<num_routes; i++) {
133  if (broadcast_system || (target_system == routes[i].sysid &&
134  (broadcast_component ||
135  target_component == routes[i].compid ||
136  !match_system))) {
137  if (in_channel != routes[i].channel && !sent_to_chan[routes[i].channel]) {
138  if (comm_get_txspace(routes[i].channel) >=
139  ((uint16_t)msg->len) + MAVLINK_NUM_NON_PAYLOAD_BYTES) {
140 #if ROUTING_DEBUG
141  ::printf("fwd msg %u from chan %u on chan %u sysid=%d compid=%d\n",
142  msg->msgid,
143  (unsigned)in_channel,
144  (unsigned)routes[i].channel,
145  (int)target_system,
146  (int)target_component);
147 #endif
148  _mavlink_resend_uart(routes[i].channel, msg);
149  }
150  sent_to_chan[routes[i].channel] = true;
151  forwarded = true;
152  }
153  }
154  }
155  if (!forwarded && match_system) {
156  process_locally = true;
157  }
158 
159  return process_locally;
160 }

213 void MAVLink_routing::learn_route(mavlink_channel_t in_channel, const mavlink_message_t* msg)
214 {
215  uint8_t i;
216  if (msg->sysid == 0 ||
217  (msg->sysid == mavlink_system.sysid &&
218  msg->compid == mavlink_system.compid)) {
219  return;
220  }
221  for (i=0; i<num_routes; i++) {
222  if (routes[i].sysid == msg->sysid &&
223  routes[i].compid == msg->compid &&
224  routes[i].channel == in_channel) {
225  if (routes[i].mavtype == 0 && msg->msgid == MAVLINK_MSG_ID_HEARTBEAT) {
226  routes[i].mavtype = mavlink_msg_heartbeat_get_type(msg);
227  }
228  break;
229  }
230  }
231  if (i == num_routes && i<MAVLINK_MAX_ROUTES) {
232  routes[i].sysid = msg->sysid;
233  routes[i].compid = msg->compid;
234  routes[i].channel = in_channel;
235  if (msg->msgid == MAVLINK_MSG_ID_HEARTBEAT) {
236  routes[i].mavtype = mavlink_msg_heartbeat_get_type(msg);
237  }
238  num_routes++;
239 #if ROUTING_DEBUG
240  ::printf("learned route %u %u via %u\n",
241  (unsigned)msg->sysid,
242  (unsigned)msg->compid,
243  (unsigned)in_channel);
244 #endif
245  }
246 }

249 /*
250  special handling for heartbeat messages. To ensure routing
251  propogation heartbeat messages need to be forwarded on all channels
252  except channels where the sysid/compid of the heartbeat could come from
253 */
254 void MAVLink_routing::handle_heartbeat(mavlink_channel_t in_channel, const mavlink_message_t* msg)
255 {
256  uint16_t mask = GCS_MAVLINK::active_channel_mask();
257 
258  // don't send on the incoming channel. This should only matter if
259  // the routing table is full
260  mask &= ~(1U<<(in_channel-MAVLINK_COMM_0));
261 
262  // mask out channels that are known sources for this sysid/compid
263  for (uint8_t i=0; i<num_routes; i++) {
264  if (routes[i].sysid == msg->sysid && routes[i].compid == msg->compid) {
265  mask &= ~(1U<<((unsigned)(routes[i].channel-MAVLINK_COMM_0)));
266  }
267  }
268 
269  if (mask == 0) {
270  // nothing to send to
271  return;
272  }
273 
274  // send on the remaining channels
275  for (uint8_t i=0; i<MAVLINK_COMM_NUM_BUFFERS; i++) {
276  if (mask & (1U<<i)) {
277  mavlink_channel_t channel = (mavlink_channel_t)(MAVLINK_COMM_0 + i);
278  if (comm_get_txspace(channel) >= ((uint16_t)msg->len) + MAVLINK_NUM_NON_PAYLOAD_BYTES) {
279 #if ROUTING_DEBUG
280  ::printf("fwd HB from chan %u on chan %u from sysid=%u compid=%u\n",
281  (unsigned)in_channel,
282  (unsigned)channel,
283  (unsigned)msg->sysid,
284  (unsigned)msg->compid);
285 #endif
286  _mavlink_resend_uart(channel, msg);
287  }
288  }
289  }
290 }
