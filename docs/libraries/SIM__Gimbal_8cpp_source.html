47 /*
48  update the gimbal state
49 */
50 void Gimbal::update(void)
51 {
52  // calculate delta time in seconds
53  uint32_t now_us = hal.scheduler->micros();
54 
55  float delta_t = (now_us - last_update_us) * 1.0e-6f;
56  last_update_us = now_us;
57 
58  Matrix3f vehicle_dcm;
59  vehicle_dcm.from_euler(radians(fdm.rollDeg), radians(fdm.pitchDeg), radians(fdm.yawDeg));
60 
61  Vector3f vehicle_gyro = Vector3f(radians(fdm.rollRate),
62  radians(fdm.pitchRate),
63  radians(fdm.yawRate));
64  Vector3f vehicle_accel_body = Vector3f(fdm.xAccel, fdm.yAccel, fdm.zAccel);
65 
66  // take a copy of the demanded rates to bypass the limiter function for testing
67  Vector3f demRateRaw = demanded_angular_rate;
68 
69  // 1) Rotate the copters rotation rates into the gimbals frame of reference
70  // copterAngRate_G = transpose(DCMgimbal)*DCMcopter*copterAngRate
71  Vector3f copterAngRate_G = dcm.transposed()*vehicle_dcm*vehicle_gyro;
72 
73  // 2) Subtract the copters body rates to obtain a copter relative rotational
74  // rate vector (X,Y,Z) in gimbal sensor frame
75  // relativeGimbalRate(X,Y,Z) = gimbalRateDemand - copterAngRate_G
76  Vector3f relativeGimbalRate = demanded_angular_rate - copterAngRate_G;
77 
78  // calculate joint angles (euler312 order)
79  // calculate copter -> gimbal rotation matrix
80  Matrix3f rotmat_copter_gimbal = dcm.transposed() * vehicle_dcm;
81 
82  joint_angles = rotmat_copter_gimbal.transposed().to_euler312();
83 
84  /* 4) For each of the three joints, calculate upper and lower rate limits
85  from the corresponding angle limits and current joint angles
86 
87  upperRatelimit = (jointAngle - lowerAngleLimit) * travelLimitGain
88  lowerRatelimit = (jointAngle - upperAngleLimit) * travelLimitGain
89 
90  travelLimitGain is equal to the inverse of the bump stop time constant and
91  should be set to something like 20 initially. If set too high it can cause
92  the rates to 'ring' when they the limiter is in force, particularly given
93  we are using a first order numerical integration.
94  */
95  Vector3f upperRatelimit = -(joint_angles - upper_joint_limits) * travelLimitGain;
96  Vector3f lowerRatelimit = -(joint_angles - lower_joint_limits) * travelLimitGain;
97 
98  /*
99  5) Calculate the gimbal joint rates (roll, elevation, azimuth)
100 
101  gimbalJointRates(roll, elev, azimuth) = Matrix*relativeGimbalRate(X,Y,Z)
102 
103  where matrix =
104  +- -+
105  | cos(elevAngle), 0, sin(elevAngle) |
106  | |
107  | sin(elevAngle) tan(rollAngle), 1, -cos(elevAngle) tan(rollAngle) |
108  | |
109  | sin(elevAngle) cos(elevAngle) |
110  | - --------------, 0, -------------- |
111  | cos(rollAngle) cos(rollAngle) |
112  +- -+
113  */
114  float rollAngle = joint_angles.x;
115  float elevAngle = joint_angles.y;
116  Matrix3f matrix = Matrix3f(Vector3f(cosf(elevAngle), 0, sinf(elevAngle)),
117  Vector3f(sinf(elevAngle)*tanf(rollAngle), 1, -cosf(elevAngle)*tanf(rollAngle)),
118  Vector3f(-sinf(elevAngle)/cosf(rollAngle), 0, cosf(elevAngle)/cosf(rollAngle)));
119  Vector3f gimbalJointRates = matrix * relativeGimbalRate;
120 
121  // 6) Apply the rate limits from 4)
122  gimbalJointRates.x = constrain_float(gimbalJointRates.x, lowerRatelimit.x, upperRatelimit.x);
123  gimbalJointRates.y = constrain_float(gimbalJointRates.y, lowerRatelimit.y, upperRatelimit.y);
124  gimbalJointRates.z = constrain_float(gimbalJointRates.z, lowerRatelimit.z, upperRatelimit.z);
125  /*
126  7) Convert the modified gimbal joint rates to body rates (still copter
127  relative)
128  relativeGimbalRate(X,Y,Z) = Matrix * gimbalJointRates(roll, elev, azimuth)
129 
130  where Matrix =
131 
132  +- -+
133  | cos(elevAngle), 0, -cos(rollAngle) sin(elevAngle) |
134  | |
135  | 0, 1, sin(rollAngle) |
136  | |
137  | sin(elevAngle), 0, cos(elevAngle) cos(rollAngle) |
138  +- -+
139  */
140  matrix = Matrix3f(Vector3f(cosf(elevAngle), 0, -cosf(rollAngle)*sinf(elevAngle)),
141  Vector3f(0, 1, sinf(rollAngle)),
142  Vector3f(sinf(elevAngle), 0, cosf(elevAngle)*cosf(rollAngle)));
143  relativeGimbalRate = matrix * gimbalJointRates;
144 
145  // 8) Add to the result from step 1) to obtain the demanded gimbal body rates
146  // in an inertial frame of reference
147  // demandedGimbalRatesInertial(X,Y,Z) = relativeGimbalRate(X,Y,Z) + copterAngRate_G
148  // Vector3f demandedGimbalRatesInertial = relativeGimbalRate + copterAngRate_G;
149 
150  // for the moment we will set gyros equal to demanded_angular_rate
151  gimbal_angular_rate = demRateRaw; // demandedGimbalRatesInertial + true_gyro_bias - supplied_gyro_bias
152 
153  // update rotation of the gimbal
154  dcm.rotate(gimbal_angular_rate*delta_t);
155  dcm.normalize();
156 
157  // calculate copter -> gimbal rotation matrix
158  rotmat_copter_gimbal = dcm.transposed() * vehicle_dcm;
159 
160  // calculate joint angles (euler312 order)
161  joint_angles = rotmat_copter_gimbal.transposed().to_euler312();
162 
163  // update observed gyro
164  gyro = gimbal_angular_rate + true_gyro_bias;
165 
166  // update delta_angle (integrate)
167  delta_angle += gyro * delta_t;
168 
169  // calculate accel in gimbal body frame
170  Vector3f copter_accel_earth = vehicle_dcm * vehicle_accel_body;
171  Vector3f accel = dcm.transposed() * copter_accel_earth;
172 
173  // integrate velocity
174  delta_velocity += accel * delta_t;
175 
176  // see if we should do a report
177  send_report();
178 }

180 /*
181  send a report to the vehicle control code over MAVLink
182 */
183 void Gimbal::send_report(void)
184 {
185  if (!mavlink.connected && mav_socket.connect(target_address, target_port)) {
186  ::printf("Gimbal connected to %s:%u\n", target_address, (unsigned)target_port);
187  mavlink.connected = true;
188  }
189  if (!mavlink.connected) {
190  return;
191  }
192 
193  // check for incoming MAVLink messages
194  uint8_t buf[100];
195  ssize_t ret;
196 
197  while ((ret=mav_socket.recv(buf, sizeof(buf), 0)) > 0) {
198  for (uint8_t i=0; i<ret; i++) {
199  mavlink_message_t msg;
200  mavlink_status_t status;
201  if (mavlink_frame_char_buffer(&mavlink.rxmsg, &mavlink.status,
202  buf[i],
203  &msg, &status) == MAVLINK_FRAMING_OK) {
204  switch (msg.msgid) {
205  case MAVLINK_MSG_ID_HEARTBEAT: {
206  if (!seen_heartbeat) {
207  seen_heartbeat = true;
208  vehicle_component_id = msg.compid;
209  vehicle_system_id = msg.sysid;
210  ::printf("Gimbal using srcSystem %u\n", (unsigned)vehicle_system_id);
211  }
212  break;
213  }
214  case MAVLINK_MSG_ID_GIMBAL_CONTROL: {
215  mavlink_gimbal_control_t pkt;
216  mavlink_msg_gimbal_control_decode(&msg, &pkt);
217  demanded_angular_rate = Vector3f(pkt.demanded_rate_x,
218  pkt.demanded_rate_y,
219  pkt.demanded_rate_z);
220  // no longer supply a bias
221  supplied_gyro_bias.zero();
222  seen_gimbal_control = true;
223  break;
224  }
225  }
226  }
227  }
228  }
229 
230  if (!seen_heartbeat) {
231  return;
232  }
233  uint32_t now = hal.scheduler->millis();
234  mavlink_message_t msg;
235  uint16_t len;
236 
237  if (now - last_heartbeat_ms >= 1000) {
238  mavlink_heartbeat_t heartbeat;
239  heartbeat.type = MAV_TYPE_GIMBAL;
240  heartbeat.autopilot = MAV_AUTOPILOT_ARDUPILOTMEGA;
241  heartbeat.base_mode = 0;
242  heartbeat.system_status = 0;
243  heartbeat.mavlink_version = 0;
244  heartbeat.custom_mode = 0;
245 
246  /*
247  save and restore sequence number for chan0, as it is used by
248  generated encode functions
249  */
250  mavlink_status_t *chan0_status = mavlink_get_channel_status(MAVLINK_COMM_0);
251  uint8_t saved_seq = chan0_status->current_tx_seq;
252  chan0_status->current_tx_seq = mavlink.seq;
253  len = mavlink_msg_heartbeat_encode(vehicle_system_id,
254  vehicle_component_id,
255  &msg, &heartbeat);
256  chan0_status->current_tx_seq = saved_seq;
257 
258  mav_socket.send(&msg.magic, len);
259  last_heartbeat_ms = now;
260  }
261 
262  /*
263  send a GIMBAL_REPORT message
264  */
265  uint32_t now_us = hal.scheduler->micros();
266  if (now_us - last_report_us > reporting_period_ms*1000UL) {
267  mavlink_gimbal_report_t gimbal_report;
268  float delta_time = (now_us - last_report_us) * 1.0e-6f;
269  last_report_us = now_us;
270  gimbal_report.target_system = vehicle_system_id;
271  gimbal_report.target_component = vehicle_component_id;
272  gimbal_report.delta_time = delta_time;
273  gimbal_report.delta_angle_x = delta_angle.x;
274  gimbal_report.delta_angle_y = delta_angle.y;
275  gimbal_report.delta_angle_z = delta_angle.z;
276  gimbal_report.delta_velocity_x = delta_velocity.x;
277  gimbal_report.delta_velocity_y = delta_velocity.y;
278  gimbal_report.delta_velocity_z = delta_velocity.z;
279  gimbal_report.joint_roll = joint_angles.x;
280  gimbal_report.joint_el = joint_angles.y;
281  gimbal_report.joint_az = joint_angles.z;
282 
283  mavlink_status_t *chan0_status = mavlink_get_channel_status(MAVLINK_COMM_0);
284  uint8_t saved_seq = chan0_status->current_tx_seq;
285  chan0_status->current_tx_seq = mavlink.seq;
286  len = mavlink_msg_gimbal_report_encode(vehicle_system_id,
287  vehicle_component_id,
288  &msg, &gimbal_report);
289  chan0_status->current_tx_seq = saved_seq;
290 
291  mav_socket.send(&msg.magic, len);
292 
293  delta_velocity.zero();
294  delta_angle.zero();
295  }
296 }
