84 dsm_decode_channel(uint16_t raw, unsigned shift, unsigned *channel, unsigned *value)
85 {
86 
87  if (raw == 0xffff)
88  return false;
89 
90  *channel = (raw >> shift) & 0xf;
91 
92  uint16_t data_mask = (1 << shift) - 1;
93  *value = raw & data_mask;
94 
95  //debug("DSM: %d 0x%04x -> %d %d", shift, raw, *channel, *value);
96 
97  return true;

105 static void
106 dsm_guess_format(bool reset, const uint8_t dsm_frame[16])
107 {
108  static uint32_t cs10;
109  static uint32_t cs11;
110  static unsigned samples;
111 
112  /* reset the 10/11 bit sniffed channel masks */
113  if (reset) {
114  cs10 = 0;
115  cs11 = 0;
116  samples = 0;
117  dsm_channel_shift = 0;
118  return;
119  }
120 
121  /* scan the channels in the current dsm_frame in both 10- and 11-bit mode */
122  for (unsigned i = 0; i < DSM_FRAME_CHANNELS; i++) {
123 
124  const uint8_t *dp = &dsm_frame[2 + (2 * i)];
125  uint16_t raw = (dp[0] << 8) | dp[1];
126  unsigned channel, value;
127 
128  /* if the channel decodes, remember the assigned number */
129  if (dsm_decode_channel(raw, 10, &channel, &value) && (channel < 31))
130  cs10 |= (1 << channel);
131 
132  if (dsm_decode_channel(raw, 11, &channel, &value) && (channel < 31))
133  cs11 |= (1 << channel);
134 
135  /* XXX if we cared, we could look for the phase bit here to decide 1 vs. 2-dsm_frame format */
136  }
137 
138  /* wait until we have seen plenty of frames - 5 should normally be enough */
139  if (samples++ < 5)
140  return;
141 
142  /*
143  * Iterate the set of sensible sniffed channel sets and see whether
144  * decoding in 10 or 11-bit mode has yielded anything we recognize.
145  *
146  * XXX Note that due to what seem to be bugs in the DSM2 high-resolution
147  * stream, we may want to sniff for longer in some cases when we think we
148  * are talking to a DSM2 receiver in high-resolution mode (so that we can
149  * reject it, ideally).
150  * See e.g. http://git.openpilot.org/cru/OPReview-116 for a discussion
151  * of this issue.
152  */
153  static uint32_t masks[] = {
154  0x3f, /* 6 channels (DX6) */
155  0x7f, /* 7 channels (DX7) */
156  0xff, /* 8 channels (DX8) */
157  0x1ff, /* 9 channels (DX9, etc.) */
158  0x3ff, /* 10 channels (DX10) */
159  0x1fff, /* 13 channels (DX10t) */
160  0x3fff /* 18 channels (DX10) */
161  };
162  unsigned votes10 = 0;
163  unsigned votes11 = 0;
164 
165  for (unsigned i = 0; i < ARRAY_SIZE(masks); i++) {
166 
167  if (cs10 == masks[i])
168  votes10++;
169 
170  if (cs11 == masks[i])
171  votes11++;
172  }
173 
174  if ((votes11 == 1) && (votes10 == 0)) {
175  dsm_channel_shift = 11;
176  debug("DSM: 11-bit format");
177  return;
178  }
179 
180  if ((votes10 == 1) && (votes11 == 0)) {
181  dsm_channel_shift = 10;
182  debug("DSM: 10-bit format");
183  return;
184  }
185 
186  /* call ourselves to reset our state ... we have to try again */
187  debug("DSM: format detect fail, 10: 0x%08x %d 11: 0x%08x %d", cs10, votes10, cs11, votes11);
188  dsm_guess_format(true, dsm_frame);
189 }

195 bool
196 dsm_decode(uint64_t frame_time, const uint8_t dsm_frame[16], uint16_t *values, uint16_t *num_values, uint16_t max_values)
197 {
198  /*
199  debug("DSM dsm_frame %02x%02x %02x%02x %02x%02x %02x%02x %02x%02x %02x%02x %02x%02x %02x%02x",
200  dsm_frame[0], dsm_frame[1], dsm_frame[2], dsm_frame[3], dsm_frame[4], dsm_frame[5], dsm_frame[6], dsm_frame[7],
201  dsm_frame[8], dsm_frame[9], dsm_frame[10], dsm_frame[11], dsm_frame[12], dsm_frame[13], dsm_frame[14], dsm_frame[15]);
202  */
203  /*
204  * If we have lost signal for at least a second, reset the
205  * format guessing heuristic.
206  */
207  if (((frame_time - dsm_last_frame_time) > 1000000) && (dsm_channel_shift != 0))
208  dsm_guess_format(true, dsm_frame);
209 
210  /* we have received something we think is a dsm_frame */
211  dsm_last_frame_time = frame_time;
212 
213  /* if we don't know the dsm_frame format, update the guessing state machine */
214  if (dsm_channel_shift == 0) {
215  dsm_guess_format(false, dsm_frame);
216  return false;
217  }
218 
219  /*
220  * The encoding of the first two bytes is uncertain, so we're
221  * going to ignore them for now.
222  *
223  * Each channel is a 16-bit unsigned value containing either a 10-
224  * or 11-bit channel value and a 4-bit channel number, shifted
225  * either 10 or 11 bits. The MSB may also be set to indicate the
226  * second dsm_frame in variants of the protocol where more than
227  * seven channels are being transmitted.
228  */
229 
230  for (unsigned i = 0; i < DSM_FRAME_CHANNELS; i++) {
231 
232  const uint8_t *dp = &dsm_frame[2 + (2 * i)];
233  uint16_t raw = (dp[0] << 8) | dp[1];
234  unsigned channel, value;
235 
236  if (!dsm_decode_channel(raw, dsm_channel_shift, &channel, &value))
237  continue;
238 
239  /* ignore channels out of range */
240  if (channel >= max_values)
241  continue;
242 
243  /* update the decoded channel count */
244  if (channel >= *num_values)
245  *num_values = channel + 1;
246 
247  /* convert 0-1024 / 0-2048 values to 1000-2000 ppm encoding. */
248  if (dsm_channel_shift == 10)
249  value *= 2;
250 
251  /*
252  * Spektrum scaling is special. There are these basic considerations
253  *
254  * * Midpoint is 1520 us
255  * * 100% travel channels are +- 400 us
256  *
257  * We obey the original Spektrum scaling (so a default setup will scale from
258  * 1100 - 1900 us), but we do not obey the weird 1520 us center point
259  * and instead (correctly) center the center around 1500 us. This is in order
260  * to get something useful without requiring the user to calibrate on a digital
261  * link for no reason.
262  */
263 
264  /* scaled integer for decent accuracy while staying efficient */
265  value = ((((int)value - 1024) * 1000) / 1700) + 1500;
266 
267  /*
268  * Store the decoded channel into the R/C input buffer, taking into
269  * account the different ideas about channel assignement that we have.
270  *
271  * Specifically, the first four channels in rc_channel_data are roll, pitch, thrust, yaw,
272  * but the first four channels from the DSM receiver are thrust, roll, pitch, yaw.
273  */
274  switch (channel) {
275  case 0:
276  channel = 2;
277  break;
278 
279  case 1:
280  channel = 0;
281  break;
282 
283  case 2:
284  channel = 1;
285 
286  default:
287  break;
288  }
289 
290  values[channel] = value;
291  }
292 
293  /*
294  * Spektrum likes to send junk in higher channel numbers to fill
295  * their packets. We don't know about a 13 channel model in their TX
296  * lines, so if we get a channel count of 13, we'll return 12 (the last
297  * data index that is stable).
298  */
299  if (*num_values == 13)
300  *num_values = 12;
301 
302 #if 0
303  if (dsm_channel_shift == 11) {
304  /* Set the 11-bit data indicator */
305  *num_values |= 0x8000;
306  }
307 #endif
308 
309  /*
310  * XXX Note that we may be in failsafe here; we need to work out how to detect that.
311  */
312  return true;
313 }
