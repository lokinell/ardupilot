92 bool AP_InertialSensor_Flymaple::_init_sensor(void)
93 {
94  // get pointer to i2c bus semaphore
95  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
96 
97  // take i2c bus sempahore
98  if (!i2c_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER))
99  return false;
100 
101  // Init the accelerometer
102  uint8_t data;
103  hal.i2c->readRegister(FLYMAPLE_ACCELEROMETER_ADDRESS, FLYMAPLE_ACCELEROMETER_ADXLREG_DEVID, &data);
104  if (data != FLYMAPLE_ACCELEROMETER_XL345_DEVID)
105  hal.scheduler->panic("AP_InertialSensor_Flymaple: could not find ADXL345 accelerometer sensor");
106  hal.i2c->writeRegister(FLYMAPLE_ACCELEROMETER_ADDRESS, FLYMAPLE_ACCELEROMETER_ADXLREG_POWER_CTL, 0x00);
107  hal.scheduler->delay(5);
108  hal.i2c->writeRegister(FLYMAPLE_ACCELEROMETER_ADDRESS, FLYMAPLE_ACCELEROMETER_ADXLREG_POWER_CTL, 0xff);
109  hal.scheduler->delay(5);
110  // Measure mode:
111  hal.i2c->writeRegister(FLYMAPLE_ACCELEROMETER_ADDRESS, FLYMAPLE_ACCELEROMETER_ADXLREG_POWER_CTL, 0x08);
112  hal.scheduler->delay(5);
113  // Full resolution, 8g:
114  // Caution, this must agree with FLYMAPLE_ACCELEROMETER_SCALE_1G
115  // In full resoution mode, the scale factor need not change
116  hal.i2c->writeRegister(FLYMAPLE_ACCELEROMETER_ADDRESS, FLYMAPLE_ACCELEROMETER_ADXLREG_DATA_FORMAT, 0x08);
117  hal.scheduler->delay(5);
118  // Normal power, 800Hz Output Data Rate, 400Hz bandwidth:
119  hal.i2c->writeRegister(FLYMAPLE_ACCELEROMETER_ADDRESS, FLYMAPLE_ACCELEROMETER_ADXLREG_BW_RATE, 0x0d);
120  hal.scheduler->delay(5);
121  // Power up default is FIFO bypass mode. FIFO is not used by the chip
122 
123  // Init the Gyro
124  // Expect to read the same as the Gyro I2C adress:
125  hal.i2c->readRegister(FLYMAPLE_GYRO_ADDRESS, FLYMAPLE_GYRO_WHO_AM_I, &data);
126  if (data != FLYMAPLE_GYRO_ADDRESS)
127  hal.scheduler->panic("AP_InertialSensor_Flymaple: could not find ITG-3200 accelerometer sensor");
128  hal.i2c->writeRegister(FLYMAPLE_GYRO_ADDRESS, FLYMAPLE_GYRO_PWR_MGM, 0x00);
129  hal.scheduler->delay(1);
130  // Sample rate divider: with 8kHz internal clock (see FLYMAPLE_GYRO_DLPF_FS),
131  // get 500Hz sample rate, 2 samples
132  hal.i2c->writeRegister(FLYMAPLE_GYRO_ADDRESS, FLYMAPLE_GYRO_SMPLRT_DIV, 0x0f);
133  hal.scheduler->delay(1);
134  // 2000 degrees/sec, 256Hz LPF, 8kHz internal sample rate
135  // This is the least amount of filtering we can configure for this device
136  hal.i2c->writeRegister(FLYMAPLE_GYRO_ADDRESS, FLYMAPLE_GYRO_DLPF_FS, 0x18);
137  hal.scheduler->delay(1);
138  // No interrupts
139  hal.i2c->writeRegister(FLYMAPLE_GYRO_ADDRESS, FLYMAPLE_GYRO_INT_CFG, 0x00);
140  hal.scheduler->delay(1);
141 
142  // Set up the filter desired
143  _set_filter_frequency(_accel_filter_cutoff());
144 
145  // give back i2c semaphore
146  i2c_sem->give();
147 
148  _gyro_instance = _imu.register_gyro();
149  _accel_instance = _imu.register_accel();
150 
151  _product_id = AP_PRODUCT_ID_FLYMAPLE;
152 
153  return true;
154 }

166 // This takes about 20us to run
167 bool AP_InertialSensor_Flymaple::update(void)
168 {
169  Vector3f accel, gyro;
170 
171  hal.scheduler->suspend_timer_procs();
172  accel = _accel_filtered;
173  gyro = _gyro_filtered;
174  _have_gyro_sample = false;
175  _have_accel_sample = false;
176  hal.scheduler->resume_timer_procs();
177 
178  _publish_accel(_accel_instance, accel);
179  _publish_gyro(_gyro_instance, gyro);
180 
181  if (_last_filter_hz != _accel_filter_cutoff()) {
182  _set_filter_frequency(_accel_filter_cutoff());
183  _last_filter_hz = _accel_filter_cutoff();
184  }
185 
186  return true;
187 }

189 // This needs to get called as often as possible.
190 // Its job is to accumulate samples as fast as is reasonable for the accel and gyro
191 // sensors.
192 // Note that this is called from gyro_sample_available() and
193 // accel_sample_available(), which is really not good enough for
194 // 800Hz, as it is common for the main loop to take more than 1.5ms
195 // before wait_for_sample() is called again. We can't just call this
196 // from a timer as timers run with interrupts disabled, and the I2C
197 // operations take too long
198 // So we are stuck with a suboptimal solution. The results are not so
199 // good in terms of timing. It may be better with the FIFOs enabled
200 void AP_InertialSensor_Flymaple::_accumulate(void)
201 {
202  // get pointer to i2c bus semaphore
203  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
204 
205  // take i2c bus sempahore
206  if (!i2c_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER))
207  return;
208 
209  // Read accelerometer
210  // ADXL345 is in the default FIFO bypass mode, so the FIFO is not used
211  uint8_t buffer[6];
212  uint32_t now = hal.scheduler->micros();
213  // This takes about 250us at 400kHz I2C speed
214  if ((now - _last_accel_timestamp) >= raw_sample_interval_us
215  && hal.i2c->readRegisters(FLYMAPLE_ACCELEROMETER_ADDRESS, FLYMAPLE_ACCELEROMETER_ADXLREG_DATAX0, 6, buffer) == 0)
216  {
217  // The order is a bit wierd here since the standard we have adopted for Flymaple
218  // sensor orientation is different to what the board designers intended
219  // Caution, to support alternative chip orientations on other bords, may
220  // need to add a chip orientation rotate
221  int16_t y = -((((int16_t)buffer[1]) << 8) | buffer[0]); // chip X axis
222  int16_t x = -((((int16_t)buffer[3]) << 8) | buffer[2]); // chip Y axis
223  int16_t z = -((((int16_t)buffer[5]) << 8) | buffer[4]); // chip Z axis
224  Vector3f accel = Vector3f(x,y,z);
225  // Adjust for chip scaling to get m/s/s
226  accel *= FLYMAPLE_ACCELEROMETER_SCALE_M_S;
227  _rotate_and_correct_accel(_accel_instance, accel);
228  _notify_new_accel_raw_sample(_accel_instance, accel);
229  _accel_filtered = _accel_filter.apply(accel);
230  _have_accel_sample = true;
231  _last_accel_timestamp = now;
232  }
233 
234  // Read gyro
235  now = hal.scheduler->micros();
236  // This takes about 250us at 400kHz I2C speed
237  if ((now - _last_gyro_timestamp) >= raw_sample_interval_us
238  && hal.i2c->readRegisters(FLYMAPLE_GYRO_ADDRESS, FLYMAPLE_GYRO_GYROX_H, 6, buffer) == 0)
239  {
240  // See above re order of samples in buffer
241  int16_t y = -((((int16_t)buffer[0]) << 8) | buffer[1]); // chip X axis
242  int16_t x = -((((int16_t)buffer[2]) << 8) | buffer[3]); // chip Y axis
243  int16_t z = -((((int16_t)buffer[4]) << 8) | buffer[5]); // chip Z axis
244  Vector3f gyro = Vector3f(x,y,z);
245  // Adjust for chip scaling to get radians/sec
246  gyro *= FLYMAPLE_GYRO_SCALE_R_S;
247  _rotate_and_correct_gyro(_gyro_instance, gyro);
248  _notify_new_gyro_raw_sample(_gyro_instance, gyro);
249  _gyro_filtered = _gyro_filter.apply(gyro);
250  _have_gyro_sample = true;
251  _last_gyro_timestamp = now;
252  }
253 
254  // give back i2c semaphore
255  i2c_sem->give();
256 }
