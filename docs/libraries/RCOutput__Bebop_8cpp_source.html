370 void RCOutput_Bebop::_run_rcout()
371 {
372  uint16_t current_period_us[BEBOP_BLDC_MOTORS_NUM];
373  uint8_t i;
374  int ret;
375  struct timespec ts;
376 
377  memset(current_period_us, 0, sizeof(current_period_us));
378 
379  while (true) {
380  pthread_mutex_lock(&_mutex);
381  ret = clock_gettime(CLOCK_MONOTONIC, &ts);
382  if (ret != 0)
383  hal.console->println("RCOutput_Bebop: bad checksum in obs data");
384 
385  if (ts.tv_nsec > (1000000000 - BEBOP_BLDC_TIMEOUT_NS))
386  {
387  ts.tv_sec += 1;
388  ts.tv_nsec = ts.tv_nsec + BEBOP_BLDC_TIMEOUT_NS - 1000000000;
389  } else {
390  ts.tv_nsec += BEBOP_BLDC_TIMEOUT_NS;
391  }
392 
393  ret = 0;
394  while ((memcmp(_period_us, current_period_us, sizeof(_period_us)) == 0) && (ret == 0))
395  ret = pthread_cond_timedwait(&_cond, &_mutex, &ts);
396 
397  memcpy(current_period_us, _period_us, sizeof(_period_us));
398  pthread_mutex_unlock(&_mutex);
399 
400  /* start propellers if the speed of the 4 motors is >= min speed
401  * min speed set to min_pwm + 50*/
402  for (i = 0; i < BEBOP_BLDC_MOTORS_NUM; i++) {
403  if (current_period_us[i] <= _min_pwm + 50)
404  break;
405  _rpm[bebop_bldc_motors[i]] = _period_us_to_rpm(current_period_us[i]);
406  }
407 
408  if (i < BEBOP_BLDC_MOTORS_NUM) {
409  /* one motor pwm value is at minimum (or under)
410  * if the motors are started, stop them*/
411  if (_state == BEBOP_BLDC_STARTED) {
412  _stop_prop();
413  _clear_error();
414  }
415  } else {
416  /* all the motor pwm values are higher than minimum
417  * if the bldc is stopped, start it*/
418  if (_state == BEBOP_BLDC_STOPPED)
419  _start_prop();
420  }
421  _set_ref_speed(_rpm);
422  }
423 }
