16 // update mount position - should be called periodically
17 void AP_Mount_Alexmos::update()
18 {
19  if (!_initialised) {
20  return;
21  }
22 
23  read_incoming(); // read the incoming messages from the gimbal
24 
25  // update based on mount mode
26  switch(get_mode()) {
27  // move mount to a "retracted" position. we do not implement a separate servo based retract mechanism
28  case MAV_MOUNT_MODE_RETRACT:
29  control_axis(_state._retract_angles.get(), true);
30  break;
31 
32  // move mount to a neutral position, typically pointing forward
33  case MAV_MOUNT_MODE_NEUTRAL:
34  control_axis(_state._neutral_angles.get(), true);
35  break;
36 
37  // point to the angles given by a mavlink message
38  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
39  // do nothing because earth-frame angle targets (i.e. _angle_ef_target_rad) should have already been set by a MOUNT_CONTROL message from GCS
40  break;
41 
42  // RC radio manual angle control, but with stabilization from the AHRS
43  case MAV_MOUNT_MODE_RC_TARGETING:
44  // update targets using pilot's rc inputs
45  update_targets_from_rc();
46  control_axis(_angle_ef_target_rad, false);
47  break;
48 
49  // point mount to a GPS point given by the mission planner
50  case MAV_MOUNT_MODE_GPS_POINT:
51  if(_frontend._ahrs.get_gps().status() >= AP_GPS::GPS_OK_FIX_2D) {
52  calc_angle_to_location(_state._roi_target, _angle_ef_target_rad, true, false);
53  control_axis(_angle_ef_target_rad, false);
54  }
55  break;
56 
57  default:
58  // we do not know this mode so do nothing
59  break;
60  }
61 }

164  send a command to the Alemox Serial API

166 void AP_Mount_Alexmos::send_command(uint8_t cmd, uint8_t* data, uint8_t size)
167 {
168  if (_port->txspace() < (size + 5)) {
169  return;
170  }
171  uint8_t checksum = 0;
172  _port->write( '>' );
173  _port->write( cmd ); // write command id
174  _port->write( size ); // write body size
175  _port->write( cmd+size ); // write header checkum
176 
177  for (uint8_t i = 0; i != size ; i++) {
178  checksum += data[i];
179  _port->write( data[i] );
180  }
181  _port->write(checksum);
182 }

218 /*
219  * detect and read the header of the incoming message from the gimbal
220  */
221 void AP_Mount_Alexmos::read_incoming()
222 {
223  uint8_t data;
224  int16_t numc;
225 
226  numc = _port->available();
227 
228  if (numc < 0 ){
229  return;
230  }
231 
232  for (int16_t i = 0; i < numc; i++) { // Process bytes received
233  data = _port->read();
234  switch (_step) {
235  case 0:
236  if ( '>' == data) {
237  _step = 1;
238  _checksum = 0; //reset checksum accumulator
239  _last_command_confirmed = false;
240  }
241  break;
242 
243  case 1: // command ID
244  _checksum = data;
245  _command_id = data;
246  _step++;
247  break;
248 
249  case 2: // Size of the body of the message
250  _checksum += data;
251  _payload_length = data;
252  _step++;
253  break;
254 
255  case 3: // checksum of the header
256  if (_checksum != data ) {
257  _step = 0;
258  _checksum = 0;
259  // checksum error
260  break;
261  }
262  _step++;
263  _checksum = 0;
264  _payload_counter = 0; // prepare to receive payload
265  break;
266 
267  case 4: // parsing body
268  _checksum += data;
269  if (_payload_counter < sizeof(_buffer)) {
270  _buffer.bytes[_payload_counter] = data;
271  }
272  if (++_payload_counter == _payload_length)
273  _step++;
274  break;
275 
276  case 5:// body checksum
277  _step = 0;
278  if (_checksum != data) {
279  break;
280  }
281  parse_body();
282  }
283  }
