171 /*
172  apply a reading in ADC counts
173  */
174 void PX4AnalogSource::_add_value(float v, float vcc5V)
175 {
176  _latest_value = v;
177  _sum_value += v;
178  if (vcc5V < 3.0f) {
179  _sum_ratiometric += v;
180  } else {
181  // this compensates for changes in the 5V rail relative to the
182  // 3.3V reference used by the ADC.
183  _sum_ratiometric += v * 5.0f / vcc5V;
184  }
185  _sum_count++;
186  if (_sum_count == 254) {
187  _sum_value /= 2;
188  _sum_ratiometric /= 2;
189  _sum_count /= 2;
190  }
191 }

214  move to the next stop pin
215  */
216 void PX4AnalogIn::next_stop_pin(void)
217 {
218  // find the next stop pin. We start one past the current stop pin
219  // and wrap completely, so we do the right thing is there is only
220  // one stop pin
221  for (uint8_t i=1; i <= PX4_ANALOG_MAX_CHANNELS; i++) {
222  uint8_t idx = (_current_stop_pin_i + i) % PX4_ANALOG_MAX_CHANNELS;
223  PX4::PX4AnalogSource *c = _channels[idx];
224  if (c && c->_stop_pin != -1) {
225  // found another stop pin
226  _stop_pin_change_time = hal.scheduler->millis();
227  _current_stop_pin_i = idx;
228 
229  // set that pin high
230  hal.gpio->pinMode(c->_stop_pin, 1);
231  hal.gpio->write(c->_stop_pin, 1);
232 
233  // set all others low
234  for (uint8_t j=0; j<PX4_ANALOG_MAX_CHANNELS; j++) {
235  PX4::PX4AnalogSource *c2 = _channels[j];
236  if (c2 && c2->_stop_pin != -1 && j != idx) {
237  hal.gpio->pinMode(c2->_stop_pin, 1);
238  hal.gpio->write(c2->_stop_pin, 0);
239  }
240  }
241  break;
242  }
243  }

249 void PX4AnalogIn::_timer_tick(void)
250 {
251  // read adc at 100Hz
252  uint32_t now = hal.scheduler->micros();
253  uint32_t delta_t = now - _last_run;
254  if (delta_t < 10000) {
255  return;
256  }
257  _last_run = now;
258 
259  struct adc_msg_s buf_adc[PX4_ANALOG_MAX_CHANNELS];
260 
261  // cope with initial setup of stop pin
262  if (_channels[_current_stop_pin_i] == NULL ||
263  _channels[_current_stop_pin_i]->_stop_pin == -1) {
264  next_stop_pin();
265  }
266 
267  /* read all channels available */
268  int ret = read(_adc_fd, &buf_adc, sizeof(buf_adc));
269  if (ret > 0) {
270  // match the incoming channels to the currently active pins
271  for (uint8_t i=0; i<ret/sizeof(buf_adc[0]); i++) {
272 #ifdef CONFIG_ARCH_BOARD_PX4FMU_V2
273  if (buf_adc[i].am_channel == 4) {
274  // record the Vcc value for later use in
275  // voltage_average_ratiometric()
276  _board_voltage = buf_adc[i].am_data * 6.6f / 4096;
277  }
278 #endif
279  }
280  for (uint8_t i=0; i<ret/sizeof(buf_adc[0]); i++) {
281  Debug("chan %u value=%u\n",
282  (unsigned)buf_adc[i].am_channel,
283  (unsigned)buf_adc[i].am_data);
284  for (uint8_t j=0; j<PX4_ANALOG_MAX_CHANNELS; j++) {
285  PX4::PX4AnalogSource *c = _channels[j];
286  if (c != NULL && buf_adc[i].am_channel == c->_pin) {
287  // add a value if either there is no stop pin, or
288  // the stop pin has been settling for enough time

293  if (c->_stop_pin != -1 && _current_stop_pin_i == j) {

296  }
297  }
298  }
299  }
