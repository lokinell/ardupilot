608 /*
609  calculate a compass heading given the attitude from DCM and the mag vector
610  */
611 float
612 Compass::calculate_heading(const Matrix3f &dcm_matrix) const
613 {
614  float cos_pitch_sq = 1.0f-(dcm_matrix.c.x*dcm_matrix.c.x);
615 
616  // Tilt compensated magnetic field Y component:
617  const Vector3f &field = get_field();
618 
619  float headY = field.y * dcm_matrix.c.z - field.z * dcm_matrix.c.y;
620 
621  // Tilt compensated magnetic field X component:
622  float headX = field.x * cos_pitch_sq - dcm_matrix.c.x * (field.y * dcm_matrix.c.y + field.z * dcm_matrix.c.z);
623 
624  // magnetic heading
625  // 6/4/11 - added constrain to keep bad values from ruining DCM Yaw - Jason S.
626  float heading = constrain_float(atan2f(-headY,headX), -3.15f, 3.15f);
627 
628  // Declination correction (if supplied)
629  if( fabsf(_declination) > 0.0f )
630  {
631  heading = heading + _declination;
632  if (heading > PI) // Angle normalization (-180 deg, 180 deg)
633  heading -= (2.0f * PI);
634  else if (heading < -PI)
635  heading += (2.0f * PI);
636  }
637 
638  return heading;
639 }

690 //
691 void Compass::setHIL(uint8_t instance, float roll, float pitch, float yaw)
692 {
693  Matrix3f R;
694 
695  // create a rotation matrix for the given attitude
696  R.from_euler(roll, pitch, yaw);
697 
698  if (!is_equal(_hil.last_declination,get_declination())) {
699  _setup_earth_field();
700  _hil.last_declination = get_declination();

703  // convert the earth frame magnetic vector to body frame, and

715  // and add in AHRS_ORIENTATION setting if not an external compass
716  _hil.field[instance].rotate(_board_orientation);
717  }

762 bool Compass::consistent() const
763 {
764  Vector3f primary_mag_field = get_field();
765  Vector3f primary_mag_field_norm;
766 
767  if (!primary_mag_field.is_zero()) {
768  primary_mag_field_norm = primary_mag_field.normalized();
769  } else {
770  return false;
771  }
772 
773  Vector2f primary_mag_field_xy = Vector2f(primary_mag_field.x,primary_mag_field.y);
774  Vector2f primary_mag_field_xy_norm;
775 
776  if (!primary_mag_field_xy.is_zero()) {
777  primary_mag_field_xy_norm = primary_mag_field_xy.normalized();
778  } else {
779  return false;
780  }
781 
782  for (uint8_t i=0; i<get_count(); i++) {
783  if (use_for_yaw(i)) {
784  Vector3f mag_field = get_field(i);
785  Vector3f mag_field_norm;
786 
787  if (!mag_field.is_zero()) {
788  mag_field_norm = mag_field.normalized();
789  } else {
790  return false;
791  }
792 
793  Vector2f mag_field_xy = Vector2f(mag_field.x,mag_field.y);
794  Vector2f mag_field_xy_norm;
795 
796  if (!mag_field_xy.is_zero()) {
797  mag_field_xy_norm = mag_field_xy.normalized();
798  } else {
799  return false;
800  }
801 
802  float xyz_ang_diff = acosf(constrain_float(mag_field_norm * primary_mag_field_norm,-1.0f,1.0f));
803  float xy_ang_diff = acosf(constrain_float(mag_field_xy_norm*primary_mag_field_xy_norm,-1.0f,1.0f));
804  float xy_len_diff = (primary_mag_field_xy-mag_field_xy).length();
805 
806  // check for gross misalignment on all axes
807  bool xyz_ang_diff_large = xyz_ang_diff > AP_COMPASS_MAX_XYZ_ANG_DIFF;
808 
809  // check for an unacceptable angle difference on the xy plane
810  bool xy_ang_diff_large = xy_ang_diff > AP_COMPASS_MAX_XY_ANG_DIFF;
811 
812  // check for an unacceptable length difference on the xy plane
813  bool xy_length_diff_large = xy_len_diff > AP_COMPASS_MAX_XY_LENGTH_DIFF;
814 
815  // check for inconsistency in the XY plane
816  if (xyz_ang_diff_large || xy_ang_diff_large || xy_length_diff_large) {
817  return false;
818  }
819  }
820  }
821  return true;
822 }
