43 void AP_OpticalFlow_Linux::init(void)
44 {
45  // only initialise once
46  if (initialised) {
47  return;
48  }
49 
50  // get pointer to i2c bus semaphore
51  AP_HAL::Semaphore *i2c_sem = hal.i2c->get_semaphore();
52  if (i2c_sem == NULL) {
53  return;
54  }
55 
56  // take i2c bus sempahore
57  if (!i2c_sem->take(200)) {
58  return;
59  }
60 
61  // read from flow sensor to ensure it is not a ll40ls Lidar (which can be on 0x42)
62  // read I2C_FRAME_SIZE bytes, the ll40ls will error whereas the flow happily returns data
63  uint8_t val[I2C_FRAME_SIZE];
64  if (hal.i2c->readRegisters(PX4FLOW_I2C_ADDRESS, 0, I2C_FRAME_SIZE, val) != 0) {
65  i2c_sem->give();
66  return;
67  }
68 
69  // success
70  initialised = true;
71  i2c_sem->give();
72 }

84 bool AP_OpticalFlow_Linux::read(optical_flow_s* report)
85 {
86  // get pointer to i2c bus semaphore
87  AP_HAL::Semaphore *i2c_sem = hal.i2c->get_semaphore();
88  if (i2c_sem == NULL) {
89  return false;
90  }
91 
92  // take i2c bus sempahore (non blocking)
93  if (!i2c_sem->take_nonblocking()) {
94  return false;
95  }
96 
97  // request measurement
98  request_measurement();
99 
100  uint8_t val[I2C_FRAME_SIZE + I2C_INTEGRAL_FRAME_SIZE] = {};
101  i2c_integral_frame f_integral;
102 
103  // Perform the writing and reading in a single command
104  if (PX4FLOW_REG == 0x00) {
105  if (hal.i2c->readRegisters(PX4FLOW_I2C_ADDRESS, 0, I2C_FRAME_SIZE + I2C_INTEGRAL_FRAME_SIZE, val) != 0) {
106  goto fail_transfer;
107  }
108  memcpy(&f_integral, &(val[I2C_FRAME_SIZE]), I2C_INTEGRAL_FRAME_SIZE);
109  }
110 
111  if (PX4FLOW_REG == 0x16) {
112  if (hal.i2c->readRegisters(PX4FLOW_I2C_ADDRESS, 0, I2C_INTEGRAL_FRAME_SIZE, val) != 0) {
113  goto fail_transfer;
114  }
115  memcpy(&f_integral, val, I2C_INTEGRAL_FRAME_SIZE);
116  }
117 
118  i2c_sem->give();
119 
120  // reduce error count
121  if (num_errors > 0) {
122  num_errors--;
123  }
124 
125  report->pixel_flow_x_integral = static_cast<float>(f_integral.pixel_flow_x_integral) / 10000.0f; //convert to radians
126  report->pixel_flow_y_integral = static_cast<float>(f_integral.pixel_flow_y_integral) / 10000.0f; //convert to radians
127  report->frame_count_since_last_readout = f_integral.frame_count_since_last_readout;
128  report->ground_distance_m = static_cast<float>(f_integral.ground_distance) / 1000.0f; // convert to meters
129  report->quality = f_integral.qual; // 0:bad, 255 max quality
130  report->gyro_x_rate_integral = static_cast<float>(f_integral.gyro_x_rate_integral) / 10000.0f; // convert to radians
131  report->gyro_y_rate_integral = static_cast<float>(f_integral.gyro_y_rate_integral) / 10000.0f; // convert to radians
132  report->gyro_z_rate_integral = static_cast<float>(f_integral.gyro_z_rate_integral) / 10000.0f; // convert to radians
133  report->integration_timespan = f_integral.integration_timespan; // microseconds
134  report->time_since_last_sonar_update = f_integral.sonar_timestamp; // microseconds
135  report->gyro_temperature = f_integral.gyro_temperature; // Temperature * 100 in centi-degrees Celsius
136  report->sensor_id = 0;
137 
138  return true;
139 
140 fail_transfer:
141  num_errors++;
142  i2c_sem->give();
143  return false;
144 }

146 // update - read latest values from sensor and fill in x,y and totals.
147 void AP_OpticalFlow_Linux::update(void)
148 {
149  optical_flow_s report;
150 
151  // return immediately if not initialised or more than half of last 40 reads have failed
152  if (!initialised || num_errors >= 20) {
153  return;
154  }
155 
156  // throttle reads to no more than 10hz
157  uint32_t now = hal.scheduler->millis();
158  if (now - last_read_ms < 100) {
159  return;
160  }
161  last_read_ms = now;
162 
163  // read the report from the sensor
164  if (!read(&report)) {
165  return;
166  }
167 
168  // process
169  struct OpticalFlow::OpticalFlow_state state;
170  state.device_id = report.sensor_id;
171  state.surface_quality = report.quality;
172  if (report.integration_timespan > 0) {
173  const Vector2f flowScaler = _flowScaler();
174  float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
175  float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
176  float integralToRate = 1e6f / float(report.integration_timespan);
177  state.flowRate.x = flowScaleFactorX * integralToRate * float(report.pixel_flow_x_integral); // rad/sec measured optically about the X sensor axis
178  state.flowRate.y = flowScaleFactorY * integralToRate * float(report.pixel_flow_y_integral); // rad/sec measured optically about the Y sensor axis
179  state.bodyRate.x = integralToRate * float(report.gyro_x_rate_integral); // rad/sec measured inertially about the X sensor axis
180  state.bodyRate.y = integralToRate * float(report.gyro_y_rate_integral); // rad/sec measured inertially about the Y sensor axis
181  } else {
182  state.flowRate.zero();
183  state.bodyRate.zero();
184  }
185 
186  // copy results to front end
187  _update_frontend(state);
188 
189 #if PX4FLOW_DEBUG
190  hal.console->printf("PX4FLOW id:%u qual:%u FlowRateX:%4.2f Y:%4.2f BodyRateX:%4.2f y:%4.2f\n",
191  (unsigned)state.device_id,
192  (unsigned)state.surface_quality,
193  (double)state.flowRate.x,
194  (double)state.flowRate.y,
195  (double)state.bodyRate.x,
196  (double)state.bodyRate.y);
197 #endif
198 }
