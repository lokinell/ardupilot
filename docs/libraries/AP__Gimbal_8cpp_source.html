88 Vector3f AP_Gimbal::getGimbalRateDemVecYaw(const Quaternion &quatEst)
89 {
90  // Define rotation from vehicle to gimbal using a 312 rotation sequence
91  Matrix3f Tvg;
92  float cosPhi = cosf(_measurement.joint_angles.x);
93  float cosTheta = cosf(_measurement.joint_angles.y);
94  float sinPhi = sinf(_measurement.joint_angles.x);
95  float sinTheta = sinf(_measurement.joint_angles.y);
96  float sinPsi = sinf(_measurement.joint_angles.z);
97  float cosPsi = cosf(_measurement.joint_angles.z);
98  Tvg[0][0] = cosTheta*cosPsi-sinPsi*sinPhi*sinTheta;
99  Tvg[1][0] = -sinPsi*cosPhi;
100  Tvg[2][0] = cosPsi*sinTheta+cosTheta*sinPsi*sinPhi;
101  Tvg[0][1] = cosTheta*sinPsi+cosPsi*sinPhi*sinTheta;
102  Tvg[1][1] = cosPsi*cosPhi;
103  Tvg[2][1] = sinPsi*sinTheta-cosTheta*cosPsi*sinPhi;
104  Tvg[0][2] = -sinTheta*cosPhi;
105  Tvg[1][2] = sinPhi;
106  Tvg[2][2] = cosTheta*cosPhi;
107 
108  // multiply the yaw joint angle by a gain to calculate a demanded vehicle frame relative rate vector required to keep the yaw joint centred
109  Vector3f gimbalRateDemVecYaw;
110  gimbalRateDemVecYaw.z = - _gimbalParams.K_gimbalRate * _measurement.joint_angles.z;
111 
112  // Get filtered vehicle turn rate in earth frame
113  vehicleYawRateFilt = (1.0f - yawRateFiltPole * _measurement.delta_time) * vehicleYawRateFilt + yawRateFiltPole * _measurement.delta_time * _ahrs.get_yaw_rate_earth();
114  Vector3f vehicle_rate_ef(0,0,vehicleYawRateFilt);
115 
116  // calculate the maximum steady state rate error corresponding to the maximum permitted yaw angle error
117  float maxRate = _gimbalParams.K_gimbalRate * yawErrorLimit;
118  float vehicle_rate_mag_ef = vehicle_rate_ef.length();
119  float excess_rate_correction = fabsf(vehicle_rate_mag_ef) - maxRate;
120  if (vehicle_rate_mag_ef > maxRate) {

126  }
127

133 Vector3f AP_Gimbal::getGimbalRateDemVecTilt(const Quaternion &quatEst)
134 {
135  // Calculate the gimbal 321 Euler angle estimates relative to earth frame
136  Vector3f eulerEst = quatEst.to_vector312();
137 
138  // Calculate a demanded quaternion using the demanded roll and pitch and estimated yaw (yaw is slaved to the vehicle)
139  Quaternion quatDem;
140  quatDem.from_vector312( _angle_ef_target_rad.x,
141  _angle_ef_target_rad.y,
142  eulerEst.z);
143 
144  //divide the demanded quaternion by the estimated to get the error
145  Quaternion quatErr = quatDem / quatEst;
146 
147  // Convert to a delta rotation using a small angle approximation
148  quatErr.normalize();
149  Vector3f deltaAngErr;
150  float scaler;
151  if (quatErr[0] >= 0.0f) {
152  scaler = 2.0f;
153  } else {
154  scaler = -2.0f;
155  }
156  deltaAngErr.x = scaler * quatErr[1];
157  deltaAngErr.y = scaler * quatErr[2];
158  deltaAngErr.z = scaler * quatErr[3];
159 
160  // multiply the angle error vector by a gain to calculate a demanded gimbal rate required to control tilt
161  Vector3f gimbalRateDemVecTilt = deltaAngErr * _gimbalParams.K_gimbalRate;
