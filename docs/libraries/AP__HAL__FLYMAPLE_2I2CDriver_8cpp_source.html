103 uint8_t FLYMAPLEI2CDriver::readRegisters(uint8_t addr, uint8_t reg,
104  uint8_t len, uint8_t* data)
105 {
106  // We conduct a write of the register number we want followed by a read
107  data[0] = reg; // Temporarily steal this for the write
108  i2c_msg msgs[2];
109  msgs[0].addr = addr;
110  msgs[0].flags = 0; // Write
111  msgs[0].length = 1;
112  msgs[0].data = data;
113  // Second transaction is a read
114  msgs[1].addr = addr;
115  msgs[1].flags = I2C_MSG_READ;
116  msgs[1].length = len;
117  msgs[1].data = data;
118  return _transfer(msgs, 2);
119 }

125 uint8_t FLYMAPLEI2CDriver::_transfer(i2c_msg *msgs, uint16 num)
126 {
127  // ALERT: patch to libmaple required for this to work else
128  // crashes next line due to a bug in latest git libmaple see http://forums.leaflabs.com/topic.php?id=13458
129  int32 result = i2c_master_xfer(I2C1, msgs, num, _timeout_ms);
130  if (result != 0) {
131  // Some sort of I2C bus fault, or absent device, reinitialise the bus
132  _reset();
133  if (!_ignore_errors) {
134  _lockup_count++;
135  }
136  }
137  return result != 0;
138 }
