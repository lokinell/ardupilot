123 // check for Failsafe conditions. This is called at 10Hz by the main
124 // ArduPlane code
125 void
126 APM_OBC::check(APM_OBC::control_mode mode, uint32_t last_heartbeat_ms, bool geofence_breached, uint32_t last_valid_rc_ms)
127 {
128  if (!_enable) {
129  return;
130  }
131  // we always check for fence breach
132  if (geofence_breached || check_altlimit()) {
133  if (!_terminate) {
134  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "Fence TERMINATE");
135  _terminate.set(1);
136  }
137  }
138 
139  // check for RC failure in manual mode
140  if (_state != STATE_PREFLIGHT &&
141  (mode == OBC_MANUAL || mode == OBC_FBW) &&
142  _rc_fail_time != 0 &&
143  (hal.scheduler->millis() - last_valid_rc_ms) > (unsigned)_rc_fail_time.get()) {
144  if (!_terminate) {
145  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "RC failure terminate");
146  _terminate.set(1);
147  }
148  }
149 
150  // tell the failsafe board if we are in manual control
151  // mode. This tells it to pass through controls from the
152  // receiver
153  if (_manual_pin != -1) {
154  hal.gpio->pinMode(_manual_pin, HAL_GPIO_OUTPUT);
155  hal.gpio->write(_manual_pin, mode==OBC_MANUAL);
156  }
157 
158  uint32_t now = hal.scheduler->millis();
159  bool gcs_link_ok = ((now - last_heartbeat_ms) < 10000);
160  bool gps_lock_ok = ((now - gps.last_fix_time_ms()) < 3000);
161 
162  switch (_state) {
163  case STATE_PREFLIGHT:
164  // we startup in preflight mode. This mode ends when
165  // we first enter auto.
166  if (mode == OBC_AUTO) {
167  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "Starting AFS_AUTO");
168  _state = STATE_AUTO;
169  }
170  break;
171 
172  case STATE_AUTO:
173  // this is the normal mode.
174  if (!gcs_link_ok) {
175  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_ERROR, "State DATA_LINK_LOSS");
176  _state = STATE_DATA_LINK_LOSS;
177  if (_wp_comms_hold) {
178  _saved_wp = mission.get_current_nav_cmd().index;
179  mission.set_current_cmd(_wp_comms_hold);
180  }
181  // if two events happen within 30s we consider it to be part of the same event
182  if (now - _last_comms_loss_ms > 30*1000UL) {
183  _comms_loss_count++;
184  _last_comms_loss_ms = now;
185  }
186  break;
187  }
188  if (!gps_lock_ok) {
189  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "State GPS_LOSS");
190  _state = STATE_GPS_LOSS;
191  if (_wp_gps_loss) {
192  _saved_wp = mission.get_current_nav_cmd().index;
193  mission.set_current_cmd(_wp_gps_loss);
194  }
195  // if two events happen within 30s we consider it to be part of the same event
196  if (now - _last_gps_loss_ms > 30*1000UL) {
197  _gps_loss_count++;
198  _last_gps_loss_ms = now;
199  }
200  break;
201  }
202  break;
203 
204  case STATE_DATA_LINK_LOSS:
205  if (!gps_lock_ok) {
206  // losing GPS lock when data link is lost
207  // leads to termination
208  if (!_terminate) {
209  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "Dual loss TERMINATE");
210  _terminate.set(1);
211  }
212  } else if (gcs_link_ok) {
213  _state = STATE_AUTO;
214  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "GCS OK");
215  // we only return to the mission if we have not exceeded AFS_MAX_COM_LOSS
216  if (_saved_wp != 0 &&
217  (_max_comms_loss <= 0 ||
218  _comms_loss_count <= _max_comms_loss)) {
219  mission.set_current_cmd(_saved_wp);
220  _saved_wp = 0;
221  }
222  }
223  break;
224 
225  case STATE_GPS_LOSS:
226  if (!gcs_link_ok) {
227  // losing GCS link when GPS lock lost
228  // leads to termination
229  if (!_terminate) {
230  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_CRITICAL, "Dual loss TERMINATE");
231  _terminate.set(1);
232  }
233  } else if (gps_lock_ok) {
234  GCS_MAVLINK::send_statustext_all(MAV_SEVERITY_INFO, "GPS OK");
235  _state = STATE_AUTO;
236  // we only return to the mission if we have not exceeded AFS_MAX_GPS_LOSS
237  if (_saved_wp != 0 &&
238  (_max_gps_loss <= 0 ||
239  _gps_loss_count <= _max_gps_loss)) {
240  mission.set_current_cmd(_saved_wp);
241  _saved_wp = 0;
242  }
243  }
244  break;
245  }
246 
247  // if we are not terminating or if there is a separate terminate
248  // pin configured then toggle the heartbeat pin at 10Hz
249  if (_heartbeat_pin != -1 && (_terminate_pin != -1 || !_terminate)) {
250  _heartbeat_pin_value = !_heartbeat_pin_value;
251  hal.gpio->pinMode(_heartbeat_pin, HAL_GPIO_OUTPUT);
252  hal.gpio->write(_heartbeat_pin, _heartbeat_pin_value);
253  }
254 
255  // set the terminate pin
256  if (_terminate_pin != -1) {
257  hal.gpio->pinMode(_terminate_pin, HAL_GPIO_OUTPUT);
258  hal.gpio->write(_terminate_pin, _terminate?1:0);
259  }
260 }

280 // check for altitude limit breach
281 bool
282 APM_OBC::check_altlimit(void)
283 {
284  if (!_enable) {
285  return false;
286  }
287  if (_amsl_limit == 0 || _qnh_pressure <= 0) {
288  // no limit set
289  return false;
290  }
291 
292  // see if the barometer is dead
293  if (hal.scheduler->millis() - baro.get_last_update() > 5000) {
294  // the barometer has been unresponsive for 5 seconds. See if we can switch to GPS
295  if (_amsl_margin_gps != -1 &&
296  gps.status() >= AP_GPS::GPS_OK_FIX_3D &&
297  gps.location().alt*0.01f <= _amsl_limit - _amsl_margin_gps) {
298  // GPS based altitude OK
299  return false;
300  }
301  // no barometer - immediate termination
302  return true;
303  }
304 
305  float alt_amsl = baro.get_altitude_difference(_qnh_pressure*100, baro.get_pressure());
306  if (alt_amsl > _amsl_limit) {
307  // pressure altitude breach
308  return true;
309  }
310 
311  // all OK
312  return false;
313 }

316  setup the IO boards failsafe values for if the FMU firmware crashes
317  */
318 void APM_OBC::setup_failsafe(void)
319 {
320  if (!_enable) {
321  return;
322  }
323  const RC_Channel *ch_roll = RC_Channel::rc_channel(rcmap.roll()-1);
324  const RC_Channel *ch_pitch = RC_Channel::rc_channel(rcmap.pitch()-1);
325  const RC_Channel *ch_yaw = RC_Channel::rc_channel(rcmap.yaw()-1);
326  const RC_Channel *ch_throttle = RC_Channel::rc_channel(rcmap.throttle()-1);
327 
328  // setup primary channel output values
329  hal.rcout->set_failsafe_pwm(1U<<(rcmap.roll()-1), ch_roll->get_limit_pwm(RC_Channel::RC_CHANNEL_LIMIT_MIN));
330  hal.rcout->set_failsafe_pwm(1U<<(rcmap.pitch()-1), ch_pitch->get_limit_pwm(RC_Channel::RC_CHANNEL_LIMIT_MAX));
331  hal.rcout->set_failsafe_pwm(1U<<(rcmap.yaw()-1), ch_yaw->get_limit_pwm(RC_Channel::RC_CHANNEL_LIMIT_MAX));
332  hal.rcout->set_failsafe_pwm(1U<<(rcmap.throttle()-1), ch_throttle->get_limit_pwm(RC_Channel::RC_CHANNEL_LIMIT_MIN));
333 
334  // and all aux channels
335  RC_Channel_aux::set_servo_failsafe(RC_Channel_aux::k_flap_auto, RC_Channel::RC_CHANNEL_LIMIT_MAX);
336  RC_Channel_aux::set_servo_failsafe(RC_Channel_aux::k_flap, RC_Channel::RC_CHANNEL_LIMIT_MAX);
337  RC_Channel_aux::set_servo_failsafe(RC_Channel_aux::k_aileron, RC_Channel::RC_CHANNEL_LIMIT_MIN);

339  RC_Channel_aux::set_servo_failsafe(RC_Channel_aux::k_elevator, RC_Channel::RC_CHANNEL_LIMIT_MAX);
340  RC_Channel_aux::set_servo_failsafe(RC_Channel_aux::k_elevator_with_input, RC_Channel::RC_CHANNEL_LIMIT_MAX);
