48 // Process bytes available from the stream
49 //
50 // The stream is assumed to contain only messages we recognise. If it
51 // contains other messages, and those messages contain the preamble
52 // bytes, it is possible for this code to fail to synchronise to the
53 // stream immediately. Without buffering the entire message and
54 // re-processing it from the top, this is unavoidable. The parser
55 // attempts to avoid this when possible.
56 //
57 bool
58 AP_GPS_SIRF::read(void)
59 {
60  uint8_t data;
61  int16_t numc;
62  bool parsed = false;
63 
64  numc = port->available();
65  while(numc--) {
66 
67  // read the next byte
68  data = port->read();
69 
70  switch(_step) {
71 
72  // Message preamble detection
73  //
74  // If we fail to match any of the expected bytes, we reset
75  // the state machine and re-consider the failed byte as
76  // the first byte of the preamble. This improves our
77  // chances of recovering from a mismatch and makes it less
78  // likely that we will be fooled by the preamble appearing
79  // as data in some other message.
80  //
81  case 1:
82  if (PREAMBLE2 == data) {
83  _step++;
84  break;
85  }
86  _step = 0;
87  // FALLTHROUGH
88  case 0:
89  if(PREAMBLE1 == data)
90  _step++;
91  break;
92 
93  // Message length
94  //
95  // We always collect the length so that we can avoid being
96  // fooled by preamble bytes in messages.
97  //
98  case 2:
99  _step++;
100  _payload_length = (uint16_t)data << 8;
101  break;
102  case 3:
103  _step++;
104  _payload_length |= data;
105  _payload_counter = 0;
106  _checksum = 0;
107  break;
108 
109  // Message header processing
110  //
111  // We sniff the message ID to determine whether we are going
112  // to gather the message bytes or just discard them.
113  //
114  case 4:
115  _step++;
116  _accumulate(data);
117  _payload_length--;
118  _gather = false;
119  switch(data) {
120  case MSG_GEONAV:
121  if (_payload_length == sizeof(sirf_geonav)) {
122  _gather = true;
123  _msg_id = data;
124  }
125  break;
126  }
127  break;
128 
129  // Receive message data
130  //
131  // Note that we are effectively guaranteed by the protocol
132  // that the checksum and postamble cannot be mistaken for
133  // the preamble, so if we are discarding bytes in this
134  // message when the payload is done we return directly
135  // to the preamble detector rather than bothering with
136  // the checksum logic.
137  //
138  case 5:
139  if (_gather) { // gather data if requested
140  _accumulate(data);
141  _buffer.bytes[_payload_counter] = data;
142  if (++_payload_counter == _payload_length)
143  _step++;
144  } else {
145  if (++_payload_counter == _payload_length)
146  _step = 0;
147  }
148  break;
149 
150  // Checksum and message processing
151  //
152  case 6:
153  _step++;
154  if ((_checksum >> 8) != data) {
155  _step = 0;
156  }
157  break;
158  case 7:
159  _step = 0;
160  if ((_checksum & 0xff) != data) {
161  break;
162  }
163  if (_gather) {
164  parsed = _parse_gps(); // Parse the new GPS packet
165  }
166  }
167  }
168  return(parsed);
169 }

171 bool
172 AP_GPS_SIRF::_parse_gps(void)
173 {
174  switch(_msg_id) {
175  case MSG_GEONAV:
176  //time = _swapl(&_buffer.nav.time);
177  // parse fix type
178  if (_buffer.nav.fix_invalid) {
179  state.status = AP_GPS::NO_FIX;
180  }else if ((_buffer.nav.fix_type & FIX_MASK) == FIX_3D) {
181  state.status = AP_GPS::GPS_OK_FIX_3D;
182  }else{
183  state.status = AP_GPS::GPS_OK_FIX_2D;
184  }
185  state.location.lat = swap_int32(_buffer.nav.latitude);
186  state.location.lng = swap_int32(_buffer.nav.longitude);
187  state.location.alt = swap_int32(_buffer.nav.altitude_msl);
188  state.ground_speed = swap_int32(_buffer.nav.ground_speed)*0.01f;
189  state.ground_course_cd = wrap_360_cd(swap_int16(_buffer.nav.ground_course));
190  state.num_sats = _buffer.nav.satellites;
191  fill_3d_velocity();
192  return true;
193  }
194  return false;
195 }

208 bool
209 AP_GPS_SIRF::_detect(struct SIRF_detect_state &state, uint8_t data)
210 {
211  switch (state.step) {
212  case 1:
213  if (PREAMBLE2 == data) {
214  state.step++;
215  break;
216  }
217  state.step = 0;
218  case 0:
219  state.payload_length = state.payload_counter = state.checksum = 0;
220  if (PREAMBLE1 == data)
221  state.step++;
222  break;
223  case 2:
224  state.step++;
225  if (data != 0) {
226  // only look for short messages
227  state.step = 0;
228  }
229  break;
230  case 3:
231  state.step++;
232  state.payload_length = data;
233  break;
234  case 4:
235  state.checksum = (state.checksum + data) & 0x7fff;
236  if (++state.payload_counter == state.payload_length)
237  state.step++;
238  break;
239  case 5:
240  state.step++;
241  if ((state.checksum >> 8) != data) {
242  state.step = 0;
243  }
244  break;
245  case 6:
246  state.step = 0;
247  if ((state.checksum & 0xff) == data) {
248  return true;
249  }
250  }
251  return false;
252 }
