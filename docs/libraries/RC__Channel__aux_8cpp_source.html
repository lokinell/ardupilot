88 void RC_Channel_aux::update_aux_servo_function(void)
89 {
90  // set auxiliary ranges
91  for (uint8_t i = 0; i < RC_AUX_MAX_CHANNELS; i++) {
92  if (_aux_channels[i] == NULL) continue;
93  RC_Channel_aux::Aux_servo_function_t function = (RC_Channel_aux::Aux_servo_function_t)_aux_channels[i]->function.get();
94  switch (function) {
95  case RC_Channel_aux::k_flap:
96  case RC_Channel_aux::k_flap_auto:
97  case RC_Channel_aux::k_egg_drop:
98  _aux_channels[i]->set_range(0,100);
99  break;
100  case RC_Channel_aux::k_aileron:
101  case RC_Channel_aux::k_aileron_with_input:
102  case RC_Channel_aux::k_elevator:
103  case RC_Channel_aux::k_elevator_with_input:
104  case RC_Channel_aux::k_dspoiler1:
105  case RC_Channel_aux::k_dspoiler2:
106  case RC_Channel_aux::k_rudder:
107  case RC_Channel_aux::k_steering:
108  case RC_Channel_aux::k_flaperon1:
109  case RC_Channel_aux::k_flaperon2:
110  _aux_channels[i]->set_angle(4500);
111  break;
112  default:
113  break;
114  }
115  }
116 
117  // create a function mask to make updates master
118  _function_mask = 0;
119  for (uint8_t i = 0; i < RC_AUX_MAX_CHANNELS; i++) {
120  if (_aux_channels[i]) {
121  RC_Channel_aux::Aux_servo_function_t function = (RC_Channel_aux::Aux_servo_function_t)_aux_channels[i]->function.get();
122  if (function < k_nr_aux_servo_functions) {
123  _function_mask |= (1UL<<(uint8_t)function);
124  }
125  }
126  }
