62 bool FLYMAPLESemaphore::_take_from_mainloop(uint32_t timeout_ms) {
63  /* Try to take immediately */
64  if (_take_nonblocking()) {
65  return true;
66  } else if (timeout_ms == 0) {
67  /* Return immediately if timeout is 0 */
68  return false;
69  }
70 
71  do {
72  /* Delay 1ms until we can successfully take, or we timed out */
73  hal.scheduler->delay(1);
74  timeout_ms--;
75  if (_take_nonblocking()) {
76  return true;
77  }
78  } while (timeout_ms > 0);
79 
80  return false;
81 }
