49 // run a full DCM update round
50 void
51 AP_AHRS_DCM::update(void)
52 {
53  float delta_t;
54 
55  if (_last_startup_ms == 0) {
56  _last_startup_ms = hal.scheduler->millis();
57  }
58 
59  // tell the IMU to grab some data
60  _ins.update();
61 
62  // ask the IMU how much time this sensor reading represents
63  delta_t = _ins.get_delta_time();
64 
65  // if the update call took more than 0.2 seconds then discard it,
66  // otherwise we may move too far. This happens when arming motors
67  // in ArduCopter
68  if (delta_t > 0.2f) {
69  memset(&_ra_sum[0], 0, sizeof(_ra_sum));
70  _ra_deltat = 0;
71  return;
72  }
73 
74  // Integrate the DCM matrix using gyro inputs
75  matrix_update(delta_t);
76 
77  // Normalize the DCM matrix
78  normalize();
79 
80  // Perform drift correction
81  drift_correction(delta_t);
82 
83  // paranoid check for bad values in the DCM matrix
84  check_matrix();
85 
86  // Calculate pitch, roll, yaw for stabilization and navigation
87  euler_angles();
88 
89  // update trig values including _cos_roll, cos_pitch
90  update_trig();

93 // update the DCM matrix using only the gyros
94 void
95 AP_AHRS_DCM::matrix_update(float _G_Dt)
96 {
97  // note that we do not include the P terms in _omega. This is
98  // because the spin_rate is calculated from _omega.length(),
99  // and including the P terms would give positive feedback into
100  // the _P_gain() calculation, which can lead to a very large P
101  // value
102  _omega.zero();
103 
104  // average across first two healthy gyros. This reduces noise on
105  // systems with more than one gyro. We don't use the 3rd gyro
106  // unless another is unhealthy as 3rd gyro on PH2 has a lot more
107  // noise
108  uint8_t healthy_count = 0;
109  Vector3f delta_angle;
110  for (uint8_t i=0; i<_ins.get_gyro_count(); i++) {
111  if (_ins.get_gyro_health(i) && healthy_count < 2) {
112  Vector3f dangle;
113  if (_ins.get_delta_angle(i, dangle)) {
114  healthy_count++;
115  delta_angle += dangle;
116  }
117  }
118  }
119  if (healthy_count > 1) {
120  delta_angle /= healthy_count;
121  }
122  if (_G_Dt > 0) {
123  _omega = delta_angle / _G_Dt;
124  _omega += _omega_I;
125  _dcm_matrix.rotate((_omega + _omega_P + _omega_yaw_P) * _G_Dt);
126  }
127 }

130 /*
131  * reset the DCM matrix and omega. Used on ground start, and on
132  * extreme errors in the matrix
133  */
134 void
135 AP_AHRS_DCM::reset(bool recover_eulers)
136 {
137  // reset the integration terms
138  _omega_I.zero();
139  _omega_P.zero();
140  _omega_yaw_P.zero();
141  _omega.zero();
142 
143  // if the caller wants us to try to recover to the current
144  // attitude then calculate the dcm matrix from the current
145  // roll/pitch/yaw values
146  if (recover_eulers && !isnan(roll) && !isnan(pitch) && !isnan(yaw)) {
147  _dcm_matrix.from_euler(roll, pitch, yaw);
148  } else {
149 
150  // Use the measured accel due to gravity to calculate an initial
151  // roll and pitch estimate
152 
153  // Get body frame accel vector
154  Vector3f initAccVec;
155  uint8_t counter = 0;
156  initAccVec = _ins.get_accel();
157 
158  // the first vector may be invalid as the filter starts up
159  while (initAccVec.length() <= 5.0f && counter++ < 10) {
160  _ins.wait_for_sample();
161  _ins.update();
162  initAccVec = _ins.get_accel();
163  }
164 
165  // normalise the acceleration vector
166  if (initAccVec.length() > 5.0f) {
167  // calculate initial pitch angle
168  pitch = atan2f(initAccVec.x, pythagorous2(initAccVec.y, initAccVec.z));
169  // calculate initial roll angle
170  roll = atan2f(-initAccVec.y, -initAccVec.z);
171  } else {
172  // If we cant use the accel vector, then align flat
173  roll = 0.0f;
174  pitch = 0.0f;
175  }
176  _dcm_matrix.from_euler(roll, pitch, 0);
177 
178  }
179 
180  _last_startup_ms = hal.scheduler->millis();
181 }

220 // renormalise one vector component of the DCM matrix
221 // this will return false if renormalization fails
222 bool
223 AP_AHRS_DCM::renorm(Vector3f const &a, Vector3f &result)
224 {
225  float renorm_val;
226 
227  // numerical errors will slowly build up over time in DCM,
228  // causing inaccuracies. We can keep ahead of those errors
229  // using the renormalization technique from the DCM IMU paper
230  // (see equations 18 to 21).
231 
232  // For APM we don't bother with the taylor expansion
233  // optimisation from the paper as on our 2560 CPU the cost of
234  // the sqrt() is 44 microseconds, and the small time saving of
235  // the taylor expansion is not worth the potential of
236  // additional error buildup.
237 
238  // Note that we can get significant renormalisation values
239  // when we have a larger delta_t due to a glitch eleswhere in
240  // APM, such as a I2c timeout or a set of EEPROM writes. While
241  // we would like to avoid these if possible, if it does happen
242  // we don't want to compound the error by making DCM less
243  // accurate.
244 
245  renorm_val = 1.0f / a.length();
246 
247  // keep the average for reporting
248  _renorm_val_sum += renorm_val;
249  _renorm_val_count++;
250 
251  if (!(renorm_val < 2.0f && renorm_val > 0.5f)) {
252  // this is larger than it should get - log it as a warning
253  if (!(renorm_val < 1.0e6f && renorm_val > 1.0e-6f)) {
254  // we are getting values which are way out of
255  // range, we will reset the matrix and hope we
256  // can recover our attitude using drift
257  // correction before we hit the ground!
258  //Serial.printf("ERROR: DCM renormalisation error. renorm_val=%f\n",
259  // renorm_val);
260  return false;
261  }
262  }
263 
264  result = a * renorm_val;
265  return true;
266 }

268 /*************************************************
269  * Direction Cosine Matrix IMU: Theory
270  * William Premerlani and Paul Bizard
271  *
272  * Numerical errors will gradually reduce the orthogonality conditions expressed by equation 5
273  * to approximations rather than identities. In effect, the axes in the two frames of reference no
274  * longer describe a rigid body. Fortunately, numerical error accumulates very slowly, so it is a
275  * simple matter to stay ahead of it.
276  * We call the process of enforcing the orthogonality conditions ÒrenormalizationÓ.
277  */
278 void
279 AP_AHRS_DCM::normalize(void)
280 {
281  float error;
282  Vector3f t0, t1, t2;
283 
284  error = _dcm_matrix.a * _dcm_matrix.b; // eq.18
285 
286  t0 = _dcm_matrix.a - (_dcm_matrix.b * (0.5f * error)); // eq.19
287  t1 = _dcm_matrix.b - (_dcm_matrix.a * (0.5f * error)); // eq.19
288  t2 = t0 % t1; // c= a x b // eq.20
289 
290  if (!renorm(t0, _dcm_matrix.a) ||
291  !renorm(t1, _dcm_matrix.b) ||
292  !renorm(t2, _dcm_matrix.c)) {
293  // Our solution is blowing up and we will force back
294  // to last euler angles
295  _last_failure_ms = hal.scheduler->millis();
296  AP_AHRS_DCM::reset(true);
297  }
298 }

301 // produce a yaw error value. The returned value is proportional
302 // to sin() of the current heading error in earth frame
303 float
304 AP_AHRS_DCM::yaw_error_compass(void)
305 {
306  const Vector3f &mag = _compass->get_field();
307  // get the mag vector in the earth frame
308  Vector2f rb = _dcm_matrix.mulXY(mag);
309 
310  if (rb.length() < FLT_EPSILON) {
311  return 0.0f;
312  }
313 
314  rb.normalize();
315  if (rb.is_inf()) {
316  // not a valid vector
317  return 0.0f;
318  }
319 
320  // update vector holding earths magnetic field (if required)
321  if( !is_equal(_last_declination,_compass->get_declination()) ) {
322  _last_declination = _compass->get_declination();
323  _mag_earth.x = cosf(_last_declination);
324  _mag_earth.y = sinf(_last_declination);
325  }
326 
327  // calculate the error term in earth frame
328  // calculate the Z component of the cross product of rb and _mag_earth
329  return rb % _mag_earth;
330 }

350 // High levels of noise on _accel_ef will cause the gain to drop and could lead to

389 // return true if we should use the compass for yaw correction
390 bool AP_AHRS_DCM::use_compass(void)
391 {
392  if (!_compass || !_compass->use_for_yaw()) {
393  // no compass available
394  return false;
395  }
396  if (!_flags.fly_forward || !have_gps()) {
397  // we don't have any alterative to the compass
398  return true;
399  }
400  if (_gps.ground_speed() < GPS_SPEED_MIN) {
401  // we are not going fast enough to use the GPS
402  return true;
403  }
404 
405  // if the current yaw differs from the GPS yaw by more than 45
406  // degrees and the estimated wind speed is less than 80% of the
407  // ground speed, then switch to GPS navigation. This will help
408  // prevent flyaways with very bad compass offsets
409  int32_t error = abs(wrap_180_cd(yaw_sensor - _gps.ground_course_cd()));
410  if (error > 4500 && _wind.length() < _gps.ground_speed()*0.8f) {
411  if (hal.scheduler->millis() - _last_consistent_heading > 2000) {
412  // start using the GPS for heading if the compass has been
413  // inconsistent with the GPS for 2 seconds
414  return false;
415  }
416  } else {
417  _last_consistent_heading = hal.scheduler->millis();
418  }
419 
420  // use the compass
421  return true;
422 }

424 // yaw drift correction using the compass or GPS
425 // this function prodoces the _omega_yaw_P vector, and also
426 // contributes to the _omega_I.z long term yaw drift estimate
427 void
428 AP_AHRS_DCM::drift_correction_yaw(void)
429 {
430  bool new_value = false;
431  float yaw_error;
432  float yaw_deltat;
433 
434  if (AP_AHRS_DCM::use_compass()) {
435  /*
436  we are using compass for yaw
437  */
438  if (_compass->last_update_usec() != _compass_last_update) {
439  yaw_deltat = (_compass->last_update_usec() - _compass_last_update) * 1.0e-6f;
440  _compass_last_update = _compass->last_update_usec();
441  // we force an additional compass read()
442  // here. This has the effect of throwing away
443  // the first compass value, which can be bad
444  if (!_flags.have_initial_yaw && _compass->read()) {
445  float heading = _compass->calculate_heading(_dcm_matrix);
446  _dcm_matrix.from_euler(roll, pitch, heading);
447  _omega_yaw_P.zero();
448  _flags.have_initial_yaw = true;
449  }
450  new_value = true;
451  yaw_error = yaw_error_compass();
452 
453  // also update the _gps_last_update, so if we later
454  // disable the compass due to significant yaw error we
455  // don't suddenly change yaw with a reset
456  _gps_last_update = _gps.last_fix_time_ms();
457  }
458  } else if (_flags.fly_forward && have_gps()) {
459  /*
460  we are using GPS for yaw
461  */
462  if (_gps.last_fix_time_ms() != _gps_last_update &&
463  _gps.ground_speed() >= GPS_SPEED_MIN) {
464  yaw_deltat = (_gps.last_fix_time_ms() - _gps_last_update) * 1.0e-3f;
465  _gps_last_update = _gps.last_fix_time_ms();
466  new_value = true;
467  float gps_course_rad = ToRad(_gps.ground_course_cd() * 0.01f);
468  float yaw_error_rad = wrap_PI(gps_course_rad - yaw);
469  yaw_error = sinf(yaw_error_rad);
470 
471  /* reset yaw to match GPS heading under any of the
472  following 3 conditions:
473 
474  1) if we have reached GPS_SPEED_MIN and have never had
475  yaw information before
476 
477  2) if the last time we got yaw information from the GPS
478  is more than 20 seconds ago, which means we may have
479  suffered from considerable gyro drift
480 
481  3) if we are over 3*GPS_SPEED_MIN (which means 9m/s)
482  and our yaw error is over 60 degrees, which means very
483  poor yaw. This can happen on bungee launch when the
484  operator pulls back the plane rapidly enough then on
485  release the GPS heading changes very rapidly
486  */
487  if (!_flags.have_initial_yaw ||
488  yaw_deltat > 20 ||
489  (_gps.ground_speed() >= 3*GPS_SPEED_MIN && fabsf(yaw_error_rad) >= 1.047f)) {
490  // reset DCM matrix based on current yaw
491  _dcm_matrix.from_euler(roll, pitch, gps_course_rad);
492  _omega_yaw_P.zero();
493  _flags.have_initial_yaw = true;
494  yaw_error = 0;
495  }
496  }
497  }
498 
499  if (!new_value) {
500  // we don't have any new yaw information
501  // slowly decay _omega_yaw_P to cope with loss
502  // of our yaw source
503  _omega_yaw_P *= 0.97f;
504  return;
505  }
506 
507  // convert the error vector to body frame
508  float error_z = _dcm_matrix.c.z * yaw_error;
509 
510  // the spin rate changes the P gain, and disables the
511  // integration at higher rates
512  float spin_rate = _omega.length();
513 
514  // sanity check _kp_yaw
515  if (_kp_yaw < AP_AHRS_YAW_P_MIN) {
516  _kp_yaw = AP_AHRS_YAW_P_MIN;
517  }
518 
519  // update the proportional control to drag the
520  // yaw back to the right value. We use a gain
521  // that depends on the spin rate. See the fastRotations.pdf
522  // paper from Bill Premerlani
523  // We also adjust the gain depending on the rate of change of horizontal velocity which
524  // is proportional to how observable the heading is from the acceerations and GPS velocity
525  // The accelration derived heading will be more reliable in turns than compass or GPS
526 
527  _omega_yaw_P.z = error_z * _P_gain(spin_rate) * _kp_yaw * _yaw_gain();
528  if (use_fast_gains()) {
529  _omega_yaw_P.z *= 8;
530  }
531 
532  // don't update the drift term if we lost the yaw reference
533  // for more than 2 seconds
534  if (yaw_deltat < 2.0f && spin_rate < ToRad(SPIN_RATE_LIMIT)) {
535  // also add to the I term
536  _omega_I_sum.z += error_z * _ki_yaw * yaw_deltat;
537  }
538 
539  _error_yaw = 0.8f * _error_yaw + 0.2f * fabsf(yaw_error);
540 }

560 // perform drift correction. This function aims to update _omega_P and
561 // _omega_I with our best estimate of the short term and long term
562 // gyro error. The _omega_P value is what pulls our attitude solution
563 // back towards the reference vector quickly. The _omega_I term is an
564 // attempt to learn the long term drift rate of the gyros.
565 //
566 // This drift correction implementation is based on a paper
567 // by Bill Premerlani from here:
568 // http://gentlenav.googlecode.com/files/RollPitchDriftCompensation.pdf
569 void
570 AP_AHRS_DCM::drift_correction(float deltat)
571 {
572  Vector3f velocity;
573  uint32_t last_correction_time;
574 
575  // perform yaw drift correction if we have a new yaw reference
576  // vector
577  drift_correction_yaw();
578 
579  // rotate accelerometer values into the earth frame
580  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
581  if (_ins.get_accel_health(i)) {
582  /*
583  by using get_delta_velocity() instead of get_accel() the
584  accel value is sampled over the right time delta for
585  each sensor, which prevents an aliasing effect
586  */
587  Vector3f delta_velocity;
588  float delta_velocity_dt;
589  _ins.get_delta_velocity(i, delta_velocity);
590  delta_velocity_dt = _ins.get_delta_velocity_dt(i);
591  if (delta_velocity_dt > 0) {
592  _accel_ef[i] = _dcm_matrix * (delta_velocity / delta_velocity_dt);
593  // integrate the accel vector in the earth frame between GPS readings
594  _ra_sum[i] += _accel_ef[i] * deltat;
595  }
596  }
597  }
598 
599  //update _accel_ef_blended
600  if (_ins.get_accel_count() == 2 && _ins.use_accel(0) && _ins.use_accel(1)) {
601  float imu1_weight_target = _active_accel_instance == 0 ? 1.0f : 0.0f;
602  // slew _imu1_weight over one second
603  _imu1_weight += constrain_float(imu1_weight_target-_imu1_weight, -deltat, deltat);
604  _accel_ef_blended = _accel_ef[0] * _imu1_weight + _accel_ef[1] * (1.0f - _imu1_weight);
605  } else {
606  _accel_ef_blended = _accel_ef[_ins.get_primary_accel()];
607  }
608 
609  // keep a sum of the deltat values, so we know how much time
610  // we have integrated over
611  _ra_deltat += deltat;
612 
613  if (!have_gps() ||
614  _gps.status() < AP_GPS::GPS_OK_FIX_3D ||
615  _gps.num_sats() < _gps_minsats) {
616  // no GPS, or not a good lock. From experience we need at
617  // least 6 satellites to get a really reliable velocity number
618  // from the GPS.
619  //
620  // As a fallback we use the fixed wing acceleration correction
621  // if we have an airspeed estimate (which we only have if
622  // _fly_forward is set), otherwise no correction
623  if (_ra_deltat < 0.2f) {
624  // not enough time has accumulated
625  return;
626  }
627  float airspeed;
628  if (airspeed_sensor_enabled()) {
629  airspeed = _airspeed->get_airspeed();
630  } else {
631  airspeed = _last_airspeed;
632  }
633  // use airspeed to estimate our ground velocity in
634  // earth frame by subtracting the wind
635  velocity = _dcm_matrix.colx() * airspeed;
636 
637  // add in wind estimate
638  velocity += _wind;
639 
640  last_correction_time = hal.scheduler->millis();
641  _have_gps_lock = false;
642  } else {
643  if (_gps.last_fix_time_ms() == _ra_sum_start) {
644  // we don't have a new GPS fix - nothing more to do
645  return;
646  }
647  velocity = _gps.velocity();
648  last_correction_time = _gps.last_fix_time_ms();
649  if (_have_gps_lock == false) {
650  // if we didn't have GPS lock in the last drift
651  // correction interval then set the velocities equal
652  _last_velocity = velocity;
653  }
654  _have_gps_lock = true;
655 
656  // keep last airspeed estimate for dead-reckoning purposes
657  Vector3f airspeed = velocity - _wind;
658  airspeed.z = 0;
659  _last_airspeed = airspeed.length();
660  }
661 
662  if (have_gps()) {
663  // use GPS for positioning with any fix, even a 2D fix
664  _last_lat = _gps.location().lat;
665  _last_lng = _gps.location().lng;
666  _position_offset_north = 0;
667  _position_offset_east = 0;
668 
669  // once we have a single GPS lock, we can update using
670  // dead-reckoning from then on
671  _have_position = true;
672  } else {
673  // update dead-reckoning position estimate
674  _position_offset_north += velocity.x * _ra_deltat;
675  _position_offset_east += velocity.y * _ra_deltat;
676  }
677 
678  // see if this is our first time through - in which case we
679  // just setup the start times and return
680  if (_ra_sum_start == 0) {
681  _ra_sum_start = last_correction_time;
682  _last_velocity = velocity;
683  return;
684  }
685 
686  // equation 9: get the corrected acceleration vector in earth frame. Units
687  // are m/s/s
688  Vector3f GA_e;
689  GA_e = Vector3f(0, 0, -1.0f);
690 
691  if (_ra_deltat <= 0) {
692  // waiting for more data
693  return;
694  }
695 
696  bool using_gps_corrections = false;
697  float ra_scale = 1.0f/(_ra_deltat*GRAVITY_MSS);
698 
699  if (_flags.correct_centrifugal && (_have_gps_lock || _flags.fly_forward)) {
700  float v_scale = gps_gain.get() * ra_scale;
701  Vector3f vdelta = (velocity - _last_velocity) * v_scale;
702  GA_e += vdelta;
703  GA_e.normalize();
704  if (GA_e.is_inf()) {
705  // wait for some non-zero acceleration information
706  _last_failure_ms = hal.scheduler->millis();
707  return;
708  }
709  using_gps_corrections = true;
710  }
711 
712  // calculate the error term in earth frame.
713  // we do this for each available accelerometer then pick the
714  // accelerometer that leads to the smallest error term. This takes
715  // advantage of the different sample rates on different
716  // accelerometers to dramatically reduce the impact of aliasing
717  // due to harmonics of vibrations that match closely the sampling
718  // rate of our accelerometers. On the Pixhawk we have the LSM303D
719  // running at 800Hz and the MPU6000 running at 1kHz, by combining
720  // the two the effects of aliasing are greatly reduced.
721  Vector3f error[INS_MAX_INSTANCES];
722  float error_dirn[INS_MAX_INSTANCES];
723  Vector3f GA_b[INS_MAX_INSTANCES];
724  int8_t besti = -1;
725  float best_error = 0;
726  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
727  if (!_ins.get_accel_health(i)) {
728  // only use healthy sensors
729  continue;
730  }
731  _ra_sum[i] *= ra_scale;
732 
733  // get the delayed ra_sum to match the GPS lag
734  if (using_gps_corrections) {
735  GA_b[i] = ra_delayed(i, _ra_sum[i]);
736  } else {
737  GA_b[i] = _ra_sum[i];
738  }
739  if (GA_b[i].is_zero()) {
740  // wait for some non-zero acceleration information
741  continue;
742  }
743  GA_b[i].normalize();
744  if (GA_b[i].is_inf()) {
745  // wait for some non-zero acceleration information
746  continue;
747  }
748  error[i] = GA_b[i] % GA_e;
749  // Take dot product to catch case vectors are opposite sign and parallel
750  error_dirn[i] = GA_b[i] * GA_e;
751  float error_length = error[i].length();
752  if (besti == -1 || error_length < best_error) {
753  besti = i;
754  best_error = error_length;
755  }
756  // Catch case where orientation is 180 degrees out
757  if (error_dirn[besti] < 0.0f) {
758  best_error = 1.0f;
759  }
760 
761  }
762 
763  if (besti == -1) {
764  // no healthy accelerometers!
765  _last_failure_ms = hal.scheduler->millis();
766  return;
767  }
768 
769  _active_accel_instance = besti;
770 
771 #define YAW_INDEPENDENT_DRIFT_CORRECTION 0
772 #if YAW_INDEPENDENT_DRIFT_CORRECTION
773  // step 2 calculate earth_error_Z
774  float earth_error_Z = error.z;
775 
776  // equation 10
777  float tilt = pythagorous2(GA_e.x, GA_e.y);
778 
779  // equation 11
780  float theta = atan2f(GA_b[besti].y, GA_b[besti].x);
781 
782  // equation 12
783  Vector3f GA_e2 = Vector3f(cosf(theta)*tilt, sinf(theta)*tilt, GA_e.z);
784 
785  // step 6
786  error = GA_b[besti] % GA_e2;
787  error.z = earth_error_Z;
788 #endif // YAW_INDEPENDENT_DRIFT_CORRECTION
789 
790  // to reduce the impact of two competing yaw controllers, we
791  // reduce the impact of the gps/accelerometers on yaw when we are
792  // flat, but still allow for yaw correction using the
793  // accelerometers at high roll angles as long as we have a GPS
794  if (AP_AHRS_DCM::use_compass()) {
795  if (have_gps() && is_equal(gps_gain,1.0f)) {
796  error[besti].z *= sinf(fabsf(roll));
797  } else {
798  error[besti].z = 0;
799  }
800  }
801 
802  // if ins is unhealthy then stop attitude drift correction and
803  // hope the gyros are OK for a while. Just slowly reduce _omega_P
804  // to prevent previous bad accels from throwing us off
805  if (!_ins.healthy()) {
806  error[besti].zero();
807  } else {
808  // convert the error term to body frame
809  error[besti] = _dcm_matrix.mul_transpose(error[besti]);
810  }
811 
812  if (error[besti].is_nan() || error[besti].is_inf()) {
813  // don't allow bad values
814  check_matrix();
815  _last_failure_ms = hal.scheduler->millis();
816  return;
817  }
818 
819  _error_rp = 0.8f * _error_rp + 0.2f * best_error;
820 
821  // base the P gain on the spin rate
822  float spin_rate = _omega.length();
823 
824  // sanity check _kp value
825  if (_kp < AP_AHRS_RP_P_MIN) {
826  _kp = AP_AHRS_RP_P_MIN;
827  }
828 
829  // we now want to calculate _omega_P and _omega_I. The
830  // _omega_P value is what drags us quickly to the
831  // accelerometer reading.
832  _omega_P = error[besti] * _P_gain(spin_rate) * _kp;
833  if (use_fast_gains()) {
834  _omega_P *= 8;
835  }
836 
837  if (_flags.fly_forward && _gps.status() >= AP_GPS::GPS_OK_FIX_2D &&
838  _gps.ground_speed() < GPS_SPEED_MIN &&
839  _ins.get_accel().x >= 7 &&
840  pitch_sensor > -3000 && pitch_sensor < 3000) {
841  // assume we are in a launch acceleration, and reduce the
842  // rp gain by 50% to reduce the impact of GPS lag on
843  // takeoff attitude when using a catapult
844  _omega_P *= 0.5f;
845  }
846 
847  // accumulate some integrator error
848  if (spin_rate < ToRad(SPIN_RATE_LIMIT)) {
849  _omega_I_sum += error[besti] * _ki * _ra_deltat;
850  _omega_I_sum_time += _ra_deltat;
851  }
852 
853  if (_omega_I_sum_time >= 5) {
854  // limit the rate of change of omega_I to the hardware
855  // reported maximum gyro drift rate. This ensures that
856  // short term errors don't cause a buildup of omega_I
857  // beyond the physical limits of the device
858  float change_limit = _gyro_drift_limit * _omega_I_sum_time;
859  _omega_I_sum.x = constrain_float(_omega_I_sum.x, -change_limit, change_limit);
860  _omega_I_sum.y = constrain_float(_omega_I_sum.y, -change_limit, change_limit);
861  _omega_I_sum.z = constrain_float(_omega_I_sum.z, -change_limit, change_limit);
862  _omega_I += _omega_I_sum;
863  _omega_I_sum.zero();
864  _omega_I_sum_time = 0;
865  }
866 
867  // zero our accumulator ready for the next GPS step
868  memset(&_ra_sum[0], 0, sizeof(_ra_sum));
869  _ra_deltat = 0;
870  _ra_sum_start = last_correction_time;
871 
872  // remember the velocity for next time
873  _last_velocity = velocity;
874 }

877 // update our wind speed estimate
878 void AP_AHRS_DCM::estimate_wind(void)
879 {
880  if (!_flags.wind_estimation) {
881  return;
882  }
883  const Vector3f &velocity = _last_velocity;
884 
885  // this is based on the wind speed estimation code from MatrixPilot by
886  // Bill Premerlani. Adaption for ArduPilot by Jon Challinger
887  // See http://gentlenav.googlecode.com/files/WindEstimation.pdf
888  Vector3f fuselageDirection = _dcm_matrix.colx();
889  Vector3f fuselageDirectionDiff = fuselageDirection - _last_fuse;
890  uint32_t now = hal.scheduler->millis();
891 
892  // scrap our data and start over if we're taking too long to get a direction change
893  if (now - _last_wind_time > 10000) {
894  _last_wind_time = now;
895  _last_fuse = fuselageDirection;
896  _last_vel = velocity;
897  return;
898  }
899 
900  float diff_length = fuselageDirectionDiff.length();
901  if (diff_length > 0.2f) {
902  // when turning, use the attitude response to estimate
903  // wind speed
904  float V;
905  Vector3f velocityDiff = velocity - _last_vel;
906 
907  // estimate airspeed it using equation 6
908  V = velocityDiff.length() / diff_length;
909 
910  _last_fuse = fuselageDirection;
911  _last_vel = velocity;
912 
913  Vector3f fuselageDirectionSum = fuselageDirection + _last_fuse;
914  Vector3f velocitySum = velocity + _last_vel;
915 
916  float theta = atan2f(velocityDiff.y, velocityDiff.x) - atan2f(fuselageDirectionDiff.y, fuselageDirectionDiff.x);
917  float sintheta = sinf(theta);
918  float costheta = cosf(theta);
919 
920  Vector3f wind = Vector3f();
921  wind.x = velocitySum.x - V * (costheta * fuselageDirectionSum.x - sintheta * fuselageDirectionSum.y);
922  wind.y = velocitySum.y - V * (sintheta * fuselageDirectionSum.x + costheta * fuselageDirectionSum.y);
923  wind.z = velocitySum.z - V * fuselageDirectionSum.z;
924  wind *= 0.5f;
925 
926  if (wind.length() < _wind.length() + 20) {
927  _wind = _wind * 0.95f + wind * 0.05f;
928  }
929 
930  _last_wind_time = now;
931  } else if (now - _last_wind_time > 2000 && airspeed_sensor_enabled()) {
932  // when flying straight use airspeed to get wind estimate if available
933  Vector3f airspeed = _dcm_matrix.colx() * _airspeed->get_airspeed();
934  Vector3f wind = velocity - (airspeed * get_EAS2TAS());
935  _wind = _wind * 0.92f + wind * 0.08f;
936  }
937 }
