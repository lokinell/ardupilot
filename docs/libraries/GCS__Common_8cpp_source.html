59 GCS_MAVLINK::setup_uart(const AP_SerialManager& serial_manager, AP_SerialManager::SerialProtocol protocol, uint8_t instance)

61  // search for serial port
62 
63  AP_HAL::UARTDriver *uart;
64  uart = serial_manager.find_serial(protocol, instance);
65  if (uart == NULL) {
66  // return immediately if not found
67  return;
68  }
69 
70  // get associated mavlink channel
71  mavlink_channel_t mav_chan;
72  if (!serial_manager.get_mavlink_channel(protocol, instance, mav_chan)) {
73  // return immediately in unlikely case mavlink channel cannot be found

77  /*

79  mode because CTS was held while powering on. This tells the
80  bootloader to wait for a firmware. It affects any SiK radio with

87  uart->set_flow_control(AP_HAL::UARTDriver::FLOW_CONTROL_DISABLE);
88  for (uint8_t i=0; i<3; i++) {
89  hal.scheduler->delay(1);
90  uart->write(0x30);
91  uart->write(0x20);
92  }
93  // since tcdrain() and TCSADRAIN may not be implemented...
94  hal.scheduler->delay(1);
95 
96  uart->set_flow_control(old_flow_control);
97 
98  // now change back to desired baudrate

125 void
126 GCS_MAVLINK::queued_param_send()
127 {
128  if (!initialised || _queued_parameter == NULL) {
129  return;
130  }
131 
132  uint16_t bytes_allowed;
133  uint8_t count;
134  uint32_t tnow = hal.scheduler->millis();
135 
136  // use at most 30% of bandwidth on parameters. The constant 26 is
137  // 1/(1000 * 1/8 * 0.001 * 0.3)
138  bytes_allowed = 57 * (tnow - _queued_parameter_send_time_ms) * 26;
139  if (bytes_allowed > comm_get_txspace(chan)) {
140  bytes_allowed = comm_get_txspace(chan);
141  }
142  count = bytes_allowed / (MAVLINK_MSG_ID_PARAM_VALUE_LEN + MAVLINK_NUM_NON_PAYLOAD_BYTES);
143 
144  // when we don't have flow control we really need to keep the
145  // param download very slow, or it tends to stall
146  if (!have_flow_control() && count > 5) {
147  count = 5;
148  }
149 
150  while (_queued_parameter != NULL && count--) {
151  AP_Param *vp;
152  float value;
153 
154  // copy the current parameter and prepare to move to the next
155  vp = _queued_parameter;
156 
157  // if the parameter can be cast to float, report it here and break out of the loop
158  value = vp->cast_to_float(_queued_parameter_type);
159 
160  char param_name[AP_MAX_NAME_SIZE];
161  vp->copy_name_token(_queued_parameter_token, param_name, sizeof(param_name), true);
162 
163  mavlink_msg_param_value_send(
164  chan,
165  param_name,
166  value,
167  mav_var_type(_queued_parameter_type),
168  _queued_parameter_count,
169  _queued_parameter_index);
170 
171  _queued_parameter = AP_Param::next_scalar(&_queued_parameter_token, &_queued_parameter_type);
172  _queued_parameter_index++;
173  }
174  _queued_parameter_send_time_ms = tnow;
175 }

287  if (!AP_Mission::mission_cmd_to_mavlink(cmd, ret_packet)) {
288  goto mission_item_send_failed;
289  }

292  if (cmd.id == (uint16_t)mission.get_current_nav_cmd().index) {
293  ret_packet.current = 1;
294  } else {

307  ret_packet.target_component = msg->compid;
308  ret_packet.seq = packet.seq;
309  ret_packet.command = cmd.id;
310 
311  _mav_finalize_message_chan_send(chan,
312  MAVLINK_MSG_ID_MISSION_ITEM,

314  MAVLINK_MSG_ID_MISSION_ITEM_LEN,
315  MAVLINK_MSG_ID_MISSION_ITEM_CRC);

658 {
659  mavlink_mission_item_t packet;
660  MAV_MISSION_RESULT result = MAV_MISSION_ACCEPTED;
661  struct AP_Mission::Mission_Command cmd = {};
662  bool mission_is_complete = false;
663 
664  mavlink_msg_mission_item_decode(msg, &packet);
665 
666  // convert mavlink packet to mission command
667  if (!AP_Mission::mavlink_to_mission_cmd(packet, cmd)) {
668  result = MAV_MISSION_INVALID;
669  goto mission_ack;
670  }
671 
672  if (packet.current == 2) {
673  // current = 2 is a flag to tell us this is a "guided mode"
674  // waypoint and not for the mission
675  handle_guided_request(cmd);
676 
677  // verify we received the command
678  result = MAV_MISSION_ACCEPTED;
679  goto mission_ack;
680  }
681 
682  if (packet.current == 3) {
683  //current = 3 is a flag to tell us this is a alt change only
684  // add home alt if needed
685  handle_change_alt_request(cmd);
686 
687  // verify we recevied the command
688  result = MAV_MISSION_ACCEPTED;
689  goto mission_ack;
690  }
691 
692  // Check if receiving waypoints (mission upload expected)
693  if (!waypoint_receiving) {
694  result = MAV_MISSION_ERROR;
695  goto mission_ack;
696  }
697 
698  // check if this is the requested waypoint
699  if (packet.seq != waypoint_request_i) {
700  result = MAV_MISSION_INVALID_SEQUENCE;
701  goto mission_ack;
702  }
703 
704  // if command index is within the existing list, replace the command
705  if (packet.seq < mission.num_commands()) {
706  if (mission.replace_cmd(packet.seq,cmd)) {
707  result = MAV_MISSION_ACCEPTED;
708  }else{
709  result = MAV_MISSION_ERROR;
710  goto mission_ack;
711  }
712  // if command is at the end of command list, add the command
713  } else if (packet.seq == mission.num_commands()) {
714  if (mission.add_cmd(cmd)) {
715  result = MAV_MISSION_ACCEPTED;
716  }else{
717  result = MAV_MISSION_ERROR;
718  goto mission_ack;
719  }
720  // if beyond the end of the command list, return an error
721  } else {
722  result = MAV_MISSION_ERROR;
723  goto mission_ack;
724  }
725 
726  // update waypoint receiving state machine
727  waypoint_timelast_receive = hal.scheduler->millis();
728  waypoint_request_i++;
729 
730  if (waypoint_request_i >= waypoint_request_last) {
731  mavlink_msg_mission_ack_send_buf(
732  msg,
733  chan,
734  msg->sysid,
735  msg->compid,
736  MAV_MISSION_ACCEPTED);
737 
738  send_text(MAV_SEVERITY_INFO,"flight plan received");
739  waypoint_receiving = false;
740  mission_is_complete = true;
741  // XXX ignores waypoint radius for individual waypoints, can
742  // only set WP_RADIUS parameter
743  } else {
744  waypoint_timelast_request = hal.scheduler->millis();
745  // if we have enough space, then send the next WP immediately
746  if (comm_get_txspace(chan) >=
747  MAVLINK_NUM_NON_PAYLOAD_BYTES+MAVLINK_MSG_ID_MISSION_ITEM_LEN) {
748  queued_waypoint_send();
749  } else {
750  send_message(MSG_NEXT_WAYPOINT);
751  }
752  }
753  return mission_is_complete;
754 
755 mission_ack:
756  // we are rejecting the mission/waypoint
757  mavlink_msg_mission_ack_send_buf(
758  msg,
759  chan,
760  msg->sysid,
761  msg->compid,
762  result);
763 
764  return mission_is_complete;

778 // send a message using mavlink, handling message queueing
779 void GCS_MAVLINK::send_message(enum ap_message id)
780 {
781  uint8_t i, nextid;
782 
783  // see if we can send the deferred messages, if any
784  while (num_deferred_messages != 0) {
785  if (!try_send_message(deferred_messages[next_deferred_message])) {
786  break;
787  }
788  next_deferred_message++;
789  if (next_deferred_message == MSG_RETRY_DEFERRED) {
790  next_deferred_message = 0;
791  }
792  num_deferred_messages--;
793  }
794 
795  if (id == MSG_RETRY_DEFERRED) {
796  return;
797  }
798 
799  // this message id might already be deferred
800  for (i=0, nextid = next_deferred_message; i < num_deferred_messages; i++) {
801  if (deferred_messages[nextid] == id) {
802  // its already deferred, discard
803  return;
804  }
805  nextid++;
806  if (nextid == MSG_RETRY_DEFERRED) {
807  nextid = 0;
808  }
809  }
810 
811  if (num_deferred_messages != 0 ||
812  !try_send_message(id)) {
813  // can't send it now, so defer it
814  if (num_deferred_messages == MSG_RETRY_DEFERRED) {
815  // the defer buffer is full, discard
816  return;
817  }
818  nextid = next_deferred_message + num_deferred_messages;
819  if (nextid >= MSG_RETRY_DEFERRED) {
820  nextid -= MSG_RETRY_DEFERRED;
821  }
822  deferred_messages[nextid] = id;
823  num_deferred_messages++;
824  }
825 }
