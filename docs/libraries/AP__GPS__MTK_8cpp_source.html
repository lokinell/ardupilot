48 // Process bytes available from the stream
49 //
50 // The stream is assumed to contain only our custom message. If it
51 // contains other messages, and those messages contain the preamble bytes,
52 // it is possible for this code to become de-synchronised. Without
53 // buffering the entire message and re-processing it from the top,
54 // this is unavoidable.
55 //
56 // The lack of a standard header length field makes it impossible to skip
57 // unrecognised messages.
58 //
59 bool
60 AP_GPS_MTK::read(void)
61 {
62  uint8_t data;
63  int16_t numc;
64  bool parsed = false;
65 
66  numc = port->available();
67  for (int16_t i = 0; i < numc; i++) { // Process bytes received
68 
69  // read the next byte
70  data = port->read();
71 
72 restart:
73  switch(_step) {
74 
75  // Message preamble, class, ID detection
76  //
77  // If we fail to match any of the expected bytes, we
78  // reset the state machine and re-consider the failed
79  // byte as the first byte of the preamble. This
80  // improves our chances of recovering from a mismatch
81  // and makes it less likely that we will be fooled by
82  // the preamble appearing as data in some other message.
83  //
84  case 0:
85  if(PREAMBLE1 == data)
86  _step++;
87  break;
88  case 1:
89  if (PREAMBLE2 == data) {
90  _step++;
91  break;
92  }
93  _step = 0;
94  goto restart;
95  case 2:
96  if (MESSAGE_CLASS == data) {
97  _step++;
98  _ck_b = _ck_a = data; // reset the checksum accumulators
99  } else {
100  _step = 0; // reset and wait for a message of the right class
101  goto restart;
102  }
103  break;
104  case 3:
105  if (MESSAGE_ID == data) {
106  _step++;
107  _ck_b += (_ck_a += data);
108  _payload_counter = 0;
109  } else {
110  _step = 0;
111  goto restart;
112  }
113  break;
114 
115  // Receive message data
116  //
117  case 4:
118  _buffer.bytes[_payload_counter++] = data;
119  _ck_b += (_ck_a += data);
120  if (_payload_counter == sizeof(_buffer))
121  _step++;
122  break;
123 
124  // Checksum and message processing
125  //
126  case 5:
127  _step++;
128  if (_ck_a != data) {
129  _step = 0;
130  }
131  break;
132  case 6:
133  _step = 0;
134  if (_ck_b != data) {
135  break;
136  }
137 
138  // set fix type
139  if (_buffer.msg.fix_type == FIX_3D) {
140  state.status = AP_GPS::GPS_OK_FIX_3D;
141  }else if (_buffer.msg.fix_type == FIX_2D) {
142  state.status = AP_GPS::GPS_OK_FIX_2D;
143  }else{
144  state.status = AP_GPS::NO_FIX;
145  }
146  state.location.lat = swap_int32(_buffer.msg.latitude) * 10;
147  state.location.lng = swap_int32(_buffer.msg.longitude) * 10;
148  state.location.alt = swap_int32(_buffer.msg.altitude);
149  state.ground_speed = swap_int32(_buffer.msg.ground_speed) * 0.01f;
150  state.ground_course_cd = wrap_360_cd(swap_int32(_buffer.msg.ground_course) / 10000);
151  state.num_sats = _buffer.msg.satellites;
152 
153  if (state.status >= AP_GPS::GPS_OK_FIX_2D) {
154  make_gps_time(0, swap_int32(_buffer.msg.utc_time)*10);
155  }
156  // we don't change _last_gps_time as we don't know the
157  // full date
158 
159  fill_3d_velocity();
160 
161  parsed = true;
162  }
163  }
164  return parsed;
165 }

168  detect a MTK GPS

170 bool
171 AP_GPS_MTK::_detect(struct MTK_detect_state &state, uint8_t data)
172 {
173  switch (state.step) {
174  case 1:
175  if (PREAMBLE2 == data) {
176  state.step++;
177  break;
178  }
179  state.step = 0;
180  case 0:
181  state.ck_b = state.ck_a = state.payload_counter = 0;
182  if(PREAMBLE1 == data)
183  state.step++;
184  break;
185  case 2:
186  if (MESSAGE_CLASS == data) {
187  state.step++;
188  state.ck_b = state.ck_a = data;
189  } else {
190  state.step = 0;
191  }
192  break;
193  case 3:
194  if (MESSAGE_ID == data) {
195  state.step++;
196  state.ck_b += (state.ck_a += data);
197  state.payload_counter = 0;
198  } else {
199  state.step = 0;
200  }
201  break;
202  case 4:
203  state.ck_b += (state.ck_a += data);
204  if (++state.payload_counter == sizeof(struct diyd_mtk_msg))
205  state.step++;
206  break;
207  case 5:
208  state.step++;
209  if (state.ck_a != data) {
210  state.step = 0;
211  }
212  break;
213  case 6:
214  state.step = 0;
215  if (state.ck_b == data) {
216  return true;
217  }
218  }
219  return false;
