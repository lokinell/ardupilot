144 // Process bytes available from the stream
145 //
146 // The stream is assumed to contain only messages we recognise. If it
147 // contains other messages, and those messages contain the preamble
148 // bytes, it is possible for this code to fail to synchronise to the
149 // stream immediately. Without buffering the entire message and
150 // re-processing it from the top, this is unavoidable. The parser
151 // attempts to avoid this when possible.
152 //
153 bool
154 AP_GPS_UBLOX::read(void)
155 {
156  uint8_t data;
157  int16_t numc;
158  bool parsed = false;
159  uint32_t millis_now = hal.scheduler->millis();
160 
161  if (need_rate_update) {
162  send_next_rate_update();
163  }else if(!_cfg_saved && gps._save_config && _num_cfg_save_tries < 5 && (millis_now - _last_cfg_sent_time) > 5000) { //save the configuration sent until now
164  _last_cfg_sent_time = millis_now;
165  _save_cfg();
166  _num_cfg_save_tries++;
167  }
168 
169  numc = port->available();
170  for (int16_t i = 0; i < numc; i++) { // Process bytes received
171 
172  // read the next byte
173  data = port->read();
174 
175  reset:
176  switch(_step) {
177 
178  // Message preamble detection
179  //
180  // If we fail to match any of the expected bytes, we reset
181  // the state machine and re-consider the failed byte as
182  // the first byte of the preamble. This improves our
183  // chances of recovering from a mismatch and makes it less
184  // likely that we will be fooled by the preamble appearing
185  // as data in some other message.
186  //
187  case 1:
188  if (PREAMBLE2 == data) {
189  _step++;
190  break;
191  }
192  _step = 0;
193  Debug("reset %u", __LINE__);
194  /* no break */
195  case 0:
196  if(PREAMBLE1 == data)
197  _step++;
198  break;
199 
200  // Message header processing
201  //
202  // We sniff the class and message ID to decide whether we
203  // are going to gather the message bytes or just discard
204  // them.
205  //
206  // We always collect the length so that we can avoid being
207  // fooled by preamble bytes in messages.
208  //
209  case 2:
210  _step++;
211  _class = data;
212  _ck_b = _ck_a = data; // reset the checksum accumulators
213  break;
214  case 3:
215  _step++;
216  _ck_b += (_ck_a += data); // checksum byte
217  _msg_id = data;
218  break;
219  case 4:
220  _step++;
221  _ck_b += (_ck_a += data); // checksum byte
222  _payload_length = data; // payload length low byte
223  break;
224  case 5:
225  _step++;
226  _ck_b += (_ck_a += data); // checksum byte
227 
228  _payload_length += (uint16_t)(data<<8);
229  if (_payload_length > sizeof(_buffer)) {
230  Debug("large payload %u", (unsigned)_payload_length);
231  // assume any payload bigger then what we know about is noise
232  _payload_length = 0;
233  _step = 0;
234  goto reset;
235  }
236  _payload_counter = 0; // prepare to receive payload
237  break;
238 
239  // Receive message data
240  //
241  case 6:
242  _ck_b += (_ck_a += data); // checksum byte
243  if (_payload_counter < sizeof(_buffer)) {
244  _buffer.bytes[_payload_counter] = data;
245  }
246  if (++_payload_counter == _payload_length)
247  _step++;
248  break;
249 
250  // Checksum and message processing
251  //
252  case 7:
253  _step++;
254  if (_ck_a != data) {
255  Debug("bad cka %x should be %x", data, _ck_a);
256  _step = 0;
257  goto reset;
258  }
259  break;
260  case 8:
261  _step = 0;
262  if (_ck_b != data) {
263  Debug("bad ckb %x should be %x", data, _ck_b);
264  break; // bad checksum
265  }
266 
267  if (_parse_gps()) {
268  parsed = true;
269  }
270  break;
271  }
272  }
273  return parsed;
274 }

413  if (_class == CLASS_CFG && _msg_id == MSG_CFG_NAV_SETTINGS) {

415  (unsigned)_buffer.nav_settings.dynModel,

425  _buffer.nav_settings.dynModel = gps._navfilter;
426  _buffer.nav_settings.mask |= 1;
427  }
428  if (gps._min_elevation != -100 &&
429  _buffer.nav_settings.minElev != gps._min_elevation) {

435  _send_message(CLASS_CFG, MSG_CFG_NAV_SETTINGS,
436  &_buffer.nav_settings,
437  sizeof(_buffer.nav_settings));
438  _cfg_saved = false; //save configuration
439  }
440  return false;
441  }
442 
443 #if UBLOX_GNSS_SETTINGS

445  uint8_t gnssCount = 0;

487 #endif

817 /*
818  detect a Ublox GPS. Adds one byte, and returns true if the stream
819  matches a UBlox
820  */
821 bool
822 AP_GPS_UBLOX::_detect(struct UBLOX_detect_state &state, uint8_t data)
823 {
824 reset:
825  switch (state.step) {
826  case 1:
827  if (PREAMBLE2 == data) {
828  state.step++;
829  break;
830  }
831  state.step = 0;
832  /* no break */
833  case 0:
834  if (PREAMBLE1 == data)
835  state.step++;
836  break;
837  case 2:
838  state.step++;
839  state.ck_b = state.ck_a = data;
840  break;
841  case 3:
842  state.step++;
843  state.ck_b += (state.ck_a += data);
844  break;
845  case 4:
846  state.step++;
847  state.ck_b += (state.ck_a += data);
848  state.payload_length = data;
849  break;
850  case 5:
851  state.step++;
852  state.ck_b += (state.ck_a += data);
853  state.payload_counter = 0;
854  break;
855  case 6:
856  state.ck_b += (state.ck_a += data);
857  if (++state.payload_counter == state.payload_length)
858  state.step++;
859  break;
860  case 7:
861  state.step++;
862  if (state.ck_a != data) {
863  state.step = 0;
864  goto reset;
865  }
866  break;
867  case 8:
868  state.step = 0;
869  if (state.ck_b == data) {
870  // a valid UBlox packet
871  return true;
872  } else {
873  goto reset;
874  }
875  }
876  return false;
877 }
