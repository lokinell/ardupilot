122 void AP_MotorsCoax::output_armed_not_stabilizing()
123 {
124  int16_t throttle_radio_output; // total throttle pwm value, summed onto throttle channel minimum, typically ~1100-1900
125  int16_t out_min = _throttle_radio_min + _min_throttle;
126  int16_t motor_out;
127 
128  int16_t min_thr = rel_pwm_to_thr_range(_spin_when_armed_ramped);
129 
130  // initialize limits flags
131  limit.roll_pitch = true;
132  limit.yaw = true;
133  limit.throttle_lower = false;
134  limit.throttle_upper = false;
135 
136  if (_throttle_control_input <= min_thr) {
137  _throttle_control_input = min_thr;
138  limit.throttle_lower = true;
139  }
140  if (_throttle_control_input >= _max_throttle) {
141  _throttle_control_input = _max_throttle;
142  limit.throttle_upper = true;
143  }
144 
145  throttle_radio_output = calc_throttle_radio_output();
146 
147  motor_out = throttle_radio_output;
148 
149  _servo1.servo_out = 0;
150  _servo1.calc_pwm();
151 
152  _servo2.servo_out = 0;
153  _servo2.calc_pwm();
154 
155  if (motor_out >= out_min) {
156  motor_out = apply_thrust_curve_and_volt_scaling(motor_out, out_min, _throttle_radio_max);
157  }
158 
159  hal.rcout->cork();
160  hal.rcout->write(AP_MOTORS_MOT_1, _servo1.radio_out);
161  hal.rcout->write(AP_MOTORS_MOT_2, _servo2.radio_out);
162  hal.rcout->write(AP_MOTORS_MOT_3, motor_out);
163  hal.rcout->write(AP_MOTORS_MOT_4, motor_out);
164  hal.rcout->push();

167 // sends commands to the motors
168 // TODO pull code that is common to output_armed_not_stabilizing into helper functions
169 void AP_MotorsCoax::output_armed_stabilizing()
170 {
171  int16_t yaw_pwm; // yaw pwm value, initially calculated by calc_yaw_pwm() but may be modified after, +/- 400
172  int16_t throttle_radio_output; // total throttle pwm value, summed onto throttle channel minimum, typically ~1100-1900
173  int16_t out_min = _throttle_radio_min + _min_throttle;
174  int16_t motor_out[4];
175 
176  // initialize limits flags
177  limit.roll_pitch = false;
178  limit.yaw = false;
179  limit.throttle_lower = false;
180  limit.throttle_upper = false;
181 
182  int16_t thr_in_min = rel_pwm_to_thr_range(_min_throttle);
183  if (_throttle_control_input <= thr_in_min) {
184  _throttle_control_input = thr_in_min;
185  limit.throttle_lower = true;
186  }
187  if (_throttle_control_input >= _max_throttle) {
188  _throttle_control_input = _max_throttle;
189  limit.throttle_upper = true;
190  }
191 
192  // calculate throttle and yaw PWM
193  throttle_radio_output = calc_throttle_radio_output();
194  yaw_pwm = calc_yaw_pwm();
195 
196  // motors
197  motor_out[AP_MOTORS_MOT_3] = _rev_yaw*yaw_pwm + throttle_radio_output;
198  motor_out[AP_MOTORS_MOT_4] = -_rev_yaw*yaw_pwm + throttle_radio_output;
199 
200  // TODO: set limits.roll_pitch and limits.yaw
201 
202  // front
203  _servo1.servo_out = _rev_roll*_roll_control_input;
204  // right
205  _servo2.servo_out = _rev_pitch*_pitch_control_input;
206 
207  _servo1.calc_pwm();
208  _servo2.calc_pwm();
209 
210  // adjust for thrust curve and voltage scaling
211  motor_out[AP_MOTORS_MOT_3] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_3], out_min, _throttle_radio_max);
212  motor_out[AP_MOTORS_MOT_4] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_4], out_min, _throttle_radio_max);
213 
214  // ensure motors don't drop below a minimum value and stop
215  motor_out[AP_MOTORS_MOT_3] = max(motor_out[AP_MOTORS_MOT_3], out_min);
216  motor_out[AP_MOTORS_MOT_4] = max(motor_out[AP_MOTORS_MOT_4], out_min);
217 
218  // send output to each motor
219  hal.rcout->cork();
220  hal.rcout->write(AP_MOTORS_MOT_1, _servo1.radio_out);
221  hal.rcout->write(AP_MOTORS_MOT_2, _servo2.radio_out);
222  hal.rcout->write(AP_MOTORS_MOT_3, motor_out[AP_MOTORS_MOT_3]);
223  hal.rcout->write(AP_MOTORS_MOT_4, motor_out[AP_MOTORS_MOT_4]);
224  hal.rcout->push();
225 }

234 // output_test - spin a motor at the pwm value specified
235 // motor_seq is the motor's sequence number from 1 to the number of motors on the frame
236 // pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
237 void AP_MotorsCoax::output_test(uint8_t motor_seq, int16_t pwm)
238 {
239  // exit immediately if not armed
240  if (!armed()) {
241  return;
242  }
243 
244  // output to motors and servos
245  switch (motor_seq) {
246  case 1:
247  // flap servo 1
248  hal.rcout->write(AP_MOTORS_MOT_1, pwm);
249  break;
250  case 2:
251  // flap servo 2
252  hal.rcout->write(AP_MOTORS_MOT_2, pwm);
253  break;
254  case 3:
255  // motor 1
256  hal.rcout->write(AP_MOTORS_MOT_3, pwm);
257  break;
258  case 4:
259  // motor 2
260  hal.rcout->write(AP_MOTORS_MOT_4, pwm);
261  break;
262  default:
263  // do nothing
264  break;
265  }
