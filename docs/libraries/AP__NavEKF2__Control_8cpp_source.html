46 void NavEKF2_core::setWindMagStateLearningMode()
47 {
48  // If we are on ground, or in constant position mode, or don't have the right vehicle and sensing to estimate wind, inhibit wind states
49  inhibitWindStates = ((!useAirspeed() && !assume_zero_sideslip()) || onGround || (PV_AidingMode == AID_NONE));
50 
51  // determine if the vehicle is manoevring
52  if (accNavMagHoriz > 0.5f) {
53  manoeuvring = true;
54  } else {
55  manoeuvring = false;
56  }
57 
58  // Determine if learning of magnetic field states has been requested by the user
59  bool magCalRequested =
60  ((frontend->_magCal == 0) && inFlight) || // when flying
61  ((frontend->_magCal == 1) && manoeuvring) || // when manoeuvring
62  ((frontend->_magCal == 3) && firstMagYawInit) || // when initial in-air yaw and field reset has completed
63  (frontend->_magCal == 4); // all the time
64 
65  // Deny mag calibration request if we aren't using the compass, it has been inhibited by the user,
66  // we do not have an absolute position reference or are on the ground (unless explicitly requested by the user)
67  // If we do nto have absolute position (eg GPS) then the earth field states cannot be learned
68  bool magCalDenied = !use_compass() || (frontend->_magCal == 2) || (PV_AidingMode == AID_NONE) || (onGround && frontend->_magCal != 4);
69 
70  // Inhibit the magnetic field calibration if not requested or denied
71  inhibitMagStates = (!magCalRequested || magCalDenied);
72 
73  // If on ground we clear the flag indicating that the magnetic field in-flight initialisation has been completed
74  // because we want it re-done for each takeoff
75  if (onGround) {
76  firstMagYawInit = false;
77  }
78 
79  // Adjust the indexing limits used to address the covariance, states and other EKF arrays to avoid unnecessary operations
80  // if we are not using those states
81  if (inhibitMagStates && inhibitWindStates) {
82  stateIndexLim = 15;
83  } else if (inhibitWindStates) {
84  stateIndexLim = 21;
85  } else {
86  stateIndexLim = 23;
87  }
88 }

90 // Set inertial navigation aiding mode
91 void NavEKF2_core::setAidingMode()
92 {
93  // Determine when to commence aiding for inertial navigation
94  // Save the previous status so we can detect when it has changed
95  prevIsAiding = isAiding;
96  // Don't allow filter to start position or velocity aiding until the tilt and yaw alignment is complete
97  bool filterIsStable = tiltAlignComplete && yawAlignComplete;
98  // If GPS useage has been prohiited then we use flow aiding provided optical flow data is present
99  bool useFlowAiding = (frontend->_fusionModeGPS == 3) && optFlowDataPresent();
100  // Start aiding if we have a source of aiding data and the filter attitude algnment is complete
101  // Latch to on
102  isAiding = ((readyToUseGPS() || useFlowAiding) && filterIsStable) || isAiding;
103 
104  // check to see if we are starting or stopping aiding and set states and modes as required
105  if (isAiding != prevIsAiding) {
106  // We have transitioned either into or out of aiding
107  // zero stored velocities used to do dead-reckoning
108  heldVelNE.zero();
109  // set various useage modes based on the condition when we start aiding. These are then held until aiding is stopped.
110  if (!isAiding) {
111  // We have ceased aiding
112  // When not aiding, estimate orientation & height fusing synthetic constant position and zero velocity measurement to constrain tilt errors
113  PV_AidingMode = AID_NONE;
114  posTimeout = true;
115  velTimeout = true;
116  // store the current position to be used to keep reporting the last known position
117  lastKnownPositionNE.x = stateStruct.position.x;
118  lastKnownPositionNE.y = stateStruct.position.y;
119  // initialise filtered altitude used to provide a takeoff reference to current baro on disarm
120  // this reduces the time required for the baro noise filter to settle before the filtered baro data can be used
121  meaHgtAtTakeOff = baroDataDelayed.hgt;
122  // reset the vertical position state to faster recover from baro errors experienced during touchdown
123  stateStruct.position.z = -meaHgtAtTakeOff;
124  } else if (frontend->_fusionModeGPS == 3) {
125  // We have commenced aiding, but GPS useage has been prohibited so use optical flow only
126  hal.console->printf("EKF2 IMU%u is using optical flow\n",(unsigned)imu_index);
127  PV_AidingMode = AID_RELATIVE; // we have optical flow data and can estimate all vehicle states
128  posTimeout = true;
129  velTimeout = true;
130  // Reset the last valid flow measurement time
131  flowValidMeaTime_ms = imuSampleTime_ms;
132  // Reset the last valid flow fusion time
133  prevFlowFuseTime_ms = imuSampleTime_ms;
134  } else {
135  // We have commenced aiding and GPS useage is allowed
136  hal.console->printf("EKF2 IMU%u is using GPS\n",(unsigned)imu_index);
137  PV_AidingMode = AID_ABSOLUTE; // we have GPS data and can estimate all vehicle states
138  posTimeout = false;
139  velTimeout = false;
140  // we need to reset the GPS timers to prevent GPS timeout logic being invoked on entry into GPS aiding
141  // this is becasue the EKF can be interrupted for an arbitrary amount of time during vehicle arming checks
142  lastTimeGpsReceived_ms = imuSampleTime_ms;
143  secondLastGpsTime_ms = imuSampleTime_ms;
144  // reset the last valid position fix time to prevent unwanted activation of GPS glitch logic
145  lastPosPassTime_ms = imuSampleTime_ms;
146  }
147  // Reset all position, velocity and covariance
148  ResetVelocity();
149  ResetPosition();
150  CovarianceInit();
151 
152  }
153 
154  // Always turn aiding off when the vehicle is disarmed
155  if (!isAiding) {
156  PV_AidingMode = AID_NONE;
157  posTimeout = true;
158  velTimeout = true;
159  }
160 }

162 // Check the alignmnent status of the tilt and yaw attitude
163 // Used during initial bootstrap alignment of the filter
164 void NavEKF2_core::checkAttitudeAlignmentStatus()
165 {
166  // Check for tilt convergence - used during initial alignment
167  float alpha = 1.0f*imuDataDelayed.delAngDT;
168  float temp=tiltErrVec.length();
169  tiltErrFilt = alpha*temp + (1.0f-alpha)*tiltErrFilt;
170  if (tiltErrFilt < 0.005f && !tiltAlignComplete) {
171  tiltAlignComplete = true;
172  hal.console->printf("EKF2 IMU%u tilt alignment complete\n",(unsigned)imu_index);
173  }
174 
175  // Once tilt has converged, align yaw using magnetic field measurements
176  if (tiltAlignComplete && !yawAlignComplete) {
177  Vector3f eulerAngles;
178  stateStruct.quat.to_euler(eulerAngles.x, eulerAngles.y, eulerAngles.z);
179  stateStruct.quat = calcQuatAndFieldStates(eulerAngles.x, eulerAngles.y);
180  StoreQuatReset();
181  yawAlignComplete = true;
182  hal.console->printf("EKF2 IMU%u yaw alignment complete\n",(unsigned)imu_index);
183  }
184 }
