53 void GCS_MAVLINK::lock_channel(mavlink_channel_t _chan, bool lock)
54 {
55  if (_chan >= MAVLINK_COMM_NUM_BUFFERS) {
56  return;
57  }
58  if (lock) {
59  mavlink_locked_mask |= (1U<<(unsigned)_chan);
60  } else {

101 uint16_t comm_get_txspace(mavlink_channel_t chan)
102 {
103  // sanity check chan
104  if (chan >= MAVLINK_COMM_NUM_BUFFERS) {
105  return 0;
106  }
107  if ((1U<<chan) & mavlink_locked_mask) {
108  return 0;
109  }
110  int16_t ret = mavlink_comm_port[chan]->txspace();
111  if (ret < 0) {
112  ret = 0;
113  }
114  return (uint16_t)ret;
