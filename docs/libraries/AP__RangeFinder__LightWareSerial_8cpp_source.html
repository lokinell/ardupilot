50 // read - return last value measured by sensor
51 bool AP_RangeFinder_LightWareSerial::get_reading(uint16_t &reading_cm)
52 {
53  if (uart == nullptr) {
54  return false;
55  }
56 
57  // read any available lines from the lidar
58  float sum = 0;
59  uint16_t count = 0;
60  int16_t nbytes = uart->available();
61  while (nbytes-- > 0) {
62  char c = uart->read();
63  if (c == '\r') {
64  linebuf[linebuf_len] = 0;
65  sum += atof(linebuf);
66  count++;
67  linebuf_len = 0;
68  } else if (isdigit(c) || c == '.') {
69  linebuf[linebuf_len++] = c;
70  if (linebuf_len == sizeof(linebuf)) {
71  // too long, discard the line
72  linebuf_len = 0;
73  }
74  }
75  }
76 
77  // we need to write a byte to prompt another reading
78  uart->write('\n');
79 
80  if (count == 0) {
81  return false;
82  }
83  reading_cm = 100 * sum / count;
84  return true;
85 }
