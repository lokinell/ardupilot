91 void AP_AHRS_NavEKF::update_EKF1(void)
92 {
93  if (!ekf1_started) {
94  // wait 1 second for DCM to output a valid tilt error estimate
95  if (start_time_ms == 0) {
96  start_time_ms = hal.scheduler->millis();
97  }
98  if (hal.scheduler->millis() - start_time_ms > startup_delay_ms) {
99  ekf1_started = EKF1.InitialiseFilterDynamic();
100  }
101  }
102  if (ekf1_started) {
103  EKF1.UpdateFilter();
104  EKF1.getRotationBodyToNED(_dcm_matrix);
105  if (active_EKF_type() == EKF_TYPE1) {
106  Vector3f eulers;
107  EKF1.getEulerAngles(eulers);
108  roll = eulers.x;
109  pitch = eulers.y;
110  yaw = eulers.z;
111 
112  update_cd_values();
113  update_trig();
114 
115  // keep _gyro_bias for get_gyro_drift()
116  EKF1.getGyroBias(_gyro_bias);
117  _gyro_bias = -_gyro_bias;
118 
119  // calculate corrected gryo estimate for get_gyro()
120  _gyro_estimate.zero();
121  uint8_t healthy_count = 0;
122  for (uint8_t i=0; i<_ins.get_gyro_count(); i++) {
123  if (_ins.get_gyro_health(i) && healthy_count < 2) {
124  _gyro_estimate += _ins.get_gyro(i);
125  healthy_count++;
126  }
127  }
128  if (healthy_count > 1) {
129  _gyro_estimate /= healthy_count;
130  }
131  _gyro_estimate += _gyro_bias;
132 
133  float abias1, abias2;
134  EKF1.getAccelZBias(abias1, abias2);
135 
136  // update _accel_ef_ekf
137  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
138  Vector3f accel = _ins.get_accel(i);
139  if (i==0) {
140  accel.z -= abias1;
141  } else if (i==1) {
142  accel.z -= abias2;
143  }
144  if (_ins.get_accel_health(i)) {
145  _accel_ef_ekf[i] = _dcm_matrix * accel;
146  }
147  }
148 
149  if(_ins.use_accel(0) && _ins.use_accel(1)) {
150  float IMU1_weighting;
151  EKF1.getIMU1Weighting(IMU1_weighting);
152  _accel_ef_ekf_blended = _accel_ef_ekf[0] * IMU1_weighting + _accel_ef_ekf[1] * (1.0f-IMU1_weighting);
153  } else {
154  _accel_ef_ekf_blended = _accel_ef_ekf[_ins.get_primary_accel()];
155  }
156  }
157  }
158 }

161 void AP_AHRS_NavEKF::update_EKF2(void)
162 {
163  if (!ekf2_started) {
164  // wait 1 second for DCM to output a valid tilt error estimate
165  if (start_time_ms == 0) {
166  start_time_ms = hal.scheduler->millis();
167  }
168  if (hal.scheduler->millis() - start_time_ms > startup_delay_ms) {
169  ekf2_started = EKF2.InitialiseFilter();
170  }
171  }
172  if (ekf2_started) {
173  EKF2.UpdateFilter();
174  EKF2.getRotationBodyToNED(_dcm_matrix);
175  if (active_EKF_type() == EKF_TYPE2) {
176  Vector3f eulers;
177  EKF2.getEulerAngles(-1,eulers);
178  roll = eulers.x;
179  pitch = eulers.y;
180  yaw = eulers.z;
181 
182  update_cd_values();
183  update_trig();
184 
185  // keep _gyro_bias for get_gyro_drift()
186  EKF2.getGyroBias(-1,_gyro_bias);
187  _gyro_bias = -_gyro_bias;
188 
189  // calculate corrected gryo estimate for get_gyro()
190  _gyro_estimate.zero();
191  uint8_t healthy_count = 0;
192  for (uint8_t i=0; i<_ins.get_gyro_count(); i++) {
193  if (_ins.get_gyro_health(i) && healthy_count < 2) {
194  _gyro_estimate += _ins.get_gyro(i);
195  healthy_count++;
196  }
197  }
198  if (healthy_count > 1) {
199  _gyro_estimate /= healthy_count;
200  }
201  _gyro_estimate += _gyro_bias;
202 
203  float abias;
204  EKF2.getAccelZBias(-1,abias);
205 
206  // This EKF uses the primary IMU
207  // Eventually we will run a separate instance of the EKF for each IMU and do the selection and blending of EKF outputs upstream
208  // update _accel_ef_ekf
209  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
210  Vector3f accel = _ins.get_accel(i);
211  if (i==_ins.get_primary_accel()) {
212  accel.z -= abias;
213  }
214  if (_ins.get_accel_health(i)) {
215  _accel_ef_ekf[i] = _dcm_matrix * accel;
216  }
217  }
218  _accel_ef_ekf_blended = _accel_ef_ekf[_ins.get_primary_accel()];
219  }
220  }

502 AP_AHRS_NavEKF::EKF_TYPE AP_AHRS_NavEKF::active_EKF_type(void) const
503 {
504  EKF_TYPE ret = EKF_TYPE_NONE;
505 
506  switch (ekf_type()) {
507  case 0:
508  return EKF_TYPE_NONE;
509 
510  case 1: {
511  // do we have an EKF yet?
512  if (!ekf1_started) {
513  return EKF_TYPE_NONE;
514  }
515  if (always_use_EKF()) {
516  uint8_t ekf_faults;
517  EKF1.getFilterFaults(ekf_faults);
518  if (ekf_faults == 0) {
519  ret = EKF_TYPE1;
520  }
521  } else if (EKF1.healthy()) {
522  ret = EKF_TYPE1;
523  }
524  break;
525  }
526 
527  case 2: {
528  // do we have an EKF2 yet?
529  if (!ekf2_started) {
530  return EKF_TYPE_NONE;
531  }
532  if (always_use_EKF()) {
533  uint8_t ekf2_faults;
534  EKF2.getFilterFaults(-1,ekf2_faults);
535  if (ekf2_faults == 0) {
536  ret = EKF_TYPE2;
537  }
538  } else if (EKF2.healthy()) {
539  ret = EKF_TYPE2;
540  }
541  break;
542  }
543  }
544 
545  if (ret != EKF_TYPE_NONE &&
546  (_vehicle_class == AHRS_VEHICLE_FIXED_WING ||
547  _vehicle_class == AHRS_VEHICLE_GROUND)) {
548  nav_filter_status filt_state;
549  if (ret == EKF_TYPE1) {
550  EKF1.getFilterStatus(filt_state);
551  } else {
552  EKF2.getFilterStatus(-1,filt_state);
553  }
554  if (hal.util->get_soft_armed() && !filt_state.flags.using_gps && _gps.status() >= AP_GPS::GPS_OK_FIX_3D) {
555  // if the EKF is not fusing GPS and we have a 3D lock, then
556  // plane and rover would prefer to use the GPS position from
557  // DCM. This is a safety net while some issues with the EKF
558  // get sorted out
559  return EKF_TYPE_NONE;
560  }
561  if (hal.util->get_soft_armed() && filt_state.flags.const_pos_mode) {
562  return EKF_TYPE_NONE;
563  }
564  if (!filt_state.flags.attitude ||
565  !filt_state.flags.horiz_vel ||
566  !filt_state.flags.vert_vel ||
567  !filt_state.flags.horiz_pos_abs ||
568  !filt_state.flags.vert_pos) {
569  return EKF_TYPE_NONE;
570  }
571  }
572  return ret;
573 }

575 /*
576  check if the AHRS subsystem is healthy
577 */
578 bool AP_AHRS_NavEKF::healthy(void) const
579 {
580  // If EKF is started we switch away if it reports unhealthy. This could be due to bad
581  // sensor data. If EKF reversion is inhibited, we only switch across if the EKF encounters
582  // an internal processing error, but not for bad sensor data.
583  switch (ekf_type()) {
584  case 0:
585  return AP_AHRS_DCM::healthy();
586 
587  case 1: {
588  bool ret = ekf1_started && EKF1.healthy();
589  if (!ret) {
590  return false;
591  }
592  if ((_vehicle_class == AHRS_VEHICLE_FIXED_WING ||
593  _vehicle_class == AHRS_VEHICLE_GROUND) &&
594  active_EKF_type() != EKF_TYPE1) {
595  // on fixed wing we want to be using EKF to be considered
596  // healthy if EKF is enabled
597  return false;
598  }
599  return true;
600  }
601 
602  case 2: {
603  bool ret = ekf2_started && EKF2.healthy();
604  if (!ret) {
605  return false;
606  }
607  if ((_vehicle_class == AHRS_VEHICLE_FIXED_WING ||
608  _vehicle_class == AHRS_VEHICLE_GROUND) &&
609  active_EKF_type() != EKF_TYPE2) {
610  // on fixed wing we want to be using EKF to be considered
611  // healthy if EKF is enabled
612  return false;
613  }
614  return true;
615  }
616  }
617 
618  return AP_AHRS_DCM::healthy();
619 }
