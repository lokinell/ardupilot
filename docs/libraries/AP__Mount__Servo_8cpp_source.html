27 // update mount position - should be called periodically
28 void AP_Mount_Servo::update()
29 {
30  static bool mount_open = 0; // 0 is closed
31 
32  // check servo map every three seconds to allow users to modify parameters
33  uint32_t now = hal.scheduler->millis();
34  if (now - _last_check_servo_map_ms > 3000) {
35  check_servo_map();
36  _last_check_servo_map_ms = now;
37  }
38 
39  switch(get_mode()) {
40  // move mount to a "retracted position" or to a position where a fourth servo can retract the entire mount into the fuselage
41  case MAV_MOUNT_MODE_RETRACT:
42  {
43  _angle_bf_output_deg = _state._retract_angles.get();
44  break;
45  }
46 
47  // move mount to a neutral position, typically pointing forward
48  case MAV_MOUNT_MODE_NEUTRAL:
49  {
50  _angle_bf_output_deg = _state._neutral_angles.get();
51  break;
52  }
53 
54  // point to the angles given by a mavlink message
55  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
56  {
57  // earth-frame angle targets (i.e. _angle_ef_target_rad) should have already been set by a MOUNT_CONTROL message from GCS
58  stabilize();
59  break;
60  }
61 
62  // RC radio manual angle control, but with stabilization from the AHRS
63  case MAV_MOUNT_MODE_RC_TARGETING:
64  {
65  // update targets using pilot's rc inputs
66  update_targets_from_rc();
67  stabilize();
68  break;
69  }
70 
71  // point mount to a GPS point given by the mission planner
72  case MAV_MOUNT_MODE_GPS_POINT:
73  {
74  if(_frontend._ahrs.get_gps().status() >= AP_GPS::GPS_OK_FIX_2D) {
75  calc_angle_to_location(_state._roi_target, _angle_ef_target_rad, _flags.tilt_control, _flags.pan_control);
76  stabilize();
77  }
78  break;
79  }
80 
81  default:
82  //do nothing
83  break;
84  }
85 
86  // move mount to a "retracted position" into the fuselage with a fourth servo
87  bool mount_open_new = (get_mode() == MAV_MOUNT_MODE_RETRACT) ? 0 : 1;
88  if (mount_open != mount_open_new) {
89  mount_open = mount_open_new;
90  move_servo(_open_idx, mount_open_new, 0, 1);
91  }
92 
93  // write the results to the servos
94  move_servo(_roll_idx, _angle_bf_output_deg.x*10, _state._roll_angle_min*0.1f, _state._roll_angle_max*0.1f);
95  move_servo(_tilt_idx, _angle_bf_output_deg.y*10, _state._tilt_angle_min*0.1f, _state._tilt_angle_max*0.1f);
96  move_servo(_pan_idx, _angle_bf_output_deg.z*10, _state._pan_angle_min*0.1f, _state._pan_angle_max*0.1f);
97 }

123 // stabilize - stabilizes the mount relative to the Earth's frame
124 // input: _angle_ef_target_rad (earth frame targets in radians)
125 // output: _angle_bf_output_deg (body frame angles in degrees)
126 void AP_Mount_Servo::stabilize()
127 {
128  // only do the full 3D frame transform if we are doing pan control
129  if (_state._stab_pan) {
130  Matrix3f m;
131  Matrix3f cam;
132  Matrix3f gimbal_target;
133  m = _frontend._ahrs.get_dcm_matrix();
134  m.transpose();
135  cam.from_euler(_angle_ef_target_rad.x, _angle_ef_target_rad.y, _angle_ef_target_rad.z);
136  gimbal_target = m * cam;
137  gimbal_target.to_euler(&_angle_bf_output_deg.x, &_angle_bf_output_deg.y, &_angle_bf_output_deg.z);
138  _angle_bf_output_deg.x = _state._stab_roll ? degrees(_angle_bf_output_deg.x) : degrees(_angle_ef_target_rad.x);
139  _angle_bf_output_deg.y = _state._stab_tilt ? degrees(_angle_bf_output_deg.y) : degrees(_angle_ef_target_rad.y);
140  _angle_bf_output_deg.z = degrees(_angle_bf_output_deg.z);
141  } else {
142  // otherwise base mount roll and tilt on the ahrs
143  // roll/tilt attitude, plus any requested angle
144  _angle_bf_output_deg.x = degrees(_angle_ef_target_rad.x);
145  _angle_bf_output_deg.y = degrees(_angle_ef_target_rad.y);
146  _angle_bf_output_deg.z = degrees(_angle_ef_target_rad.z);
147  if (_state._stab_roll) {
148  _angle_bf_output_deg.x -= degrees(_frontend._ahrs.roll);
149  }
150  if (_state._stab_tilt) {
151  _angle_bf_output_deg.y -= degrees(_frontend._ahrs.pitch);
152  }
153 
154  // lead filter
155  const Vector3f &gyro = _frontend._ahrs.get_gyro();
156 
157  if (_state._stab_roll && !is_zero(_state._roll_stb_lead) && fabsf(_frontend._ahrs.pitch) < M_PI_F/3.0f) {
158  // Compute rate of change of euler roll angle
159  float roll_rate = gyro.x + (_frontend._ahrs.sin_pitch() / _frontend._ahrs.cos_pitch()) * (gyro.y * _frontend._ahrs.sin_roll() + gyro.z * _frontend._ahrs.cos_roll());
160  _angle_bf_output_deg.x -= degrees(roll_rate) * _state._roll_stb_lead;
161  }
162 
163  if (_state._stab_tilt && !is_zero(_state._pitch_stb_lead)) {
164  // Compute rate of change of euler pitch angle
165  float pitch_rate = _frontend._ahrs.cos_pitch() * gyro.y - _frontend._ahrs.sin_roll() * gyro.z;
166  _angle_bf_output_deg.y -= degrees(pitch_rate) * _state._pitch_stb_lead;
167  }
168  }
169 }

171 // closest_limit - returns closest angle to 'angle' taking into account limits. all angles are in degrees * 10
172 int16_t AP_Mount_Servo::closest_limit(int16_t angle, int16_t angle_min, int16_t angle_max)
173 {
174  // Make sure the angle lies in the interval [-180 .. 180[ degrees
175  while (angle < -1800) angle += 3600;
176  while (angle >= 1800) angle -= 3600;
177 
178  // Make sure the angle limits lie in the interval [-180 .. 180[ degrees
179  while (angle_min < -1800) angle_min += 3600;
180  while (angle_min >= 1800) angle_min -= 3600;
181  while (angle_max < -1800) angle_max += 3600;
182  while (angle_max >= 1800) angle_max -= 3600;
183  // TODO call this function somehow, otherwise this will never work
184  //set_range(min, max);
185 
186  // If the angle is outside servo limits, saturate the angle to the closest limit
187  // On a circle the closest angular position must be carefully calculated to account for wrap-around
188  if ((angle < angle_min) && (angle > angle_max)) {
189  // angle error if min limit is used
190  int16_t err_min = angle_min - angle + (angle<angle_min ? 0 : 3600); // add 360 degrees if on the "wrong side"
191  // angle error if max limit is used
192  int16_t err_max = angle - angle_max + (angle>angle_max ? 0 : 3600); // add 360 degrees if on the "wrong side"
193  angle = err_min<err_max ? angle_min : angle_max;
194  }
195 
196  return angle;
197 }
