120 void RCOutput_PCA9685::set_freq(uint32_t chmask, uint16_t freq_hz)
121 {
122 
123  /* Correctly finish last pulses */
124  for (int i = 0; i < (PWM_CHAN_COUNT - _channel_offset); i++) {
125  write(i, _pulses_buffer[i]);
126  }
127 
128  if (!_i2c_sem->take(10)) {
129  return;
130  }
131 
132  /* Shutdown before sleeping.
133  * see p.14 of PCA9685 product datasheet
134  */
135  hal.i2c->writeRegister(_addr, PCA9685_RA_ALL_LED_OFF_H, PCA9685_ALL_LED_OFF_H_SHUT);
136 
137  /* Put PCA9685 to sleep (required to write prescaler) */
138  hal.i2c->writeRegister(_addr, PCA9685_RA_MODE1, PCA9685_MODE1_SLEEP_BIT);
139 
140  /* Calculate prescale and save frequency using this value: it may be
141  * different from @freq_hz due to rounding/ceiling. We use ceil() rather
142  * than round() so the resulting frequency is never greater than @freq_hz
143  */
144  uint8_t prescale = ceil(_osc_clock / (4096 * freq_hz)) - 1;
145  _frequency = _osc_clock / (4096 * (prescale + 1));
146 
147  /* Write prescale value to match frequency */
148  hal.i2c->writeRegister(_addr, PCA9685_RA_PRE_SCALE, prescale);
149 
150  if (_external_clock) {
151  /* Enable external clocking */
152  hal.i2c->writeRegister(_addr, PCA9685_RA_MODE1,
153  PCA9685_MODE1_SLEEP_BIT | PCA9685_MODE1_EXTCLK_BIT);
154  }
155 
156  /* Restart the device to apply new settings and enable auto-incremented write */
157  hal.i2c->writeRegister(_addr, PCA9685_RA_MODE1,
158  PCA9685_MODE1_RESTART_BIT | PCA9685_MODE1_AI_BIT);
159 
160  _i2c_sem->give();
161 }

196 void RCOutput_PCA9685::push()
197 {
198  _corking = false;
199 
200  if (_pending_write_mask == 0)
201  return;
202 
203  // Calculate the number of channels for this transfer.
204  uint8_t max_ch = (sizeof(unsigned) * 8) - __builtin_clz(_pending_write_mask);
205  uint8_t min_ch = __builtin_ctz(_pending_write_mask);
206 
207  /*
208  * scratch buffer size is always for all the channels, but we write only
209  * from min_ch to max_ch
210  */
211  uint8_t data[PWM_CHAN_COUNT * 4] = { };
212 
213  for (unsigned ch = min_ch; ch < max_ch; ch++) {
214  uint16_t period_us = _pulses_buffer[ch];
215  uint16_t length = 0;
216 
217  if (period_us)
218  length = round((period_us * 4096) / (1000000.f / _frequency)) - 1;
219 
220  uint8_t *d = &data[ch * 4];
221  *d++ = 0;
222  *d++ = 0;
223  *d++ = length & 0xFF;
224  *d++ = length >> 8;
225  }
226 
227  if (!_i2c_sem->take_nonblocking()) {
228  return;
229  }
230 
231  hal.i2c->writeRegisters(_addr,
232  PCA9685_RA_LED0_ON_L + 4 * (_channel_offset + min_ch),
233  (max_ch - min_ch) * 4,
234  &data[min_ch * 4]);
235 
236  _i2c_sem->give();
237 
238  _pending_write_mask = 0;
239 }
