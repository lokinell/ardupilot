133 void GPIO_RPI::setPWM0Period(uint32_t time_us)
134 {
135  // stop clock and waiting for busy flag doesn't work, so kill clock
136  *(clk + PWMCLK_CNTL) = 0x5A000000 | (1 << 5);
137  usleep(10);
138 
139  // set frequency
140  // DIVI is the integer part of the divisor
141  // the fractional part (DIVF) drops clock cycles to get the output frequency, bad for servo motors
142  // 320 bits for one cycle of 20 milliseconds = 62.5 us per bit = 16 kHz
143  int idiv = (int) (19200000.0f / (320000000.0f / time_us));
144  if (idiv < 1 || idiv > 0x1000) {
145  return;
146  }
147  *(clk + PWMCLK_DIV) = 0x5A000000 | (idiv<<12);
148 
149  // source=osc and enable clock
150  *(clk + PWMCLK_CNTL) = 0x5A000011;
151 
152  // disable PWM
153  *(pwm + PWM_CTL) = 0;
154 
155  // needs some time until the PWM module gets disabled, without the delay the PWM module crashs
156  usleep(10);
157 
158  // filled with 0 for 20 milliseconds = 320 bits
159  *(pwm + PWM_RNG1) = 320;
160 
161  // init with 0%
162  setPWM0Duty(0);
163 
164  // start PWM1 in serializer mode
165  *(pwm + PWM_CTL) = 3;
166 }
