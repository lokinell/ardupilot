58 /*
59  run one tick
60  this will run as many scheduler tasks as we can in the specified time
61  */
62 void AP_Scheduler::run(uint16_t time_available)
63 {
64  uint32_t run_started_usec = hal.scheduler->micros();
65  uint32_t now = run_started_usec;
66 
67  for (uint8_t i=0; i<_num_tasks; i++) {
68  uint16_t dt = _tick_counter - _last_run[i];
69  uint16_t interval_ticks = pgm_read_word(&_tasks[i].interval_ticks);
70  if (dt >= interval_ticks) {
71  // this task is due to run. Do we have enough time to run it?
72  _task_time_allowed = pgm_read_word(&_tasks[i].max_time_micros);
73 
74  if (dt >= interval_ticks*2) {
75  // we've slipped a whole run of this task!
76  if (_debug > 1) {
77  hal.console->printf("Scheduler slip task[%u-%s] (%u/%u/%u)\n",
78  (unsigned)i,
79  _tasks[i].name,
80  (unsigned)dt,
81  (unsigned)interval_ticks,
82  (unsigned)_task_time_allowed);
83  }
84  }
85 
86  if (_task_time_allowed <= time_available) {
87  // run it
88  _task_time_started = now;
89  task_fn_t func;
90  pgm_read_block(&_tasks[i].function, &func, sizeof(func));
91  current_task = i;
92  func();
93  current_task = -1;
94 
95  // record the tick counter when we ran. This drives
96  // when we next run the event
97  _last_run[i] = _tick_counter;
98 
99  // work out how long the event actually took
100  now = hal.scheduler->micros();
101  uint32_t time_taken = now - _task_time_started;
102 
103  if (time_taken > _task_time_allowed) {
104  // the event overran!
105  if (_debug > 2) {
106  hal.console->printf("Scheduler overrun task[%u-%s] (%u/%u)\n",
107  (unsigned)i,
108  _tasks[i].name,
109  (unsigned)time_taken,
110  (unsigned)_task_time_allowed);
111  }
112  }
113  if (time_taken >= time_available) {
114  goto update_spare_ticks;
115  }
116  time_available -= time_taken;
117  }
118  }
119  }
120 
121  // update number of spare microseconds
122  _spare_micros += time_available;
123 
124 update_spare_ticks:
125  _spare_ticks++;
126  if (_spare_ticks == 32) {
127  _spare_ticks /= 2;
128  _spare_micros /= 2;
129  }
130 }

132 /*
133  return number of micros until the current task reaches its deadline
134  */
135 uint16_t AP_Scheduler::time_available_usec(void)
136 {
137  uint32_t dt = hal.scheduler->micros() - _task_time_started;
138  if (dt > _task_time_allowed) {
139  return 0;
140  }
141  return _task_time_allowed - dt;
142 }

144 /*

147 float AP_Scheduler::load_average(uint32_t tick_time_usec) const
148 {
149  if (_spare_ticks == 0) {

151  }
152  uint32_t used_time = tick_time_usec - (_spare_micros/_spare_ticks);
