73 uint8_t FLYMAPLEStorage::read_byte(uint16_t loc){
74 //hal.console->printf("read_byte %d\n", loc);
75  uint16_t eeprom_index = loc >> 10;
76  uint16_t eeprom_offset = loc & 0x3ff;
77  if (eeprom_index >= num_eeprom_blocks)
78  {
79  hal.console->printf("FLYMAPLEStorage::read_byte loc %d out of range\n", loc);
80  return 0xff; // What else?
81  }
82 
83  // 'bytes' are packed 2 per word
84  // Read existing dataword and change upper or lower byte
85  uint16_t data = eeprom[eeprom_index].read(eeprom_offset >> 1);
86  if (eeprom_offset & 1)
87  return data >> 8; // Odd, upper byte
88  else
89  return data & 0xff; // Even lower byte
90 }

99 void FLYMAPLEStorage::write_byte(uint16_t loc, uint8_t value)
100 {
101 // hal.console->printf("write_byte %d, %d\n", loc, value);
102 
103  uint16_t eeprom_index = loc >> 10;
104  uint16_t eeprom_offset = loc & 0x3ff;
105  if (eeprom_index >= num_eeprom_blocks)
106  {
107  hal.console->printf("FLYMAPLEStorage::write_byte loc %d out of range\n", loc);
108  return;
109  }
110 
111  // 'bytes' are packed 2 per word
112  // Read existing data word and change upper or lower byte
113  uint16_t data = eeprom[eeprom_index].read(eeprom_offset >> 1);
114  if (eeprom_offset & 1)
115  data = (data & 0x00ff) | (value << 8); // Odd, upper byte
116  else
117  data = (data & 0xff00) | value; // Even, lower byte
118  eeprom[eeprom_index].write(eeprom_offset >> 1, data);
119 }
