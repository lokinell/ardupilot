138  }

141  readGpsData();
142

144  if (RecallGPS() && PV_AidingMode == AID_ABSOLUTE) {
145  // Don't fuse velocity data if GPS doesn't support it
146  if (frontend->_fusionModeGPS <= 1) {
147  fuseVelData = true;
148  } else {
149  fuseVelData = false;
150  }

152  } else {
153  fuseVelData = false;

158  readHgtData();

172  }
