211 // update inflight calculaton that determines if GPS data is good enough for reliable navigation
212 void NavEKF2_core::calcGpsGoodForFlight(void)
213 {
214  // use a simple criteria based on the GPS receivers claimed speed accuracy and the EKF innovation consistency checks
215 
216  // set up varaibles and constants used by filter that is applied to GPS speed accuracy
217  const float alpha1 = 0.2f; // coefficient for first stage LPF applied to raw speed accuracy data
218  const float tau = 10.0f; // time constant (sec) of peak hold decay
219  if (lastGpsCheckTime_ms == 0) {
220  lastGpsCheckTime_ms = imuSampleTime_ms;
221  }
222  float dtLPF = (imuSampleTime_ms - lastGpsCheckTime_ms) * 1e-3f;
223  lastGpsCheckTime_ms = imuSampleTime_ms;
224  float alpha2 = constrain_float(dtLPF/tau,0.0f,1.0f);
225 
226  // get the receivers reported speed accuracy
227  float gpsSpdAccRaw;
228  if (!_ahrs->get_gps().speed_accuracy(gpsSpdAccRaw)) {
229  gpsSpdAccRaw = 0.0f;
230  }
231 
232  // filter the raw speed accuracy using a LPF
233  sAccFilterState1 = constrain_float((alpha1 * gpsSpdAccRaw + (1.0f - alpha1) * sAccFilterState1),0.0f,10.0f);
234 
235  // apply a peak hold filter to the LPF output
236  sAccFilterState2 = max(sAccFilterState1,((1.0f - alpha2) * sAccFilterState2));
237 
238  // Apply a threshold test with hysteresis to the filtered GPS speed accuracy data
239  if (sAccFilterState2 > 1.5f ) {
240  gpsSpdAccPass = false;
241  } else if(sAccFilterState2 < 1.0f) {
242  gpsSpdAccPass = true;
243  }
244 
245  // Apply a threshold test with hysteresis to the normalised position and velocity innovations
246  // Require a fail for one second and a pass for 10 seconds to transition
247  if (lastInnovFailTime_ms == 0) {
248  lastInnovFailTime_ms = imuSampleTime_ms;
249  lastInnovPassTime_ms = imuSampleTime_ms;
250  }
251  if (velTestRatio < 1.0f && posTestRatio < 1.0f) {
252  lastInnovPassTime_ms = imuSampleTime_ms;
253  } else if (velTestRatio > 0.7f || posTestRatio > 0.7f) {
254  lastInnovFailTime_ms = imuSampleTime_ms;
255  }
256  if ((imuSampleTime_ms - lastInnovPassTime_ms) > 1000) {
257  ekfInnovationsPass = false;
258  } else if ((imuSampleTime_ms - lastInnovFailTime_ms) > 10000) {
259  ekfInnovationsPass = true;
260  }
261 
262  // both GPS speed accuracy and EKF innovations must pass
263  gpsAccuracyGood = gpsSpdAccPass && ekfInnovationsPass;
264 }

266 // Detect if we are in flight or on ground
267 void NavEKF2_core::detectFlight()
268 {
269  /*
270  If we are a fly forward type vehicle (eg plane), then in-air status can be determined through a combination of speed and height criteria.
271  Because of the differing certainty requirements of algorithms that need the in-flight / on-ground status we use two booleans where
272  onGround indicates a high certainty we are not flying and inFlight indicates a high certainty we are flying. It is possible for
273  both onGround and inFlight to be false if the status is uncertain, but they cannot both be true.
274 
275  If we are a plane as indicated by the assume_zero_sideslip() status, then different logic is used
276 
277  TODO - this logic should be moved out of the EKF and into the flight vehicle code.
278  */
279 
280  if (assume_zero_sideslip()) {
281  // To be confident we are in the air we use a criteria which combines arm status, ground speed, airspeed and height change
282  float gndSpdSq = sq(gpsDataDelayed.vel.x) + sq(gpsDataDelayed.vel.y);
283  bool highGndSpd = false;
284  bool highAirSpd = false;
285  bool largeHgtChange = false;
286 
287  // trigger at 8 m/s airspeed
288  if (_ahrs->airspeed_sensor_enabled()) {
289  const AP_Airspeed *airspeed = _ahrs->get_airspeed();
290  if (airspeed->get_airspeed() * airspeed->get_EAS2TAS() > 10.0f) {
291  highAirSpd = true;
292  }
293  }
294 
295  // trigger at 10 m/s GPS velocity, but not if GPS is reporting bad velocity errors
296  if (gndSpdSq > 100.0f && gpsSpdAccuracy < 1.0f) {
297  highGndSpd = true;
298  }
299 
300  // trigger if more than 10m away from initial height
301  if (fabsf(baroDataDelayed.hgt) > 10.0f) {
302  largeHgtChange = true;
303  }
304 
305  // Determine to a high certainty we are flying
306  if (motorsArmed && highGndSpd && (highAirSpd || largeHgtChange)) {
307  onGround = false;
308  inFlight = true;
309  }
310 
311  // if is possible we are in flight, set the time this condition was last detected
312  if (motorsArmed && (highGndSpd || highAirSpd || largeHgtChange)) {
313  airborneDetectTime_ms = imuSampleTime_ms;
314  onGround = false;
315  }
316 
317  // Determine to a high certainty we are not flying
318  // after 5 seconds of not detecting a possible flight condition or we are disarmed, we transition to on-ground mode
319  if(!motorsArmed || ((imuSampleTime_ms - airborneDetectTime_ms) > 5000)) {
320  onGround = true;
321  inFlight = false;
322  }
323  } else {
324  // Non fly forward vehicle, so can only use height and motor arm status
325 
326  // If the motors are armed then we could be flying and if they are not armed then we are definitely not flying
327  if (motorsArmed) {
328  onGround = false;
329  } else {
330  inFlight = false;
331  onGround = true;
332  }
333 
334  // If height has increased since exiting on-ground, then we definitely are flying
335  if (!onGround && ((stateStruct.position.z - posDownAtTakeoff) < -1.5f)) {
336  inFlight = true;
337  }
338 
339  // If rangefinder has increased since exiting on-ground, then we definitely are flying
340  if (!onGround && ((rngMea - rngAtStartOfFlight) > 0.5f)) {
341  inFlight = true;
342  }
343 
344  }
345 
346  // store current on-ground and in-air status for next time
347  prevOnGround = onGround;
348  prevInFlight = inFlight;
349 
350  // Store vehicle height and range prior to takeoff for use in post takeoff checks
351  if (onGround && prevOnGround) {
352  // store vertical position at start of flight to use as a reference for ground relative checks
353  posDownAtTakeoff = stateStruct.position.z;
354  // store the range finder measurement which will be used as a reference to detect when we have taken off
355  rngAtStartOfFlight = rngMea;
356  }
357 
358 }
