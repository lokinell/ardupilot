104  // exit immediately if we are disabled (perhaps set pwm values back to defaults)
105  if (!_enabled) {
106  return;

113 
114  // get horizontal velocity
115  const Vector3f &velocity = _inav->get_velocity();
116  ground_speed = pythagorous2(velocity.x,velocity.y);
117 
118  // get the current time
119  now = hal.scheduler->millis();
120 
121  // check our speed vs the minimum
122  if (ground_speed >= _speed_min) {
123  // if we are not already spraying
124  if (!_flags.spraying) {
125  // set the timer if this is the first time we've surpassed the min speed
126  if (_speed_over_min_time == 0) {
127  _speed_over_min_time = now;
128  }else{
129  // check if we've been over the speed long enough to engage the sprayer
130  if((now - _speed_over_min_time) > AC_SPRAYER_DEFAULT_TURN_ON_DELAY) {
131  _flags.spraying = true;
132  _speed_over_min_time = 0;
133  }
134  }
135  }
136  // reset the speed under timer
137  _speed_under_min_time = 0;
138  }else{
139  // we are under the min speed. If we are spraying
140  if (_flags.spraying) {
141  // set the timer if this is the first time we've dropped below the min speed
142  if (_speed_under_min_time == 0) {
143  _speed_under_min_time = now;
144  }else{
145  // check if we've been over the speed long enough to engage the sprayer
146  if((now - _speed_under_min_time) > AC_SPRAYER_DEFAULT_SHUT_OFF_DELAY) {
147  _flags.spraying = false;
148  _speed_under_min_time = 0;
149  }
150  }
151  }
152  // reset the speed over timer
153  _speed_over_min_time = 0;
154  }
155 
156  // if testing pump output speed as if travelling at 1m/s
157  if (_flags.testing) {
158  ground_speed = 100.0f;
159  }
160 
161  // if spraying or testing update the pump servo position
162  if (_flags.spraying || _flags.testing) {
163  RC_Channel_aux::move_servo(RC_Channel_aux::k_sprayer_pump, min(max(ground_speed * _pump_pct_1ms, 100 *_pump_min_pct),10000),0,10000);
164  RC_Channel_aux::set_radio(RC_Channel_aux::k_sprayer_spinner, _spinner_pwm);
