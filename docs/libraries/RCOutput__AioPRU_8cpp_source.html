37 {

40  uint32_t *ctrl;

42  signal(SIGBUS,catch_sigbus);

44  mem_fd = open("/dev/mem", O_RDWR|O_SYNC);
45 
46  pwm = (struct pwm*) mmap(0, 0x1000, PROT_READ|PROT_WRITE, MAP_SHARED, mem_fd, RCOUT_PRUSS_RAM_BASE);
47  iram = (uint32_t*)mmap(0, 0x2000, PROT_READ|PROT_WRITE, MAP_SHARED, mem_fd, RCOUT_PRUSS_IRAM_BASE);
48  ctrl = (uint32_t*)mmap(0, 0x1000, PROT_READ|PROT_WRITE, MAP_SHARED, mem_fd, RCOUT_PRUSS_CTRL_BASE);
49 
50  close(mem_fd);
51

53  *ctrl = 0;
54 
55  // Load firmware
56  memcpy(iram, PRUcode, sizeof(PRUcode));
57 
58  // Start PRU 1

60

62  // overrides this when necessary
63  set_freq(0xFFFFFFFF, 50);
64 }
