174 void AC_Circle::get_closest_point_on_circle(Vector3f &result)
175 {
176  // return center if radius is zero
177  if (_radius <= 0) {
178  result = _center;
179  return;
180  }
181 
182  // get current position
183  const Vector3f &curr_pos = _inav.get_position();
184 
185  // calc vector from current location to circle center
186  Vector2f vec; // vector from circle center to current location
187  vec.x = (curr_pos.x - _center.x);
188  vec.y = (curr_pos.y - _center.y);
189  float dist = pythagorous2(vec.x, vec.y);
190 
191  // if current location is exactly at the center of the circle return edge directly behind vehicle
192  if (is_zero(dist)) {

194  result.y = _center.y - _radius * _ahrs.sin_yaw();
195  result.z = _center.z;
196  return;
197  }
198 
199  // calculate closest point on edge of circle

201  result.y = _center.y + vec.y / dist * _radius;
