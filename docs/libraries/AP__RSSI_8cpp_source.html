106 // Read the receiver RSSI value as a float 0.0f - 1.0f.
107 // 0.0 represents weakest signal, 1.0 represents maximum signal.
108 float AP_RSSI::read_receiver_rssi()
109 {
110  // Default to 0 RSSI
111  float receiver_rssi = 0.0f;
112 
113  switch (rssi_type) {
114  case RssiType::RSSI_DISABLED :
115  receiver_rssi = 0.0f;
116  break;
117  case RssiType::RSSI_ANALOG_PIN :
118  receiver_rssi = read_pin_rssi();
119  break;
120  case RssiType::RSSI_RC_CHANNEL_VALUE :
121  receiver_rssi = read_channel_rssi();
122  break;
123  default :
124  receiver_rssi = 0.0f;
125  break;
126  }
127 
128  return receiver_rssi;
129 }

158 // Scale and constrain a float rssi value to 0.0 to 1.0 range
159 float AP_RSSI::scale_and_constrain_float_rssi(float current_rssi_value, float low_rssi_range, float high_rssi_range)
160 {
161  float rssi_value_range = fabsf(high_rssi_range - low_rssi_range);
162  if (is_zero(rssi_value_range)) {
163  // User range isn't meaningful, return 0 for RSSI (and avoid divide by zero)
164  return 0.0f;
165  }
166  // Note that user-supplied ranges may be inverted and we accommodate that here.
167  // (Some radio receivers put out inverted ranges for RSSI-type values).
168  bool range_is_inverted = (high_rssi_range < low_rssi_range);
169  // Constrain to the possible range - values outside are clipped to ends
170  current_rssi_value = constrain_float(current_rssi_value,
171  range_is_inverted ? high_rssi_range : low_rssi_range,
172  range_is_inverted ? low_rssi_range : high_rssi_range);
173 
174  if (range_is_inverted)
175  {
176  // Swap values so we can treat them as low->high uniformly in the code that follows
177  current_rssi_value = high_rssi_range + abs(current_rssi_value - low_rssi_range);
178  std::swap(low_rssi_range, high_rssi_range);
179  }
180 
181  // Scale the value down to a 0.0 - 1.0 range
182  float rssi_value_scaled = (current_rssi_value - low_rssi_range) / rssi_value_range;
183  // Make absolutely sure the value is clipped to the 0.0 - 1.0 range. This should handle things if the
184  // value retrieved falls outside the user-supplied range.
185  return constrain_float(rssi_value_scaled, 0.0f, 1.0f);
186 }
