138 // Read Sensor data
139 bool AP_Compass_HMC5843::read_raw()
140 {
141  struct AP_HMC5843_SerialBus::raw_value rv;
142 
143  if (_bus->read_raw(&rv) != 0) {
144  _bus->set_high_speed(false);
145  _retry_time = hal.scheduler->millis() + 1000;
146  return false;
147  }
148 
149  int16_t rx, ry, rz;
150  rx = (((int16_t)rv.val[0]) << 8) | rv.val[1];
151  if (_product_id == AP_COMPASS_TYPE_HMC5883L) {
152  rz = (((int16_t)rv.val[2]) << 8) | rv.val[3];
153  ry = (((int16_t)rv.val[4]) << 8) | rv.val[5];
154  } else {
155  ry = (((int16_t)rv.val[2]) << 8) | rv.val[3];
156  rz = (((int16_t)rv.val[4]) << 8) | rv.val[5];
157  }
158  if (rx == -4096 || ry == -4096 || rz == -4096) {
159  // no valid data available
160  return false;
161  }
162 
163  _mag_x = -rx;
164  _mag_y = ry;
165  _mag_z = -rz;
166 
167  return true;
168 }

171 // accumulate a reading from the magnetometer
172 void AP_Compass_HMC5843::accumulate(void)
173 {
174  if (!_initialised) {
175  // someone has tried to enable a compass for the first time
176  // mid-flight .... we can't do that yet (especially as we won't
177  // have the right orientation!)
178  return;
179  }
180 
181  uint32_t tnow = hal.scheduler->micros();
182  if (_accum_count != 0 && (tnow - _last_accum_time) < 13333) {
183  // the compass gets new data at 75Hz
184  return;
185  }
186 
187  if (!_bus_sem->take(1)) {
188  // the bus is busy - try again later
189  return;
190  }
191  bool result = read_raw();
192  _bus_sem->give();
193 
194  if (result) {
195  // the _mag_N values are in the range -2048 to 2047, so we can
196  // accumulate up to 15 of them in an int16_t. Let's make it 14
197  // for ease of calculation. We expect to do reads at 10Hz, and
198  // we get new data at most 75Hz, so we don't expect to
199  // accumulate more than 8 before a read
200  // get raw_field - sensor frame, uncorrected
201  Vector3f raw_field = Vector3f(_mag_x, _mag_y, _mag_z);
202  raw_field *= _gain_multiple;
203 
204  // rotate raw_field from sensor frame to body frame
205  rotate_field(raw_field, _compass_instance);
206 
207  // publish raw_field (uncorrected point sample) for calibration use
208  publish_raw_field(raw_field, tnow, _compass_instance);
209 
210  // correct raw_field for known errors
211  correct_field(raw_field, _compass_instance);
212 
213  // publish raw_field (corrected point sample) for EKF use
214  publish_unfiltered_field(raw_field, tnow, _compass_instance);
215 
216  _mag_x_accum += raw_field.x;
217  _mag_y_accum += raw_field.y;
218  _mag_z_accum += raw_field.z;
219  _accum_count++;
220  if (_accum_count == 14) {
221  _mag_x_accum /= 2;
222  _mag_y_accum /= 2;
223  _mag_z_accum /= 2;
224  _accum_count = 7;
225  }
226  _last_accum_time = tnow;
227  }
228 }

244 bool AP_Compass_HMC5843::_detect_version()
245 {
246  _base_config = 0x0;
247 
248  if (!write_register(ConfigRegA, SampleAveraging_8<<5 | DataOutputRate_75HZ<<2 | NormalOperation) ||
249  !read_register(ConfigRegA, &_base_config)) {
250  return false;
251  }
252  if (_base_config == (SampleAveraging_8<<5 | DataOutputRate_75HZ<<2 | NormalOperation)) {
253  /* a 5883L supports the sample averaging config */
254  _product_id = AP_COMPASS_TYPE_HMC5883L;
255  return true;
256  } else if (_base_config == (NormalOperation | DataOutputRate_75HZ<<2)) {
257  _product_id = AP_COMPASS_TYPE_HMC5843;
258  return true;
259  } else {
260  /* not behaving like either supported compass type */
261  return false;
262  }
263 }

266 bool
267 AP_Compass_HMC5843::init()
268 {
269  uint8_t calibration_gain = 0x20;
270  uint16_t expected_x = 715;
271  uint16_t expected_yz = 715;
272  _gain_multiple = (1.0f / 1300) * 1000;
273 
274  _bus_sem = _bus->get_semaphore();
275  hal.scheduler->suspend_timer_procs();
276 
277  if (!_bus_sem || !_bus_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
278  hal.console->printf("HMC5843: Unable to get bus semaphore\n");
279  goto fail_sem;
280  }
281 
282  if (!_bus->configure()) {
283  hal.console->printf("HMC5843: Could not configure the bus\n");
284  goto errout;
285  }
286 
287  if (!_detect_version()) {
288  hal.console->printf("HMC5843: Could not detect version\n");
289  goto errout;
290  }
291 
292  if (_product_id == AP_COMPASS_TYPE_HMC5883L) {
293  calibration_gain = 0x60;
294  /*
295  note that the HMC5883 datasheet gives the x and y expected
296  values as 766 and the z as 713. Experiments have shown the x
297  axis is around 766, and the y and z closer to 713.
298  */
299  expected_x = 766;
300  expected_yz = 713;
301  _gain_multiple = (1.0f / 1090) * 1000;
302  }
303 
304  if (!_calibrate(calibration_gain, expected_x, expected_yz)) {
305  hal.console->printf("HMC5843: Could not calibrate sensor\n");
306  goto errout;
307  }
308 
309  // leave test mode
310  if (!re_initialise()) {
311  goto errout;
312  }
313 
314  if (!_bus->start_measurements()) {
315  hal.console->printf("HMC5843: Could not start measurements on bus\n");
316  goto errout;
317  }
318  _initialised = true;
319 
320  _bus_sem->give();
321  hal.scheduler->resume_timer_procs();
322 
323  // perform an initial read
324  read();
325 
326 #if 0
327  hal.console->printf("CalX: %.2f CalY: %.2f CalZ: %.2f\n",
328  _scaling[0], _scaling[1], _scaling[2]);
329 #endif
330 
331  _compass_instance = register_compass();
332  set_dev_id(_compass_instance, _product_id);
333 
334 #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX && CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_RASPILOT
335  set_external(_compass_instance, true);
336 #endif
337 
338  return true;
339 
340 errout:
341  _bus_sem->give();
342 fail_sem:
343  hal.scheduler->resume_timer_procs();
344  return false;
345 }

347 bool AP_Compass_HMC5843::_calibrate(uint8_t calibration_gain,
348  uint16_t expected_x,
349  uint16_t expected_yz)
350 {
351  int numAttempts = 0, good_count = 0;
352  bool success = false;
353 
354  while (success == 0 && numAttempts < 25 && good_count < 5)
355  {
356  numAttempts++;
357 
358  // force positiveBias (compass should return 715 for all channels)
359  if (!write_register(ConfigRegA, PositiveBiasConfig))
360  continue; // compass not responding on the bus
361 
362  hal.scheduler->delay(50);
363 
364  // set gains
365  if (!write_register(ConfigRegB, calibration_gain) ||
366  !write_register(ModeRegister, SingleConversion))
367  continue;
368 
369  // read values from the compass
370  hal.scheduler->delay(50);
371  if (!read_raw())
372  continue; // we didn't read valid values
373 
374  hal.scheduler->delay(10);
375 
376  float cal[3];
377 
378  // hal.console->printf("mag %d %d %d\n", _mag_x, _mag_y, _mag_z);
379 
380  cal[0] = fabsf(expected_x / (float)_mag_x);
381  cal[1] = fabsf(expected_yz / (float)_mag_y);
382  cal[2] = fabsf(expected_yz / (float)_mag_z);
383 
384  // hal.console->printf("cal=%.2f %.2f %.2f\n", cal[0], cal[1], cal[2]);
385 
386  // we throw away the first two samples as the compass may
387  // still be changing its state from the application of the
388  // strap excitation. After that we accept values in a
389  // reasonable range
390 
391  if (numAttempts <= 2) {
392  continue;
393  }
394 
395 #define IS_CALIBRATION_VALUE_VALID(val) (val > 0.7f && val < 1.35f)
396 
397  if (IS_CALIBRATION_VALUE_VALID(cal[0]) &&
398  IS_CALIBRATION_VALUE_VALID(cal[1]) &&
399  IS_CALIBRATION_VALUE_VALID(cal[2])) {
400  // hal.console->printf("car=%.2f %.2f %.2f good\n", cal[0], cal[1], cal[2]);
401  good_count++;
402 
403  _scaling[0] += cal[0];
404  _scaling[1] += cal[1];
405  _scaling[2] += cal[2];
406  }
407 
408 #undef IS_CALIBRATION_VALUE_VALID
409 
410 #if 0
411  /* useful for debugging */
412  hal.console->printf("MagX: %d MagY: %d MagZ: %d\n", (int)_mag_x, (int)_mag_y, (int)_mag_z);
413  hal.console->printf("CalX: %.2f CalY: %.2f CalZ: %.2f\n", cal[0], cal[1], cal[2]);
414 #endif
415  }
416 
417  if (good_count >= 5) {
418  _scaling[0] = _scaling[0] / good_count;
419  _scaling[1] = _scaling[1] / good_count;
420  _scaling[2] = _scaling[2] / good_count;
421  success = true;
422  } else {
423  /* best guess */
424  _scaling[0] = 1.0;
425  _scaling[1] = 1.0;
426  _scaling[2] = 1.0;
427  }
428 
429  return success;
430 }

432 // Read Sensor data
433 void AP_Compass_HMC5843::read()
434 {
435  if (!_initialised) {
436  // someone has tried to enable a compass for the first time
437  // mid-flight .... we can't do that yet (especially as we won't
438  // have the right orientation!)
439  return;
440  }
441  if (_retry_time != 0) {
442  if (hal.scheduler->millis() < _retry_time) {
443  return;
444  }
445  if (!re_initialise()) {
446  _retry_time = hal.scheduler->millis() + 1000;
447  _bus->set_high_speed(false);
448  return;
449  }
450  }
451 
452  if (_accum_count == 0) {
453  accumulate();
454  if (_retry_time != 0) {
455  _bus->set_high_speed(false);
456  return;
457  }
458  }
459 
460  Vector3f field(_mag_x_accum * _scaling[0],
461  _mag_y_accum * _scaling[1],
462  _mag_z_accum * _scaling[2]);
463  field /= _accum_count;
464 
465  _accum_count = 0;
466  _mag_x_accum = _mag_y_accum = _mag_z_accum = 0;
467 
468  // rotate to the desired orientation
469  if (_product_id == AP_COMPASS_TYPE_HMC5883L) {
470  field.rotate(ROTATION_YAW_90);
471  }
472 
473  publish_filtered_field(field, _compass_instance);
474  _retry_time = 0;
475 }
