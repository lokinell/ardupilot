31 bool ToshibaLED_I2C::hw_init()
32 {
33  // get pointer to i2c bus semaphore
34  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
35 
36  // take i2c bus sempahore
37  if (!i2c_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
38  return false;
39  }
40 
41  // disable recording of i2c lockup errors
42  hal.i2c->ignore_errors(true);
43 
44  // enable the led
45  bool ret = (hal.i2c->writeRegister(TOSHIBA_LED_ADDRESS, TOSHIBA_LED_ENABLE, 0x03) == 0);
46 
47  // update the red, green and blue values to zero
48  uint8_t val[3] = { _led_off, _led_off, _led_off };
49  ret &= (hal.i2c->writeRegisters(TOSHIBA_LED_ADDRESS, TOSHIBA_LED_PWM0, 3, val) == 0);
50 
51  // re-enable recording of i2c lockup errors
52  hal.i2c->ignore_errors(false);
53 
54  // give back i2c semaphore
55  i2c_sem->give();
56 
57  return ret;
58 }
