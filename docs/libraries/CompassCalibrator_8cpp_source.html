17 /*
18  * The intention of a magnetometer in a compass application is to measure
19  * Earth's magnetic field. Measurements other than those of Earth's magnetic
20  * field are considered errors. This algorithm computes a set of correction
21  * parameters that null out errors from various sources:
22  *
23  * - Sensor bias error
24  * - "Hard iron" error caused by materials fixed to the vehicle body that
25  * produce static magnetic fields.
26  * - Sensor scale-factor error
27  * - Sensor cross-axis sensitivity
28  * - "Soft iron" error caused by materials fixed to the vehicle body that
29  * distort magnetic fields.
30  *
31  * This is done by taking a set of samples that are assumed to be the product
32  * of rotation in earth's magnetic field and fitting an offset ellipsoid to
33  * them, determining the correction to be applied to adjust the samples into an
34  * origin-centered sphere.
35  *
36  * The state machine of this library is described entirely by the
37  * compass_cal_status_t enum, and all state transitions are managed by the
38  * set_status function. Normally, the library is in the NOT_STARTED state. When
39  * the start function is called, the state transitions to WAITING_TO_START,
40  * until two conditions are met: the delay as elapsed, and the memory for the
41  * sample buffer has been successfully allocated.
42  * Once these conditions are met, the state transitions to RUNNING_STEP_ONE, and
43  * samples are collected via calls to the new_sample function. These samples are
44  * accepted or rejected based on distance to the nearest sample. The samples are
45  * assumed to cover the surface of a sphere, and the radius of that sphere is
46  * initialized to a conservative value. Based on a circle-packing pattern, the
47  * minimum distance is set such that some percentage of the surface of that
48  * sphere must be covered by samples.
49  *
50  * Once the sample buffer is full, a sphere fitting algorithm is run, which
51  * computes a new sphere radius. The sample buffer is thinned of samples which
52  * no longer meet the acceptance criteria, and the state transitions to
53  * RUNNING_STEP_TWO. Samples continue to be collected until the buffer is full
54  * again, the full ellipsoid fit is run, and the state transitions to either
55  * SUCCESS or FAILED.
56  *
57  * The fitting algorithm used is Levenberg-Marquardt. See also:
58  * http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm
59  */

103 float CompassCalibrator::get_completion_percent() const {
104  // first sampling step is 1/3rd of the progress bar
105  // never return more than 99% unless _status is COMPASS_CAL_SUCCESS
106  switch(_status) {
107  case COMPASS_CAL_NOT_STARTED:
108  case COMPASS_CAL_WAITING_TO_START:
109  return 0.0f;
110  case COMPASS_CAL_RUNNING_STEP_ONE:
111  return 33.3f * _samples_collected/COMPASS_CAL_NUM_SAMPLES;
112  case COMPASS_CAL_RUNNING_STEP_TWO:
113  return 33.3f + 65.7f*((float)(_samples_collected-_samples_thinned)/(COMPASS_CAL_NUM_SAMPLES-_samples_thinned));
114  case COMPASS_CAL_SUCCESS:
115  return 100.0f;
116  case COMPASS_CAL_FAILED:
117  default:
118  return 0.0f;
119  };

145 void CompassCalibrator::update(bool &failure) {
146  failure = false;
147 
148  if(!fitting()) {
149  return;
150  }
151 
152  if(_status == COMPASS_CAL_RUNNING_STEP_ONE) {
153  if (_fit_step >= 10) {
154  if(_fitness == _initial_fitness || isnan(_fitness)) { //if true, means that fitness is diverging instead of converging
155  set_status(COMPASS_CAL_FAILED);
156  failure = true;
157  }
158  set_status(COMPASS_CAL_RUNNING_STEP_TWO);
159  } else {
160  run_sphere_fit();
161  _fit_step++;
162  }
163  } else if(_status == COMPASS_CAL_RUNNING_STEP_TWO) {
164  if (_fit_step >= 35) {
165  if(fit_acceptable()) {
166  set_status(COMPASS_CAL_SUCCESS);
167  } else {
168  set_status(COMPASS_CAL_FAILED);
169  failure = true;
170  }
171  } else if (_fit_step < 15) {
172  run_sphere_fit();
173  _fit_step++;
174  } else {
175  run_ellipsoid_fit();
176  _fit_step++;
177  }
178  }
179 }

216 bool CompassCalibrator::set_status(compass_cal_status_t status) {
217  if (status != COMPASS_CAL_NOT_STARTED && _status == status) {
218  return true;
219  }
220 
221  switch(status) {
222  case COMPASS_CAL_NOT_STARTED:
223  reset_state();
224  _status = COMPASS_CAL_NOT_STARTED;
225 
226  if(_sample_buffer != NULL) {
227  free(_sample_buffer);
228  _sample_buffer = NULL;
229  }
230  return true;
231 
232  case COMPASS_CAL_WAITING_TO_START:
233  reset_state();
234  _status = COMPASS_CAL_WAITING_TO_START;
235 
236  set_status(COMPASS_CAL_RUNNING_STEP_ONE);
237  return true;
238 
239  case COMPASS_CAL_RUNNING_STEP_ONE:
240  if(_status != COMPASS_CAL_WAITING_TO_START) {
241  return false;
242  }
243 
244  if(_attempt == 1 && (hal.scheduler->millis()-_start_time_ms)*1.0e-3f < _delay_start_sec) {
245  return false;
246  }
247 
248  if (_sample_buffer == NULL) {
249  _sample_buffer =
250  (CompassSample*) malloc(sizeof(CompassSample) *
251  COMPASS_CAL_NUM_SAMPLES);
252  }
253 
254  if(_sample_buffer != NULL) {
255  initialize_fit();
256  _status = COMPASS_CAL_RUNNING_STEP_ONE;
257  return true;
258  }
259 
260  return false;
261 
262  case COMPASS_CAL_RUNNING_STEP_TWO:
263  if(_status != COMPASS_CAL_RUNNING_STEP_ONE) {
264  return false;
265  }
266  thin_samples();
267  initialize_fit();
268  _status = COMPASS_CAL_RUNNING_STEP_TWO;
269  return true;
270 
271  case COMPASS_CAL_SUCCESS:
272  if(_status != COMPASS_CAL_RUNNING_STEP_TWO) {
273  return false;
274  }
275 
276  if(_sample_buffer != NULL) {
277  free(_sample_buffer);
278  _sample_buffer = NULL;
279  }
280 
281  _status = COMPASS_CAL_SUCCESS;
282  return true;
283 
284  case COMPASS_CAL_FAILED:
285  if(_status == COMPASS_CAL_NOT_STARTED) {
286  return false;
287  }
288 
289  if(_retry && set_status(COMPASS_CAL_WAITING_TO_START)) {
290  _attempt++;
291  return true;
292  }
293 
294  if(_sample_buffer != NULL) {
295  free(_sample_buffer);
296  _sample_buffer = NULL;
297  }
298 
299  _status = COMPASS_CAL_FAILED;
300  return true;
301 
302  default:
303  return false;
304  };

307 bool CompassCalibrator::fit_acceptable() {
308  if( !isnan(_fitness) &&
309  _params.radius > 150 && _params.radius < 950 && //Earth's magnetic field strength range: 250-850mG
310  fabsf(_params.offset.x) < 1000 &&
311  fabsf(_params.offset.y) < 1000 &&
312  fabsf(_params.offset.z) < 1000 &&
313  _params.diag.x > 0.2f && _params.diag.x < 5.0f &&
314  _params.diag.y > 0.2f && _params.diag.y < 5.0f &&
315  _params.diag.z > 0.2f && _params.diag.z < 5.0f &&
316  fabsf(_params.offdiag.x) < 1.0f && //absolute of sine/cosine output cannot be greater than 1
317  fabsf(_params.offdiag.y) < 1.0f &&
318  fabsf(_params.offdiag.z) < 1.0f ){
319 
320  return _fitness <= sq(_tolerance);
321  }
322  return false;

325 void CompassCalibrator::thin_samples() {
326  if(_sample_buffer == NULL) {
327  return;
328  }
329 
330  _samples_thinned = 0;
331  // shuffle the samples http://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle
332  // this is so that adjacent samples don't get sequentially eliminated
333  for(uint16_t i=_samples_collected-1; i>=1; i--) {
334  uint16_t j = get_random() % (i+1);
335  CompassSample temp = _sample_buffer[i];
336  _sample_buffer[i] = _sample_buffer[j];
337  _sample_buffer[j] = temp;
338  }
339 
340  for(uint16_t i=0; i < _samples_collected; i++) {
341  if(!accept_sample(_sample_buffer[i])) {
342  _sample_buffer[i] = _sample_buffer[_samples_collected-1];
343  _samples_collected --;
344  _samples_thinned ++;
345  }
346  }
347 }

349 /*
350  * The sample acceptance distance is determined as follows:
351  * For any regular polyhedron with triangular faces, the angle theta subtended
352  * by two closest points is defined as
353  *
354  * theta = arccos(cos(A)/(1-cos(A)))
355  *
356  * Where:
357  * A = (4pi/F + pi)/3
358  * and
359  * F = 2V - 4 is the number of faces for the polyhedron in consideration,
360  * which depends on the number of vertices V
361  *
362  * The above equation was proved after solving for spherical triangular excess
363  * and related equations.
364  */
365 bool CompassCalibrator::accept_sample(const Vector3f& sample)
366 {
367  static const uint16_t faces = (2 * COMPASS_CAL_NUM_SAMPLES - 4);
368  static const float a = (4.0f * M_PI_F / (3.0f * faces)) + M_PI_F / 3.0f;
369  static const float theta = 0.5f * acosf(cosf(a) / (1.0f - cosf(a)));
370 
371  if(_sample_buffer == NULL) {
372  return false;
373  }
374 
375  float min_distance = _params.radius * 2*sinf(theta/2);
376 
377  for (uint16_t i = 0; i<_samples_collected; i++){
378  float distance = (sample - _sample_buffer[i].get()).length();
379  if(distance < min_distance) {
380  return false;
381  }
382  }
383  return true;
384 }

391  Matrix3f softiron(
392  params.diag.x , params.offdiag.x , params.offdiag.y,

404 float CompassCalibrator::calc_mean_squared_residuals(const param_t& params) const
405 {
406  if(_sample_buffer == NULL || _samples_collected == 0) {
407  return 1.0e30f;
408  }
409  float sum = 0.0f;
410  for(uint16_t i=0; i < _samples_collected; i++){
411  Vector3f sample = _sample_buffer[i].get();
412  float resid = calc_residual(sample, params);
413  sum += sq(resid);
414  }
415  sum /= _samples_collected;
416  return sum;

419 void CompassCalibrator::calc_sphere_jacob(const Vector3f& sample, const param_t& params, float* ret) const{
420  const Vector3f &offset = params.offset;
421  const Vector3f &diag = params.diag;
422  const Vector3f &offdiag = params.offdiag;
423  Matrix3f softiron(
424  diag.x , offdiag.x , offdiag.y,
425  offdiag.x , diag.y , offdiag.z,
426  offdiag.y , offdiag.z , diag.z
427  );
428 
429  float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
430  float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
431  float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
432  float length = (softiron*(sample+offset)).length();
433 
434  // 0: partial derivative (radius wrt fitness fn) fn operated on sample
435  ret[0] = 1.0f;
436  // 1-3: partial derivative (offsets wrt fitness fn) fn operated on sample
437  ret[1] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
438  ret[2] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
439  ret[3] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
440 }

442 void CompassCalibrator::run_sphere_fit()
443 {
444  if(_sample_buffer == NULL) {
445  return;
446  }
447 
448  const float lma_damping = 10.0f;
449 
450  float fitness = _fitness;
451  float fit1, fit2;
452  param_t fit1_params, fit2_params;
453  fit1_params = fit2_params = _params;
454 
455  float JTJ[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS];
456  float JTJ2[COMPASS_CAL_NUM_SPHERE_PARAMS*COMPASS_CAL_NUM_SPHERE_PARAMS];
457  float JTFI[COMPASS_CAL_NUM_SPHERE_PARAMS];
458 
459  memset(&JTJ,0,sizeof(JTJ));
460  memset(&JTJ2,0,sizeof(JTJ2));
461  memset(&JTFI,0,sizeof(JTFI));
462  // Gauss Newton Part common for all kind of extensions including LM
463  for(uint16_t k = 0; k<_samples_collected; k++) {
464  Vector3f sample = _sample_buffer[k].get();
465 
466  float sphere_jacob[COMPASS_CAL_NUM_SPHERE_PARAMS];
467 
468  calc_sphere_jacob(sample, fit1_params, sphere_jacob);
469 
470  for(uint8_t i = 0;i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
471  // compute JTJ
472  for(uint8_t j = 0; j < COMPASS_CAL_NUM_SPHERE_PARAMS; j++) {
473  JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j];
474  JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+j] += sphere_jacob[i] * sphere_jacob[j]; //a backup JTJ for LM
475  }
476  // compute JTFI
477  JTFI[i] += sphere_jacob[i] * calc_residual(sample, fit1_params);
478  }
479  }
480 
481 
482  //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
483  //refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
484  for(uint8_t i = 0; i < COMPASS_CAL_NUM_SPHERE_PARAMS; i++) {
485  JTJ[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda;
486  JTJ2[i*COMPASS_CAL_NUM_SPHERE_PARAMS+i] += _sphere_lambda/lma_damping;
487  }
488 
489  if(!inverse(JTJ, JTJ, 4)) {
490  return;
491  }
492 
493  if(!inverse(JTJ2, JTJ2, 4)) {
494  return;
495  }
496 
497  for(uint8_t row=0; row < COMPASS_CAL_NUM_SPHERE_PARAMS; row++) {
498  for(uint8_t col=0; col < COMPASS_CAL_NUM_SPHERE_PARAMS; col++) {
499  fit1_params.get_sphere_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
500  fit2_params.get_sphere_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_SPHERE_PARAMS+col];
501  }
502  }
503 
504  fit1 = calc_mean_squared_residuals(fit1_params);
505  fit2 = calc_mean_squared_residuals(fit2_params);
506 
507  if(fit1 > _fitness && fit2 > _fitness){
508  _sphere_lambda *= lma_damping;
509  } else if(fit2 < _fitness && fit2 < fit1) {
510  _sphere_lambda /= lma_damping;
511  fit1_params = fit2_params;
512  fitness = fit2;
513  } else if(fit1 < _fitness){
514  fitness = fit1;
515  }
516  //--------------------Levenberg-Marquardt-part-ends-here--------------------------------//
517 
518  if(!isnan(fitness) && fitness < _fitness) {
519  _fitness = fitness;
520  _params = fit1_params;
521  }
522 }

526 void CompassCalibrator::calc_ellipsoid_jacob(const Vector3f& sample, const param_t& params, float* ret) const{
527  const Vector3f &offset = params.offset;
528  const Vector3f &diag = params.diag;
529  const Vector3f &offdiag = params.offdiag;
530  Matrix3f softiron(
531  diag.x , offdiag.x , offdiag.y,
532  offdiag.x , diag.y , offdiag.z,
533  offdiag.y , offdiag.z , diag.z
534  );
535 
536  float A = (diag.x * (sample.x + offset.x)) + (offdiag.x * (sample.y + offset.y)) + (offdiag.y * (sample.z + offset.z));
537  float B = (offdiag.x * (sample.x + offset.x)) + (diag.y * (sample.y + offset.y)) + (offdiag.z * (sample.z + offset.z));
538  float C = (offdiag.y * (sample.x + offset.x)) + (offdiag.z * (sample.y + offset.y)) + (diag.z * (sample.z + offset.z));
539  float length = (softiron*(sample+offset)).length();
540 
541  // 0-2: partial derivative (offset wrt fitness fn) fn operated on sample
542  ret[0] = -1.0f * (((diag.x * A) + (offdiag.x * B) + (offdiag.y * C))/length);
543  ret[1] = -1.0f * (((offdiag.x * A) + (diag.y * B) + (offdiag.z * C))/length);
544  ret[2] = -1.0f * (((offdiag.y * A) + (offdiag.z * B) + (diag.z * C))/length);
545  // 3-5: partial derivative (diag offset wrt fitness fn) fn operated on sample
546  ret[3] = -1.0f * ((sample.x + offset.x) * A)/length;
547  ret[4] = -1.0f * ((sample.y + offset.y) * B)/length;
548  ret[5] = -1.0f * ((sample.z + offset.z) * C)/length;
549  // 6-8: partial derivative (off-diag offset wrt fitness fn) fn operated on sample
550  ret[6] = -1.0f * (((sample.y + offset.y) * A) + ((sample.x + offset.x) * B))/length;
551  ret[7] = -1.0f * (((sample.z + offset.z) * A) + ((sample.x + offset.x) * C))/length;
552  ret[8] = -1.0f * (((sample.z + offset.z) * B) + ((sample.y + offset.y) * C))/length;
553 }

555 void CompassCalibrator::run_ellipsoid_fit()
556 {
557  if(_sample_buffer == NULL) {
558  return;
559  }
560 
561  const float lma_damping = 10.0f;
562 
563 
564  float fitness = _fitness;
565  float fit1, fit2;
566  param_t fit1_params, fit2_params;
567  fit1_params = fit2_params = _params;
568 
569 
570  float JTJ[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS];
571  float JTJ2[COMPASS_CAL_NUM_ELLIPSOID_PARAMS*COMPASS_CAL_NUM_ELLIPSOID_PARAMS];
572  float JTFI[COMPASS_CAL_NUM_ELLIPSOID_PARAMS];
573 
574  memset(&JTJ,0,sizeof(JTJ));
575  memset(&JTJ2,0,sizeof(JTJ2));
576  memset(&JTFI,0,sizeof(JTFI));
577  // Gauss Newton Part common for all kind of extensions including LM
578  for(uint16_t k = 0; k<_samples_collected; k++) {
579  Vector3f sample = _sample_buffer[k].get();
580 
581  float ellipsoid_jacob[COMPASS_CAL_NUM_ELLIPSOID_PARAMS];
582 
583  calc_ellipsoid_jacob(sample, fit1_params, ellipsoid_jacob);
584 
585  for(uint8_t i = 0;i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
586  // compute JTJ
587  for(uint8_t j = 0; j < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; j++) {
588  JTJ [i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
589  JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+j] += ellipsoid_jacob[i] * ellipsoid_jacob[j];
590  }
591  // compute JTFI
592  JTFI[i] += ellipsoid_jacob[i] * calc_residual(sample, fit1_params);
593  }
594  }
595 
596 
597 
598  //------------------------Levenberg-Marquardt-part-starts-here---------------------------------//
599  //refer: http://en.wikipedia.org/wiki/Levenberg%E2%80%93Marquardt_algorithm#Choice_of_damping_parameter
600  for(uint8_t i = 0; i < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; i++) {
601  JTJ[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda;
602  JTJ2[i*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+i] += _ellipsoid_lambda/lma_damping;
603  }
604 
605  if(!inverse(JTJ, JTJ, 9)) {
606  return;
607  }
608 
609  if(!inverse(JTJ2, JTJ2, 9)) {
610  return;
611  }
612 
613  for(uint8_t row=0; row < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; row++) {
614  for(uint8_t col=0; col < COMPASS_CAL_NUM_ELLIPSOID_PARAMS; col++) {
615  fit1_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
616  fit2_params.get_ellipsoid_params()[row] -= JTFI[col] * JTJ2[row*COMPASS_CAL_NUM_ELLIPSOID_PARAMS+col];
617  }
618  }
619 
620  fit1 = calc_mean_squared_residuals(fit1_params);
621  fit2 = calc_mean_squared_residuals(fit2_params);
622 
623  if(fit1 > _fitness && fit2 > _fitness){
624  _ellipsoid_lambda *= lma_damping;
625  } else if(fit2 < _fitness && fit2 < fit1) {
626  _ellipsoid_lambda /= lma_damping;
627  fit1_params = fit2_params;
628  fitness = fit2;
629  } else if(fit1 < _fitness){
630  fitness = fit1;
631  }
632  //--------------------Levenberg-part-ends-here--------------------------------//
633 
634  if(fitness < _fitness) {
635  _fitness = fitness;
636  _params = fit1_params;
637  }
638 }
