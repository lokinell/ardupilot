77 bool AP_Limit_Geofence::triggered() {
78 
79  // reset trigger before checking
80  _triggered = false;
81 
82  // never trigger while disabled
83  if (!_enabled) return false;
84 
85  // if Geofence is required and we don't know where we are, trigger.
86  if (_required && (!_gps || !_gps->fix || !_home || !_current_loc)) {
87 
88  // TRIGGER
89  _triggered = true;
90  }
91 
92  uint32_t distance;
93 
94  // simple mode, pointers to current and home exist.
95  if (_simple && _current_loc && _home) {
96  distance = (uint32_t) get_distance(*_current_loc, *_home);
97  if (distance > 0 && distance > (uint16_t) _radius) {
98 
99  // TRIGGER
100  _triggered = true;
101  }
102  }
103  else {
104 
105  // COMPLEX GEOFENCE mode
106 
107  // check boundary and update if necessary
108  if (!_boundary_uptodate) {
109  update_boundary();
110  }
111 
112  // if boundary is correct, and current_loc exists check if we
113  // are inside the fence.
114  if (boundary_correct() && _current_loc) {
115  Vector2l location;
116  location.x = _current_loc->lat;
117  location.y = _current_loc->lng;
118  // trigger if outside
119  if (Polygon_outside(location, &_boundary[1], _fence_total-1)) {
120  // TRIGGER
121  _triggered = true;
122  }
123  } else {
124  // boundary incorrect
125  // If geofence is required and our boundary fence is incorrect,
126  // we trigger.
127  if (_required) {
128  // TRIGGER
129  _triggered = true;
130  }
131  }
132  }
133  return _triggered;
134 }
