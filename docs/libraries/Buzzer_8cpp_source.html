44 // update - updates led according to timed_updated. Should be called at 50Hz
45 void Buzzer::update()
46 {
47  // return immediately if disabled
48  if (!AP_Notify::flags.external_leds) {
49  return;
50  }
51 
52  // check for arming failed event
53  if (AP_Notify::events.arming_failed) {
54  // arming failed buzz
55  play_pattern(SINGLE_BUZZ);
56  }
57 
58  // reduce 50hz call down to 10hz
59  _counter++;
60  if (_counter < 5) {
61  return;
62  }
63  _counter = 0;
64 
65  // complete currently played pattern
66  if (_pattern != NONE) {
67  _pattern_counter++;
68  switch (_pattern) {
69  case SINGLE_BUZZ:
70  // buzz for 10th of a second
71  if (_pattern_counter == 1) {
72  on(true);
73  }else{
74  on(false);
75  _pattern = NONE;
76  }
77  return;
78  case DOUBLE_BUZZ:
79  // buzz for 10th of a second
80  switch (_pattern_counter) {
81  case 1:
82  on(true);
83  break;
84  case 2:
85  on(false);
86  break;
87  case 3:
88  on(true);
89  break;
90  case 4:
91  default:
92  on(false);
93  _pattern = NONE;
94  break;
95  }
96  return;
97  case ARMING_BUZZ:
98  // record start time
99  if (_pattern_counter == 1) {
100  _arming_buzz_start_ms = hal.scheduler->millis();
101  on(true);
102  } else {
103  // turn off buzzer after 3 seconds
104  if (hal.scheduler->millis() - _arming_buzz_start_ms >= BUZZER_ARMING_BUZZ_MS) {
105  _arming_buzz_start_ms = 0;
106  on(false);
107  _pattern = NONE;
108  }
109  }
110  return;
111  case BARO_GLITCH:
112  // four fast tones
113  switch (_pattern_counter) {
114  case 1:
115  case 3:
116  case 5:
117  case 7:
118  case 9:
119  on(true);
120  break;
121  case 2:
122  case 4:
123  case 6:
124  case 8:
125  on(false);
126  break;
127  case 10:
128  on(false);
129  _pattern = NONE;
130  break;
131  default:
132  // do nothing
133  break;
134  }
135  return;
136  case EKF_BAD:
137  // four tones getting shorter)
138  switch (_pattern_counter) {
139  case 1:
140  case 5:
141  case 8:
142  case 10:
143  on(true);
144  break;
145  case 4:
146  case 7:
147  case 9:
148  on(false);
149  break;
150  case 11:
151  on(false);
152  _pattern = NONE;
153  break;
154  default:
155  // do nothing
156  break;
157  }
158  return;
159  default:
160  // do nothing
161  break;
162  }
163  }
164 
165  // check if armed status has changed
166  if (_flags.armed != AP_Notify::flags.armed) {
167  _flags.armed = AP_Notify::flags.armed;
168  if (_flags.armed) {
169  // double buzz when armed
170  play_pattern(ARMING_BUZZ);
171  }else{
172  // single buzz when disarmed
173  play_pattern(SINGLE_BUZZ);
174  }
175  return;
176  }
177 
178  // check ekf bad
179  if (_flags.ekf_bad != AP_Notify::flags.ekf_bad) {
180  _flags.ekf_bad = AP_Notify::flags.ekf_bad;
181  if (_flags.ekf_bad) {
182  // ekf bad warning buzz
183  play_pattern(EKF_BAD);
184  }
185  return;
186  }
187 
188  // if battery failsafe constantly single buzz
189  if (AP_Notify::flags.failsafe_battery) {
190  play_pattern(SINGLE_BUZZ);
191  }
192 }
