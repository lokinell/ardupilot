61 // output - update value to send to ESC/Servo
62 void AP_MotorsHeli_RSC::output(RotorControlState state)
63 {
64  switch (state){
65  case ROTOR_CONTROL_STOP:
66  // set rotor ramp to decrease speed to zero, this happens instantly inside update_rotor_ramp()
67  update_rotor_ramp(0.0f);
68 
69  // control output forced to zero
70  _control_output = 0;
71  break;
72 
73  case ROTOR_CONTROL_IDLE:
74  // set rotor ramp to decrease speed to zero
75  update_rotor_ramp(0.0f);
76 
77  // set rotor control speed to idle speed parameter, this happens instantly and ignore ramping
78  _control_output = _idle_output;
79  break;
80 
81  case ROTOR_CONTROL_ACTIVE:
82  // set main rotor ramp to increase to full speed
83  update_rotor_ramp(1.0f);
84 
85  if ((_control_mode == ROTOR_CONTROL_MODE_SPEED_PASSTHROUGH) || (_control_mode == ROTOR_CONTROL_MODE_SPEED_SETPOINT)) {
86  // set control rotor speed to ramp slewed value between idle and desired speed
87  _control_output = _idle_output + (_rotor_ramp_output * (_desired_speed - _idle_output));
88  } else if (_control_mode == ROTOR_CONTROL_MODE_OPEN_LOOP_POWER_OUTPUT) {
89  // throttle output depending on estimated power demand. Output is ramped up from idle speed during rotor runup.
90  _control_output = _idle_output + (_rotor_ramp_output * ((_power_output_low - _idle_output) + (_power_output_range * _load_feedforward)));
91  }
92  break;
93  }
94 
95  // update rotor speed run-up estimate
96  update_rotor_runup();
97 
98  // output to rsc servo
99  write_rsc(_control_output);
100 }

102 // update_rotor_ramp - slews rotor output scalar between 0 and 1, outputs float scalar to _rotor_ramp_output
103 void AP_MotorsHeli_RSC::update_rotor_ramp(float rotor_ramp_input)
104 {
105  // ramp output upwards towards target
106  if (_rotor_ramp_output < rotor_ramp_input) {
107  // allow control output to jump to estimated speed
108  if (_rotor_ramp_output < _rotor_runup_output) {
109  _rotor_ramp_output = _rotor_runup_output;
110  }
111  // ramp up slowly to target
112  _rotor_ramp_output += _ramp_increment;
113  if (_rotor_ramp_output > rotor_ramp_input) {
114  _rotor_ramp_output = rotor_ramp_input;
115  }
116  }else{
117  // ramping down happens instantly
118  _rotor_ramp_output = rotor_ramp_input;
119  }
120 }

122 // update_rotor_runup - function to slew rotor runup scalar, outputs float scalar to _rotor_runup_ouptut
123 void AP_MotorsHeli_RSC::update_rotor_runup()
124 {
125  // ramp speed estimate towards control out
126  if (_rotor_runup_output < _rotor_ramp_output) {
127  _rotor_runup_output += _runup_increment;
128  if (_rotor_runup_output > _rotor_ramp_output) {
129  _rotor_runup_output = _rotor_ramp_output;
130  }
131  }else{
132  _rotor_runup_output -= _runup_increment;
133  if (_rotor_runup_output < _rotor_ramp_output) {
134  _rotor_runup_output = _rotor_ramp_output;
135  }
136  }
137 
138  // update run-up complete flag
139 
140  // if control mode is disabled, then run-up complete always returns true
141  if ( _control_mode == ROTOR_CONTROL_MODE_DISABLED ){
142  _runup_complete = true;
143  return;
144  }
145 
146  // if rotor ramp and runup are both at full speed, then run-up has been completed
147  if (!_runup_complete && (_rotor_ramp_output >= 1.0f) && (_rotor_runup_output >= 1.0f)) {
148  _runup_complete = true;
149  }
150  // if rotor speed is less than critical speed, then run-up is not complete
151  // this will prevent the case where the target rotor speed is less than critical speed
152  if (_runup_complete && (get_rotor_speed() <= _critical_speed)) {
153  _runup_complete = false;
154  }
155 }
