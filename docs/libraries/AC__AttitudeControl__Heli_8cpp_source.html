127 // rate_bf_to_motor_roll_pitch - ask the rate controller to calculate the motor outputs to achieve the target rate in centi-degrees / second
128 void AC_AttitudeControl_Heli::rate_bf_to_motor_roll_pitch(float rate_roll_target_cds, float rate_pitch_target_cds)
129 {
130  float roll_pd, roll_i, roll_ff; // used to capture pid values
131  float pitch_pd, pitch_i, pitch_ff; // used to capture pid values
132  float rate_roll_error, rate_pitch_error; // simply target_rate - current_rate
133  float roll_out, pitch_out;
134  const Vector3f& gyro = _ahrs.get_gyro(); // get current rates
135 
136  // calculate error
137  rate_roll_error = rate_roll_target_cds - gyro.x * AC_ATTITUDE_CONTROL_DEGX100;
138  rate_pitch_error = rate_pitch_target_cds - gyro.y * AC_ATTITUDE_CONTROL_DEGX100;
139 
140  // input to PID controller
141  _pid_rate_roll.set_input_filter_all(rate_roll_error);
142  _pid_rate_roll.set_desired_rate(rate_roll_target_cds);
143  _pid_rate_pitch.set_input_filter_all(rate_pitch_error);
144  _pid_rate_pitch.set_desired_rate(rate_pitch_target_cds);
145 
146  // call p and d controllers
147  roll_pd = _pid_rate_roll.get_p() + _pid_rate_roll.get_d();
148  pitch_pd = _pid_rate_pitch.get_p() + _pid_rate_pitch.get_d();
149 
150  // get roll i term
151  roll_i = _pid_rate_roll.get_integrator();
152 
153  // update i term as long as we haven't breached the limits or the I term will certainly reduce
154  if (!_flags_heli.limit_roll || ((roll_i>0&&rate_roll_error<0)||(roll_i<0&&rate_roll_error>0))){
155  if (_flags_heli.leaky_i){
156  roll_i = ((AC_HELI_PID&)_pid_rate_roll).get_leaky_i(AC_ATTITUDE_HELI_RATE_INTEGRATOR_LEAK_RATE);
157  }else{
158  roll_i = _pid_rate_roll.get_i();
159  }
160  }
161 
162  // get pitch i term
163  pitch_i = _pid_rate_pitch.get_integrator();
164 
165  // update i term as long as we haven't breached the limits or the I term will certainly reduce
166  if (!_flags_heli.limit_pitch || ((pitch_i>0&&rate_pitch_error<0)||(pitch_i<0&&rate_pitch_error>0))){
167  if (_flags_heli.leaky_i) {
168  pitch_i = ((AC_HELI_PID&)_pid_rate_pitch).get_leaky_i(AC_ATTITUDE_HELI_RATE_INTEGRATOR_LEAK_RATE);
169  }else{
170  pitch_i = _pid_rate_pitch.get_i();
171  }
172  }
173 
174  roll_ff = roll_feedforward_filter.apply(((AC_HELI_PID&)_pid_rate_roll).get_vff(rate_roll_target_cds), _dt);
175  pitch_ff = pitch_feedforward_filter.apply(((AC_HELI_PID&)_pid_rate_pitch).get_vff(rate_pitch_target_cds), _dt);
176 
177  // add feed forward and final output
178  roll_out = roll_pd + roll_i + roll_ff;
179  pitch_out = pitch_pd + pitch_i + pitch_ff;
180 
181  // constrain output and update limit flags
182  if (fabsf(roll_out) > AC_ATTITUDE_RATE_RP_CONTROLLER_OUT_MAX) {
183  roll_out = constrain_float(roll_out,-AC_ATTITUDE_RATE_RP_CONTROLLER_OUT_MAX,AC_ATTITUDE_RATE_RP_CONTROLLER_OUT_MAX);
184  _flags_heli.limit_roll = true;
185  }else{
186  _flags_heli.limit_roll = false;
187  }
188  if (fabsf(pitch_out) > AC_ATTITUDE_RATE_RP_CONTROLLER_OUT_MAX) {
189  pitch_out = constrain_float(pitch_out,-AC_ATTITUDE_RATE_RP_CONTROLLER_OUT_MAX,AC_ATTITUDE_RATE_RP_CONTROLLER_OUT_MAX);
190  _flags_heli.limit_pitch = true;
191  }else{
192  _flags_heli.limit_pitch = false;
193  }
194 
195  // output to motors
196  _motors.set_roll(roll_out);
197  _motors.set_pitch(pitch_out);
198 
199  // Piro-Comp, or Pirouette Compensation is a pre-compensation calculation, which basically rotates the Roll and Pitch Rate I-terms as the
200  // helicopter rotates in yaw. Much of the built-up I-term is needed to tip the disk into the incoming wind. Fast yawing can create an instability
201  // as the built-up I-term in one axis must be reduced, while the other increases. This helps solve that by rotating the I-terms before the error occurs.
202  // It does assume that the rotor aerodynamics and mechanics are essentially symmetrical about the main shaft, which is a generally valid assumption.
203  if (_piro_comp_enabled){
204 
205  int32_t piro_roll_i, piro_pitch_i; // used to hold I-terms while doing piro comp
206 
207  piro_roll_i = roll_i;
208  piro_pitch_i = pitch_i;
209 
210  Vector2f yawratevector;
211  yawratevector.x = cosf(-_ahrs.get_gyro().z * _dt);
212  yawratevector.y = sinf(-_ahrs.get_gyro().z * _dt);
213  yawratevector.normalize();
214 
215  roll_i = piro_roll_i * yawratevector.x - piro_pitch_i * yawratevector.y;
216  pitch_i = piro_pitch_i * yawratevector.x + piro_roll_i * yawratevector.y;
217 
218  _pid_rate_pitch.set_integrator(pitch_i);
219  _pid_rate_roll.set_integrator(roll_i);
220  }
221 
222 }

224 // rate_bf_to_motor_yaw - ask the rate controller to calculate the motor outputs to achieve the target rate in centi-degrees / second
225 float AC_AttitudeControl_Heli::rate_bf_to_motor_yaw(float rate_target_cds)
226 {
227  float pd,i,vff,aff; // used to capture pid values for logging
228  float current_rate; // this iteration's rate
229  float rate_error; // simply target_rate - current_rate
230  float yaw_out;
231 
232  // get current rate
233  // To-Do: make getting gyro rates more efficient?
234  current_rate = (_ahrs.get_gyro().z * AC_ATTITUDE_CONTROL_DEGX100);
235 
236  // calculate error and call pid controller
237  rate_error = rate_target_cds - current_rate;
238 
239  // send input to PID controller
240  _pid_rate_yaw.set_input_filter_all(rate_error);
241  _pid_rate_yaw.set_desired_rate(rate_target_cds);
242 
243  // get p and d
244  pd = _pid_rate_yaw.get_p() + _pid_rate_yaw.get_d();
245 
246  // get i term
247  i = _pid_rate_yaw.get_integrator();
248 
249  // update i term as long as we haven't breached the limits or the I term will certainly reduce
250  if (!_flags_heli.limit_yaw || ((i>0&&rate_error<0)||(i<0&&rate_error>0))) {
251  if (((AP_MotorsHeli&)_motors).rotor_runup_complete()) {
252  i = _pid_rate_yaw.get_i();
253  } else {
254  i = ((AC_HELI_PID&)_pid_rate_yaw).get_leaky_i(AC_ATTITUDE_HELI_RATE_INTEGRATOR_LEAK_RATE); // If motor is not running use leaky I-term to avoid excessive build-up
255  }
256  }
257 
258  vff = yaw_velocity_feedforward_filter.apply(((AC_HELI_PID&)_pid_rate_yaw).get_vff(rate_target_cds), _dt);
259  aff = yaw_acceleration_feedforward_filter.apply(((AC_HELI_PID&)_pid_rate_yaw).get_aff(rate_target_cds), _dt);
260 
261  // add feed forward
262  yaw_out = pd + i + vff + aff;
263 
264  // constrain output and update limit flag
265  if (fabsf(yaw_out) > AC_ATTITUDE_RATE_YAW_CONTROLLER_OUT_MAX) {
266  yaw_out = constrain_float(yaw_out,-AC_ATTITUDE_RATE_YAW_CONTROLLER_OUT_MAX,AC_ATTITUDE_RATE_YAW_CONTROLLER_OUT_MAX);
267  _flags_heli.limit_yaw = true;
268  }else{
269  _flags_heli.limit_yaw = false;
270  }
271 
272  // output to motors
273  return yaw_out;
274 }
