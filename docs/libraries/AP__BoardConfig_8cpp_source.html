104 void AP_BoardConfig::init()
105 {
106 #if CONFIG_HAL_BOARD == HAL_BOARD_PX4
107  /* configurre the FMU driver for the right number of PWMs */
108 
109  // ensure only valid values are set, rounding up
110  if (_pwm_count > 6) _pwm_count.set(6);
111  if (_pwm_count < 0) _pwm_count.set(0);
112  if (_pwm_count == 1) _pwm_count.set(2);
113  if (_pwm_count == 3) _pwm_count.set(4);
114  if (_pwm_count == 5) _pwm_count.set(6);
115 
116  int fd = open("/dev/px4fmu", 0);
117  if (fd == -1) {
118  hal.scheduler->panic("Unable to open /dev/px4fmu");
119  }
120  if (ioctl(fd, PWM_SERVO_SET_COUNT, _pwm_count.get()) != 0) {
121  hal.console->printf("RCOutput: Unable to setup alt PWM to %u channels\n", _pwm_count.get());
122  }
123  close(fd);
124 
125  hal.uartC->set_flow_control((AP_HAL::UARTDriver::flow_control)_ser1_rtscts.get());
126  if (hal.uartD != NULL) {
127  hal.uartD->set_flow_control((AP_HAL::UARTDriver::flow_control)_ser2_rtscts.get());
128  }
129 
130  if (_safety_enable.get() == 0) {
131  hal.rcout->force_safety_off();
132  }
133 
134  if (_sbus_out_enable.get() == 1) {
135  fd = open("/dev/px4io", 0);
136  if (fd == -1 || ioctl(fd, SBUS_SET_PROTO_VERSION, 1) != 0) {
137  hal.console->printf("SBUS: Unable to setup SBUS output\n");
138  }
139  if (fd != -1) {
140  close(fd);
141  }
142  }
143 
144 #if !defined(CONFIG_ARCH_BOARD_PX4FMU_V1)
145  if (_can_enable == 1) {
146  const char *args[] = { "uavcan", "start", NULL };
147  int ret = uavcan_main(3, args);
148  if (ret != 0) {
149  hal.console->printf("UAVCAN: failed to start\n");
150  } else {
151  hal.console->printf("UAVCAN: started\n");
152  // give some time for CAN bus initialisation
153  hal.scheduler->delay(500);
154  }
155  }
156 #endif
157 
158 #elif CONFIG_HAL_BOARD == HAL_BOARD_VRBRAIN
159  /* configure the VRBRAIN driver for the right number of PWMs */
160 
161 #endif
162 }
