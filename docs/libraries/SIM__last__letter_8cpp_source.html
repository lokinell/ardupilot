101 /*
102  receive an update from the FDM
103  This is a blocking function
104  */
105 void last_letter::recv_fdm(const struct sitl_input &input)
106 {
107  fdm_packet pkt;
108 
109  /*
110  we re-send the servo packet every 0.1 seconds until we get a
111  reply. This allows us to cope with some packet loss to the FDM
112  */
113  while (sock.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt)) {
114  send_servos(input);
115  }
116 
117  accel_body = Vector3f(pkt.xAccel, pkt.yAccel, pkt.zAccel);
118  gyro = Vector3f(pkt.rollRate, pkt.pitchRate, pkt.yawRate);
119  velocity_ef = Vector3f(pkt.speedN, pkt.speedE, pkt.speedD);
120  location.lat = pkt.latitude * 1.0e7;
121  location.lng = pkt.longitude * 1.0e7;
122  location.alt = pkt.altitude*1.0e2;
123  dcm.from_euler(pkt.roll, pkt.pitch, pkt.yaw);
124 
125  airspeed = pkt.airspeed;
126 
127  // auto-adjust to last_letter frame rate
128  uint64_t deltat_us = pkt.timestamp_us - last_timestamp_us;
129  time_now_us += deltat_us;
130 
131  if (deltat_us < 1.0e4 && deltat_us > 0) {
132  adjust_frame_time(1.0e6/deltat_us);
133  }
134  last_timestamp_us = pkt.timestamp_us;
135 }
