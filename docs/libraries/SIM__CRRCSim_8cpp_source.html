45 /*
46  decode and send servos for heli
47 */
48 void CRRCSim::send_servos_heli(const struct sitl_input &input)
49 {
50  float swash1 = (input.servos[0]-1000) / 1000.0f;
51  float swash2 = (input.servos[1]-1000) / 1000.0f;
52  float swash3 = (input.servos[2]-1000) / 1000.0f;
53  float tail_rotor = (input.servos[3]-1000) / 1000.0f;
54  float rsc = (input.servos[7]-1000) / 1000.0f;
55 
56  float col_pitch = (swash1+swash2+swash3)/3.0 - 0.5f;
57  float roll_rate = (swash1 - swash2)/2;
58  float pitch_rate = -((swash1 + swash2)/2.0 - swash3)/2;
59  float yaw_rate = -(tail_rotor - 0.5);
60 
61  servo_packet pkt;
62  pkt.roll_rate = constrain_float(roll_rate, -0.5, 0.5);
63  pkt.pitch_rate = constrain_float(pitch_rate, -0.5, 0.5);
64  pkt.throttle = constrain_float(rsc, 0, 1);
65  pkt.yaw_rate = constrain_float(yaw_rate, -0.5, 0.5);
66  pkt.col_pitch = constrain_float(col_pitch, -0.5, 0.5);
67 
68  sock.sendto(&pkt, sizeof(pkt), "127.0.0.1", 9002);
69 }

103 /*
104  receive an update from the FDM
105  This is a blocking function
106  */
107 void CRRCSim::recv_fdm(const struct sitl_input &input)
108 {
109  fdm_packet pkt;
110 
111  /*
112  we re-send the servo packet every 0.1 seconds until we get a
113  reply. This allows us to cope with some packet loss to the FDM
114  */
115  while (sock.recv(&pkt, sizeof(pkt), 100) != sizeof(pkt)) {
116  send_servos(input);
117  }
118 
119  accel_body = Vector3f(pkt.xAccel, pkt.yAccel, pkt.zAccel);
120  gyro = Vector3f(pkt.rollRate, pkt.pitchRate, pkt.yawRate);
121  velocity_ef = Vector3f(pkt.speedN, pkt.speedE, pkt.speedD);
122 
123  Location loc1, loc2;
124  loc2.lat = pkt.latitude * 1.0e7;
125  loc2.lng = pkt.longitude * 1.0e7;
126  memset(&loc1, 0, sizeof(loc1));
127  Vector2f posdelta = location_diff(loc1, loc2);
128  position.x = posdelta.x;
129  position.y = posdelta.y;
130  position.z = -pkt.altitude;
131 
132  airspeed = pkt.airspeed;
133 
134  dcm.from_euler(pkt.roll, pkt.pitch, pkt.yaw);
135 
136  // auto-adjust to crrcsim frame rate
137  double deltat = pkt.timestamp - last_timestamp;
138  time_now_us += deltat * 1.0e6;
139 
140  if (deltat < 0.01 && deltat > 0) {
141  adjust_frame_time(1.0/deltat);
142  }
143  last_timestamp = pkt.timestamp;
144 }
