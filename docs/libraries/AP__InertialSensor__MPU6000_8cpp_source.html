268 void AP_MPU6000_BusDriver_SPI::read_data_transaction(uint8_t *samples,
269  AP_HAL::DigitalSource *_drdy_pin,
270  uint8_t &n_samples)
271 {
272  /* one register address followed by seven 2-byte registers */
273  struct PACKED {
274  uint8_t cmd;
275  uint8_t int_status;
276  uint8_t d[14];
277  } rx, tx = { cmd : MPUREG_INT_STATUS | 0x80, };
278 
279  _spi->transaction((const uint8_t *)&tx, (uint8_t *)&rx, sizeof(rx));
280 
281  /*
282  detect a bad SPI bus transaction by looking for all 14 bytes
283  zero. This can happen with some boards with hw that end up
284  needing a lower bus speed
285  */
286  uint8_t i;
287  for (i=0; i<14; i++) {
288  if (rx.d[i] != 0) break;
289  }
290  if (i == 14) {
291  // likely a bad bus transaction
292  if (++_error_count > 4) {
293  set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
294  }
295  }
296 
297  n_samples = 1;
298  /* remove cmd from data sample */
299  memcpy(&samples[0], &rx.d[0], 14);
300 
301  return;
302 }

364 void AP_MPU6000_BusDriver_I2C::read_data_transaction(uint8_t *samples,
365  AP_HAL::DigitalSource *_drdy_pin,
366  uint8_t &n_samples)
367 {
368  uint16_t bytes_read;
369  uint8_t ret = 0;
370 
371  ret = _i2c->readRegisters(_addr, MPUREG_FIFO_COUNTH, 2, _rx);
372  if(ret != 0) {
373  hal.console->printf("MPU6000: error in i2c read\n");
374  n_samples = 0;
375  return;
376  }
377 
378  bytes_read = uint16_val(_rx, 0);
379 
380  n_samples = bytes_read / MPU6000_SAMPLE_SIZE;
381 
382  if(n_samples > 3) {
383  hal.console->printf("bytes_read = %d, n_samples %d > 3, dropping samples\n",
384  bytes_read, n_samples);
385 
386  /* Too many samples, do a FIFO RESET */
387  write8(MPUREG_USER_CTRL, 0);
388  write8(MPUREG_USER_CTRL, BIT_USER_CTRL_FIFO_RESET | BIT_USER_CTRL_SIG_COND_RESET);
389  write8(MPUREG_USER_CTRL, BIT_USER_CTRL_FIFO_EN);
390  n_samples = 0;
391  return;
392  }
393  else if (n_samples == 0) {
394  /* Not enough data in FIFO */
395  return;
396  }
397  else {
398  ret = _i2c->readRegisters(_addr, MPUREG_FIFO_R_W, n_samples * MPU6000_SAMPLE_SIZE, _rx);
399  }
400 
401  if(ret != 0) {
402  hal.console->printf("MPU6000: error in i2c read %d bytes\n",
403  n_samples * MPU6000_SAMPLE_SIZE);
404  n_samples = 0;
405  return;
406  }
407 
408  memcpy(samples, _rx, n_samples * MPU6000_SAMPLE_SIZE);
409 
410  return;
411 }

689 void AP_InertialSensor_MPU6000::_accumulate(uint8_t *samples, uint8_t n_samples)
690 {
691  for(uint8_t i=0; i < n_samples; i++) {
692  uint8_t *data = samples + MPU6000_SAMPLE_SIZE * i;
693  Vector3f accel, gyro;
694  float temp;
695 
696  accel = Vector3f(int16_val(data, 1),
697  int16_val(data, 0),
698  -int16_val(data, 2));
699  accel *= MPU6000_ACCEL_SCALE_1G;
700 
701  gyro = Vector3f(int16_val(data, 5),
702  int16_val(data, 4),
703  -int16_val(data, 6));
704  gyro *= _gyro_scale;
705

707  /* scaling/offset values from the datasheet */
708  temp = temp/340 + 36.53;

710 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_PXF
711  accel.rotate(ROTATION_PITCH_180_YAW_90);
712  gyro.rotate(ROTATION_PITCH_180_YAW_90);

714  accel.rotate(ROTATION_YAW_270);
715  gyro.rotate(ROTATION_YAW_270);

717  accel.rotate(ROTATION_YAW_90);
718  gyro.rotate(ROTATION_YAW_90);
719 #endif
720 
721  _rotate_and_correct_accel(_accel_instance, accel);
722  _rotate_and_correct_gyro(_gyro_instance, gyro);

803 bool AP_InertialSensor_MPU6000::_hardware_init(void)
804 {
805  if (!_bus_sem->take(100)) {
806  hal.scheduler->panic("MPU6000: Unable to get semaphore");
807  }
808 
809  // initially run the bus at low speed
810  _bus->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
811 
812  // Chip reset
813  uint8_t tries;
814  for (tries = 0; tries<5; tries++) {
815  uint8_t user_ctrl = _register_read(MPUREG_USER_CTRL);
816 
817  /* First disable the master I2C to avoid hanging the slaves on the
818  * aulixiliar I2C bus - it will be enabled again if the AuxiliaryBus
819  * is used */
820  if (user_ctrl & BIT_USER_CTRL_I2C_MST_EN) {
821  _register_write(MPUREG_USER_CTRL, user_ctrl & ~BIT_USER_CTRL_I2C_MST_EN);
822  hal.scheduler->delay(10);
823  }
824 
825  /* reset device */
826  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
827  hal.scheduler->delay(100);
828 
829  /* bus-dependent initialization*/
830  _bus->init();
831 
832  // Wake up device and select GyroZ clock. Note that the
833  // MPU6000 starts up in sleep mode, and it can take some time
834  // for it to come out of sleep
835  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_ZGYRO);
836  hal.scheduler->delay(5);
837 
838  // check it has woken up
839  if (_register_read(MPUREG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_ZGYRO)
840  break;
841 
842  hal.scheduler->delay(10);
843  if (_data_ready())
844  break;
845 
846 #if MPU6000_DEBUG
847  _dump_registers();
848 #endif
849  }
850  if (tries == 5) {
851  hal.console->println("Failed to boot MPU6000 5 times");
852  goto fail_tries;
853  }
854 
855  _bus->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_HIGH);
856  _bus_sem->give();
857 
858  return true;
859 
860 fail_tries:
861  _bus->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_HIGH);
862  _bus_sem->give();
863  return false;
864 }
