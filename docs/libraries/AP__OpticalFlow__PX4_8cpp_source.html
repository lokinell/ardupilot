57 // update - read latest values from sensor and fill in x,y and totals.
58 void AP_OpticalFlow_PX4::update(void)
59 {
60  // return immediately if not initialised
61  if (_fd == -1) {
62  return;
63  }
64 
65  struct optical_flow_s report;
66  while (::read(_fd, &report, sizeof(optical_flow_s)) == sizeof(optical_flow_s) &&
67  report.timestamp != _last_timestamp) {
68  struct OpticalFlow::OpticalFlow_state state;
69  state.device_id = report.sensor_id;
70  state.surface_quality = report.quality;
71  if (report.integration_timespan > 0) {
72  float yawAngleRad = _yawAngleRad();
73  float cosYaw = cosf(yawAngleRad);
74  float sinYaw = sinf(yawAngleRad);
75  const Vector2f flowScaler = _flowScaler();
76  float flowScaleFactorX = 1.0f + 0.001f * flowScaler.x;
77  float flowScaleFactorY = 1.0f + 0.001f * flowScaler.y;
78  float integralToRate = 1e6f / float(report.integration_timespan);
79  // rotate sensor measurements from sensor to body frame through sensor yaw angle
80  state.flowRate.x = flowScaleFactorX * integralToRate * (cosYaw * float(report.pixel_flow_x_integral) - sinYaw * float(report.pixel_flow_y_integral)); // rad/sec measured optically about the X body axis
81  state.flowRate.y = flowScaleFactorY * integralToRate * (sinYaw * float(report.pixel_flow_x_integral) + cosYaw * float(report.pixel_flow_y_integral)); // rad/sec measured optically about the Y body axis
82  state.bodyRate.x = integralToRate * (cosYaw * float(report.gyro_x_rate_integral) - sinYaw * float(report.gyro_y_rate_integral)); // rad/sec measured inertially about the X body axis
83  state.bodyRate.y = integralToRate * (sinYaw * float(report.gyro_x_rate_integral) + cosYaw * float(report.gyro_y_rate_integral)); // rad/sec measured inertially about the Y body axis
84  } else {
85  state.flowRate.zero();
86  state.bodyRate.zero();
87  }
88  _last_timestamp = report.timestamp;
89 
90  _update_frontend(state);
91  }
92 }
