77  // The PX4Flow sensor outputs flow rates with the following axis and sign conventions:
78  // A positive X rate is produced by a positive sensor rotation about the X axis
79  // A positive Y rate is produced by a positive sensor rotation about the Y axis
80  // This filter uses a different definition of optical flow rates to the sensor with a positive optical flow rate produced by a
81  // negative rotation about that axis. For example a positive rotation of the flight vehicle about its X (roll) axis would produce a negative X flow rate
82  flowMeaTime_ms = imuSampleTime_ms;
83  // calculate bias errors on flow sensor gyro rates, but protect against spikes in data
84  // reset the accumulated body delta angle and time

86  if (delTimeOF > 0.01f) {
87  flowGyroBias.x = 0.99f * flowGyroBias.x + 0.01f * constrain_float((rawGyroRates.x - delAngBodyOF.x/delTimeOF),-0.1f,0.1f);
88  flowGyroBias.y = 0.99f * flowGyroBias.y + 0.01f * constrain_float((rawGyroRates.y - delAngBodyOF.y/delTimeOF),-0.1f,0.1f);
89  delAngBodyOF.zero();

91  }
92  // check for takeoff if relying on optical flow and zero measurements until takeoff detected
93  // if we haven't taken off - constrain position and velocity states
94  if (frontend->_fusionModeGPS == 3) {
95  detectOptFlowTakeoff();
96  }
97  // calculate rotation matrices at mid sample time for flow observations
98  stateStruct.quat.rotation_matrix(Tbn_flow);
99  Tnb_flow = Tbn_flow.transposed();

101  if ((rawFlowQuality > 0) && rawFlowRates.length() < 4.2f && rawGyroRates.length() < 4.2f) {
102  // correct flow sensor rates for bias

104  omegaAcrossFlowTime.y = rawGyroRates.y - flowGyroBias.y;

288  * for storage in the data buffers used by the EKF. If the IMU data arrives at

293 void NavEKF2_core::readIMUData()
294 {
295  const AP_InertialSensor &ins = _ahrs->get_ins();
296 
297  // average IMU sampling rate
298  dtIMUavg = 1.0f/ins.get_sample_rate();
299 
300  // the imu sample time is used as a common time reference throughout the filter
301  imuSampleTime_ms = hal.scheduler->millis();
302 
303  // use the nominated imu or primary if not available
304  if (ins.use_accel(imu_index)) {
305  readDeltaVelocity(imu_index, imuDataNew.delVel, imuDataNew.delVelDT);
306  } else {
307  readDeltaVelocity(ins.get_primary_accel(), imuDataNew.delVel, imuDataNew.delVelDT);
308  }
309 
310  // Get delta angle data from primary gyro or primary if not available
311  if (ins.use_gyro(imu_index)) {
312  readDeltaAngle(imu_index, imuDataNew.delAng);
313  } else {
314  readDeltaAngle(ins.get_primary_gyro(), imuDataNew.delAng);
315  }
316  imuDataNew.delAngDT = max(ins.get_delta_time(),1.0e-4f);
317 
318  // Get current time stamp
319  imuDataNew.time_ms = imuSampleTime_ms;
320 
321  // remove gyro scale factor errors
322  imuDataNew.delAng.x = imuDataNew.delAng.x * stateStruct.gyro_scale.x;
323  imuDataNew.delAng.y = imuDataNew.delAng.y * stateStruct.gyro_scale.y;
324  imuDataNew.delAng.z = imuDataNew.delAng.z * stateStruct.gyro_scale.z;
325 
326  // remove sensor bias errors
327  imuDataNew.delAng -= stateStruct.gyro_bias;
328  imuDataNew.delVel.z -= stateStruct.accel_zbias;
329

331  imuDataDownSampledNew.delAngDT += imuDataNew.delAngDT;
332  imuDataDownSampledNew.delVelDT += imuDataNew.delVelDT;
333 
334  // Rotate quaternon atitude from previous to new and normalise.
335  // Accumulation using quaternions prevents introduction of coning errors due to downsampling
336  Quaternion deltaQuat;
337  deltaQuat.rotate(imuDataNew.delAng);
338  imuQuatDownSampleNew = imuQuatDownSampleNew*deltaQuat;
339  imuQuatDownSampleNew.normalize();
340

342  // This prevents introduction of sculling errors due to downsampling
343  Matrix3f deltaRotMat;
344  deltaQuat.inverse().rotation_matrix(deltaRotMat);
345  imuDataDownSampledNew.delVel = deltaRotMat*imuDataDownSampledNew.delVel;
346

348  imuDataDownSampledNew.delVel += imuDataNew.delVel;
349 
350  // Keep track of the number of IMU frames since the last state prediction
351  framesSincePredict++;
352

354  // to be used by the state prediction, ignoring the frontend permission if more than 20msec has lapsed

356  // convert the accumulated quaternion to an equivalent delta angle
357  imuQuatDownSampleNew.to_axis_angle(imuDataDownSampledNew.delAng);
358  // Time stamp the data
359  imuDataDownSampledNew.time_ms = imuSampleTime_ms;

361  StoreIMU();
362  // zero the accumulated IMU data and quaternion
363  imuDataDownSampledNew.delAng.zero();
364  imuDataDownSampledNew.delVel.zero();
365  imuDataDownSampledNew.delAngDT = 0.0f;
366  imuDataDownSampledNew.delVelDT = 0.0f;
367  imuQuatDownSampleNew[0] = 1.0f;
368  imuQuatDownSampleNew[3] = imuQuatDownSampleNew[2] = imuQuatDownSampleNew[1] = 0.0f;
369  // reset the counter used to let the frontend know how many frames have elapsed since we started a new update cycle
370  framesSincePredict = 0;
371  // set the flag to let the filter know it has new IMU data nad needs to run
372  runUpdates = true;
373  } else {
374  // we don't have new IMU data in the buffer so don't run filter updates on this time step
375  runUpdates = false;
376  }
377 
378  // extract the oldest available data from the FIFO buffer
379  imuDataDelayed = storedIMU[fifoIndexDelayed];

559  velTimeout = true;

563  // If we don't have airspeed or sideslip assumption or optical flow to constrain drift, then go into constant position mode.
564  // If we can do optical flow nav (valid flow data and height above ground estimate), then go into flow nav mode.
565  if (PV_AidingMode == AID_ABSOLUTE && !useAirspeed() && !assume_zero_sideslip()) {

648 // check for new altitude measurement data and update stored measurement if available
649 void NavEKF2_core::readHgtData()
650 {
651  // check to see if baro measurement has changed so we know if a new measurement has arrived
652  // do not accept data at a faster rate than 14Hz to avoid overflowing the FIFO buffer
653  if (frontend->_baro.get_last_update() - lastHgtReceived_ms > 70) {
654  // Don't use Baro height if operating in optical flow mode as we use range finder instead
655  if (frontend->_fusionModeGPS == 3 && frontend->_altSource == 1) {
656  if ((imuSampleTime_ms - rngValidMeaTime_ms) < 2000) {
657  // adjust range finder measurement to allow for effect of vehicle tilt and height of sensor
658  baroDataNew.hgt = max(rngMea * Tnb_flow.c.z, rngOnGnd);
659  // calculate offset to baro data that enables baro to be used as a backup
660  // filter offset to reduce effect of baro noise and other transient errors on estimate
661  baroHgtOffset = 0.1f * (frontend->_baro.get_altitude() + stateStruct.position.z) + 0.9f * baroHgtOffset;
662  } else if (isAiding && takeOffDetected) {
663  // we have lost range finder measurements and are in optical flow flight
664  // use baro measurement and correct for baro offset - failsafe use only as baro will drift
665  baroDataNew.hgt = max(frontend->_baro.get_altitude() - baroHgtOffset, rngOnGnd);
666  } else {
667  // If we are on ground and have no range finder reading, assume the nominal on-ground height
668  baroDataNew.hgt = rngOnGnd;
669  // calculate offset to baro data that enables baro to be used as a backup
670  // filter offset to reduce effect of baro noise and other transient errors on estimate
671  baroHgtOffset = 0.1f * (frontend->_baro.get_altitude() + stateStruct.position.z) + 0.9f * baroHgtOffset;
672  }
673  } else {
674  // Normal operation is to use baro measurement
675  baroDataNew.hgt = frontend->_baro.get_altitude();
676  }
677 
678  // filtered baro data used to provide a reference for takeoff
679  // it is is reset to last height measurement on disarming in performArmingChecks()
680  if (!getTakeoffExpected()) {
681  const float gndHgtFiltTC = 0.5f;
682  const float dtBaro = frontend->hgtAvg_ms*1.0e-3f;
683  float alpha = constrain_float(dtBaro / (dtBaro+gndHgtFiltTC),0.0f,1.0f);
684  meaHgtAtTakeOff += (baroDataDelayed.hgt-meaHgtAtTakeOff)*alpha;
685  } else if (isAiding && getTakeoffExpected()) {
686  // If we are in takeoff mode, the height measurement is limited to be no less than the measurement at start of takeoff
687  // This prevents negative baro disturbances due to copter downwash corrupting the EKF altitude during initial ascent
688  baroDataNew.hgt = max(baroDataNew.hgt, meaHgtAtTakeOff);
689  }
690 
691  // time stamp used to check for new measurement
692  lastHgtReceived_ms = frontend->_baro.get_last_update();
693 
694  // estimate of time height measurement was taken, allowing for delays
695  baroDataNew.time_ms = lastHgtReceived_ms - frontend->_hgtDelay_ms;
696 
697  // Correct for the average intersampling delay due to the filter updaterate
698  baroDataNew.time_ms -= localFilterTimeStep_ms/2;
699 
700  // Prevent time delay exceeding age of oldest IMU data in the buffer
701  baroDataNew.time_ms = max(baroDataNew.time_ms,imuDataDelayed.time_ms);
702 
703  // save baro measurement to buffer to be fused later
704  StoreBaro();
705  }
