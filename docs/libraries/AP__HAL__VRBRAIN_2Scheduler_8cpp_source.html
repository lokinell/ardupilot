109 void VRBRAINScheduler::delay_microseconds(uint16_t usec)
110 {
111  perf_begin(_perf_delay);
112  if (usec >= 500) {
113  delay_microseconds_semaphore(usec);

115  return;
116  }

118  uint64_t dt;

120  up_udelay(usec - dt);
