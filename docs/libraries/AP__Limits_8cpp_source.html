150 bool AP_Limits::check_triggered(bool only_required) {
151 
152  // check all enabled modules for triggered limits
153  AP_Limit_Module *mod = _modules_head;
154 
155  // reset bit fields
156  mods_triggered = 0;
157  mods_enabled = 0;
158  mods_required = 0;
159 
160  while (mod) {
161 
162  unsigned module_id = mod->get_module_id();
163 
164  // We check enabled, triggered and required across all modules
165  // We set all the bit-fields each time we check, keeping them up to date
166 
167  if (mod->enabled()) {
168  mods_enabled |= module_id;
169 
170  if (mod->required()) mods_required |= module_id;
171  if (mod->triggered()) mods_triggered |= module_id;
172  }
173 
174  mod = mod->next();
175  }
176 
177  if (only_required) {
178  // just modules that are both required AND triggered. (binary AND)
179  return (mods_required & mods_triggered) != 0;
180  } else {
181  return mods_triggered != 0;
182  }
183 }
