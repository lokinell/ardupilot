68 float AP_BattMonitor_Bebop::_filter_voltage(float vbat_raw)
69 {
70  static const float a[2] = {
71  1.0f, -9.9686333183343789e-01f
72  };
73  static const float b[2] = {
74  1.5683340832810533e-03f, 1.5683340832810533e-03f
75  };
76  float vbat;
77  static int only_once = 1;
78 
79  /* on first time reset filter with first raw value */
80  if (only_once) {
81  vbat = vbat_raw;
82  _prev_vbat_raw = vbat_raw;
83  _prev_vbat = vbat_raw;
84  only_once = 0;
85  } else if (vbat_raw > 0.0f) {
86  /* 1st order fitler */
87  vbat = b[0] * vbat_raw +
88  b[1] * _prev_vbat_raw - a[1] * _prev_vbat;
89  _prev_vbat_raw = vbat_raw;
90  _prev_vbat = vbat;
91  } else {
92  vbat = _prev_vbat;
93  }
94 
95  return vbat;
96 }

98 float AP_BattMonitor_Bebop::_compute_compensation(const uint16_t *rpm,
99  float vbat_raw)
100 {
101  float vbat, res;
102  size_t i, j;
103 
104  vbat = vbat_raw;
105  for (i = 0; i < BEBOP_BLDC_MOTORS_NUM; i++) {
106  res = 0;
107  for (j = 0; j < ARRAY_SIZE(bat_comp_polynomial_coeffs); j++)
108  res = res * rpm[i] + bat_comp_polynomial_coeffs[j];
109 
110  vbat -= res;
111  }
112 
113  return vbat;
114 }

116 float AP_BattMonitor_Bebop::_compute_battery_percentage(float vbat)
117 {
118  float percent = 0.0f;
119  int i;
120 
121  if (vbat <= bat_lut[0].voltage) {
122  percent = 0.0f;
123  } else if (vbat >= bat_lut[BATTERY_PERCENT_LUT_SIZE - 1].voltage) {
124  percent = 100.0f;
125  } else {
126  i = 0;
127  while (vbat >= bat_lut[i].voltage)
128  i++;
129 
130  percent += bat_lut[i - 1].percent +
131  (vbat - bat_lut[i - 1].voltage) *
132  (bat_lut[i].percent - bat_lut[i - 1].percent) /
133  (bat_lut[i].voltage - bat_lut[i - 1].voltage);
134  }
135 
136  return percent;
137 }

139 void AP_BattMonitor_Bebop::read(void)
140 {
141  int ret;
142  uint32_t tnow;
143  BebopBLDC_ObsData data;
144  float remaining, vbat, vbat_raw;
145 
146  auto rcout = Linux::RCOutput_Bebop::from(hal.rcout);
147  tnow = hal.scheduler->micros();
148 
149  ret = rcout->read_obs_data(data);
150  if (ret < 0) {
151  _state.healthy = false;
152  return;
153  }
154 
155  /* get battery voltage observed by cypress */
156  vbat_raw = (float)data.batt_mv / 1000.0f;
157 
158  /* do not compute battery status on ramping or braking transition */
159  if (data.status == BEBOP_BLDC_STATUS_RAMPING ||
160  data.status == BEBOP_BLDC_STATUS_STOPPING)
161  return;
162 
163  /* if motors are spinning compute polynomial compensation */
164  if (data.status == BEBOP_BLDC_STATUS_SPINNING_1 ||
165  data.status == BEBOP_BLDC_STATUS_SPINNING_2) {
166  vbat = _compute_compensation(data.rpm, vbat_raw);
167  /* otherwise compute constant compensation */
168  } else {
169  vbat = vbat_raw - BATTERY_VOLTAGE_COMPENSATION_LANDED;
170  }
171 
172  /* filter raw value */
173  vbat = _filter_voltage(vbat);
174 
175  /* ensure battery voltage/percent will not grow up during use */
176  if (vbat > _battery_voltage_max) {
177  vbat = _battery_voltage_max;
178  } else if (vbat < 0.0f) {
179  vbat = 0.0f;
180  _battery_voltage_max = 0.0f;
181  } else {
182  _battery_voltage_max = vbat;
183  }
184 
185  /* compute remaining battery percent */
186  remaining = _compute_battery_percentage(vbat);
187 
188  /* fillup battery state */
189  _state.voltage = vbat;
190  _state.last_time_micros = tnow;
191  _state.healthy = true;
192  _state.current_total_mah = BATTERY_CAPACITY -
193  (remaining * BATTERY_CAPACITY) / 100.0f;
194 }
