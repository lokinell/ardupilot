164  apply a reading in ADC counts

166 void VRBRAINAnalogSource::_add_value(float v, float vcc5V)
167 {
168  _latest_value = v;
169  _sum_value += v;
170  if (vcc5V < 3.0f) {
171  _sum_ratiometric += v;
172  } else {
173  // this compensates for changes in the 5V rail relative to the
174  // 3.3V reference used by the ADC.
175  _sum_ratiometric += v * 5.0f / vcc5V;
176  }
177  _sum_count++;
178  if (_sum_count == 254) {
179  _sum_value /= 2;
180  _sum_ratiometric /= 2;
181  _sum_count /= 2;
182  }
183 }

206  move to the next stop pin
207  */
208 void VRBRAINAnalogIn::next_stop_pin(void)
209 {
210  // find the next stop pin. We start one past the current stop pin
211  // and wrap completely, so we do the right thing is there is only
212  // one stop pin
213  for (uint8_t i=1; i <= VRBRAIN_ANALOG_MAX_CHANNELS; i++) {
214  uint8_t idx = (_current_stop_pin_i + i) % VRBRAIN_ANALOG_MAX_CHANNELS;
215  VRBRAIN::VRBRAINAnalogSource *c = _channels[idx];
216  if (c && c->_stop_pin != -1) {
217  // found another stop pin
218  _stop_pin_change_time = hal.scheduler->millis();
219  _current_stop_pin_i = idx;
220 
221  // set that pin high
222  hal.gpio->pinMode(c->_stop_pin, 1);
223  hal.gpio->write(c->_stop_pin, 1);
224 
225  // set all others low
226  for (uint8_t j=0; j<VRBRAIN_ANALOG_MAX_CHANNELS; j++) {
227  VRBRAIN::VRBRAINAnalogSource *c2 = _channels[j];
228  if (c2 && c2->_stop_pin != -1 && j != idx) {
229  hal.gpio->pinMode(c2->_stop_pin, 1);
230  hal.gpio->write(c2->_stop_pin, 0);
231  }
232  }
233  break;
234  }
235  }

239  called at 1kHz
240  */
241 void VRBRAINAnalogIn::_timer_tick(void)
242 {
243  // read adc at 100Hz
244  uint32_t now = hal.scheduler->micros();
245  uint32_t delta_t = now - _last_run;
246  if (delta_t < 10000) {
247  return;
248  }
249  _last_run = now;
250 
251  struct adc_msg_s buf_adc[VRBRAIN_ANALOG_MAX_CHANNELS];
252 
253  // cope with initial setup of stop pin
254  if (_channels[_current_stop_pin_i] == NULL ||
255  _channels[_current_stop_pin_i]->_stop_pin == -1) {
256  next_stop_pin();
257  }
258 
259  /* read all channels available */
260  int ret = read(_adc_fd, &buf_adc, sizeof(buf_adc));
261  if (ret > 0) {
262  // match the incoming channels to the currently active pins
263  for (uint8_t i=0; i<ret/sizeof(buf_adc[0]); i++) {
264  Debug("chan %u value=%u\n",
265  (unsigned)buf_adc[i].am_channel,
266  (unsigned)buf_adc[i].am_data);
267  for (uint8_t j=0; j<VRBRAIN_ANALOG_MAX_CHANNELS; j++) {
268  VRBRAIN::VRBRAINAnalogSource *c = _channels[j];
269  if (c != NULL && buf_adc[i].am_channel == c->_pin) {
270  // add a value if either there is no stop pin, or
271  // the stop pin has been settling for enough time
272  if (c->_stop_pin == -1 ||
273  (_current_stop_pin_i == j &&
274  hal.scheduler->millis() - _stop_pin_change_time > c->_settle_time_ms)) {
275  c->_add_value(buf_adc[i].am_data, _board_voltage);
276  if (c->_stop_pin != -1 && _current_stop_pin_i == j) {
277  next_stop_pin();
278  }
279  }
280  }
281  }
282  }
283  }
284 
285  // check for new battery data on FMUv1
286  if (_battery_handle != -1) {
287  struct battery_status_s battery;
288  bool updated = false;
289  if (orb_check(_battery_handle, &updated) == 0 && updated) {
290  orb_copy(ORB_ID(battery_status), _battery_handle, &battery);
291  if (battery.timestamp != _battery_timestamp) {
292  _battery_timestamp = battery.timestamp;
293  for (uint8_t j=0; j<VRBRAIN_ANALOG_MAX_CHANNELS; j++) {
294  VRBRAIN::VRBRAINAnalogSource *c = _channels[j];
295  if (c == NULL) continue;
296  if (c->_pin == VRBRAIN_ANALOG_ORB_BATTERY_VOLTAGE_PIN) {
297  c->_add_value(battery.voltage_v / VRBRAIN_VOLTAGE_SCALING, 0);
298  }
299  if (c->_pin == VRBRAIN_ANALOG_ORB_BATTERY_CURRENT_PIN) {
300  // scale it back to voltage, knowing that the
301  // px4io code scales by 90.0/5.0
302  c->_add_value(battery.current_a * (5.0f/90.0f) / VRBRAIN_VOLTAGE_SCALING, 0);
303  }
304  }
305  }
306  }
307  }
