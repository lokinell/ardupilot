35 void Balloon::update(const struct sitl_input &input)
36 {
37  if (!released && input.servos[6] > 1800) {
38  ::printf("Balloon released\n");
39  released = true;
40  }
41 
42  if (!burst && input.servos[7] > 1800) {
43  ::printf("Balloon burst\n");
44  burst = true;
45  }
46 
47  float delta_time = frame_time_us * 1.0e-6f;
48 
49  // rotational air resistance
50  Vector3f rot_accel = -gyro * radians(400) / terminal_rotation_rate;
51 
52  // update rotational rates in body frame
53  gyro += rot_accel * delta_time;
54 
55  // update attitude
56  dcm.rotate(gyro * delta_time);
57  dcm.normalize();
58 
59  // air resistance
60  Vector3f air_resistance = -velocity_ef * (GRAVITY_MSS/terminal_velocity);
61 
62  float lift_accel = 0;
63  if (!burst && released) {
64  float air_resistance_at_climb_rate = climb_rate * (GRAVITY_MSS/terminal_velocity);
65  lift_accel = air_resistance_at_climb_rate + GRAVITY_MSS * dcm.c.z;
66  }
67 
68  accel_body = Vector3f(0, 0, -lift_accel);
69  Vector3f accel_earth = dcm * accel_body;
70  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
71  accel_earth += air_resistance;
72 
73  // if we're on the ground, then our vertical acceleration is limited
74  // to zero. This effectively adds the force of the ground on the aircraft
75  if (on_ground(position) && accel_earth.z > 0) {
76  accel_earth.z = 0;
77  }
78 
79  // work out acceleration as seen by the accelerometers. It sees the kinematic
80  // acceleration (ie. real movement), plus gravity
81  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
82 
83  // add some noise
84  add_noise(velocity_ef.length() / terminal_velocity);
85 
86  // new velocity vector
87  velocity_ef += accel_earth * delta_time;
88 
89  // new position vector
90  Vector3f old_position = position;
91  position += velocity_ef * delta_time;
92 
93  if (position.z < -burst_altitude) {
94  ::printf("Balloon burst at %.1f\n", -position.z);
95  burst = true;
96  }
97 
98  // assume zero wind for now
99  airspeed = velocity_ef.length();
100

102  if (on_ground(position)) {

104  printf("Hit ground at %f m/s\n", velocity_ef.z);
105 
106  velocity_ef.zero();
107 
108  // zero roll/pitch, but keep yaw
109  float r, p, y;
110  dcm.to_euler(&r, &p, &y);
111  dcm.from_euler(0, 0, y);

114  }
