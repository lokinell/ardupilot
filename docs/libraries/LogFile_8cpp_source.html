486  print_mode_fn print_mode,
487  AP_HAL::BetterStream *port)

490  uint16_t page = start_page;

492 
493  if (df_BufferIdx != 0) {
494  FinishWrite();
495  hal.scheduler->delay(100);
496  }
497 
498  StartRead(start_page);
499 
500  while (true) {
501  uint8_t data;
502  if (!ReadBlock(&data, 1)) {
503  break;
504  }
505

507  switch(log_step) {
508  case 0:
509  if (data == HEAD_BYTE1) {
510  log_step++;
511  }
512  break;
513 
514  case 1:
515  if (data == HEAD_BYTE2) {
516  log_step++;
517  } else {
518  log_step = 0;
519  }
520  break;
521 
522  case 2:
523  log_step = 0;
524  if (first_entry && data != LOG_FORMAT_MSG) {
525  _print_log_formats(port);
526  }
527  first_entry = false;
528  _print_log_entry(data, print_mode, port);
529  break;
530  }

532  if (new_page != page) {
533  if (new_page == end_page+1 || new_page == start_page) {
