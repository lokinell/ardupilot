80 template <class T, class U, uint8_t FILTER_SIZE>
81 T AverageFilter<T,U,FILTER_SIZE>:: apply(T sample)
82 {
83  U result = 0;
84 
85  // call parent's apply function to get the sample into the array
86  FilterWithBuffer<T,FILTER_SIZE>::apply(sample);
87 
88  // increment the number of samples so far
89  _num_samples++;
90  if( _num_samples > FILTER_SIZE || _num_samples == 0 )
91  _num_samples = FILTER_SIZE;
92 
93  // get sum of all values - there is a risk of overflow here that we ignore
94  for(uint8_t i=0; i<FILTER_SIZE; i++)
95  result += FilterWithBuffer<T,FILTER_SIZE>::samples[i];
96 
97  return (T)(result / _num_samples);
98 }
