49 PWM_Sysfs::PWM_Sysfs(uint8_t chip, uint8_t channel)
50  : _chip(chip)
51  , _channel(channel)
52  , _duty_cycle_fd(-1)
53  , _nsec_duty_cycle_value(0)
54 {
55  char path[PWM_PATH_MAX];
56  int r;
57 
58  r = snprintf(path, sizeof(path), PWM_BASE_PATH "%u/export", _chip);
59  if (r < 0 || r >= (int)sizeof(path)) {
60  hal.scheduler->panic("LinuxPWM_Sysfs: chip=%u channel=%u "
61  "Error formatting pwm export: %s",
62  _chip, _channel, strerror(errno));
63  }
64 
65  /*
66  * Not checking for write errors here because if the PWM channel was
67  * already exported, the write will return a EBUSY and we can catch up
68  * any other error when trying to open the duty_cycle file descriptor.
69  */
70  Util::from(hal.util)->write_file(path, "%u", _channel);
71 
72  r = snprintf(path, sizeof(path), PWM_BASE_PATH "%u/pwm%u/duty_cycle",
73  _chip, _channel);
74  if (r < 0 || r >= (int)sizeof(path)) {
75  hal.scheduler->panic("LinuxPWM_Sysfs: chip=%u channel=%u "
76  "Error formatting channel duty cycle: %s",
77  _chip, _channel, strerror(errno));
78  }
79 
80  _duty_cycle_fd = ::open(path, O_RDWR | O_CLOEXEC);
81  if (_duty_cycle_fd < 0) {
82  hal.scheduler->panic("LinuxPWM_Sysfs: chip=%u channel=%u "
83  "Unable to open file %s: %s",
84  _chip, _channel, path, strerror(errno));
85  }
86 }
