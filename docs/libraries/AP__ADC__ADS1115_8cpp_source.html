198 void AP_ADC_ADS1115::_update()
199 {
200  /* TODO: make update rate configurable */
201  if (hal.scheduler->micros() - _last_update_timestamp < 100000) {
202  return;
203  }
204 
205  union {
206  uint8_t b[2];
207  int16_t w;
208  } word, config;
209 
210  if (!_i2c_sem->take_nonblocking()) {
211  return;
212  }
213 
214  if ( hal.i2c->readRegisters(ADS1115_ADDRESS_ADDR_GND, ADS1115_RA_CONFIG, 2, config.b) != 0 ) {
215  error("i2c->readRegisters failed in ADS1115");
216  _i2c_sem->give();
217  return;
218  }
219 
220  /* check rdy bit */
221  if ((config.b[1] & 0x80) != 0x80 ) {
222  _i2c_sem->give();
223  return;
224  }
225 
226  if ( hal.i2c->readRegisters(ADS1115_ADDRESS_ADDR_GND, ADS1115_RA_CONVERSION, 2, word.b) != 0 ) {
227  _i2c_sem->give();
228  return;
229  }
230 
231  word.w = word.b[0] << 8 | word.b[1]; /* Exchange MSB and LSB */
232 
233  float sample = _convert_register_data_to_mv(word.w);
234 
235  _samples[_channel_to_read].data = sample;
236  _samples[_channel_to_read].id = _channel_to_read;
237 
238  /* select next channel */
239  _channel_to_read = (_channel_to_read + 1) % _channels_number;
240  _start_conversion(_channel_to_read);
241 
242  _i2c_sem->give();
243 
244  _last_update_timestamp = hal.scheduler->micros();
245 }
