171 void SPIDeviceManager::cs_assert(enum AP_HAL::SPIDevice type)
172 {
173  uint16_t bus = 0, i;
174  for (i=0; i<LINUX_SPI_DEVICE_NUM_DEVICES; i++) {
175  if (_device[i]._type == type) {
176  bus = _device[i]._bus;
177  break;
178  }
179  }
180  if (i == LINUX_SPI_DEVICE_NUM_DEVICES) {
181  hal.scheduler->panic("Bad device type");
182  }
183 
184  // Kernel-mode CS handling
185  if(_device[i]._cs_pin == SPI_CS_KERNEL)
186  return;
187 
188  for (i=0; i<LINUX_SPI_DEVICE_NUM_DEVICES; i++) {
189  if (_device[i]._bus != bus) {
190  // not the same bus
191  continue;
192  }
193  if (_device[i]._type != type) {
194  if (_device[i]._cs->read() != 1) {
195  hal.console->printf("two CS enabled at once i=%u %u and %u\n",
196  (unsigned)i, (unsigned)type, (unsigned)_device[i]._type);
197  }
198  }
199  }
200  for (i=0; i<LINUX_SPI_DEVICE_NUM_DEVICES; i++) {
201  if (_device[i]._type == type) {
202  _device[i]._cs->write(0);
203  }
204  }
205 }

233 bool SPIDeviceManager::transaction(SPIDeviceDriver &driver, const uint8_t *tx, uint8_t *rx, uint16_t len)
234 {
235  int r;
236  // we set the mode before we assert the CS line so that the bus is
237  // in the correct idle state before the chip is selected
238  r = ioctl(driver._fd, SPI_IOC_WR_MODE, &driver._mode);
239  if (r == -1) {
240  hal.console->printf("SPI: error on setting mode\n");
241  return false;
242  }
243 
244  cs_assert(driver._type);
245  struct spi_ioc_transfer spi[1];
246  memset(spi, 0, sizeof(spi));
247  spi[0].tx_buf = (uint64_t)tx;
248  spi[0].rx_buf = (uint64_t)rx;
249  spi[0].len = len;
250  spi[0].delay_usecs = 0;
251  spi[0].speed_hz = driver._speed;
252  spi[0].bits_per_word = driver._bitsPerWord;
253  spi[0].cs_change = 0;
254 
255  if (rx != NULL) {
256  // keep valgrind happy
257  memset(rx, 0, len);
258  }
259 
260  r = ioctl(driver._fd, SPI_IOC_MESSAGE(1), &spi);
261  cs_release(driver._type);
262 
263  if (r == -1) {
264  hal.console->printf("SPI: error on doing transaction\n");
265  return false;
266  }
267 
268  return true;
269 }
