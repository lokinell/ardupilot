167 // output_test - spin a motor at the pwm value specified
168 // motor_seq is the motor's sequence number from 1 to the number of motors on the frame
169 // pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
170 void AP_MotorsHeli_Single::output_test(uint8_t motor_seq, int16_t pwm)
171 {
172  // exit immediately if not armed
173  if (!armed()) {
174  return;
175  }
176 
177  // output to motors and servos
178  switch (motor_seq) {
179  case 1:
180  // swash servo 1
181  hal.rcout->write(AP_MOTORS_MOT_1, pwm);
182  break;
183  case 2:
184  // swash servo 2

186  break;
187  case 3:
188  // swash servo 3

192  // external gyro & tail servo

197  write_aux(_ext_gyro_gain_std);
198  }
199  }
200  hal.rcout->write(AP_MOTORS_MOT_4, pwm);
201  break;
202  case 5:
203  // main rotor
204  hal.rcout->write(AP_MOTORS_HELI_SINGLE_RSC, pwm);

221 // calculate_scalars - recalculates various scalers used.
222 void AP_MotorsHeli_Single::calculate_scalars()
223 {
224  // range check collective min, max and mid
225  if( _collective_min >= _collective_max ) {
226  _collective_min = AP_MOTORS_HELI_COLLECTIVE_MIN;
227  _collective_max = AP_MOTORS_HELI_COLLECTIVE_MAX;
228  }
229  _collective_mid = constrain_int16(_collective_mid, _collective_min, _collective_max);
230 
231  // calculate collective mid point as a number from 0 to 1000
232  _collective_mid_pwm = ((float)(_collective_mid-_collective_min))/((float)(_collective_max-_collective_min))*1000.0f;
233 
234  // calculate maximum collective pitch range from full positive pitch to zero pitch
235  _collective_range = 1000 - _collective_mid_pwm;
236 
237  // determine roll, pitch and collective input scaling
238  _roll_scaler = (float)_cyclic_max/4500.0f;
239  _pitch_scaler = (float)_cyclic_max/4500.0f;
240  _collective_scalar = ((float)(_collective_max-_collective_min))/1000.0f;
241 
242  // calculate factors based on swash type and servo position
243  calculate_roll_pitch_collective_factors();
244 
245  // send setpoints to main rotor controller and trigger recalculation of scalars
246  _main_rotor.set_control_mode(static_cast<RotorControlMode>(_rsc_mode.get()));
247  _main_rotor.set_ramp_time(_rsc_ramp_time);
248  _main_rotor.set_runup_time(_rsc_runup_time);
249  _main_rotor.set_critical_speed(_rsc_critical);
250  _main_rotor.set_idle_output(_rsc_idle_output);
251  _main_rotor.set_power_output_range(_rsc_power_low, _rsc_power_high);
252  _main_rotor.recalc_scalers();
253 
254  // send setpoints to tail rotor controller and trigger recalculation of scalars
255  if (_tail_type == AP_MOTORS_HELI_SINGLE_TAILTYPE_DIRECTDRIVE_VARPITCH) {
256  _tail_rotor.set_control_mode(ROTOR_CONTROL_MODE_SPEED_SETPOINT);
257  _tail_rotor.set_ramp_time(AP_MOTORS_HELI_SINGLE_DDVPT_RAMP_TIME);
258  _tail_rotor.set_runup_time(AP_MOTORS_HELI_SINGLE_DDVPT_RUNUP_TIME);
259  _tail_rotor.set_critical_speed(_rsc_critical);
260  _tail_rotor.set_idle_output(_rsc_idle_output);
261  } else {
262  _tail_rotor.set_control_mode(ROTOR_CONTROL_MODE_DISABLED);
263  _tail_rotor.set_ramp_time(0);
264  _tail_rotor.set_runup_time(0);
265  _tail_rotor.set_critical_speed(0);
266  _tail_rotor.set_idle_output(0);
267  }
268  _tail_rotor.recalc_scalers();
269 }

271 // calculate_roll_pitch_collective_factors - calculate factors based on swash type and servo position
272 void AP_MotorsHeli_Single::calculate_roll_pitch_collective_factors()
273 {
274  if (_swash_type == AP_MOTORS_HELI_SINGLE_SWASH_CCPM) { //CCPM Swashplate, perform control mixing
275 
276  // roll factors
277  _rollFactor[CH_1] = cosf(radians(_servo1_pos + 90 - (_phase_angle + _delta_phase_angle)));
278  _rollFactor[CH_2] = cosf(radians(_servo2_pos + 90 - (_phase_angle + _delta_phase_angle)));
279  _rollFactor[CH_3] = cosf(radians(_servo3_pos + 90 - (_phase_angle + _delta_phase_angle)));
280 
281  // pitch factors
282  _pitchFactor[CH_1] = cosf(radians(_servo1_pos - (_phase_angle + _delta_phase_angle)));
283  _pitchFactor[CH_2] = cosf(radians(_servo2_pos - (_phase_angle + _delta_phase_angle)));
284  _pitchFactor[CH_3] = cosf(radians(_servo3_pos - (_phase_angle + _delta_phase_angle)));
285 
286  // collective factors
287  _collectiveFactor[CH_1] = 1;
288  _collectiveFactor[CH_2] = 1;
289  _collectiveFactor[CH_3] = 1;
290 
291  }else{ //H1 Swashplate, keep servo outputs seperated
292 
293  // roll factors
294  _rollFactor[CH_1] = 1;
295  _rollFactor[CH_2] = 0;
296  _rollFactor[CH_3] = 0;
297 
298  // pitch factors
299  _pitchFactor[CH_1] = 0;
300  _pitchFactor[CH_2] = 1;
301  _pitchFactor[CH_3] = 0;
302 
303  // collective factors
304  _collectiveFactor[CH_1] = 0;
305  _collectiveFactor[CH_2] = 0;
306  _collectiveFactor[CH_3] = 1;
307  }
308 }

346 //
347 // move_actuators - moves swash plate and tail rotor
348 // - expected ranges:
349 // roll : -4500 ~ 4500
350 // pitch: -4500 ~ 4500
351 // collective: 0 ~ 1000
352 // yaw: -4500 ~ 4500
353 //
354 void AP_MotorsHeli_Single::move_actuators(int16_t roll_out, int16_t pitch_out, int16_t coll_in, int16_t yaw_out)
355 {
356  int16_t yaw_offset = 0;
357  int16_t coll_out_scaled;
358 
359  // initialize limits flag
360  limit.roll_pitch = false;
361  limit.yaw = false;
362  limit.throttle_lower = false;
363  limit.throttle_upper = false;
364 
365  // rescale roll_out and pitch_out into the min and max ranges to provide linear motion
366  // across the input range instead of stopping when the input hits the constrain value
367  // these calculations are based on an assumption of the user specified cyclic_max
368  // coming into this equation at 4500 or less, and based on the original assumption of the
369  // total _servo_x.servo_out range being -4500 to 4500.
370 
371  float total_out = pythagorous2((float)pitch_out, (float)roll_out);
372 
373  if (total_out > _cyclic_max) {
374  float ratio = (float)_cyclic_max / total_out;
375  roll_out *= ratio;
376  pitch_out *= ratio;
377  limit.roll_pitch = true;
378  }
379 
380  // constrain collective input
381  _collective_out = coll_in;
382  if (_collective_out <= 0) {
383  _collective_out = 0;
384  limit.throttle_lower = true;
385  }
386  if (_collective_out >= 1000) {
387  _collective_out = 1000;
388  limit.throttle_upper = true;
389  }
390 
391  // ensure not below landed/landing collective
392  if (_heliflags.landing_collective && _collective_out < _land_collective_min) {
393  _collective_out = _land_collective_min;
394  limit.throttle_lower = true;
395  }
396 
397  // scale collective pitch
398  coll_out_scaled = _collective_out * _collective_scalar + _collective_min - 1000;
399 
400  // if servo output not in manual mode, process pre-compensation factors
401  if (_servo_mode == SERVO_CONTROL_MODE_AUTOMATED) {
402  // rudder feed forward based on collective
403  // the feed-forward is not required when the motor is stopped or at idle, and thus not creating torque
404  // also not required if we are using external gyro
405  if ((_main_rotor.get_control_output() > _rsc_idle_output) && _tail_type != AP_MOTORS_HELI_SINGLE_TAILTYPE_SERVO_EXTGYRO) {
406  // sanity check collective_yaw_effect
407  _collective_yaw_effect = constrain_float(_collective_yaw_effect, -AP_MOTORS_HELI_SINGLE_COLYAW_RANGE, AP_MOTORS_HELI_SINGLE_COLYAW_RANGE);
408  yaw_offset = _collective_yaw_effect * abs(_collective_out - _collective_mid_pwm);
409  }
410  } else {
411  yaw_offset = 0;
412  }
413 
414  // feed power estimate into main rotor controller
415  // ToDo: include tail rotor power?
416  // ToDo: add main rotor cyclic power?
417  _main_rotor_power = ((float)(abs(_collective_out - _collective_mid_pwm)) / _collective_range);
418  _main_rotor.set_motor_load(_main_rotor_power);
419 
420  // swashplate servos
421  _swash_servo_1.servo_out = (_rollFactor[CH_1] * roll_out + _pitchFactor[CH_1] * pitch_out)/10 + _collectiveFactor[CH_1] * coll_out_scaled + (_swash_servo_1.radio_trim-1500);
422  _swash_servo_2.servo_out = (_rollFactor[CH_2] * roll_out + _pitchFactor[CH_2] * pitch_out)/10 + _collectiveFactor[CH_2] * coll_out_scaled + (_swash_servo_2.radio_trim-1500);
423  if (_swash_type == AP_MOTORS_HELI_SINGLE_SWASH_H1) {
424  _swash_servo_1.servo_out += 500;
425  _swash_servo_2.servo_out += 500;
426  }
427  _swash_servo_3.servo_out = (_rollFactor[CH_3] * roll_out + _pitchFactor[CH_3] * pitch_out)/10 + _collectiveFactor[CH_3] * coll_out_scaled + (_swash_servo_3.radio_trim-1500);
428 
429  // use servo_out to calculate pwm_out and radio_out
430  _swash_servo_1.calc_pwm();
431  _swash_servo_2.calc_pwm();
432  _swash_servo_3.calc_pwm();
433 
434  hal.rcout->cork();
435 
436  // actually move the servos
437  hal.rcout->write(AP_MOTORS_MOT_1, _swash_servo_1.radio_out);
438  hal.rcout->write(AP_MOTORS_MOT_2, _swash_servo_2.radio_out);
439  hal.rcout->write(AP_MOTORS_MOT_3, _swash_servo_3.radio_out);
440 
441  // update the yaw rate using the tail rotor/servo
442  move_yaw(yaw_out + yaw_offset);
443 
444  hal.rcout->push();
445 }

482 // servo_test - move servos through full range of movement
483 void AP_MotorsHeli_Single::servo_test()
484 {
485  _servo_test_cycle_time += 1.0f / _loop_rate;
486 
487  if ((_servo_test_cycle_time >= 0.0f && _servo_test_cycle_time < 0.5f)|| // Tilt swash back
488  (_servo_test_cycle_time >= 6.0f && _servo_test_cycle_time < 6.5f)){
489  _pitch_test += (4500 / (_loop_rate/2));
490  _oscillate_angle += 8 * M_PI_F / _loop_rate;
491  _yaw_test = 2250 * sinf(_oscillate_angle);
492  } else if ((_servo_test_cycle_time >= 0.5f && _servo_test_cycle_time < 4.5f)|| // Roll swash around
493  (_servo_test_cycle_time >= 6.5f && _servo_test_cycle_time < 10.5f)){
494  _oscillate_angle += M_PI_F / (2 * _loop_rate);
495  _roll_test = 4500 * sinf(_oscillate_angle);
496  _pitch_test = 4500 * cosf(_oscillate_angle);
