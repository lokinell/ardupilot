81 bool FLYMAPLEGPIO::attach_interrupt(uint8_t interrupt_num, AP_HAL::Proc p, uint8_t mode)
82 {
83  // Flymaple can only handle RISING, FALLING and CHANGE
84  ExtIntTriggerMode flymaple_interrupt_mode;
85  if (mode == HAL_GPIO_INTERRUPT_FALLING)
86  flymaple_interrupt_mode = FALLING;
87  else if (mode == HAL_GPIO_INTERRUPT_RISING)
88  flymaple_interrupt_mode = RISING;
89  else
90  return false;
91 
92  // REVISIT: Assumes pin and interrupt number are the same. Are they?
93  attachInterrupt(interrupt_num, p, flymaple_interrupt_mode);
94  return true;
95 }
