85 void GCS_MAVLINK::handle_log_request_data(mavlink_message_t *msg, DataFlash_Class &dataflash)
86 {
87  mavlink_log_request_data_t packet;
88  mavlink_msg_log_request_data_decode(msg, &packet);
89 
90  _log_listing = false;
91  if (!_log_sending || _log_num_data != packet.id) {
92  _log_sending = false;
93 
94  uint16_t num_logs = dataflash.get_num_logs();
95  if (packet.id > num_logs || packet.id < 1) {
96  return;

98 
99  uint32_t time_utc, size;
100  dataflash.get_log_info(packet.id, size, time_utc);
101  _log_num_data = packet.id;
102  _log_data_size = size;
103 
104  uint16_t end;
105  dataflash.get_log_boundaries(packet.id, _log_data_page, end);
106  }
107 
108  _log_data_offset = packet.ofs;
109  if (_log_data_offset >= _log_data_size) {
110  _log_data_remaining = 0;
111  } else {
112  _log_data_remaining = _log_data_size - _log_data_offset;
113  }
114  if (_log_data_remaining > packet.count) {
115  _log_data_remaining = packet.count;
116  }
117  _log_sending = true;
118

146 void GCS_MAVLINK::handle_log_send(DataFlash_Class &dataflash)
147 {
148  if (_log_listing) {
149  handle_log_send_listing(dataflash);
150  }
151  if (!_log_sending) {
152  return;
153  }
154  uint8_t num_sends = 1;
155  if (chan == MAVLINK_COMM_0 && hal.gpio->usb_connected()) {
156  // when on USB we can send a lot more data
157  num_sends = 40;
158  } else if (have_flow_control()) {
159 #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX
160  num_sends = 80;
161 #else
162  num_sends = 10;
163 #endif
164  }
165 
166 #if CONFIG_HAL_BOARD == HAL_BOARD_SITL
167  // assume USB speeds in SITL for the purposes of log download
168  num_sends = 40;
169 #endif
170 
171  for (uint8_t i=0; i<num_sends; i++) {
172  if (_log_sending) {
173  if (!handle_log_send_data(dataflash)) break;
174  }
175  }
176 }

181 void GCS_MAVLINK::handle_log_send_listing(DataFlash_Class &dataflash)
182 {
183  uint16_t txspace = comm_get_txspace(chan);
184  if (txspace < MAVLINK_NUM_NON_PAYLOAD_BYTES+MAVLINK_MSG_ID_LOG_ENTRY_LEN) {
185  // no space
186  return;
187  }
188  if (hal.scheduler->millis() - last_heartbeat_time > 3000) {
189  // give a heartbeat a chance
190  return;
191  }
192 
193  uint32_t size, time_utc;
194  if (_log_next_list_entry == 0) {
195  size = 0;
196  time_utc = 0;
197  } else {
198  dataflash.get_log_info(_log_next_list_entry, size, time_utc);
199  }
200  mavlink_msg_log_entry_send(chan, _log_next_list_entry, _log_num_logs, _log_last_list_entry, time_utc, size);
201  if (_log_next_list_entry == _log_last_list_entry) {
202  _log_listing = false;
203  } else {
204  _log_next_list_entry++;
205  }
206 }

211 bool GCS_MAVLINK::handle_log_send_data(DataFlash_Class &dataflash)
212 {
213  uint16_t txspace = comm_get_txspace(chan);
214  if (txspace < MAVLINK_NUM_NON_PAYLOAD_BYTES+MAVLINK_MSG_ID_LOG_DATA_LEN) {
215  // no space
216  return false;
217  }
218  if (hal.scheduler->millis() - last_heartbeat_time > 3000) {
219  // give a heartbeat a chance
220  return false;
221  }
222 
223  int16_t ret = 0;
224  uint32_t len = _log_data_remaining;
225  mavlink_log_data_t packet;
226 
227  if (len > 90) {
228  len = 90;
229  }
230  ret = dataflash.get_log_data(_log_num_data, _log_data_page, _log_data_offset, len, packet.data);
231  if (ret < 0) {
232  // report as EOF on error
233  ret = 0;
234  }
235  if (ret < 90) {
236  memset(&packet.data[ret], 0, 90-ret);
237  }
238 
239  packet.ofs = _log_data_offset;
240  packet.id = _log_num_data;
241  packet.count = ret;
242  _mav_finalize_message_chan_send(chan, MAVLINK_MSG_ID_LOG_DATA, (const char *)&packet,
243  MAVLINK_MSG_ID_LOG_DATA_LEN, MAVLINK_MSG_ID_LOG_DATA_CRC);
244 
245  _log_data_offset += len;
246  _log_data_remaining -= len;
247  if (ret < 90 || _log_data_remaining == 0) {
248  _log_sending = false;
249  }
250  return true;
251 }
