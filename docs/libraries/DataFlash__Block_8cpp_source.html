73  BlockWrite(df_BufferNum, df_BufferIdx, &ph, sizeof(ph), pBuffer, n);
74  df_BufferIdx += n + sizeof(ph);

181 void DataFlash_Block::EraseAll()
182 {

185  BlockErase(j);
