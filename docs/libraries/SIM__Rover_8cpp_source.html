49 /*
50  return turning circle (diameter) in meters for steering angle proportion in degrees
51 */
52 float SimRover::turn_circle(float steering)
53 {
54  if (fabsf(steering) < 1.0e-6) {
55  return 0;
56  }
57  return turning_circle * sinf(radians(35)) / sinf(radians(steering*35));

60 /*
61  return yaw rate in degrees/second given steering_angle and speed
62 */
63 float SimRover::calc_yaw_rate(float steering, float speed)
64 {
65  if (skid_steering) {
66  return steering * skid_turn_rate;
67  }
68  if (fabsf(steering) < 1.0e-6 or fabsf(speed) < 1.0e-6) {
69  return 0;
70  }
71  float d = turn_circle(steering);
72  float c = M_PI_F * d;
73  float t = c / speed;
74  float rate = 360.0f / t;
75  return rate;
76 }

88 /*
89  update the rover simulation by one time step
90  */
91 void SimRover::update(const struct sitl_input &input)
92 {
93  float steering, throttle;
94 
95  // if in skid steering mode the steering and throttle values are used for motor1 and motor2
96  if (skid_steering) {
97  float motor1 = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
98  float motor2 = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
99  steering = motor1 - motor2;
100  throttle = 0.5*(motor1 + motor2);
101  } else {
102  steering = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
103  throttle = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
104  }
105 
106  // how much time has passed?
107  float delta_time = frame_time_us * 1.0e-6f;
108 
109  // speed in m/s in body frame
110  Vector3f velocity_body = dcm.transposed() * velocity_ef;
111 
112  // speed along x axis, +ve is forward
113  float speed = velocity_body.x;
114 
115  // yaw rate in degrees/s
116  float yaw_rate = calc_yaw_rate(steering, speed);
117 
118  // target speed with current throttle
119  float target_speed = throttle * max_speed;
120 
121  // linear acceleration in m/s/s - very crude model
122  float accel = max_accel * (target_speed - speed) / max_speed;
123 
124  gyro = Vector3f(0,0,radians(yaw_rate));
125 
126  // update attitude
127  dcm.rotate(gyro * delta_time);
128  dcm.normalize();
129 
130  // accel in body frame due to motor
131  accel_body = Vector3f(accel, 0, 0);
132 
133  // add in accel due to direction change
134  accel_body.y += radians(yaw_rate) * speed;
135 
136  // now in earth frame
137  Vector3f accel_earth = dcm * accel_body;
138  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
139 
140  // we are on the ground, so our vertical accel is zero
141  accel_earth.z = 0;
142 
143  // work out acceleration as seen by the accelerometers. It sees the kinematic
144  // acceleration (ie. real movement), plus gravity
145  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
146 
147  // new velocity vector
148  velocity_ef += accel_earth * delta_time;
149 
150  // new position vector
151  position += velocity_ef * delta_time;
152  position.z = -home.alt*0.01f;
153 
154  // update lat/lon/altitude
155  update_position();
156 }
