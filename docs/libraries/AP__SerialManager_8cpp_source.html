114 // init - // init - initialise serial ports
115 void AP_SerialManager::init()
116 {
117  // initialise pointers to serial ports
118  state[1].uart = hal.uartC; // serial1, uartC, normally telem1
119  state[2].uart = hal.uartD; // serial2, uartD, normally telem2
120  state[3].uart = hal.uartB; // serial3, uartB, normally 1st GPS
121  state[4].uart = hal.uartE; // serial4, uartE, normally 2nd GPS
122 
123  // initialise serial ports
124  for (uint8_t i=1; i<SERIALMANAGER_NUM_PORTS; i++) {
125  if (state[i].uart != NULL) {
126  switch (state[i].protocol) {
127  case SerialProtocol_Console:
128  state[i].uart->begin(map_baudrate(state[i].baud),
129  AP_SERIALMANAGER_CONSOLE_BUFSIZE_RX,
130  AP_SERIALMANAGER_CONSOLE_BUFSIZE_TX);
131  break;
132  case SerialProtocol_MAVLink:
133  case SerialProtocol_MAVLink2:
134  state[i].uart->begin(map_baudrate(state[i].baud),
135  AP_SERIALMANAGER_MAVLINK_BUFSIZE_RX,
136  AP_SERIALMANAGER_MAVLINK_BUFSIZE_TX);
137  break;
138  case SerialProtocol_FRSky_DPort:
139  // Note baudrate is hardcoded to 57600
140  state[i].baud = AP_SERIALMANAGER_FRSKY_DPORT_BAUD/1000; // update baud param in case user looks at it
141  state[i].uart->begin(AP_SERIALMANAGER_FRSKY_DPORT_BAUD,
142  AP_SERIALMANAGER_FRSKY_BUFSIZE_RX,
143  AP_SERIALMANAGER_FRSKY_BUFSIZE_TX);
144  break;
145  case SerialProtocol_FRSky_SPort:
146  // Note baudrate is hardcoded to 9600
147  state[i].baud = AP_SERIALMANAGER_FRSKY_SPORT_BAUD/1000; // update baud param in case user looks at it
148  // begin is handled by AP_Frsky_telem library
149  break;
150  case SerialProtocol_GPS:
151  case SerialProtocol_GPS2:
152  state[i].uart->begin(map_baudrate(state[i].baud),
153  AP_SERIALMANAGER_GPS_BUFSIZE_RX,
154  AP_SERIALMANAGER_GPS_BUFSIZE_TX);
155  break;
156  case SerialProtocol_AlexMos:
157  // Note baudrate is hardcoded to 115200
158  state[i].baud = AP_SERIALMANAGER_ALEXMOS_BAUD / 1000; // update baud param in case user looks at it
159  state[i].uart->begin(AP_SERIALMANAGER_ALEXMOS_BAUD,
160  AP_SERIALMANAGER_ALEXMOS_BUFSIZE_RX,
161  AP_SERIALMANAGER_ALEXMOS_BUFSIZE_TX);
162  break;
163  case SerialProtocol_SToRM32:
164  // Note baudrate is hardcoded to 115200
165  state[i].baud = AP_SERIALMANAGER_SToRM32_BAUD / 1000; // update baud param in case user looks at it
166  state[i].uart->begin(map_baudrate(state[i].baud),
167  AP_SERIALMANAGER_SToRM32_BUFSIZE_RX,
168  AP_SERIALMANAGER_SToRM32_BUFSIZE_TX);
169  break;
170  }
171  }
172  }

175 // find_serial - searches available serial ports for the first instance that allows the given protocol
176 // instance should be zero if searching for the first instance, 1 for the second, etc
177 // returns uart on success, NULL if a serial port cannot be found
178 AP_HAL::UARTDriver *AP_SerialManager::find_serial(enum SerialProtocol protocol, uint8_t instance) const
179 {
180  uint8_t found_instance = 0;
181 
182  // search for matching protocol
183  for(uint8_t i=0; i<SERIALMANAGER_NUM_PORTS; i++) {
184  if (protocol_match(protocol, (enum SerialProtocol)state[i].protocol.get())) {
185  if (found_instance == instance) {
186  return state[i].uart;
187  }
188  found_instance++;
189  }
190  }
191 
192  // if we got this far we did not find the uart
193  return NULL;
194 }

217 // get_mavlink_channel - provides the mavlink channel associated with a given protocol
218 // instance should be zero if searching for the first instance, 1 for the second, etc
219 // returns true if a channel is found, false if not
220 bool AP_SerialManager::get_mavlink_channel(enum SerialProtocol protocol, uint8_t instance, mavlink_channel_t &mav_chan) const
221 {
222  // check for console
223  if (protocol == SerialProtocol_Console) {
224  mav_chan = MAVLINK_COMM_0;
225  return true;
226  }
227 
228  // check for MAVLink
229  if (protocol_match(protocol, SerialProtocol_MAVLink)) {
230  if (instance == 0) {
231  mav_chan = MAVLINK_COMM_1;
232  return true;
233  }
234  if (instance == 1) {
235  mav_chan = MAVLINK_COMM_2;
236  return true;
237  }
238  if (instance == 2) {
239  mav_chan = MAVLINK_COMM_3;
240  return true;
241  }
242  }
243 
244  // report failure
245  return false;
246 }

277 /*
278  * map from a 16 bit EEPROM baud rate to a real baud rate.
279  * For PX4 we can do 1.5MBit, although 921600 is more reliable.
280  */
281 uint32_t AP_SerialManager::map_baudrate(int32_t rate) const
282 {
283  if (rate <= 0) {
284  rate = 57;
285  }
286  switch (rate) {
287  case 1: return 1200;
288  case 2: return 2400;
289  case 4: return 4800;
290  case 9: return 9600;
291  case 19: return 19200;
292  case 38: return 38400;
293  case 57: return 57600;
294  case 100: return 100000;
295  case 111: return 111100;
296  case 115: return 115200;
297  case 230: return 230400;
298  case 460: return 460800;
299  case 500: return 500000;
300  case 921: return 921600;
301  case 1500: return 1500000;
302  }
303 
304  if (rate > 2000) {
305  // assume it is a direct baudrate. This allows for users to
306  // set an exact baudrate as long as it is over 2000 baud
307  return (uint32_t)rate;
308  }
309 
310  // otherwise allow any other kbaud rate
311  return rate*1000;
312 }

314 // protocol_match - returns true if the protocols match
315 bool AP_SerialManager::protocol_match(enum SerialProtocol protocol1, enum SerialProtocol protocol2) const
316 {
317  // check for obvious match
318  if (protocol1 == protocol2) {
319  return true;
320  }
321 
322  // mavlink match
323  if (((protocol1 == SerialProtocol_MAVLink) || (protocol1 == SerialProtocol_MAVLink2)) &&
324  ((protocol2 == SerialProtocol_MAVLink) || (protocol2 == SerialProtocol_MAVLink2))) {
325  return true;
326  }
327 
328  // gps match
329  if (((protocol1 == SerialProtocol_GPS) || (protocol1 == SerialProtocol_GPS2)) &&
330  ((protocol2 == SerialProtocol_GPS) || (protocol2 == SerialProtocol_GPS2))) {
331  return true;
332  }
333 
334  return false;
335 }
