161 void PX4Scheduler::delay(uint16_t ms)
162 {
163  if (in_timerprocess()) {
164  ::printf("ERROR: delay() from timer process\n");
165  return;
166  }
167  perf_begin(_perf_delay);
168  uint64_t start = micros64();
169 
170  while ((micros64() - start)/1000 < ms &&
171  !_px4_thread_should_exit) {
172  delay_microseconds_semaphore(1000);
173  if (_min_delay_cb_ms <= ms) {
174  if (_delay_cb) {
175  _delay_cb();
176  }
177  }
178  }
179  perf_end(_perf_delay);
180  if (_px4_thread_should_exit) {
181  exit(1);
182  }
183 }
