24 // select fusion of optical flow measurements
25 void NavEKF2_core::SelectFlowFusion()
26 {
27  // Check if the magnetometer has been fused on that time step and the filter is running at faster than 200 Hz
28  // If so, don't fuse measurements on this time step to reduce frame over-runs
29  // Only allow one time slip to prevent high rate magnetometer data preventing fusion of other measurements
30  if (magFusePerformed && dtIMUavg < 0.005f && !optFlowFusionDelayed) {
31  optFlowFusionDelayed = true;
32  return;
33  } else {
34  optFlowFusionDelayed = false;
35  }
36 
37  // start performance timer
38  hal.util->perf_begin(_perf_FuseOptFlow);
39  // Perform Data Checks
40  // Check if the optical flow data is still valid
41  flowDataValid = ((imuSampleTime_ms - flowValidMeaTime_ms) < 1000);
42  // Check if the optical flow sensor has timed out
43  bool flowSensorTimeout = ((imuSampleTime_ms - flowValidMeaTime_ms) > 5000);
44  // Check if the fusion has timed out (flow measurements have been rejected for too long)
45  bool flowFusionTimeout = ((imuSampleTime_ms - prevFlowFuseTime_ms) > 5000);
46  // check is the terrain offset estimate is still valid
47  gndOffsetValid = ((imuSampleTime_ms - gndHgtValidTime_ms) < 5000);
48  // Perform tilt check
49  bool tiltOK = (Tnb_flow.c.z > frontend->DCM33FlowMin);
50  // Constrain measurements to zero if we are using optical flow and are on the ground
51  if (frontend->_fusionModeGPS == 3 && !takeOffDetected && isAiding) {
52  ofDataDelayed.flowRadXYcomp.zero();
53  ofDataDelayed.flowRadXY.zero();
54  flowDataValid = true;
55  }
56 
57  // If the flow measurements have been rejected for too long and we are relying on them, then revert to constant position mode
58  if ((flowSensorTimeout || flowFusionTimeout) && PV_AidingMode == AID_RELATIVE) {
59  PV_AidingMode = AID_NONE;
60  // reset the velocity
61  ResetVelocity();
62  // store the current position to be used to as a sythetic position measurement
63  lastKnownPositionNE.x = stateStruct.position.x;
64  lastKnownPositionNE.y = stateStruct.position.y;
65  // reset the position
66  ResetPosition();
67  }
68 
69  // if we do have valid flow measurements, fuse data into a 1-state EKF to estimate terrain height
70  // we don't do terrain height estimation in optical flow only mode as the ground becomes our zero height reference
71  if ((newDataFlow || newDataRng) && tiltOK) {
72  // fuse range data into the terrain estimator if available
73  fuseRngData = newDataRng;
74  // fuse optical flow data into the terrain estimator if available and if there is no range data (range data is better)
75  fuseOptFlowData = (newDataFlow && !fuseRngData);
76  // Estimate the terrain offset (runs a one state EKF)
77  EstimateTerrainOffset();
78  // Indicate we have used the range data
79  newDataRng = false;
80  }
81 
82  // Fuse optical flow data into the main filter if not excessively tilted and we are in the correct mode
83  if (newDataFlow && tiltOK && PV_AidingMode == AID_RELATIVE)
84  {
85  // Set the flow noise used by the fusion processes
86  R_LOS = sq(max(frontend->_flowNoise, 0.05f));
87  // Fuse the optical flow X and Y axis data into the main filter sequentially
88  FuseOptFlow();
89  // reset flag to indicate that no new flow data is available for fusion
90  newDataFlow = false;
91  }
92 
93  // stop the performance timer
94  hal.util->perf_end(_perf_FuseOptFlow);
95 }

122  // limit distance to prevent intialisation afer bad gps causing bad numerical conditioning

143  float q3 = stateStruct.quat[3]; // quaternion at optical flow measurement time

179  }

191  float H_OPT;
192

194  float flowRngPred = max((terrainState - stateStruct.position[2]),rngOnGnd) / Tnb_flow.c.z;
195 
196  // constrain terrain height to be below the vehicle
197  terrainState = max(terrainState, stateStruct.position[2] + rngOnGnd);
198

203  // get predicted sensed angular optical rates relative to X and Y sensor axes
204  losPred = relVelSensor.length()/flowRngPred;
205 
206  // calculate innovations
207  auxFlowObsInnov = losPred - sqrtf(sq(flowRadXYcomp[0]) + sq(flowRadXYcomp[1]));
208 
209  // calculate observation jacobian
210  float t3 = sq(q0);
211  float t4 = sq(q1);
212  float t5 = sq(q2);
213  float t6 = sq(q3);
214  float t10 = q0*q3*2.0f;
215  float t11 = q1*q2*2.0f;
216  float t14 = t3+t4-t5-t6;

218  float t16 = t10+t11;

240  float t32 = sq(t12);

242

244  auxFlowObsInnovVar = H_OPT*Popt*H_OPT + R_LOS;
245 
246  // calculate Kalman gain
247  K_OPT = Popt*H_OPT/auxFlowObsInnovVar;
248 
249  // calculate the innovation consistency test ratio
250  auxFlowTestRatio = sq(auxFlowObsInnov) / (sq(frontend->_flowInnovGate) * auxFlowObsInnovVar);
251 
252  // don't fuse if optical flow data is outside valid range
253  if (max(flowRadXY[0],flowRadXY[1]) < frontend->_maxFlowRate) {
254 
255  // correct the state
256  terrainState -= K_OPT * auxFlowObsInnov;
257 
258  // constrain the state
259  terrainState = max(terrainState, stateStruct.position[2] + rngOnGnd);
260 
261  // correct the covariance
262  Popt = Popt - K_OPT * H_OPT * Popt;
263 
264  // prevent the state variances from becoming negative
265  Popt = max(Popt,0.0f);
266  }
267  }
268  }
269 
270  // stop the performance timer
271  hal.util->perf_end(_perf_TerrainOffset);
