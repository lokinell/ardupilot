179 // update_max_throttle - updates the limits on _max_throttle if necessary taking into account slow_start_throttle flag
180 void AP_MotorsMulticopter::update_max_throttle()
181 {
182  // ramp up minimum spin speed if necessary
183  if (_multicopter_flags.slow_start_low_end) {
184  _spin_when_armed_ramped += AP_MOTOR_SLOW_START_LOW_END_INCREMENT;
185  if (_spin_when_armed_ramped >= _spin_when_armed) {
186  _spin_when_armed_ramped = _spin_when_armed;
187  _multicopter_flags.slow_start_low_end = false;
188  }
189  }
190 
191  // implement slow start
192  if (_multicopter_flags.slow_start) {
193  // increase slow start throttle
194  _max_throttle += AP_MOTOR_SLOW_START_INCREMENT;
195 
196  // turn off slow start if we've reached max throttle
197  if (_max_throttle >= _throttle_control_input) {
198  _max_throttle = AP_MOTORS_DEFAULT_MAX_THROTTLE;
199  _multicopter_flags.slow_start = false;
200  }
201  return;
202  }
203 
204  // current limit throttle
205  current_limit_max_throttle();
206 }

208 // current_limit_max_throttle - limits maximum throttle based on current
209 void AP_MotorsMulticopter::current_limit_max_throttle()
210 {
211  // return maximum if current limiting is disabled
212  if (_batt_current_max <= 0) {
213  _throttle_limit = 1.0f;
214  _max_throttle = AP_MOTORS_DEFAULT_MAX_THROTTLE;
215  return;
216  }
217 
218  // remove throttle limit if throttle is at zero or disarmed
219  if(_throttle_control_input <= 0 || !_flags.armed) {
220  _throttle_limit = 1.0f;
221  }
222 
223  // limit throttle if over current
224  if (_batt_current > _batt_current_max*1.25f) {
225  // Fast drop for extreme over current (1 second)
226  _throttle_limit -= 1.0f/_loop_rate;
227  } else if(_batt_current > _batt_current_max) {
228  // Slow drop for extreme over current (5 second)
229  _throttle_limit -= 0.2f/_loop_rate;
230  } else {
231  // Increase throttle limit (2 second)
232  _throttle_limit += 0.5f/_loop_rate;
233  }
234 
235  // throttle limit drops to 20% between hover and full throttle
236  _throttle_limit = constrain_float(_throttle_limit, 0.2f, 1.0f);
237 
238  // limit max throttle
239  _max_throttle = _hover_out + ((1000-_hover_out)*_throttle_limit);
240 }

242 // apply_thrust_curve_and_volt_scaling - returns throttle curve adjusted pwm value (i.e. 1000 ~ 2000)
243 int16_t AP_MotorsMulticopter::apply_thrust_curve_and_volt_scaling(int16_t pwm_out, int16_t pwm_min, int16_t pwm_max) const
244 {
245  // convert to 0.0 to 1.0 ratio
246  float throttle_ratio = ((float)(pwm_out-pwm_min))/((float)(pwm_max-pwm_min));
247 
248  // apply thrust curve - domain 0.0 to 1.0, range 0.0 to 1.0
249  if (_thrust_curve_expo > 0.0f){
250  throttle_ratio = ((_thrust_curve_expo-1.0f) + safe_sqrt((1.0f-_thrust_curve_expo)*(1.0f-_thrust_curve_expo) + 4.0f*_thrust_curve_expo*_lift_max*throttle_ratio))/(2.0f*_thrust_curve_expo*_batt_voltage_filt.get());
251  }
252 
253  // scale to maximum thrust point
254  throttle_ratio *= _thrust_curve_max;
255 
256  // convert back to pwm range, constrain and return
257  return (int16_t)constrain_float(throttle_ratio*(pwm_max-pwm_min)+pwm_min, pwm_min, (pwm_max-pwm_min)*_thrust_curve_max+pwm_min);
258 }

260 // update_lift_max from battery voltage - used for voltage compensation
261 void AP_MotorsMulticopter::update_lift_max_from_batt_voltage()
262 {
263  // sanity check battery_voltage_min is not too small
264  // if disabled or misconfigured exit immediately
265  if((_batt_voltage_max <= 0) || (_batt_voltage_min >= _batt_voltage_max) || (_batt_voltage < 0.25f*_batt_voltage_min)) {
266  _batt_voltage_filt.reset(1.0f);
267  _lift_max = 1.0f;
268  return;
269  }
270 
271  _batt_voltage_min = max(_batt_voltage_min, _batt_voltage_max * 0.6f);
272 
273  // add current based voltage sag to battery voltage
274  float batt_voltage = _batt_voltage + _batt_current * _batt_resistance;
275  batt_voltage = constrain_float(batt_voltage, _batt_voltage_min, _batt_voltage_max);
276 
277  // filter at 0.5 Hz
278  float bvf = _batt_voltage_filt.apply(batt_voltage/_batt_voltage_max, 1.0f/_loop_rate);
279 
280  // calculate lift max
281  _lift_max = bvf*(1-_thrust_curve_expo) + _thrust_curve_expo*bvf*bvf;
282 }

284 // update_battery_resistance - calculate battery resistance when throttle is above hover_out
285 void AP_MotorsMulticopter::update_battery_resistance()
286 {
287  // if motors are stopped, reset resting voltage and current
288  if (_throttle_control_input <= 0 || !_flags.armed) {
289  _batt_voltage_resting = _batt_voltage;
290  _batt_current_resting = _batt_current;
291  _batt_timer = 0;
292  } else {
293  // update battery resistance when throttle is over hover throttle
294  if ((_batt_timer < 400) && ((_batt_current_resting*2.0f) < _batt_current)) {
295  if (_throttle_control_input >= _hover_out) {
296  // filter reaches 90% in 1/4 the test time
297  _batt_resistance += 0.05f*(( (_batt_voltage_resting-_batt_voltage)/(_batt_current-_batt_current_resting) ) - _batt_resistance);
298  _batt_timer += 1;
299  } else {
300  // initialize battery resistance to prevent change in resting voltage estimate
301  _batt_resistance = ((_batt_voltage_resting-_batt_voltage)/(_batt_current-_batt_current_resting));
302  }
303  }
304  }
305 }

307 // update_throttle_thr_mix - slew set_throttle_thr_mix to requested value
308 void AP_MotorsMulticopter::update_throttle_thr_mix()
309 {
310  // slew _throttle_thr_mix to _throttle_thr_mix_desired
311  if (_throttle_thr_mix < _throttle_thr_mix_desired) {
312  // increase quickly (i.e. from 0.1 to 0.9 in 0.4 seconds)
313  _throttle_thr_mix += min(2.0f/_loop_rate, _throttle_thr_mix_desired-_throttle_thr_mix);
314  } else if (_throttle_thr_mix > _throttle_thr_mix_desired) {
315  // reduce more slowly (from 0.9 to 0.1 in 1.6 seconds)
316  _throttle_thr_mix -= min(0.5f/_loop_rate, _throttle_thr_mix-_throttle_thr_mix_desired);
317  }
318  _throttle_thr_mix = constrain_float(_throttle_thr_mix, 0.1f, 1.0f);
319 }

327 float AP_MotorsMulticopter::get_compensation_gain() const
328 {
329  // avoid divide by zero

331  return 1.0f;
332  }

334  float ret = 1.0f / _lift_max;
335

338  ret *= 1.0f / constrain_float(_air_density_ratio,0.5f,1.25f);
339  }
