43  hal.console->printf("GPIO: Unable to setup GPIO LED RED\n");

48  hal.scheduler->panic("Unable to open " TONEALARM0_DEVICE_PATH);
49  }
50

52  if (_gpio_fmu_fd == -1) {
53  hal.scheduler->panic("Unable to open GPIO");

60

68 void PX4GPIO::pinMode(uint8_t pin, uint8_t output)
69 {
70  switch (pin) {
71  case PX4_GPIO_FMU_SERVO_PIN(0) ... PX4_GPIO_FMU_SERVO_PIN(5):
72  uint32_t pinmask = 1U<<(pin-PX4_GPIO_FMU_SERVO_PIN(0));
73  if (output) {
74  uint8_t old_value = read(pin);
75  if (old_value) {
76  ioctl(_gpio_fmu_fd, GPIO_SET_OUTPUT_HIGH, pinmask);
77  } else {
78  ioctl(_gpio_fmu_fd, GPIO_SET_OUTPUT_LOW, pinmask);
79  }
80  } else {
81  ioctl(_gpio_fmu_fd, GPIO_SET_INPUT, pinmask);
82  }
83  break;
84  }

158 void PX4GPIO::write(uint8_t pin, uint8_t value)
159 {
160  switch (pin) {
161 
162 #ifdef CONFIG_ARCH_BOARD_PX4FMU_V1
163  case HAL_GPIO_A_LED_PIN: // Arming LED
164  if (value == LOW) {
165  ioctl(_led_fd, LED_OFF, LED_RED);
166  } else {
167  ioctl(_led_fd, LED_ON, LED_RED);
168  }
169  break;
170 
171  case HAL_GPIO_B_LED_PIN: // not used yet
172  break;
173 
174  case HAL_GPIO_C_LED_PIN: // GPS LED
175  if (value == LOW) {
176  ioctl(_led_fd, LED_OFF, LED_BLUE);
177  } else {
178  ioctl(_led_fd, LED_ON, LED_BLUE);
179  }
180  break;
181 #endif
182 
183  case PX4_GPIO_PIEZO_PIN: // Piezo beeper
184  if (value == LOW) { // this is inverted

189  }
190  break;

194  ioctl(_gpio_fmu_fd, value==LOW?GPIO_CLEAR:GPIO_SET, GPIO_EXT_1);
