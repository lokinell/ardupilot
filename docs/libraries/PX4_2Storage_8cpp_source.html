208 void PX4Storage::_timer_tick(void)
209 {
210  if (!_initialised || _dirty_mask == 0) {
211  return;
212  }
213  perf_begin(_perf_storage);
214 
215  if (_fd == -1) {
216  _fd = open(MTD_PARAMS_FILE, O_WRONLY);
217  if (_fd == -1) {
218  perf_end(_perf_storage);
219  perf_count(_perf_errors);
220  return;
221  }
222  }
223 
224  // write out the first dirty set of lines. We don't write more
225  // than one to keep the latency of this call to a minimum
226  uint8_t i, n;
227  for (i=0; i<PX4_STORAGE_NUM_LINES; i++) {
228  if (_dirty_mask & (1<<i)) {
229  break;
230  }
231  }
232  if (i == PX4_STORAGE_NUM_LINES) {
233  // this shouldn't be possible
234  perf_end(_perf_storage);
235  perf_count(_perf_errors);
236  return;
237  }
238  uint32_t write_mask = (1U<<i);
239  // see how many lines to write
240  for (n=1; (i+n) < PX4_STORAGE_NUM_LINES &&
241  n < (PX4_STORAGE_MAX_WRITE>>PX4_STORAGE_LINE_SHIFT); n++) {
242  if (!(_dirty_mask & (1<<(n+i)))) {
243  break;
244  }
245  // mark that line clean
246  write_mask |= (1<<(n+i));
247  }
248 
249  /*
250  write the lines. This also updates _dirty_mask. Note that
251  because this is a SCHED_FIFO thread it will not be preempted
252  by the main task except during blocking calls. This means we
253  don't need a semaphore around the _dirty_mask updates.
254  */
255  if (lseek(_fd, i<<PX4_STORAGE_LINE_SHIFT, SEEK_SET) == (i<<PX4_STORAGE_LINE_SHIFT)) {
256  _dirty_mask &= ~write_mask;
257  if (write(_fd, &_buffer[i<<PX4_STORAGE_LINE_SHIFT], n<<PX4_STORAGE_LINE_SHIFT) != n<<PX4_STORAGE_LINE_SHIFT) {
258  // write error - likely EINTR
259  _dirty_mask |= write_mask;
260  close(_fd);
261  _fd = -1;
262  perf_count(_perf_errors);
263  }
264  }
265  perf_end(_perf_storage);
266 }
