91 /*
92  steering rate controller. Returns servo out -4500 to 4500 given
93  desired yaw rate in degrees/sec. Positive yaw rate means clockwise yaw.
94 */
95 int32_t AP_SteerController::get_steering_out_rate(float desired_rate)
96 {
97  uint32_t tnow = hal.scheduler->millis();
98  uint32_t dt = tnow - _last_t;
99  if (_last_t == 0 || dt > 1000) {
100  dt = 0;
101  }
102  _last_t = tnow;
103 
104  float speed = _ahrs.groundspeed();
105  if (speed < _minspeed) {
106  // assume a minimum speed. This stops osciallations when first starting to move
107  speed = _minspeed;
108  }
109 
110  // this is a linear approximation of the inverse steering
111  // equation for a ground vehicle. It returns steering as an angle from -45 to 45
112  float scaler = 1.0f / speed;
113 
114  _pid_info.desired = desired_rate;
115 
116  // Calculate the steering rate error (deg/sec) and apply gain scaler
117  float rate_error = (desired_rate - ToDeg(_ahrs.get_gyro().z)) * scaler;
118 
119  // Calculate equivalent gains so that values for K_P and K_I can be taken across from the old PID law
120  // No conversion is required for K_D
121  float ki_rate = _K_I * _tau * 45.0f;
122  float kp_ff = max((_K_P - _K_I * _tau) * _tau - _K_D , 0) * 45.0f;
123  float k_ff = _K_FF * 45.0f;
124  float delta_time = (float)dt * 0.001f;
125 
126  // Multiply roll rate error by _ki_rate and integrate
127  // Don't integrate if in stabilise mode as the integrator will wind up against the pilots inputs
128  if (ki_rate > 0 && speed >= _minspeed) {
129  // only integrate if gain and time step are positive.
130  if (dt > 0) {
131  float integrator_delta = rate_error * ki_rate * delta_time * scaler;
132  // prevent the integrator from increasing if steering defln demand is above the upper limit
133  if (_last_out < -45) {
134  integrator_delta = max(integrator_delta , 0);
135  } else if (_last_out > 45) {
136  // prevent the integrator from decreasing if steering defln demand is below the lower limit
137  integrator_delta = min(integrator_delta, 0);
138  }
139  _pid_info.I += integrator_delta;
140  }
141  } else {
142  _pid_info.I = 0;
143  }
144 
145  // Scale the integration limit
146  float intLimScaled = _imax * 0.01f;
147 
148  // Constrain the integrator state
149  _pid_info.I = constrain_float(_pid_info.I, -intLimScaled, intLimScaled);
150 
151  _pid_info.D = rate_error * _K_D * 4.0f;
152  _pid_info.P = (ToRad(desired_rate) * kp_ff) * scaler;
153  _pid_info.FF = (ToRad(desired_rate) * k_ff) * scaler;
154 
155  // Calculate the demanded control surface deflection
156  _last_out = _pid_info.D + _pid_info.FF + _pid_info.P + _pid_info.I;
157 
158  // Convert to centi-degrees and constrain
159  return constrain_float(_last_out * 100, -4500, 4500);
160 }

167 int32_t AP_SteerController::get_steering_out_lat_accel(float desired_accel)
