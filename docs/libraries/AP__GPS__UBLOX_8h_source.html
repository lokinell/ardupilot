29 /*
30  * try to put a UBlox into binary mode. This is in two parts.
31  *
32  * First we send a ubx binary message that enables the NAV_SOL message
33  * at rate 1. Then we send a NMEA message to set the baud rate to our
34  * desired rate. The reason for doing the NMEA message second is if we
35  * send it first the second message will be ignored for a baud rate
36  * change.
37  * The reason we need the NAV_SOL rate message at all is some uBlox
38  * modules are configured with all ubx binary messages off, which
39  * would mean we would never detect it.
40  */
41 #define UBLOX_SET_BINARY "\265\142\006\001\003\000\001\006\001\022\117$PUBX,41,1,0003,0001,38400,0*26\r\n"
42 #define UBLOX_SET_BINARY_RAW_BAUD "\265\142\006\001\003\000\001\006\001\022\117$PUBX,41,1,0003,0001,115200,0*1E\r\n"
