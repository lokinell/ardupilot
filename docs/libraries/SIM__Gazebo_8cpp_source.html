48 void Gazebo::send_servos(const struct sitl_input &input)
49 {
50  servo_packet pkt;
51  pkt.motor_speed[0] = (input.servos[0]-1000) / 1000.0f;
52  pkt.motor_speed[1] = (input.servos[1]-1000) / 1000.0f;
53  pkt.motor_speed[2] = (input.servos[2]-1000) / 1000.0f;
54  pkt.motor_speed[3] = (input.servos[3]-1000) / 1000.0f;
55  sock.sendto(&pkt, sizeof(pkt), "127.0.0.1", 9002);

85  pkt.imu_orientation_quat[1],

88  quat.rotation_matrix(dcm);
