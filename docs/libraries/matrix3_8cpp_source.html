24 // create a rotation matrix given some euler angles
25 // this is based on http://gentlenav.googlecode.com/files/EulerAngles.pdf
26 template <typename T>
27 void Matrix3<T>::from_euler(float roll, float pitch, float yaw)
28 {
29  float cp = cosf(pitch);
30  float sp = sinf(pitch);
31  float sr = sinf(roll);
32  float cr = cosf(roll);
33  float sy = sinf(yaw);
34  float cy = cosf(yaw);
35 
36  a.x = cp * cy;
37  a.y = (sr * sp * cy) - (cr * sy);
38  a.z = (cr * sp * cy) + (sr * sy);
39  b.x = cp * sy;
40  b.y = (sr * sp * sy) + (cr * cy);
41  b.z = (cr * sp * sy) - (sr * cy);
42  c.x = -sp;
43  c.y = sr * cp;
44  c.z = cr * cp;
45 }

76 /*
77  fill the matrix from Euler angles in radians in 312 convention
78 */
79 template <typename T>
80 void Matrix3<T>::from_euler312(float roll, float pitch, float yaw)
81 {
82  float c3 = cosf(pitch);
83  float s3 = sinf(pitch);
84  float s2 = sinf(roll);
85  float c2 = cosf(roll);
86  float s1 = sinf(yaw);
87  float c1 = cosf(yaw);
88 
89  a.x = c1 * c3 - s1 * s2 * s3;
90  b.y = c1 * c2;
91  c.z = c3 * c2;
92  a.y = -c2*s1;
93  a.z = s3*c1 + c3*s2*s1;
94  b.x = c3*s1 + s3*s2*c1;
95  b.z = s1*s3 - s2*c1*c3;
96  c.x = -s3*c2;
97  c.y = s2;
98 }

160 template <typename T>
161 void Matrix3<T>::normalize(void)
162 {
163  float error = a * b;
164  Vector3<T> t0 = a - (b * (0.5f * error));
165  Vector3<T> t1 = b - (a * (0.5f * error));
166  Vector3<T> t2 = t0 % t1;
167  a = t0 * (1.0f / t0.length());
168  b = t1 * (1.0f / t1.length());
169  c = t2 * (1.0f / t2.length());
170 }
