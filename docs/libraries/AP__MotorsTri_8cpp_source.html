139 void AP_MotorsTri::output_armed_not_stabilizing()
140 {
141  int16_t throttle_radio_output; // total throttle pwm value, summed onto throttle channel minimum, typically ~1100-1900
142  int16_t out_min = _throttle_radio_min + _min_throttle;
143  int16_t out_max = _throttle_radio_max;
144  int16_t motor_out[AP_MOTORS_MOT_4+1];
145 
146  // initialize limits flags
147  limit.roll_pitch = true;
148  limit.yaw = true;
149  limit.throttle_lower = false;
150  limit.throttle_upper = false;
151 
152  int16_t thr_in_min = rel_pwm_to_thr_range(_spin_when_armed_ramped);
153  if (_throttle_control_input <= thr_in_min) {
154  _throttle_control_input = thr_in_min;
155  limit.throttle_lower = true;
156  }
157 
158  if (_throttle_control_input >= _hover_out) {
159  _throttle_control_input = _hover_out;
160  limit.throttle_upper = true;
161  }
162 
163  throttle_radio_output = calc_throttle_radio_output();
164 
165  motor_out[AP_MOTORS_MOT_1] = throttle_radio_output;
166  motor_out[AP_MOTORS_MOT_2] = throttle_radio_output;
167  motor_out[AP_MOTORS_MOT_4] = throttle_radio_output;
168 
169  if(throttle_radio_output >= out_min) {
170  // adjust for thrust curve and voltage scaling
171  motor_out[AP_MOTORS_MOT_1] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_1], out_min, out_max);
172  motor_out[AP_MOTORS_MOT_2] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_2], out_min, out_max);
173  motor_out[AP_MOTORS_MOT_4] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_4], out_min, out_max);
174  }
175 
176  hal.rcout->cork();
177 
178  // send output to each motor
179  hal.rcout->write(AP_MOTORS_MOT_1, motor_out[AP_MOTORS_MOT_1]);
180  hal.rcout->write(AP_MOTORS_MOT_2, motor_out[AP_MOTORS_MOT_2]);
181  hal.rcout->write(AP_MOTORS_MOT_4, motor_out[AP_MOTORS_MOT_4]);
182 
183  // send centering signal to yaw servo
184  hal.rcout->write(AP_MOTORS_CH_TRI_YAW, _yaw_servo_trim);
185 
186  hal.rcout->push();
187 }

190 // TODO pull code that is common to output_armed_not_stabilizing into helper functions
191 void AP_MotorsTri::output_armed_stabilizing()
192 {
193  int16_t roll_pwm; // roll pwm value, initially calculated by calc_roll_pwm() but may be modified after, +/- 400
194  int16_t pitch_pwm; // pitch pwm value, initially calculated by calc_roll_pwm() but may be modified after, +/- 400
195  int16_t throttle_radio_output; // total throttle pwm value, summed onto throttle channel minimum, typically ~1100-1900
196  int16_t yaw_radio_output; // final yaw pwm value sent to motors, typically ~1100-1900
197  int16_t out_min = _throttle_radio_min + _min_throttle;
198  int16_t out_max = _throttle_radio_max;
199  int16_t motor_out[AP_MOTORS_MOT_4+1];
200 
201  // initialize limits flags
202  limit.roll_pitch = false;
203  limit.yaw = false;
204  limit.throttle_lower = false;
205  limit.throttle_upper = false;
206 
207  // Throttle is 0 to 1000 only
208  int16_t thr_in_min = rel_pwm_to_thr_range(_min_throttle);
209  if (_throttle_control_input <= thr_in_min) {
210  _throttle_control_input = thr_in_min;
211  limit.throttle_lower = true;
212  }
213  if (_throttle_control_input >= _max_throttle) {
214  _throttle_control_input = _max_throttle;
215  limit.throttle_upper = true;
216  }
217 
218  // tricopters limit throttle to 80%
219  // To-Do: implement improved stability patch and remove this limit
220  if (_throttle_control_input > 800) {
221  _throttle_control_input = 800;
222  limit.throttle_upper = true;
223  }
224 
225  roll_pwm = calc_roll_pwm();
226  pitch_pwm = calc_pitch_pwm();
227  throttle_radio_output = calc_throttle_radio_output();
228  yaw_radio_output = calc_yaw_radio_output();
229 
230  // if we are not sending a throttle output, we cut the motors
231  if( is_zero(_throttle_control_input) ) {
232  // range check spin_when_armed
233  if (_spin_when_armed_ramped < 0) {
234  _spin_when_armed_ramped = 0;
235  }
236  if (_spin_when_armed_ramped > _min_throttle) {
237  _spin_when_armed_ramped = _min_throttle;
238  }
239  motor_out[AP_MOTORS_MOT_1] = _throttle_radio_min + _spin_when_armed_ramped;
240  motor_out[AP_MOTORS_MOT_2] = _throttle_radio_min + _spin_when_armed_ramped;
241  motor_out[AP_MOTORS_MOT_4] = _throttle_radio_min + _spin_when_armed_ramped;
242 
243  }else{
244  int16_t roll_out = (float)(roll_pwm * 0.866f);
245  int16_t pitch_out = pitch_pwm / 2;
246 
247  // check if throttle is below limit
248  if (_throttle_control_input <= _min_throttle) {
249  limit.throttle_lower = true;
250  _throttle_control_input = _min_throttle;
251  throttle_radio_output = calc_throttle_radio_output();
252  }
253 
254  // TODO: set limits.roll_pitch and limits.yaw
255 
256  //left front
257  motor_out[AP_MOTORS_MOT_2] = throttle_radio_output + roll_out + pitch_out;
258  //right front
259  motor_out[AP_MOTORS_MOT_1] = throttle_radio_output - roll_out + pitch_out;
260  // rear
261  motor_out[AP_MOTORS_MOT_4] = throttle_radio_output - pitch_pwm;
262 
263  // Tridge's stability patch
264  if(motor_out[AP_MOTORS_MOT_1] > out_max) {
265  motor_out[AP_MOTORS_MOT_2] -= (motor_out[AP_MOTORS_MOT_1] - out_max);
266  motor_out[AP_MOTORS_MOT_4] -= (motor_out[AP_MOTORS_MOT_1] - out_max);
267  motor_out[AP_MOTORS_MOT_1] = out_max;
268  }
269 
270  if(motor_out[AP_MOTORS_MOT_2] > out_max) {
271  motor_out[AP_MOTORS_MOT_1] -= (motor_out[AP_MOTORS_MOT_2] - out_max);
272  motor_out[AP_MOTORS_MOT_4] -= (motor_out[AP_MOTORS_MOT_2] - out_max);
273  motor_out[AP_MOTORS_MOT_2] = out_max;
274  }
275 
276  if(motor_out[AP_MOTORS_MOT_4] > out_max) {
277  motor_out[AP_MOTORS_MOT_1] -= (motor_out[AP_MOTORS_MOT_4] - out_max);
278  motor_out[AP_MOTORS_MOT_2] -= (motor_out[AP_MOTORS_MOT_4] - out_max);
279  motor_out[AP_MOTORS_MOT_4] = out_max;
280  }
281 
282  // adjust for thrust curve and voltage scaling
283  motor_out[AP_MOTORS_MOT_1] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_1], out_min, out_max);
284  motor_out[AP_MOTORS_MOT_2] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_2], out_min, out_max);
285  motor_out[AP_MOTORS_MOT_4] = apply_thrust_curve_and_volt_scaling(motor_out[AP_MOTORS_MOT_4], out_min, out_max);
286 
287  // ensure motors don't drop below a minimum value and stop
288  motor_out[AP_MOTORS_MOT_1] = max(motor_out[AP_MOTORS_MOT_1], out_min);
289  motor_out[AP_MOTORS_MOT_2] = max(motor_out[AP_MOTORS_MOT_2], out_min);
290  motor_out[AP_MOTORS_MOT_4] = max(motor_out[AP_MOTORS_MOT_4], out_min);
291  }
292 
293  hal.rcout->cork();
294 
295  // send output to each motor
296  hal.rcout->write(AP_MOTORS_MOT_1, motor_out[AP_MOTORS_MOT_1]);
297  hal.rcout->write(AP_MOTORS_MOT_2, motor_out[AP_MOTORS_MOT_2]);
298  hal.rcout->write(AP_MOTORS_MOT_4, motor_out[AP_MOTORS_MOT_4]);
299 
300  // send out to yaw command to tail servo
301  hal.rcout->write(AP_MOTORS_CH_TRI_YAW, yaw_radio_output);
302 
303  hal.rcout->push();
304 }

347 // calc_yaw_radio_output - calculate final radio output for yaw channel
348 int16_t AP_MotorsTri::calc_yaw_radio_output()
349 {
350  int16_t ret;
351 
352  if (_yaw_servo_reverse < 0) {
353  if (_yaw_control_input >= 0){
354  ret = (_yaw_servo_trim - (_yaw_control_input/4500 * (_yaw_servo_trim - _yaw_servo_min)));
355  } else {
356  ret = (_yaw_servo_trim - (_yaw_control_input/4500 * (_yaw_servo_max - _yaw_servo_trim)));
357  }
358  } else {
359  if (_yaw_control_input >= 0){
360  ret = ((_yaw_control_input/4500 * (_yaw_servo_max - _yaw_servo_trim)) + _yaw_servo_trim);
361  } else {
362  ret = ((_yaw_control_input/4500 * (_yaw_servo_trim - _yaw_servo_min)) + _yaw_servo_trim);
363  }
364  }
365 
366  return ret;
367 }
