53  update the helicopter simulation by one time step
54  */
55 void Helicopter::update(const struct sitl_input &input)
56 {
57  // how much time has passed?
58  float delta_time = frame_time_us * 1.0e-6f;
59 
60  float rsc = (input.servos[7]-1000) / 1000.0f;
61  // ignition only for gas helis
62  bool ignition_enabled = gas_heli?(input.servos[5] > 1500):true;
63 
64  float thrust = 0;
65  float roll_rate = 0;
66  float pitch_rate = 0;
67  float yaw_rate = 0;
68  float torque_effect_accel = 0;
69  float lateral_x_thrust = 0;
70  float lateral_y_thrust = 0;
71 
72  float swash1 = (input.servos[0]-1000) / 1000.0f;
73  float swash2 = (input.servos[1]-1000) / 1000.0f;
74  float swash3 = (input.servos[2]-1000) / 1000.0f;
75 
76  if (!ignition_enabled) {
77  rsc = 0;
78  }
79  float rsc_scale = rsc/rsc_setpoint;
80 
81  switch (frame_type) {
82  case HELI_FRAME_CONVENTIONAL: {
83  // simulate a traditional helicopter
84 
85  float tail_rotor = (input.servos[3]-1000) / 1000.0f;
86 
87  thrust = (rsc/rsc_setpoint) * (swash1+swash2+swash3) / 3.0f;
88  torque_effect_accel = (rsc_scale+thrust) * rotor_rot_accel;
89 
90  roll_rate = swash1 - swash2;
91  pitch_rate = (swash1+swash2) / 2.0f - swash3;
92  yaw_rate = tail_rotor - 0.5f;
93 
94  lateral_y_thrust = yaw_rate * rsc_scale * tail_thrust_scale;
95  break;
96  }
97 
98  case HELI_FRAME_DUAL: {
99  // simulate a tandem helicopter
100 
101  float swash4 = (input.servos[3]-1000) / 1000.0f;
102  float swash5 = (input.servos[4]-1000) / 1000.0f;
103  float swash6 = (input.servos[5]-1000) / 1000.0f;
104 
105  thrust = (rsc / rsc_setpoint) * (swash1+swash2+swash3+swash4+swash5+swash6) / 6.0f;
106  torque_effect_accel = (rsc_scale + rsc / rsc_setpoint) * rotor_rot_accel * ((swash1+swash2+swash3) - (swash4+swash5+swash6));
107 
108  roll_rate = (swash1-swash2) + (swash4-swash5);
109  pitch_rate = (swash1+swash2+swash3) - (swash4+swash5+swash6);
110  yaw_rate = (swash1-swash2) + (swash5-swash4);
111  break;
112  }
113 
114  case HELI_FRAME_COMPOUND: {
115  // simulate a compound helicopter
116 
117  float right_rotor = (input.servos[3]-1000) / 1000.0f;
118  float left_rotor = (input.servos[4]-1000) / 1000.0f;
119 
120  thrust = (rsc/rsc_setpoint) * (swash1+swash2+swash3) / 3.0f;
121  torque_effect_accel = (rsc_scale+thrust) * rotor_rot_accel;
122 
123  roll_rate = swash1 - swash2;
124  pitch_rate = (swash1+swash2) / 2.0f - swash3;
125  yaw_rate = right_rotor - left_rotor;
126 
127  lateral_x_thrust = (left_rotor+right_rotor-1) * rsc_scale * tail_thrust_scale;
128  break;
129  }
130  }
131 
132  roll_rate *= rsc_scale;
133  pitch_rate *= rsc_scale;
134  yaw_rate *= rsc_scale;
135 
136  // rotational acceleration, in rad/s/s, in body frame
137  Vector3f rot_accel;
138  rot_accel.x = roll_rate * roll_rate_max;
139  rot_accel.y = pitch_rate * pitch_rate_max;
140  rot_accel.z = yaw_rate * yaw_rate_max;
141 
142  // rotational air resistance
143  rot_accel.x -= gyro.x * radians(5000.0) / terminal_rotation_rate;
144  rot_accel.y -= gyro.y * radians(5000.0) / terminal_rotation_rate;
145  rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
146 
147  // torque effect on tail
148  rot_accel.z += torque_effect_accel;
149 
150  // update rotational rates in body frame
151  gyro += rot_accel * delta_time;
152 
153  // update attitude
154  dcm.rotate(gyro * delta_time);
155  dcm.normalize();
156 
157  // air resistance
158  Vector3f air_resistance = -velocity_ef * (GRAVITY_MSS/terminal_velocity);
159 
160  // scale thrust to newtons
161  thrust *= thrust_scale;
162 
163  accel_body = Vector3f(lateral_x_thrust, lateral_y_thrust, -thrust / mass);
164  Vector3f accel_earth = dcm * accel_body;
165  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
166  accel_earth += air_resistance;
167 
168  // if we're on the ground, then our vertical acceleration is limited
169  // to zero. This effectively adds the force of the ground on the aircraft
170  if (on_ground(position) && accel_earth.z > 0) {
171  accel_earth.z = 0;
172  }
173 
174  // work out acceleration as seen by the accelerometers. It sees the kinematic
175  // acceleration (ie. real movement), plus gravity
176  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
177 
178  // add some noise
179  add_noise(thrust / thrust_scale);
180 
181  // new velocity vector
182  velocity_ef += accel_earth * delta_time;
183 
184  // new position vector
185  Vector3f old_position = position;
186  position += velocity_ef * delta_time;
187 
188  // assume zero wind for now
189  airspeed = velocity_ef.length();
190 
191  // constrain height to the ground
192  if (on_ground(position)) {
193  if (!on_ground(old_position)) {
194  printf("Hit ground at %f m/s\n", velocity_ef.z);
195 
196  velocity_ef.zero();
197 
198  // zero roll/pitch, but keep yaw
199  float r, p, y;
200  dcm.to_euler(&r, &p, &y);
201  dcm.from_euler(0, 0, y);
202 
203  position.z = -(ground_level + frame_height - home.alt*0.01f);
204  }
205  }
206 
207  // update lat/lon/altitude
208  update_position();
209 }
