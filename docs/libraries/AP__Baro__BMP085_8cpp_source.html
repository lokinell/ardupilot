49 AP_Baro_BMP085::AP_Baro_BMP085(AP_Baro &baro) :
50  AP_Baro_Backend(baro),
51  _instance(0),
52  _temp_sum(0),
53  _press_sum(0),
54  _count(0),
55  BMP085_State(0),
56  ac1(0), ac2(0), ac3(0), b1(0), b2(0), mb(0), mc(0), md(0),
57  ac4(0), ac5(0), ac6(0),
58  _retry_time(0)
59 {
60  uint8_t buff[22];
61 
62  // get pointer to i2c bus semaphore
63  AP_HAL::Semaphore *i2c_sem = hal.i2c->get_semaphore();
64 
65  // take i2c bus sempahore
66  if (!i2c_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
67  hal.scheduler->panic("BMP085: unable to get semaphore");
68  }
69 
70  // End Of Conversion (PC7) input
71  if (BMP085_EOC >= 0) {
72  hal.gpio->pinMode(BMP085_EOC, HAL_GPIO_INPUT);
73  }
74 
75  // We read the calibration data registers
76  if (hal.i2c->readRegisters(BMP085_ADDRESS, 0xAA, 22, buff) != 0) {
77  hal.scheduler->panic("BMP085: bad calibration registers");
78  }
79 
80  ac1 = ((int16_t)buff[0] << 8) | buff[1];
81  ac2 = ((int16_t)buff[2] << 8) | buff[3];
82  ac3 = ((int16_t)buff[4] << 8) | buff[5];
83  ac4 = ((int16_t)buff[6] << 8) | buff[7];
84  ac5 = ((int16_t)buff[8] << 8) | buff[9];
85  ac6 = ((int16_t)buff[10] << 8) | buff[11];
86  b1 = ((int16_t)buff[12] << 8) | buff[13];
87  b2 = ((int16_t)buff[14] << 8) | buff[15];
88  mb = ((int16_t)buff[16] << 8) | buff[17];
89  mc = ((int16_t)buff[18] << 8) | buff[19];
90  md = ((int16_t)buff[20] << 8) | buff[21];
91 
92  _last_press_read_command_time = 0;
93  _last_temp_read_command_time = 0;
94 
95  _instance = _frontend.register_sensor();
96 
97  //Send a command to read Temp
98  Command_ReadTemp();
99 
100  BMP085_State = 0;
101 
102  i2c_sem->give();
103 }

105 // Read the sensor. This is a state machine
106 // acumulate a new sensor reading
107 void AP_Baro_BMP085::accumulate(void)
108 {
109  // get pointer to i2c bus semaphore
110  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
111 
112  if (!BMP_DATA_READY()) {
113  return;
114  }
115 
116  // take i2c bus sempahore
117  if (!i2c_sem->take(1))
118  return;
119 
120  if (BMP085_State == 0) {
121  ReadTemp();
122  } else {
123  if (ReadPress()) {
124  Calculate();
125  }
126  }
127  BMP085_State++;
128  if (BMP085_State == 5) {
129  BMP085_State = 0;
130  Command_ReadTemp();
131  } else {
132  Command_ReadPress();
133  }
134 
135  i2c_sem->give();
136 }

142 void AP_Baro_BMP085::update(void)
143 {
144  if (_count == 0 && BMP_DATA_READY()) {
145  accumulate();
146  }
147  if (_count == 0) {
148  return;
149  }
150 
151  float temperature = 0.1f * _temp_sum / _count;
152  float pressure = _press_sum / _count;
153 
154  _count = 0;
155  _temp_sum = 0;
156  _press_sum = 0;
157 
158  _copy_to_frontend(_instance, pressure, temperature);
159 }

170 // Read Raw Pressure values
171 bool AP_Baro_BMP085::ReadPress()
172 {
173  uint8_t buf[3];
174 
175  if (hal.i2c->readRegisters(BMP085_ADDRESS, 0xF6, 3, buf) != 0) {
176  _retry_time = hal.scheduler->millis() + 1000;
177  hal.i2c->setHighSpeed(false);
178  return false;
179  }
180 
181  RawPress = (((uint32_t)buf[0] << 16)
182  | ((uint32_t)buf[1] << 8)
183  | ((uint32_t)buf[2])) >> (8 - OVERSAMPLING);
184  return true;
185 }

196 {

211 // Calculate Temperature and Pressure in real units.
212 void AP_Baro_BMP085::Calculate()
213 {
214  int32_t x1, x2, x3, b3, b5, b6, p;
215  uint32_t b4, b7;
216  int32_t tmp;
217 
218  // See Datasheet page 13 for this formulas
219  // Based also on Jee Labs BMP085 example code. Thanks for share.
220  // Temperature calculations
221  x1 = ((int32_t)RawTemp - ac6) * ac5 >> 15;
222  x2 = ((int32_t) mc << 11) / (x1 + md);
223  b5 = x1 + x2;
224  _temp_sum += (b5 + 8) >> 4;
225 
226  // Pressure calculations
227  b6 = b5 - 4000;
228  x1 = (b2 * (b6 * b6 >> 12)) >> 11;
229  x2 = ac2 * b6 >> 11;
230  x3 = x1 + x2;
231  //b3 = (((int32_t) ac1 * 4 + x3)<<OVERSAMPLING + 2) >> 2; // BAD
232  //b3 = ((int32_t) ac1 * 4 + x3 + 2) >> 2; //OK for OVERSAMPLING=0
233  tmp = ac1;
234  tmp = (tmp*4 + x3)<<OVERSAMPLING;
235  b3 = (tmp+2)/4;
236  x1 = ac3 * b6 >> 13;
237  x2 = (b1 * (b6 * b6 >> 12)) >> 16;
238  x3 = ((x1 + x2) + 2) >> 2;
239  b4 = (ac4 * (uint32_t) (x3 + 32768)) >> 15;
240  b7 = ((uint32_t) RawPress - b3) * (50000 >> OVERSAMPLING);
241  p = b7 < 0x80000000 ? (b7 * 2) / b4 : (b7 / b4) * 2;
242 
243  x1 = (p >> 8) * (p >> 8);
244  x1 = (x1 * 3038) >> 16;
245  x2 = (-7357 * p) >> 16;
246  _press_sum += p + ((x1 + x2 + 3791) >> 4);
247 
248  _count++;
249  if (_count == 254) {
250  _temp_sum *= 0.5f;
251  _press_sum *= 0.5f;
252  _count /= 2;
253  }
254 }
