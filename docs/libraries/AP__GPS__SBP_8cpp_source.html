103 //This attempts to reads all SBP messages from the incoming port.
104 //Returns true if a new message was read, false if we failed to read a message.
105 void
106 AP_GPS_SBP::_sbp_process()
107 {
108 
109  while (port->available() > 0) {
110  uint8_t temp = port->read();
111  uint16_t crc;
112 
113 
114  //This switch reads one character at a time,
115  //parsing it into buffers until a full message is dispatched
116  switch(parser_state.state) {
117  case sbp_parser_state_t::WAITING:
118  if (temp == SBP_PREAMBLE) {
119  parser_state.n_read = 0;
120  parser_state.state = sbp_parser_state_t::GET_TYPE;
121  }
122  break;
123 
124  case sbp_parser_state_t::GET_TYPE:
125  *((uint8_t*)&(parser_state.msg_type) + parser_state.n_read) = temp;
126  parser_state.n_read += 1;
127  if (parser_state.n_read >= 2) {
128  parser_state.n_read = 0;
129  parser_state.state = sbp_parser_state_t::GET_SENDER;
130  }
131  break;
132 
133  case sbp_parser_state_t::GET_SENDER:
134  *((uint8_t*)&(parser_state.sender_id) + parser_state.n_read) = temp;
135  parser_state.n_read += 1;
136  if (parser_state.n_read >= 2) {
137  parser_state.n_read = 0;
138  parser_state.state = sbp_parser_state_t::GET_LEN;
139  }
140  break;
141 
142  case sbp_parser_state_t::GET_LEN:
143  parser_state.msg_len = temp;
144  parser_state.n_read = 0;
145  parser_state.state = sbp_parser_state_t::GET_MSG;
146  break;
147 
148  case sbp_parser_state_t::GET_MSG:
149  *((uint8_t*)&(parser_state.msg_buff) + parser_state.n_read) = temp;
150  parser_state.n_read += 1;
151  if (parser_state.n_read >= parser_state.msg_len) {
152  parser_state.n_read = 0;
153  parser_state.state = sbp_parser_state_t::GET_CRC;
154  }
155  break;
156 
157  case sbp_parser_state_t::GET_CRC:
158  *((uint8_t*)&(parser_state.crc) + parser_state.n_read) = temp;
159  parser_state.n_read += 1;
160  if (parser_state.n_read >= 2) {
161  parser_state.state = sbp_parser_state_t::WAITING;
162 
163  crc = crc16_ccitt((uint8_t*)&(parser_state.msg_type), 2, 0);
164  crc = crc16_ccitt((uint8_t*)&(parser_state.sender_id), 2, crc);
165  crc = crc16_ccitt(&(parser_state.msg_len), 1, crc);
166  crc = crc16_ccitt(parser_state.msg_buff, parser_state.msg_len, crc);
167  if (parser_state.crc == crc) {
168  _sbp_process_message();
169  } else {
170  Debug("CRC Error Occurred!");
171  crc_error_counter += 1;
172  }
173 
174  parser_state.state = sbp_parser_state_t::WAITING;
175  }
176  break;
177 
178  default:
179  parser_state.state = sbp_parser_state_t::WAITING;
180  break;
181  }
182  }

186 //INVARIANT: A fully received message with correct CRC is currently in parser_state
187 void
188 AP_GPS_SBP::_sbp_process_message() {
189  switch(parser_state.msg_type) {
190  case SBP_HEARTBEAT_MSGTYPE:
191  last_heatbeat_received_ms = hal.scheduler->millis();
192  break;
193 
194  case SBP_GPS_TIME_MSGTYPE:
195  memcpy(&last_gps_time, parser_state.msg_buff, sizeof(last_gps_time));
196  break;
197 
198  case SBP_VEL_NED_MSGTYPE:
199  memcpy(&last_vel_ned, parser_state.msg_buff, sizeof(last_vel_ned));
200  break;
201 
202  case SBP_POS_LLH_MSGTYPE: {
203  struct sbp_pos_llh_t *pos_llh = (struct sbp_pos_llh_t*)parser_state.msg_buff;
204  // Check if this is a single point or RTK solution
205  // flags = 0 -> single point
206  if (pos_llh->flags == 0) {
207  last_pos_llh_spp = *pos_llh;
208  } else if (pos_llh->flags == 1 || pos_llh->flags == 2) {
209  last_pos_llh_rtk = *pos_llh;
210  }
211  break;
212  }
213 
214  case SBP_DOPS_MSGTYPE:
215  memcpy(&last_dops, parser_state.msg_buff, sizeof(last_dops));
216  break;
217 
218  case SBP_TRACKING_STATE_MSGTYPE:
219  //INTENTIONALLY BLANK
220  //Currenly unhandled, but logged after switch statement.
221  break;
222 
223  case SBP_IAR_STATE_MSGTYPE: {
224  sbp_iar_state_t *iar = (struct sbp_iar_state_t*)parser_state.msg_buff;
225  last_iar_num_hypotheses = iar->num_hypotheses;
226  break;
227  }
228 
229  default:
230  // log anyway if it's an unsupported message.
231  // The log mask will be used to adjust or suppress logging
232  break;
233  }
234 
235  logging_log_raw_sbp(parser_state.msg_type, parser_state.sender_id, parser_state.msg_len, parser_state.msg_buff);
236 }

238 bool
239 AP_GPS_SBP::_attempt_state_update()
240 {
241 
242  // If we currently have heartbeats
243  // - NO FIX
244  //
245  // If we have a full update available, save it
246  //
247  uint32_t now = hal.scheduler->millis();
248  bool ret = false;
249 
250  if (now - last_heatbeat_received_ms > SBP_TIMEOUT_HEATBEAT) {
251 
252  state.status = AP_GPS::NO_GPS;
253  Debug("No Heartbeats from Piksi! Driver Ready to Die!");
254  ret = false;
255 
256  } else if (last_pos_llh_rtk.tow == last_vel_ned.tow
257  && abs((int32_t) (last_gps_time.tow - last_vel_ned.tow)) < 10000
258  && abs((int32_t) (last_dops.tow - last_vel_ned.tow)) < 60000
259  && last_vel_ned.tow > last_full_update_tow) {
260 
261  // Use the RTK position
262  sbp_pos_llh_t *pos_llh = &last_pos_llh_rtk;
263 
264  // Update time state
265  state.time_week = last_gps_time.wn;
266  state.time_week_ms = last_vel_ned.tow;
267 
268  state.hdop = last_dops.hdop;
269 
270  // Update velocity state
271  state.velocity[0] = (float)(last_vel_ned.n / 1000.0);
272  state.velocity[1] = (float)(last_vel_ned.e / 1000.0);
273  state.velocity[2] = (float)(last_vel_ned.d / 1000.0);
274 
275  float ground_vector_sq = state.velocity[0]*state.velocity[0] + state.velocity[1]*state.velocity[1];
276  state.ground_speed = safe_sqrt(ground_vector_sq);
277 
278  state.ground_course_cd = wrap_360_cd((int32_t) 100*ToDeg(atan2f(state.velocity[1], state.velocity[0])));
279 
280  // Update position state
281 
282  state.location.lat = (int32_t) (pos_llh->lat*1e7);
283  state.location.lng = (int32_t) (pos_llh->lon*1e7);
284  state.location.alt = (int32_t) (pos_llh->height*1e2);
285  state.num_sats = pos_llh->n_sats;
286 
287  if (pos_llh->flags == 0)
288  state.status = AP_GPS::GPS_OK_FIX_3D;
289  else if (pos_llh->flags == 2)
290  state.status = AP_GPS::GPS_OK_FIX_3D_DGPS;
291  else if (pos_llh->flags == 1)
292  state.status = AP_GPS::GPS_OK_FIX_3D_RTK;
293 
294 
295  last_full_update_tow = last_vel_ned.tow;
296  last_full_update_cpu_ms = now;
297 
298  logging_log_full_update();
299  ret = true;
300 
301  } else if (now - last_full_update_cpu_ms > SBP_TIMEOUT_PVT) {
302 
303  //INVARIANT: If we currently have a fix, ONLY return true after a full update.
304 
305  state.status = AP_GPS::NO_FIX;
306  ret = true;
307 
308  } else {
309 
310  //No timeouts yet, no data yet, nothing has happened.
311  ret = false;
312 
313  }
314 
315 
316  return ret;
317 
318 }

322 bool
323 AP_GPS_SBP::_detect(struct SBP_detect_state &state, uint8_t data)
324 {
325  // This switch reads one character at a time, if we find something that
326  // looks like our preamble we'll try to read the full message length,
327  // calculating the CRC. If the CRC matches, we have an SBP GPS!
328 
329  switch(state.state) {
330  case SBP_detect_state::WAITING:
331  if (data == SBP_PREAMBLE) {
332  state.n_read = 0;
333  state.crc_so_far = 0;
334  state.state = SBP_detect_state::GET_TYPE;
335  }
336  break;
337 
338  case SBP_detect_state::GET_TYPE:
339  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
340  state.n_read += 1;
341  if (state.n_read >= 2) {
342  state.n_read = 0;
343  state.state = SBP_detect_state::GET_SENDER;
344  }
345  break;
346 
347  case SBP_detect_state::GET_SENDER:
348  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
349  state.n_read += 1;
350  if (state.n_read >= 2) {
351  state.n_read = 0;
352  state.state = SBP_detect_state::GET_LEN;
353  }
354  break;
355 
356  case SBP_detect_state::GET_LEN:
357  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
358  state.msg_len = data;
359  state.n_read = 0;
360  state.state = SBP_detect_state::GET_MSG;
361  break;
362 
363  case SBP_detect_state::GET_MSG:
364  state.crc_so_far = crc16_ccitt(&data, 1, state.crc_so_far);
365  state.n_read += 1;
366  if (state.n_read >= state.msg_len) {
367  state.n_read = 0;
368  state.state = SBP_detect_state::GET_CRC;
369  }
370  break;
371 
372  case SBP_detect_state::GET_CRC:
373  *((uint8_t*)&(state.crc) + state.n_read) = data;
374  state.n_read += 1;
375  if (state.n_read >= 2) {
376  state.state = SBP_detect_state::WAITING;
377  return state.crc == state.crc_so_far;
378  }
379  break;
380 
381  default:
382  state.state = SBP_detect_state::WAITING;
383  break;
384  }
385  return false;
386 }
