69 void HAL_FLYMAPLE::run(int argc, char* const argv[], Callbacks* callbacks) const
70 {
71  assert(callbacks);
72 
73  /* initialize all drivers and private members here.
74  * up to the programmer to do this in the correct order.
75  * Scheduler should likely come first. */
76  scheduler->init(NULL);
77 
78  /* uartA is the serial port used for the console, so lets make sure
79  * it is initialized at boot */
80  uartA->begin(115200);
81 
82  rcin->init(NULL);
83  rcout->init(NULL);
84  spi->init(NULL);
85  i2c->begin();
86  i2c->setTimeout(100);
87  analogin->init(NULL);
88  storage->init(NULL); // Uses EEPROM.*, flash_stm* copied from AeroQuad_v3.2
89 
90  callbacks->setup();
91  scheduler->system_initialized();
92 
93  for (;;) {
94  callbacks->loop();
95  }
96 }
