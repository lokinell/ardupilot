114 /*
115  start an autotune session
116 */
117 void AP_AutoTune::start(void)
118 {
119  running = true;
120  state = DEMAND_UNSATURATED;
121  uint32_t now = hal.scheduler->millis();
122 
123  state_enter_ms = now;
124  last_save_ms = now;
125 
126  last_save = current;
127  restore = current;
128 
129  uint8_t level = aparm.autotune_level;
130  if (level > ARRAY_SIZE(tuning_table)) {
131  level = ARRAY_SIZE(tuning_table);
132  }
133  if (level < 1) {
134  level = 1;
135  }
136 
137  current.rmax.set(pgm_read_float(&tuning_table[level-1].rmax));
138  // D gain is scaled to a fixed ratio of P gain
139  current.D.set( pgm_read_float(&tuning_table[level-1].Dratio) * current.P);
140  current.tau.set( pgm_read_float(&tuning_table[level-1].tau));
141 
142  current.imax = constrain_float(current.imax, AUTOTUNE_MIN_IMAX, AUTOTUNE_MAX_IMAX);
143 
144  // force a fixed ratio of I to D gain on the rate feedback path
145  current.I = 0.5f * current.D / current.tau;
146 
147  next_save = current;
148 
149  Debug("START P -> %.3f\n", current.P.get());
150 }

164 /*
165  one update cycle of the autotuner
166  */
167 void AP_AutoTune::update(float desired_rate, float achieved_rate, float servo_out)
168 {
169  if (!running) {
170  return;
171  }
172  check_save();
173 
174  // see what state we are in
175  ATState new_state;
176  float abs_desired_rate = fabsf(desired_rate);
177  uint32_t now = hal.scheduler->millis();
178 
179  if (fabsf(servo_out) >= 45) {
180  // we have saturated the servo demand (not including
181  // integrator), we cannot get any information that would allow
182  // us to increase the gain
183  saturated_surfaces = true;
184  }
185 
186  if (abs_desired_rate < 0.8f * current.rmax) {
187  // we are not demanding max rate
188  new_state = DEMAND_UNSATURATED;
189  } else if (fabsf(achieved_rate) > abs_desired_rate) {
190  new_state = desired_rate > 0 ? DEMAND_OVER_POS : DEMAND_OVER_NEG;
191  } else {
192  new_state = desired_rate > 0 ? DEMAND_UNDER_POS : DEMAND_UNDER_NEG;
193  }
194  if (new_state != state) {
195  check_state_exit(now - state_enter_ms);
196  state = new_state;
197  state_enter_ms = now;
198  saturated_surfaces = false;
199  }
200  if (state != DEMAND_UNSATURATED) {
201  write_log(servo_out, desired_rate, achieved_rate);
202  }
203 }

205 /*
206  called when we change state to see if we should change gains
207  */
208 void AP_AutoTune::check_state_exit(uint32_t state_time_ms)
209 {
210  switch (state) {
211  case DEMAND_UNSATURATED:
212  break;
213  case DEMAND_UNDER_POS:
214  case DEMAND_UNDER_NEG:
215  // we increase P if we have not saturated the surfaces during
216  // this state, and we have
217  if (state_time_ms >= AUTOTUNE_UNDERSHOOT_TIME && !saturated_surfaces) {
218  current.P.set(current.P * (100+AUTOTUNE_INCREASE_STEP) * 0.01f);
219  if (current.P > AUTOTUNE_MAX_P) {
220  current.P = AUTOTUNE_MAX_P;
221  }
222  Debug("UNDER P -> %.3f\n", current.P.get());
223  }
224  current.D.set( pgm_read_float(&tuning_table[aparm.autotune_level-1].Dratio) * current.P);
225  break;
226  case DEMAND_OVER_POS:
227  case DEMAND_OVER_NEG:
228  if (state_time_ms >= AUTOTUNE_OVERSHOOT_TIME) {
229  current.P.set(current.P * (100-AUTOTUNE_DECREASE_STEP) * 0.01f);
230  if (current.P < AUTOTUNE_MIN_P) {
231  current.P = AUTOTUNE_MIN_P;
232  }
233  Debug("OVER P -> %.3f\n", current.P.get());
234  }
235  current.D.set( pgm_read_float(&tuning_table[aparm.autotune_level-1].Dratio) * current.P);
236  break;
237  }
238 }

240 /*
241  see if we should save new values
242  */
243 void AP_AutoTune::check_save(void)
244 {
245  if (hal.scheduler->millis() - last_save_ms < AUTOTUNE_SAVE_PERIOD) {
246  return;
247  }
248 
249  // save the next_save values, which are the autotune value from
250  // the last save period. This means the pilot has
251  // AUTOTUNE_SAVE_PERIOD milliseconds to decide they don't like the
252  // gains and switch out of autotune
253  ATGains tmp = current;
254 
255  save_gains(next_save);
256  Debug("SAVE P -> %.3f\n", current.P.get());
257 
258  // restore our current gains
259  current = tmp;
260 
261  // if the pilot exits autotune they get these saved values
262  restore = next_save;
263 
264  // the next values to save will be the ones we are flying now
265  next_save = current;
266  last_save_ms = hal.scheduler->millis();
267 }
