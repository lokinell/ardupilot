511 // Update Filter States - this should be called whenever new IMU data is available
512 void NavEKF2::UpdateFilter(void)
513 {
514  if (!core) {
515  return;
516  }
517 
518  const AP_InertialSensor &ins = _ahrs->get_ins();
519 
520  for (uint8_t i=0; i<num_cores; i++) {
521  // if the previous core has only recently finished a new state prediction cycle, then
522  // dont start a new cycle to allow time for fusion operations to complete if the update
523  // rate is higher than 200Hz
524  bool statePredictEnabled;
525  if ((i > 0) && (core[i-1].getFramesSincePredict() < 2) && (ins.get_sample_rate() > 200)) {
526  statePredictEnabled = false;
527  } else {
528  statePredictEnabled = true;
529  }
530  core[i].UpdateFilter(statePredictEnabled);
531  }

533  // If the current core selected has a bad fault score or is unhealthy, switch to a healthy core with the lowest fault score
534  if (core[primary].faultScore() > 0.0f || !core[primary].healthy()) {
535  float score = 1e9f;
536  for (uint8_t i=0; i<num_cores; i++) {
537  if (core[i].healthy()) {
538  float tempScore = core[i].faultScore();
539  if (tempScore < score) {
540  primary = i;
541  score = tempScore;
542  }
543  }
544  }
