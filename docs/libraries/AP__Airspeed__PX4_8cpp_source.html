48 // read the airspeed sensor
49 bool AP_Airspeed_PX4::get_differential_pressure(float &pressure)
50 {
51  if (_fd == -1) {
52  return false;
53  }
54 
55  // read from the PX4 airspeed sensor
56  float psum = 0;
57  float tsum = 0;
58  uint16_t count = 0;
59  struct differential_pressure_s report;
60 
61  while (::read(_fd, &report, sizeof(report)) == sizeof(report) &&
62  report.timestamp != _last_timestamp) {
63  psum += report.differential_pressure_raw_pa;
64  tsum += report.temperature;
65  count++;
66  _last_timestamp = report.timestamp;
67  }
68  if (count == 0) {
69  return false;
70  }
71  pressure = psum / count;
72  _temperature = tsum / count;
73  return true;
74 }
