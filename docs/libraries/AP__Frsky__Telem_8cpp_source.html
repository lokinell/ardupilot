96 /*
97  send_frames - sends updates down telemetry link for both DPORT and SPORT protocols
98  should be called by main program at 50hz to allow poll for serial bytes
99  coming from the receiver for the SPort protocol
100 */
101 void AP_Frsky_Telem::send_frames(uint8_t control_mode)
102 {
103  // return immediately if not initialised
104  if (!_initialised_uart) {
105  return;
106  }
107 
108  if (_protocol == FrSkySPORT) {
109  if (!_mode_data_ready) {
110  _mode=control_mode;
111  _mode_data_ready = true;
112  }
113  if (!_baro_data_ready) {
114  calc_baro_alt();
115  _baro_data_ready = true;
116  }
117  if (!_gps_data_ready) {
118  calc_gps_position();
119  _gps_data_ready = true;
120  }
121  if (!_sats_data_ready) {
122  calc_gps_sats();
123  _sats_data_ready = true;
124  }
125  if (!_battery_data_ready) {
126  calc_battery();
127  _battery_data_ready = true;
128  }
129  } else {
130  _mode=control_mode;
131  send_hub_frame();
132  }
133 }

152 /*
153  send_hub_frame - send frame1 and frame2 when protocol is FrSkyDPORT
154  frame 1 is sent every 200ms with baro alt, nb sats, batt volts and amp, control_mode
155  frame 2 is sent every second with gps position data
156 */
157 void AP_Frsky_Telem::send_hub_frame()
158 {
159  uint32_t now = hal.scheduler->millis();
160 
161  // send frame1 every 200ms
162  if (now - _last_frame1_ms > 200) {
163  _last_frame1_ms = now;
164  calc_gps_sats();
165  send_gps_sats();
166  send_mode();
167 
168  calc_battery();
169  send_batt_remain();
170  send_batt_volts();
171  send_current();
172 
173  calc_baro_alt();
174  send_baro_alt_m();
175  send_baro_alt_cm();
176  }
177  // send frame2 every second
178  if (now - _last_frame2_ms > 1000) {
179  _last_frame2_ms = now;
180  send_heading();
181  calc_gps_position();
182  if (_pos_gps_ok) {
183  send_gps_lat_dd();
184  send_gps_lat_mm();
185  send_gps_lat_ns();
186  send_gps_lon_dd();
187  send_gps_lon_mm();
188  send_gps_lon_ew();
189  send_gps_speed_meter();
190  send_gps_speed_cm();
191  send_gps_alt_meter();
192  send_gps_alt_cm();
193  }
194  }
195 }

197 /*
198  sport_tick - main call to send updates to transmitter when protocol is FrSkySPORT
199  called by scheduler at a high rate
200 */
201 void AP_Frsky_Telem::sport_tick(void)
202 {
203  // check UART has been initialised
204  if (!_initialised_uart) {
205  init_uart_for_sport();
206  }
207 
208  int16_t numc;
209  numc = _port->available();
210 
211  // check if available is negative
212  if (numc < 0) {
213  return;
214  }
215 
216  // this is the constant for hub data frame
217  if (_port->txspace() < 19) {
218  return;
219  }
220 
221  for (int16_t i = 0; i < numc; i++) {
222  int16_t readbyte = _port->read();
223  if (_sport_status == 0) {
224  if (readbyte == SPORT_START_FRAME) {
225  _sport_status = 1;
226  }
227  } else {
228  switch (readbyte) {
229  case DATA_ID_FAS:
230  if (_battery_data_ready) {
231  switch (_fas_call) {
232  case 0:
233  send_batt_volts();
234  break;
235  case 1:
236  send_current();
237  break;
238  }
239  _fas_call++;
240  if (_fas_call > 1) {
241  _fas_call = 0;
242  }
243  _battery_data_ready = false;
244  }
245  break;
246  case DATA_ID_GPS:
247  if (_gps_data_ready) {
248  switch (_gps_call) {
249  case 0:
250  send_gps_lat_dd();
251  break;
252  case 1:
253  send_gps_lat_mm();
254  break;
255  case 2:
256  send_gps_lat_ns();
257  break;
258  case 3:
259  send_gps_lon_dd();
260  break;
261  case 4:
262  send_gps_lon_mm();
263  break;
264  case 5:
265  send_gps_lon_ew();
266  break;
267  case 6:
268  send_gps_speed_meter();
269  break;
270  case 7:
271  send_gps_speed_cm();
272  break;
273  case 8:
274  send_gps_alt_meter();
275  break;
276  case 9:
277  send_gps_alt_cm();
278  break;
279  case 10:
280  send_heading();
281  break;
282  }
283 
284  _gps_call++;
285  if (_gps_call > 10) {
286  _gps_call = 0;
287  _gps_data_ready = false;
288  }
289  }
290  break;
291  case DATA_ID_VARIO:
292  if (_baro_data_ready) {
293  switch (_vario_call) {
294  case 0 :
295  send_baro_alt_m();
296  break;
297  case 1:
298  send_baro_alt_cm();
299  break;
300  }
301  _vario_call ++;
302  if (_vario_call > 1) {
303  _vario_call = 0;
304  _baro_data_ready = false;
305  }
306  }
307  break;
308  case DATA_ID_SP2UR:
309  switch (_various_call) {
310  case 0 :
311  if ( _sats_data_ready ) {
312  send_gps_sats();
313  _sats_data_ready = false;
314  }
315  break;
316  case 1:
317  if ( _mode_data_ready ) {
318  send_mode();
319  _mode_data_ready = false;
320  }
321  break;
322  }
323  _various_call++;
324  if (_various_call > 1) {
325  _various_call = 0;
326  }
327  break;
328  }
329  _sport_status = 0;
330  }
331  }

357 /*
358  send 1 byte and do the byte stuffing Frsky stuff
359  This can send more than 1 byte eventually
360 */
361 void AP_Frsky_Telem::frsky_send_byte(uint8_t value)
362 {
363  if (_protocol == FrSkyDPORT) {
364  const uint8_t x5E[] = { 0x5D, 0x3E };
365  const uint8_t x5D[] = { 0x5D, 0x3D };
366  switch (value) {
367  case 0x5E:
368  _port->write( x5E, sizeof(x5E));
369  break;
370 
371  case 0x5D:
372  _port->write( x5D, sizeof(x5D));
373  break;
374 
375  default:
376  _port->write(&value, sizeof(value));
377  break;
378  }
379  } else {
380  //SPORT
381  calc_crc(value);
382  const uint8_t x7E[] = { 0x7D, 0x5E };
383  const uint8_t x7D[] = { 0x7D, 0x5D };
384  switch (value) {
385  case 0x7E:
386  _port->write( x7E, sizeof(x7E));
387  break;
388 
389  case 0x7D:
390  _port->write( x7D, sizeof(x7D));
391  break;
392 
393  default:
394  _port->write(&value, sizeof(value));
395  break;
396  }
397  }

422 void AP_Frsky_Telem::frsky_send_data(uint8_t id, int16_t data)
423 {
424  static const uint8_t zero = 0x0;
425 
426  /* Cast data to unsigned, because signed shift might behave incorrectly */
427  uint16_t udata = data;
428 
429  if (_protocol == FrSkySPORT) {
430  frsky_send_sport_prim();
431  frsky_send_byte(id);
432  frsky_send_byte(zero);
433  } else {
434  frsky_send_hub_startstop();
435  frsky_send_byte(id);
436  }
437 
438  frsky_send_byte(udata); /* LSB */
439  frsky_send_byte(udata >> 8); /* MSB */
440 
441  if (_protocol == FrSkySPORT) {
442  //Sport expect 32 bits data but we use only 16 bits data, so we send 0 for MSB
443  frsky_send_byte(zero);
444  frsky_send_byte(zero);
445  send_crc();
446  }
447 }

450  * calc_baro_alt : send altitude in Meters based on ahrs estimate
451  */
452 void AP_Frsky_Telem::calc_baro_alt()
453 {
454  struct Location loc;
455  float baro_alt = 0; // in meters
456  bool posok = _ahrs.get_position(loc);
457  if (posok) {
458  baro_alt = loc.alt * 0.01f; // convert to meters
459  if (!loc.flags.relative_alt) {
460  baro_alt -= _ahrs.get_home().alt * 0.01f; // subtract home if set
461  }
462  }
463  /*
464  Note that this isn't actually barometric altitude, it is the
465  inertial nav estimate of altitude above home.
466  */
467  _baro_alt_meters = (int16_t)baro_alt;
468  _baro_alt_cm = (baro_alt - abs(_baro_alt_meters)) * 100;
469 }

484 void AP_Frsky_Telem::calc_gps_position()
485 {
486  _course_in_degrees = (_ahrs.yaw_sensor / 100) % 360;
487 
488  const AP_GPS &gps = _ahrs.get_gps();
489  float lat;
490  float lon ;
491  float alt ;
492  float speed;
493  _pos_gps_ok = (gps.status() >= 3);
494  if (_pos_gps_ok) {
495  Location loc = gps.location();//get gps instance 0
496 
497  lat = frsky_format_gps(fabsf(loc.lat/10000000.0f));
498  _latdddmm = lat;
499  _latmmmm = (lat - _latdddmm) * 10000;
500  _lat_ns = (loc.lat < 0) ? 'S' : 'N';
501 
502  lon = frsky_format_gps(fabsf(loc.lng/10000000.0f));
503  _londddmm = lon;
504  _lonmmmm = (lon - _londddmm) * 10000;
505  _lon_ew = (loc.lng < 0) ? 'W' : 'E';
506 
507  alt = loc.alt * 0.01f;
508  _alt_gps_meters = (int16_t)alt;
509  _alt_gps_cm = (alt - abs(_alt_gps_meters)) * 100;
510 
511  speed = gps.ground_speed();
512  _speed_in_meter = speed;
513  _speed_in_centimeter = (speed - _speed_in_meter) * 100;
514  } else {
515  _latdddmm = 0;
516  _latmmmm = 0;
517  _lat_ns = 0;
518  _londddmm = 0;
519  _lonmmmm = 0;
520  _alt_gps_meters = 0;
521  _alt_gps_cm = 0;
522  _speed_in_meter = 0;
523  _speed_in_centimeter = 0;
524  }
525 }
