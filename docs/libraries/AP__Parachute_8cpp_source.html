85 void AP_Parachute::update()
86 {
87  // exit immediately if not enabled or parachute not to be released
88  if (_enabled <= 0) {
89  return;
90  }
91 
92  // calc time since release
93  uint32_t time_diff = hal.scheduler->millis() - _release_time;
94 
95  // check if we should release parachute
96  if ((_release_time != 0) && !_release_in_progress) {
97  if (time_diff >= AP_PARACHUTE_RELEASE_DELAY_MS) {
98  if (_release_type == AP_PARACHUTE_TRIGGER_TYPE_SERVO) {
99  // move servo
100  RC_Channel_aux::set_radio(RC_Channel_aux::k_parachute_release, _servo_on_pwm);
101  }else if (_release_type <= AP_PARACHUTE_TRIGGER_TYPE_RELAY_3) {
102  // set relay
103  _relay.on(_release_type);
104  }
105  _release_in_progress = true;
106  _released = true;
107  }
108  }else if ((_release_time == 0) || time_diff >= AP_PARACHUTE_RELEASE_DELAY_MS + AP_PARACHUTE_RELEASE_DURATION_MS) {
109  if (_release_type == AP_PARACHUTE_TRIGGER_TYPE_SERVO) {
110  // move servo back to off position
111  RC_Channel_aux::set_radio(RC_Channel_aux::k_parachute_release, _servo_off_pwm);
112  }else if (_release_type <= AP_PARACHUTE_TRIGGER_TYPE_RELAY_3) {
113  // set relay back to zero volts
114  _relay.off(_release_type);
115  }
116  // reset released flag and release_time
117  _release_in_progress = false;
118  _release_time = 0;
119  // update AP_Notify
120  AP_Notify::flags.parachute_release = 0;
121  }
122 }
