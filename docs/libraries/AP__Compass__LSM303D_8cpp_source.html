286 bool
287 AP_Compass_LSM303D::init()
288 {
289  // TODO: support users without data ready pin
290  if (LSM303D_DRDY_M_PIN < 0)
291  return false;
292 
293  hal.scheduler->suspend_timer_procs();
294 
295  _spi = hal.spi->device(AP_HAL::SPIDevice_LSM303D);
296  _spi_sem = _spi->get_semaphore();
297 
298  _drdy_pin_m = hal.gpio->channel(LSM303D_DRDY_M_PIN);
299  _drdy_pin_m->mode(HAL_GPIO_INPUT);
300 
301  // Test WHOAMI
302  uint8_t whoami = _register_read(ADDR_WHO_AM_I);
303  if (whoami != WHO_I_AM) {
304  hal.console->printf("LSM303D: unexpected WHOAMI 0x%x\n", (unsigned)whoami);
305  hal.scheduler->panic("LSM303D: bad WHOAMI");
306  }
307 
308  uint8_t tries = 0;
309  do {
310  // TODO: don't try to init 25 times
311  bool success = _hardware_init();
312  if (success) {
313  hal.scheduler->delay(5+2);
314  if (!_spi_sem->take(100)) {
315  hal.scheduler->panic("LSM303D: Unable to get semaphore");
316  }
317  if (_data_ready()) {
318  _spi_sem->give();
319  break;
320  } else {
321  hal.console->println(
322  "LSM303D startup failed: no data ready");
323  }
324  _spi_sem->give();
325  }
326  if (tries++ > 5) {
327  hal.scheduler->panic("PANIC: failed to boot LSM303D 5 times");
328  }
329  } while (1);
330 
331  _scaling[0] = 1.0;
332  _scaling[1] = 1.0;
333  _scaling[2] = 1.0;
334 
335  /* register the compass instance in the frontend */
336  _compass_instance = register_compass();
337  set_dev_id(_compass_instance, get_dev_id());
338 #if CONFIG_HAL_BOARD == HAL_BOARD_LINUX && CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_RASPILOT
339  set_external(_compass_instance, false);
340 #endif
341 
342  hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_Compass_LSM303D::_update, void));
343 
344  _spi_sem->give();
345  hal.scheduler->resume_timer_procs();
346  _initialised = true;
347 
348  return _initialised;
349 }

356 bool AP_Compass_LSM303D::_hardware_init(void)
357 {
358  if (!_spi_sem->take(100)) {
359  hal.scheduler->panic("LSM303D: Unable to get semaphore");
360  }
361 
362  // initially run the bus at low speed
363  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
364 
365  // ensure the chip doesn't interpret any other bus traffic as I2C
366  _disable_i2c();
367 
368  /* enable mag */
369  _reg7_expected = REG7_CONT_MODE_M;
370  _register_write(ADDR_CTRL_REG7, _reg7_expected);
371  _register_write(ADDR_CTRL_REG5, REG5_RES_HIGH_M);
372  _register_write(ADDR_CTRL_REG4, 0x04); // DRDY on MAG on INT2
373 
374  _mag_set_range(LSM303D_MAG_DEFAULT_RANGE_GA);
375  _mag_set_samplerate(LSM303D_MAG_DEFAULT_RATE);
376 
377  // TODO: Software filtering
378 
379  // now that we have initialised, we set the SPI bus speed to high
380  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_HIGH);
381  _spi_sem->give();
382 
383  return true;
384 }

402 void AP_Compass_LSM303D::_collect_samples()
403 {
404  if (!_initialised) {
405  return;
406  }
407 
408  if (!_read_raw()) {
409  error("_read_raw() failed\n");
410  } else {
411  Vector3f raw_field = Vector3f(_mag_x, _mag_y, _mag_z) * _mag_range_scale;
412  uint32_t time_us = hal.scheduler->micros();
413 
414  // rotate raw_field from sensor frame to body frame
415  rotate_field(raw_field, _compass_instance);
416 
417  // publish raw_field (uncorrected point sample) for _scaling use
418  publish_raw_field(raw_field, time_us, _compass_instance);
419 
420  // correct raw_field for known errors
421  correct_field(raw_field, _compass_instance);
422 
423  // publish raw_field (corrected point sample) for EKF use
424  publish_unfiltered_field(raw_field, time_us, _compass_instance);
425 
426  _mag_x_accum += raw_field.x;
427  _mag_y_accum += raw_field.y;
428  _mag_z_accum += raw_field.z;
429  _accum_count++;
430  if (_accum_count == 10) {
431  _mag_x_accum /= 2;
432  _mag_y_accum /= 2;
433  _mag_z_accum /= 2;
434  _accum_count = 5;
435  }
436  }

440 void AP_Compass_LSM303D::read()
441 {
442  if (!_initialised) {
443  // someone has tried to enable a compass for the first time
444  // mid-flight .... we can't do that yet (especially as we won't
445  // have the right orientation!)
446  return;
447  }
448 
449  if (_accum_count == 0) {
450  /* We're not ready to publish*/
451  return;
452  }
453 
454  hal.scheduler->suspend_timer_procs();
455  Vector3f field(_mag_x_accum * _scaling[0],
456  _mag_y_accum * _scaling[1],
457  _mag_z_accum * _scaling[2]);
458  field /= _accum_count;
459 
460  _accum_count = 0;
461  _mag_x_accum = _mag_y_accum = _mag_z_accum = 0;
462  hal.scheduler->resume_timer_procs();
463 
464  publish_filtered_field(field, _compass_instance);
465 }
