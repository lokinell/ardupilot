121 AP_Declination::get_declination(float lat, float lon)
122 {
123  int16_t decSW, decSE, decNW, decNE, lonmin, latmin;
124  uint8_t latmin_index,lonmin_index;
125  float decmin, decmax;
126 
127  // Constrain to valid inputs
128  lat = constrain_float(lat, -90, 90);
129  lon = constrain_float(lon, -180, 180);
130 
131  latmin = floorf(lat/5)*5;
132  lonmin = floorf(lon/5)*5;
133 
134  latmin_index= (90+latmin)/5;
135  lonmin_index= (180+lonmin)/5;
136 
137  decSW = get_lookup_value(latmin_index, lonmin_index);
138  decSE = get_lookup_value(latmin_index, lonmin_index+1);
139  decNE = get_lookup_value(latmin_index+1, lonmin_index+1);
140  decNW = get_lookup_value(latmin_index+1, lonmin_index);
141 
142  /* approximate declination within the grid using bilinear interpolation */
143  decmin = (lon - lonmin) / 5 * (decSE - decSW) + decSW;
144  decmax = (lon - lonmin) / 5 * (decNE - decNW) + decNW;
145  return (lat - latmin) / 5 * (decmax - decmin) + decmin;
146 }

149 AP_Declination::get_lookup_value(uint8_t x, uint8_t y)
150 {
151  // return value
152  int16_t val = 0;
153 
154  // These are exception indicies
155  if(x <= 6 || x >= 34)
156  {
157  // If the x index is in the upper range we need to translate it
158  // to match the 10 indicies in the exceptions lookup table
159  if(x >= 34) x -= 27;
160 
161  // Read the unsigned value from the array
162  val = PGM_UINT8(&exceptions[x][y]);
163 
164  // Read the 8 bit compressed sign values
165  uint8_t sign = PGM_UINT8(&exception_signs[x][y/8]);
166 
167  // Check the sign bit for this index
168  if(sign & (0x80 >> y%8))
169  val = -val;
170 
171  return val;
172  }
173 
174  // Because the values were removed from the start of the
175  // original array (0-6) to the exception array, all the indicies
176  // in this main lookup need to be shifted left 7
177  // EX: User enters 7 -> 7 is the first row in this array so it needs to be zero
178  if(x >= 7) x -= 7;
179 
180  // If we are looking for the first value we can just use the
181  // row start value from declination_keys
182  if(y == 0) return PGM_UINT8(&declination_keys[0][x]);
183 
184  // Init vars
185  row_value stval;
186  int16_t offset = 0;
187 
188  // These will never exceed the second dimension length of 73
189  uint8_t current_virtual_index = 0, r;
190 
191  // This could be the length of the array or less (1075 or less)
192  uint16_t start_index = 0, i;
193 
194  // Init value to row start
195  val = PGM_UINT8(&declination_keys[0][x]);
196 
197  // Find the first element in the 1D array
198  // that corresponds with the target row
199  for(i = 0; i < x; i++) {
200  start_index += PGM_UINT8(&declination_keys[1][i]);
201  }
202 
203  // Traverse the row until we find our value
204  for(i = start_index; i < (start_index + PGM_UINT8(&declination_keys[1][x])) && current_virtual_index <= y; i++) {
205 
206  // Pull out the row_value struct
207  memcpy((void*) &stval, (const char *)&declination_values[i], sizeof(struct row_value));
208 
209  // Pull the first offset and determine sign
210  offset = stval.abs_offset;
211  offset = (stval.offset_sign == 1) ? -offset : offset;
212 
213  // Add offset for each repeat
214  // This will at least run once for zero repeat
215  for(r = 0; r <= stval.repeats && current_virtual_index <= y; r++) {
216  val += offset;
217  current_virtual_index++;
218  }
219  }
