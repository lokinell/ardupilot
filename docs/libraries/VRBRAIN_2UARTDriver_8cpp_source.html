44 void VRBRAINUARTDriver::begin(uint32_t b, uint16_t rxS, uint16_t txS)
45 {
46  if (strcmp(_devpath, "/dev/null") == 0) {
47  // leave uninitialised
48  return;
49  }
50 
51  uint16_t min_tx_buffer = 1024;
52  uint16_t min_rx_buffer = 512;
53  if (strcmp(_devpath, "/dev/ttyACM0") == 0) {
54  min_tx_buffer = 16384;
55  min_rx_buffer = 1024;
56  }
57  // on VRBRAIN we have enough memory to have a larger transmit and
58  // receive buffer for all ports. This means we don't get delays
59  // while waiting to write GPS config packets
60  if (txS < min_tx_buffer) {
61  txS = min_tx_buffer;
62  }
63  if (rxS < min_rx_buffer) {
64  rxS = min_rx_buffer;
65  }
66 
67  /*
68  allocate the read buffer
69  we allocate buffers before we successfully open the device as we
70  want to allocate in the early stages of boot, and cause minimum
71  thrashing of the heap once we are up. The ttyACM0 driver may not
72  connect for some time after boot
73  */
74  if (rxS != 0 && rxS != _readbuf_size) {
75  _initialised = false;
76  while (_in_timer) {
77  hal.scheduler->delay(1);
78  }
79  _readbuf_size = rxS;
80  if (_readbuf != NULL) {
81  free(_readbuf);
82  }
83  _readbuf = (uint8_t *)malloc(_readbuf_size);
84  _readbuf_head = 0;
85  _readbuf_tail = 0;
86  }
87 
88  if (b != 0) {
89  _baudrate = b;
90  }
91 
92  /*
93  allocate the write buffer
94  */
95  if (txS != 0 && txS != _writebuf_size) {
96  _initialised = false;
97  while (_in_timer) {
98  hal.scheduler->delay(1);
99  }
100  _writebuf_size = txS;
101  if (_writebuf != NULL) {
102  free(_writebuf);
103  }
104  _writebuf = (uint8_t *)malloc(_writebuf_size+16);
105  _writebuf_head = 0;
106  _writebuf_tail = 0;
107  }
108 
109  if (_fd == -1) {
110  _fd = open(_devpath, O_RDWR);
111  if (_fd == -1) {
112  return;
113  }
114  }
115 
116  if (_baudrate != 0) {
117  // set the baud rate
118  struct termios t;
119  tcgetattr(_fd, &t);
120  cfsetspeed(&t, _baudrate);
121  // disable LF -> CR/LF
122  t.c_oflag &= ~ONLCR;
123  tcsetattr(_fd, TCSANOW, &t);
124 
125  // separately setup IFLOW if we can. We do this as a 2nd call
126  // as if the port has no RTS pin then the tcsetattr() call
127  // will fail, and if done as one call then it would fail to
128  // set the baudrate.
129  tcgetattr(_fd, &t);
130  t.c_cflag |= CRTS_IFLOW;
131  tcsetattr(_fd, TCSANOW, &t);
132  }
133 
134  if (_writebuf_size != 0 && _readbuf_size != 0 && _fd != -1) {
135  if (!_initialised) {
136  ::printf("initialised %s OK %u %u\n", _devpath,
137  (unsigned)_writebuf_size, (unsigned)_readbuf_size);
138  }
139  _initialised = true;

306 size_t VRBRAINUARTDriver::write(const uint8_t *buffer, size_t size)
307 {
308  if (!_initialised) {
309  try_initialise();
310  return 0;
311  }
312  if (hal.scheduler->in_timerprocess()) {
313  // not allowed from timers
314  return 0;
315  }
316 
317  if (!_nonblocking_writes) {
318  /*
319  use the per-byte delay loop in write() above for blocking writes
320  */
321  size_t ret = 0;
322  while (size--) {
323  if (write(*buffer++) != 1) break;
324  ret++;
325  }
326  return ret;
327  }
328 
329  uint16_t _head, space;
330  space = BUF_SPACE(_writebuf);
331  if (space == 0) {
332  return 0;
333  }
334  if (size > space) {
335  size = space;
336  }
337  if (_writebuf_tail < _head) {
338  // perform as single memcpy
339  assert(_writebuf_tail+size <= _writebuf_size);
340  memcpy(&_writebuf[_writebuf_tail], buffer, size);
341  BUF_ADVANCETAIL(_writebuf, size);
342  return size;
343  }
344 
345  // perform as two memcpy calls
346  uint16_t n = _writebuf_size - _writebuf_tail;
347  if (n > size) n = size;
348  assert(_writebuf_tail+n <= _writebuf_size);
349  memcpy(&_writebuf[_writebuf_tail], buffer, n);
350  BUF_ADVANCETAIL(_writebuf, n);
351  buffer += n;
352  n = size - n;
353  if (n > 0) {
354  assert(_writebuf_tail+n <= _writebuf_size);
355  memcpy(&_writebuf[_writebuf_tail], buffer, n);
356  BUF_ADVANCETAIL(_writebuf, n);
357  }

362  try writing n bytes, handling an unresponsive port

364 int VRBRAINUARTDriver::_write_fd(const uint8_t *buf, uint16_t n)
365 {
366  int ret = 0;
367 
368  // the FIONWRITE check is to cope with broken O_NONBLOCK behaviour
369  // in NuttX on ttyACM0
370 
371  // FIONWRITE is also used for auto flow control detection
372  // Assume output flow control is not working if:
373  // port is configured for auto flow control
374  // and this is not the first write since flow control turned on
375  // and no data has been removed from the buffer since flow control turned on
376  // and more than .5 seconds elapsed after writing a total of > 5 characters
377  //
378 
379  int nwrite = 0;
380 
381  if (ioctl(_fd, FIONWRITE, (unsigned long)&nwrite) == 0) {
382  if (_flow_control == FLOW_CONTROL_AUTO) {
383  if (_first_write_time == 0) {
384  if (_total_written == 0) {
385  // save the remaining buffer bytes for comparison next write
386  _os_start_auto_space = nwrite;
387  }
388  } else {
389  if (_os_start_auto_space - nwrite + 1 >= _total_written &&
390  (hal.scheduler->micros64() - _first_write_time) > 500*1000UL) {
391  // it doesn't look like hw flow control is working
392  ::printf("disabling flow control on %s _total_written=%u\n",
393  _devpath, (unsigned)_total_written);
394  set_flow_control(FLOW_CONTROL_DISABLE);
395  }
396  }
397  }
398  if (nwrite > n) {
399  nwrite = n;
400  }
401  if (nwrite > 0) {
402  ret = ::write(_fd, buf, nwrite);
403  }
404  }
405 
406  if (ret > 0) {
407  BUF_ADVANCEHEAD(_writebuf, ret);
408  _last_write_time = hal.scheduler->micros64();
409  _total_written += ret;
410  if (! _first_write_time && _total_written > 5) {
411  _first_write_time = _last_write_time;
412  }
413  return ret;
414  }
415 
416  if (hal.scheduler->micros64() - _last_write_time > 2000 &&
417  _flow_control == FLOW_CONTROL_DISABLE) {
418 #if 0
419  // this trick is disabled for now, as it sometimes blocks on
420  // re-opening the ttyACM0 port, which would cause a crash
421  if (hal.scheduler->micros64() - _last_write_time > 2000000) {
422  // we haven't done a successful write for 2 seconds - try
423  // reopening the port
424  _initialised = false;
425  ::close(_fd);
426  _fd = ::open(_devpath, O_RDWR);
427  if (_fd == -1) {
428  fprintf(stdout, "Failed to reopen UART device %s - %s\n",
429  _devpath, strerror(errno));
430  // leave it uninitialised
431  return n;
432  }
433 
434  _last_write_time = hal.scheduler->micros64();
435  _initialised = true;
436  }
437 #else
438  _last_write_time = hal.scheduler->micros64();
439 #endif
440  // we haven't done a successful write for 2ms, which means the
441  // port is running at less than 500 bytes/sec. Start
442  // discarding bytes, even if this is a blocking port. This
443  // prevents the ttyACM0 port blocking startup if the endpoint
444  // is not connected
445  BUF_ADVANCEHEAD(_writebuf, n);
446  return n;
447  }
448  return ret;
449 }

482 void VRBRAINUARTDriver::_timer_tick(void)
483 {
484  uint16_t n;
485 
486  if (!_initialised) return;
487 
488  // don't try IO on a disconnected USB port
489  if (strcmp(_devpath, "/dev/ttyACM0") == 0 && !hal.gpio->usb_connected()) {
490  return;
491  }
492 
493  _in_timer = true;
494 
495  // write any pending bytes
496  uint16_t _tail;
497  n = BUF_AVAILABLE(_writebuf);
498  if (n > 0) {
499  uint16_t n1 = _writebuf_size - _writebuf_head;
500  perf_begin(_perf_uart);
501  if (n1 >= n) {
502  // do as a single write
503  _write_fd(&_writebuf[_writebuf_head], n);
504  } else {
505  // split into two writes
506  int ret = _write_fd(&_writebuf[_writebuf_head], n1);
507  if (ret == n1 && n > n1) {
508  _write_fd(&_writebuf[_writebuf_head], n - n1);
509  }
510  }
511  perf_end(_perf_uart);
512  }
513 
514  // try to fill the read buffer
515  uint16_t _head;
516  n = BUF_SPACE(_readbuf);
517  if (n > 0) {
518  uint16_t n1 = _readbuf_size - _readbuf_tail;
519  perf_begin(_perf_uart);
520  if (n1 >= n) {
521  // one read will do
522  assert(_readbuf_tail+n <= _readbuf_size);
523  _read_fd(&_readbuf[_readbuf_tail], n);
524  } else {
525  assert(_readbuf_tail+n1 <= _readbuf_size);
526  int ret = _read_fd(&_readbuf[_readbuf_tail], n1);
527  if (ret == n1 && n > n1) {
528  assert(_readbuf_tail+(n-n1) <= _readbuf_size);
529  _read_fd(&_readbuf[_readbuf_tail], n - n1);
530  }
531  }
532  perf_end(_perf_uart);
533  }
534
