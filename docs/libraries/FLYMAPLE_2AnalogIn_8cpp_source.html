52 void FLYMAPLEAnalogIn::_register_channel(FLYMAPLEAnalogSource* ch) {
53  if (_num_channels >= FLYMAPLE_INPUT_MAX_CHANNELS) {
54  for(;;) {
55  hal.console->print(
56  "Error: AP_HAL_FLYMAPLE::FLYMAPLEAnalogIn out of channels\r\n");
57  hal.scheduler->delay(1000);
58  }
59  }
60  _channels[_num_channels] = ch;
61  /* Need to lock to increment _num_channels as it is used
62  * by the interrupt to access _channels */
63  noInterrupts();
64  _num_channels++;
65  interrupts();
66  // Start conversions:
67  adc_reg_map *regs = ADC1->regs;
68  regs->CR2 |= ADC_CR2_SWSTART;
69 }

71 void FLYMAPLEAnalogIn::_timer_event(void)
72 {
73  adc_reg_map *regs = ADC1->regs;
74  if (_channels[_active_channel]->_pin == ANALOG_INPUT_NONE) {
75  _channels[_active_channel]->new_sample(0);
76  goto next_channel;
77  }
78 
79  if (!(regs->SR & ADC_SR_EOC))
80  {
81  /* ADC Conversion is still running - this should not happen, as we
82  * are called at 1khz. */
83  return;
84  }
85 
86  if (_num_channels == 0)
87  {
88  /* No channels are registered - nothing to be done. */
89  return;
90  }
91 
92  _channel_repeat_count++;
93  if (_channel_repeat_count < CHANNEL_READ_REPEAT ||
94  !_channels[_active_channel]->reading_settled())
95  {
96  /* Start a new conversion, throw away the current conversion */
97  regs->CR2 |= ADC_CR2_SWSTART;
98  return;
99  }
100 
101  _channel_repeat_count = 0;
102  {
103  // Need this block because of declared variabled and goto
104  uint16_t sample = (uint16)(regs->DR & ADC_DR_DATA);
105  /* Give the active channel a new sample */
106  _channels[_active_channel]->new_sample( sample );
107  }
108 next_channel:
109  /* stop the previous channel, if a stop pin is defined */
110  _channels[_active_channel]->stop_read();
111  /* Move to the next channel */
112  _active_channel = (_active_channel + 1) % _num_channels;
113  /* Setup the next channel's conversion */
114  _channels[_active_channel]->setup_read();
115  /* Start conversion */
116  regs->CR2 |= ADC_CR2_SWSTART;
117 }
