69 /*
70  initialise the AP_RPM class.
71  */
72 void AP_RPM::init(void)
73 {
74  if (num_instances != 0) {
75  // init called a 2nd time?
76  return;
77  }
78  for (uint8_t i=0; i<RPM_MAX_INSTANCES; i++) {
79 #if CONFIG_HAL_BOARD == HAL_BOARD_PX4 || CONFIG_HAL_BOARD == HAL_BOARD_VRBRAIN
80  uint8_t type = _type[num_instances];
81  uint8_t instance = num_instances;
82 
83  if (type == RPM_TYPE_PX4_PWM) {
84  state[instance].instance = instance;
85  drivers[instance] = new AP_RPM_PX4_PWM(*this, instance, state[instance]);
86  }
87 #endif
88  if (drivers[i] != NULL) {
89  // we loaded a driver for this instance, so it must be
90  // present (although it may not be healthy)
91  num_instances = i+1;
92  }
93  }
94 }

106  }
107  drivers[i]->update();
