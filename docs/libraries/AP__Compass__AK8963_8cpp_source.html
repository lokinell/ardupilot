127 bool AP_Compass_AK8963::init()
128 {
129  _bus_sem = _bus->get_semaphore();
130 
131  hal.scheduler->suspend_timer_procs();
132 
133  if (!_bus_sem->take(100)) {
134  hal.console->printf("AK8963: Unable to get bus semaphore");
135  goto fail_sem;
136  }
137 
138  if (!_check_id()) {
139  hal.console->printf("AK8963: Wrong id\n");
140  goto fail;
141  }
142 
143  if (!_calibrate()) {
144  hal.console->printf("AK8963: Could not read calibration data\n");
145  goto fail;
146  }
147 
148  if (!_setup_mode()) {
149  hal.console->printf("AK8963: Could not setup mode\n");
150  goto fail;
151  }
152 
153  if (!_bus->start_measurements()) {
154  hal.console->printf("AK8963: Could not start measurements\n");
155  goto fail;
156  }
157 
158  _initialized = true;
159 
160  /* register the compass instance in the frontend */
161  _compass_instance = register_compass();
162  set_dev_id(_compass_instance, _bus->get_dev_id());
163  hal.scheduler->register_timer_process(FUNCTOR_BIND_MEMBER(&AP_Compass_AK8963::_update, void));
164 
165  _bus_sem->give();
166  hal.scheduler->resume_timer_procs();
167 
168  return true;
169 
170 fail:
171  _bus_sem->give();
172 fail_sem:
173  hal.scheduler->resume_timer_procs();
174 
175  return false;
176 }

225 void AP_Compass_AK8963::_update()
226 {
227  struct AP_AK8963_SerialBus::raw_value rv;
228  float mag_x, mag_y, mag_z;
229  // get raw_field - sensor frame, uncorrected
230  Vector3f raw_field;
231  uint32_t time_us = hal.scheduler->micros();
232 
233 
234  if (hal.scheduler->micros() - _last_update_timestamp < 10000) {
235  goto end;
236  }
237 
238  if (!_sem_take_nonblocking()) {
239  goto end;
240  }
241 
242  _bus->read_raw(&rv);
243 
244  /* Check for overflow. See AK8963's datasheet, section
245  * 6.4.3.6 - Magnetic Sensor Overflow. */
246  if ((rv.st2 & 0x08)) {
247  goto fail;
248  }
249 
250  mag_x = (float) rv.val[0];
251  mag_y = (float) rv.val[1];
252  mag_z = (float) rv.val[2];
253 
254  if (is_zero(mag_x) && is_zero(mag_y) && is_zero(mag_z)) {
255  goto fail;
256  }
257 
258  raw_field = Vector3f(mag_x, mag_y, mag_z);
259 
260  _make_factory_sensitivity_adjustment(raw_field);
261  _make_adc_sensitivity_adjustment(raw_field);
262  raw_field *= AK8963_MILLIGAUSS_SCALE;
263 
264  // rotate raw_field from sensor frame to body frame
265  rotate_field(raw_field, _compass_instance);
266 
267  // publish raw_field (uncorrected point sample) for calibration use
268  publish_raw_field(raw_field, time_us, _compass_instance);
269 
270  // correct raw_field for known errors
271  correct_field(raw_field, _compass_instance);
272 
273  // publish raw_field (corrected point sample) for EKF use
274  publish_unfiltered_field(raw_field, time_us, _compass_instance);
275 
276  _mag_x_accum += raw_field.x;
277  _mag_y_accum += raw_field.y;
278  _mag_z_accum += raw_field.z;
279  _accum_count++;
280  if (_accum_count == 10) {
281  _mag_x_accum /= 2;
282  _mag_y_accum /= 2;
283  _mag_z_accum /= 2;
284  _accum_count = 5;
285  }
286 
287  _last_update_timestamp = hal.scheduler->micros();
288 fail:
289  _sem_give();
290 end:
291  return;
292 }

320 bool AP_Compass_AK8963::_calibrate()
321 {
322  /* Enable FUSE-mode in order to be able to read calibration data */
323  _bus->register_write(AK8963_CNTL1, AK8963_FUSE_MODE | AK8963_16BIT_ADC);
324 
325  uint8_t response[3];
326  _bus->register_read(AK8963_ASAX, response, 3);
327 
328  for (int i = 0; i < 3; i++) {
329  float data = response[i];
330  _magnetometer_ASA[i] = ((data - 128) / 256 + 1);
331  error("%d: %lf\n", i, _magnetometer_ASA[i]);
332  }
333 
334  return true;
335 }
