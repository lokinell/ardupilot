147  if(_backend->get_frame_of_reference() == MAV_FRAME_LOCAL_NED){

151  }else{ // assume MAV_FRAME_BODY_NED (i.e. a hard-mounted sensor)
152  // subtract vehicle lean angles
153  x_rad = _angle_to_target.x - _ahrs.roll;
154  y_rad = -_angle_to_target.y + _ahrs.pitch;
155  }

163
