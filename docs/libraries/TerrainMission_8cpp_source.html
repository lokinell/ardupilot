32  check that we have fetched all mission terrain data
33  */
34 void AP_Terrain::update_mission_data(void)
35 {
36  if (last_mission_change_ms != mission.last_change_time_ms() ||
37  last_mission_spacing != grid_spacing) {
38  // the mission has changed - start again
39  next_mission_index = 1;
40  next_mission_pos = 0;
41  last_mission_change_ms = mission.last_change_time_ms();
42  last_mission_spacing = grid_spacing;
43  }
44  if (next_mission_index == 0) {
45  // nothing to do
46  return;
47  }
48 
49  uint16_t pending, loaded;
50  get_statistics(pending, loaded);
51  if (pending && ahrs.get_gps().status() >= AP_GPS::GPS_OK_FIX_3D) {
52  // wait till we have fully filled the current set of grids
53  return;
54  }
55 
56  // don't do more than 20 waypoints at a time, to prevent too much
57  // CPU usage
58  for (uint8_t i=0; i<20; i++) {
59  // get next mission command
60  AP_Mission::Mission_Command cmd;
61  if (!mission.read_cmd_from_storage(next_mission_index, cmd)) {
62  // nothing more to do
63  next_mission_index = 0;
64  return;
65  }
66 
67  // we only want nav waypoint commands. That should be enough to
68  // prefill the terrain data and makes many things much simpler
69  while ((cmd.id != MAV_CMD_NAV_WAYPOINT &&
70  cmd.id != MAV_CMD_NAV_SPLINE_WAYPOINT) ||
71  (cmd.content.location.lat == 0 && cmd.content.location.lng == 0)) {
72  next_mission_index++;
73  if (!mission.read_cmd_from_storage(next_mission_index, cmd)) {
74  // nothing more to do
75  next_mission_index = 0;
76  next_mission_pos = 0;
77  return;
78  }
79  }
80 
81  // we will fetch 5 points around the waypoint. Four at 10 grid
82  // spacings away at 45, 135, 225 and 315 degrees, and the
83  // point itself
84  if (next_mission_pos != 4) {
85  location_update(cmd.content.location, 45+90*next_mission_pos, grid_spacing.get() * 10);
86  }
87 
88  // we have a mission command to check
89  float height;
90  if (!height_amsl(cmd.content.location, height)) {
91  // if we can't get data for a mission item then return and
92  // check again next time
93  return;
94  }
95 
96  next_mission_pos++;
97  if (next_mission_pos == 5) {
98 #if TERRAIN_DEBUG
99  hal.console->printf("checked waypoint %u\n", (unsigned)next_mission_index);
100 #endif
101 
102  // move to next waypoint
103  next_mission_index++;
104  next_mission_pos = 0;
105  }
106  }
107 }
