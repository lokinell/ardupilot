73 int32_t AP_YawController::get_servo_out(float scaler, bool disable_integrator)
74 {
75  uint32_t tnow = hal.scheduler->millis();
76  uint32_t dt = tnow - _last_t;
77  if (_last_t == 0 || dt > 1000) {
78  dt = 0;
79  }
80  _last_t = tnow;
81 
82 
83  int16_t aspd_min = aparm.airspeed_min;
84  if (aspd_min < 1) {
85  aspd_min = 1;
86  }
87 
88  float delta_time = (float) dt / 1000.0f;
89 
90  // Calculate yaw rate required to keep up with a constant height coordinated turn
91  float aspeed;
92  float rate_offset;
93  float bank_angle = _ahrs.roll;
94  // limit bank angle between +- 80 deg if right way up
95  if (fabsf(bank_angle) < 1.5707964f) {
96  bank_angle = constrain_float(bank_angle,-1.3962634f,1.3962634f);
97  }
98  if (!_ahrs.airspeed_estimate(&aspeed)) {
99  // If no airspeed available use average of min and max
100  aspeed = 0.5f*(float(aspd_min) + float(aparm.airspeed_max));
101  }
102  rate_offset = (GRAVITY_MSS / max(aspeed , float(aspd_min))) * tanf(bank_angle) * cosf(bank_angle) * _K_FF;
103 
104  // Get body rate vector (radians/sec)
105  float omega_z = _ahrs.get_gyro().z;
106 
107  // Get the accln vector (m/s^2)
108  float accel_y = _ahrs.get_ins().get_accel().y;
109 
110  // Subtract the steady turn component of rate from the measured rate
111  // to calculate the rate relative to the turn requirement in degrees/sec
112  float rate_hp_in = ToDeg(omega_z - rate_offset);
113 
114  // Apply a high-pass filter to the rate to washout any steady state error
115  // due to bias errors in rate_offset
116  // Use a cut-off frequency of omega = 0.2 rad/sec
117  // Could make this adjustable by replacing 0.9960080 with (1 - omega * dt)
118  float rate_hp_out = 0.9960080f * _last_rate_hp_out + rate_hp_in - _last_rate_hp_in;
119  _last_rate_hp_out = rate_hp_out;
120  _last_rate_hp_in = rate_hp_in;
121 
122  //Calculate input to integrator
123  float integ_in = - _K_I * (_K_A * accel_y + rate_hp_out);
124 
125  // Apply integrator, but clamp input to prevent control saturation and freeze integrator below min FBW speed
126  // Don't integrate if in stabilise mode as the integrator will wind up against the pilots inputs
127  // Don't integrate if _K_D is zero as integrator will keep winding up
128  if (!disable_integrator && _K_D > 0) {
129  //only integrate if airspeed above min value
130  if (aspeed > float(aspd_min))
131  {
132  // prevent the integrator from increasing if surface defln demand is above the upper limit
133  if (_last_out < -45) {
134  _integrator += max(integ_in * delta_time , 0);
135  } else if (_last_out > 45) {
136  // prevent the integrator from decreasing if surface defln demand is below the lower limit
137  _integrator += min(integ_in * delta_time , 0);
138  } else {
139  _integrator += integ_in * delta_time;
140  }
141  }
142  } else {
143  _integrator = 0;
144  }
145 
146  if (_K_D < 0.0001f) {
147  // yaw damping is disabled, and the integrator is scaled by damping, so return 0
148  return 0;
149  }
150 
151  // Scale the integration limit
152  float intLimScaled = _imax * 0.01f / (_K_D * scaler * scaler);
153 
154  // Constrain the integrator state
155  _integrator = constrain_float(_integrator, -intLimScaled, intLimScaled);
156 
157  // Protect against increases to _K_D during in-flight tuning from creating large control transients
158  // due to stored integrator values
159  if (_K_D > _K_D_last && _K_D > 0) {
160  _integrator = _K_D_last/_K_D * _integrator;
161  }
162  _K_D_last = _K_D;
163 
164  // Calculate demanded rudder deflection, +Ve deflection yaws nose right
165  // Save to last value before application of limiter so that integrator limiting
166  // can detect exceedance next frame
167  // Scale using inverse dynamic pressure (1/V^2)
168  _pid_info.I = _K_D * _integrator * scaler * scaler;
169  _pid_info.D = _K_D * (-rate_hp_out) * scaler * scaler;
170  _last_out = _pid_info.I + _pid_info.D;
171 
172  // Convert to centi-degrees and constrain
173  return constrain_float(_last_out * 100, -4500, 4500);
174 }
