253 void RCInput_RPI::init_ctrl_data()
254 {
255  uint32_t phys_fifo_addr;
256  uint32_t dest = 0;
257  uint32_t cbp = 0;
258  dma_cb_t* cbp_curr;
259  //Set fifo addr (for delay)
260  phys_fifo_addr = ((pcm_base + 0x04) & 0x00FFFFFF) | 0x7e000000;
261 
262  //Init dma control blocks.
263  /*We are transferring 1 byte of GPIO register. Every 56th iteration we are
264  sampling TIMER register, which length is 8 bytes. So, for every 56 samples of GPIO we need
265  56 * 1 + 8 = 64 bytes of buffer. Value 56 was selected specially to have a 64-byte "block"
266  TIMER - GPIO. So, we have integer count of such "blocks" at one virtual page. (4096 / 64 = 64
267  "blocks" per page. As minimum, we must have 2 virtual pages of buffer (to have integer count of
268  vitual pages for control blocks): for every 56 iterations (64 bytes of buffer) we need 56 control blocks for GPIO
269  sampling, 56 control blocks for setting frequency and 1 control block for sampling timer, so,
270  we need 56 + 56 + 1 = 113 control blocks. For integer value, we need 113 pages of control blocks.
271  Each control block length is 32 bytes. In 113 pages we will have (113 * 4096 / 32) = 113 * 128 control
272  blocks. 113 * 128 control blocks = 64 * 128 bytes of buffer = 2 pages of buffer.
273  So, for 56 * 64 * 2 iteration we init DMA for sampling GPIO
274  and timer to (64 * 64 * 2) = 8192 bytes = 2 pages of buffer.
275  */
276  // fprintf(stderr, "ERROR SEARCH1\n");
277 
278  uint32_t i = 0;
279  for (i = 0; i < 56 * 128 * RCIN_RPI_BUFFER_LENGTH; i++) // 8 * 56 * 128 == 57344
280  {
281  //Transfer timer every 56th sample
282  if(i % 56 == 0) {
283  cbp_curr = (dma_cb_t*)con_blocks->get_page(con_blocks->_virt_pages, cbp);
284 
285  init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP | RCIN_RPI_DMA_DEST_INC | RCIN_RPI_DMA_SRC_INC, RCIN_RPI_TIMER_BASE,
286  (uintptr_t) circle_buffer->get_page(circle_buffer->_phys_pages, dest),
287  8,
288  0,
289  (uintptr_t) con_blocks->get_page(con_blocks->_phys_pages,
290  cbp + sizeof(dma_cb_t) ) );
291 
292  dest += 8;
293  cbp += sizeof(dma_cb_t);
294  }
295 
296  // Transfer GPIO (1 byte)
297  cbp_curr = (dma_cb_t*)con_blocks->get_page(con_blocks->_virt_pages, cbp);
298  init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP, RCIN_RPI_GPIO_LEV0_ADDR,
299  (uintptr_t) circle_buffer->get_page(circle_buffer->_phys_pages, dest),
300  1,
301  0,
302  (uintptr_t) con_blocks->get_page(con_blocks->_phys_pages,
303  cbp + sizeof(dma_cb_t) ) );
304 
305  dest += 1;
306  cbp += sizeof(dma_cb_t);
307 
308  // Delay (for setting sampling frequency)
309  /* DMA is waiting data request signal (DREQ) from PCM. PCM is set for 1 MhZ freqency, so,
310  each sample of GPIO is limited by writing to PCA queue.
311  */
312  cbp_curr = (dma_cb_t*)con_blocks->get_page(con_blocks->_virt_pages, cbp);
313  init_dma_cb(&cbp_curr, RCIN_RPI_DMA_NO_WIDE_BURSTS | RCIN_RPI_DMA_WAIT_RESP | RCIN_RPI_DMA_D_DREQ | RCIN_RPI_DMA_PER_MAP(2),
314  RCIN_RPI_TIMER_BASE, phys_fifo_addr,
315  4,
316  0,
317  (uintptr_t)con_blocks->get_page(con_blocks->_phys_pages,
318  cbp + sizeof(dma_cb_t) ) );
319 
320  cbp += sizeof(dma_cb_t);
321  }
322  //Make last control block point to the first (to make circle)
323  cbp -= sizeof(dma_cb_t);
324  ((dma_cb_t*)con_blocks->get_page(con_blocks->_virt_pages, cbp))->next = (uintptr_t) con_blocks->get_page(con_blocks->_phys_pages, 0);
325 }

332 void RCInput_RPI::init_PCM()
333 {
334  pcm_reg[RCIN_RPI_PCM_CS_A] = 1; // Disable Rx+Tx, Enable PCM block
335  hal.scheduler->delay_microseconds(100);
336  clk_reg[RCIN_RPI_PCMCLK_CNTL] = 0x5A000006; // Source=PLLD (500MHz)
337  hal.scheduler->delay_microseconds(100);
338  clk_reg[RCIN_RPI_PCMCLK_DIV] = 0x5A000000 | ((50000/RCIN_RPI_SAMPLE_FREQ)<<12); // Set pcm div. If we need to configure DMA frequency.
339  hal.scheduler->delay_microseconds(100);
340  clk_reg[RCIN_RPI_PCMCLK_CNTL] = 0x5A000016; // Source=PLLD and enable
341  hal.scheduler->delay_microseconds(100);
342  pcm_reg[RCIN_RPI_PCM_TXC_A] = 0<<31 | 1<<30 | 0<<20 | 0<<16; // 1 channel, 8 bits
343  hal.scheduler->delay_microseconds(100);
344  pcm_reg[RCIN_RPI_PCM_MODE_A] = (10 - 1) << 10; //PCM mode
345  hal.scheduler->delay_microseconds(100);
346  pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<4 | 1<<3; // Clear FIFOs
347  hal.scheduler->delay_microseconds(100);
348  pcm_reg[RCIN_RPI_PCM_DREQ_A] = 64<<24 | 64<<8; // DMA Req when one slot is free?
349  hal.scheduler->delay_microseconds(100);
350  pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<9; // Enable DMA
351  hal.scheduler->delay_microseconds(100);
352  pcm_reg[RCIN_RPI_PCM_CS_A] |= 1<<2; // Enable Tx
353  hal.scheduler->delay_microseconds(100);

455 void RCInput_RPI::_timer_tick()
456 {
457  int j;
458  void* x;
459 
460  //Now we are getting address in which DMAC is writing at current moment
461  dma_cb_t* ad = (dma_cb_t*) con_blocks->get_virt_addr(dma_reg[RCIN_RPI_DMA_CONBLK_AD | RCIN_RPI_DMA_CHANNEL << 8]);
462  for(j = 1; j >= -1; j--){
463  x = circle_buffer->get_virt_addr((ad + j)->dst);
464  if(x != NULL) {
465  break;}
466  }
467 
468  //How many bytes have DMA transfered (and we can process)?
469  counter = circle_buffer->bytes_available(curr_pointer, circle_buffer->get_offset(circle_buffer->_virt_pages, (uintptr_t)x));
470  //We can't stay in method for a long time, because it may lead to delays
471  if (counter > RCIN_RPI_MAX_COUNTER) {
472  counter = RCIN_RPI_MAX_COUNTER;
473  }
474 
475  //Processing ready bytes
476  for (;counter > 0x40;counter--) {
477  //Is it timer samle?
478  if (curr_pointer % (64) == 0) {
479  curr_tick = *((uint64_t*) circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer));
480  curr_pointer+=8;
481  counter-=8;
482  }
483  //Reading required bit
484  curr_signal = *((uint8_t*) circle_buffer->get_page(circle_buffer->_virt_pages, curr_pointer)) & 0x10 ? 1 : 0;
485  //If the signal changed
486  if (curr_signal != last_signal) {
487  delta_time = curr_tick - prev_tick;
488  prev_tick = curr_tick;
489  switch (state) {
490  case RCIN_RPI_INITIAL_STATE:
491  state = RCIN_RPI_ZERO_STATE;
492  break;
493  case RCIN_RPI_ZERO_STATE:
494  if (curr_signal == 0) {
495  width_s0 = (uint16_t) delta_time;
496  state = RCIN_RPI_ONE_STATE;
497  break;
498  }
499  else
500  break;
501  case RCIN_RPI_ONE_STATE:
502  if (curr_signal == 1) {
503  width_s1 = (uint16_t) delta_time;
504  state = RCIN_RPI_ZERO_STATE;
505  _process_rc_pulse(width_s0, width_s1);
506  break;
507  }
508  else
509  break;
510  }
511  }
512  last_signal = curr_signal;
513  curr_pointer++;
514  if (curr_pointer >= circle_buffer->get_page_count()*PAGE_SIZE) {
515  curr_pointer = 0;
516  }
517  curr_tick+=curr_tick_inc;
518  }
519 }
