13 /*
14  * A compass measurement is expected to pass through the following functions:
15  * 1. rotate_field - this rotates the measurement in-place from sensor frame
16  * to body frame
17  * 2. publish_raw_field - this provides an uncorrected point-sample for
18  * calibration libraries
19  * 3. correct_field - this corrects the measurement in-place for hard iron,
20  * soft iron, motor interference, and non-orthagonality errors
21  * 4. publish_unfiltered_field - this (optionally) provides a corrected
22  * point sample for fusion into the EKF
23  * 5. publish_filtered_field - legacy filtered magnetic field
24  *
25  * All those functions expect the mag field to be in milligauss.
26  */

60 void AP_Compass_Backend::correct_field(Vector3f &mag, uint8_t i)
61 {
62  Compass::mag_state &state = _compass._state[i];
63 
64  if (state.diagonals.get().is_zero()) {
65  state.diagonals.set(Vector3f(1.0f,1.0f,1.0f));
66  }
67 
68  const Vector3f &offsets = state.offset.get();
69  const Vector3f &diagonals = state.diagonals.get();
70  const Vector3f &offdiagonals = state.offdiagonals.get();
71  const Vector3f &mot = state.motor_compensation.get();
72 
73  /*
74  * note that _motor_offset[] is kept even if compensation is not
75  * being applied so it can be logged correctly
76  */
77  mag += offsets;
78  if(_compass._motor_comp_type != AP_COMPASS_MOT_COMP_DISABLED && !is_zero(_compass._thr_or_curr)) {
79  state.motor_offset = mot * _compass._thr_or_curr;
80  mag += state.motor_offset;
81  } else {
82  state.motor_offset.zero();
83  }
84 
85  Matrix3f mat(
86  diagonals.x, offdiagonals.x, offdiagonals.y,
87  offdiagonals.x, diagonals.y, offdiagonals.z,
88  offdiagonals.y, offdiagonals.z, diagonals.z
89  );
90 
91  mag = mat * mag;
