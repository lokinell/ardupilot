202 void SITLScheduler::_run_timer_procs(bool called_from_isr)
203 {
204  if (_in_timer_proc) {
205  // the timer calls took longer than the period of the
206  // timer. This is bad, and may indicate a serious
207  // driver failure. We can't just call the drivers
208  // again, as we could run out of stack. So we only
209  // call the _failsafe call. It's job is to detect if
210  // the drivers or the main loop are indeed dead and to
211  // activate whatever failsafe it thinks may help if
212  // need be. We assume the failsafe code can't
213  // block. If it does then we will recurse and die when
214  // we run out of stack
215  if (_failsafe != NULL) {
216  _failsafe();
217  }
218  return;
219  }
220  _in_timer_proc = true;
221 
222  if (!_timer_suspended) {
223  // now call the timer based drivers
224  for (int i = 0; i < _num_timer_procs; i++) {
225  if (_timer_proc[i]) {
226  _timer_proc[i]();
227  }
228  }
229  } else if (called_from_isr) {
230  _timer_event_missed = true;
231  }
232 
233  // and the failsafe, if one is setup
234  if (_failsafe != NULL) {
235  _failsafe();
236  }
237 
238  _in_timer_proc = false;
239 }
