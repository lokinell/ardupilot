60 void FLYMAPLEScheduler::init(void* machtnichts)
61 {
62  delay_us(2000000); // Wait for startup so we have time to connect a new USB console
63  // 1kHz interrupts from systick for normal timers
64  systick_attach_callback(_timer_procs_timer_event);
65 
66  // Set up Maple hardware timer for 1khz failsafe timer
67  // ref: http://leaflabs.com/docs/lang/api/hardwaretimer.html#lang-hardwaretimer
68  _failsafe_timer.pause();
69  _failsafe_timer.setPeriod(1000); // 1000us = 1kHz
70  _failsafe_timer.setChannelMode(TIMER_CH1, TIMER_OUTPUT_COMPARE);// Set up an interrupt on channel 1
71  _failsafe_timer.setCompare(TIMER_CH1, 1); // Interrupt 1 count after each update
72  _failsafe_timer.attachInterrupt(TIMER_CH1, _failsafe_timer_event);
73  _failsafe_timer.refresh();// Refresh the timer's count, prescale, and overflow
74  _failsafe_timer.resume(); // Start the timer counting
75  // We run this timer at a higher priority, so that a broken timer handler (ie one that hangs)
76  // will not prevent the failsafe timer interrupt.
77  // Caution: the timer number must agree with the HardwareTimer number
78  nvic_irq_set_priority(NVIC_TIMER2, 0x14);
79 }

82 void FLYMAPLEScheduler::delay(uint16_t ms)
83 {
84  uint32_t start = libmaple_micros();
85 
86  while (ms > 0) {
87  while ((libmaple_micros() - start) >= 1000) {
88  ms--;
89  if (ms == 0) break;
90  start += 1000;
91  }
92  if (_min_delay_cb_ms <= ms) {
93  if (_delay_cb) {
94  _delay_cb();
95  }
96  }
97  }
98 }

131 void FLYMAPLEScheduler::register_timer_process(AP_HAL::MemberProc proc)
132 {
133  for (int i = 0; i < _num_timer_procs; i++) {
134  if (_timer_proc[i] == proc) {
135  return;
136  }
137  }
138 
139  if (_num_timer_procs < FLYMAPLE_SCHEDULER_MAX_TIMER_PROCS) {
140  /* this write to _timer_proc can be outside the critical section
141  * because that memory won't be used until _num_timer_procs is
142  * incremented. */
143  _timer_proc[_num_timer_procs] = proc;
144  /* _num_timer_procs is used from interrupt, and multiple bytes long. */
145  noInterrupts();
146  _num_timer_procs++;
147  interrupts();
148  }
149 }
