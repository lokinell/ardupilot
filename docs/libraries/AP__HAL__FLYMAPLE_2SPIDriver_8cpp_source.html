51 bool FLYMAPLESPIDeviceDriver::transaction(const uint8_t *tx, uint8_t *rx, uint16_t len)
52 {
53  cs_assert();
54  if (rx == NULL) {
55  for (uint16_t i = 0; i < len; i++) {
56  transfer(tx[i]);
57  }
58  } else {
59  for (uint16_t i = 0; i < len; i++) {
60  rx[i] = transfer(tx[i]);
61  }
62  }
63  cs_release();
64  return true;
65 }
