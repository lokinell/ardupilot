120 void AP_MotorsMatrix::output_armed_not_stabilizing()
121 {
122  uint8_t i;
123  int16_t throttle_radio_output; // total throttle pwm value, summed onto throttle channel minimum, typically ~1100-1900
124  int16_t motor_out[AP_MOTORS_MAX_NUM_MOTORS]; // final outputs sent to the motors
125  int16_t out_min_pwm = _throttle_radio_min + _min_throttle; // minimum pwm value we can send to the motors
126  int16_t out_max_pwm = _throttle_radio_max; // maximum pwm value we can send to the motors
127 
128  // initialize limits flags
129  limit.roll_pitch = true;
130  limit.yaw = true;
131  limit.throttle_lower = false;
132  limit.throttle_upper = false;
133 
134  int16_t thr_in_min = rel_pwm_to_thr_range(_spin_when_armed_ramped);
135  if (_throttle_control_input <= thr_in_min) {
136  _throttle_control_input = thr_in_min;
137  limit.throttle_lower = true;
138  }
139 
140  if (_throttle_control_input >= _hover_out) {
141  _throttle_control_input = _hover_out;
142  limit.throttle_upper = true;
143  }
144 
145  throttle_radio_output = calc_throttle_radio_output();
146 
147  // set output throttle
148  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
149  if (motor_enabled[i]) {
150  motor_out[i] = throttle_radio_output;
151  }
152  }
153 
154  if(throttle_radio_output >= out_min_pwm) {
155  // apply thrust curve and voltage scaling
156  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
157  if (motor_enabled[i]) {
158  motor_out[i] = apply_thrust_curve_and_volt_scaling(motor_out[i], out_min_pwm, out_max_pwm);
159  }
160  }
161  }
162 
163  // send output to each motor
164  hal.rcout->cork();
165  for( i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++ ) {
166  if( motor_enabled[i] ) {
167  hal.rcout->write(i, motor_out[i]);
168  }
169  }
170  hal.rcout->push();

173 // output_armed - sends commands to the motors
174 // includes new scaling stability patch
175 // TODO pull code that is common to output_armed_not_stabilizing into helper functions
176 void AP_MotorsMatrix::output_armed_stabilizing()
177 {
178  int8_t i;
179  int16_t roll_pwm; // roll pwm value, initially calculated by calc_roll_pwm() but may be modified after, +/- 400
180  int16_t pitch_pwm; // pitch pwm value, initially calculated by calc_roll_pwm() but may be modified after, +/- 400
181  int16_t yaw_pwm; // yaw pwm value, initially calculated by calc_yaw_pwm() but may be modified after, +/- 400
182  int16_t throttle_radio_output; // total throttle pwm value, summed onto throttle channel minimum, typically ~1100-1900
183  int16_t out_min_pwm = _throttle_radio_min + _min_throttle; // minimum pwm value we can send to the motors
184  int16_t out_max_pwm = _throttle_radio_max; // maximum pwm value we can send to the motors
185  int16_t out_mid_pwm = (out_min_pwm+out_max_pwm)/2; // mid pwm value we can send to the motors
186  int16_t out_best_thr_pwm; // the is the best throttle we can come up which provides good control without climbing
187  float rpy_scale = 1.0; // this is used to scale the roll, pitch and yaw to fit within the motor limits
188 
189  int16_t rpy_out[AP_MOTORS_MAX_NUM_MOTORS]; // buffer so we don't have to multiply coefficients multiple times.
190  int16_t motor_out[AP_MOTORS_MAX_NUM_MOTORS]; // final outputs sent to the motors
191 
192  int16_t rpy_low = 0; // lowest motor value
193  int16_t rpy_high = 0; // highest motor value
194  int16_t yaw_allowed; // amount of yaw we can fit in
195  int16_t thr_adj; // the difference between the pilot's desired throttle and out_best_thr_pwm (the throttle that is actually provided)
196 
197  // initialize limits flags
198  limit.roll_pitch = false;
199  limit.yaw = false;
200  limit.throttle_lower = false;
201  limit.throttle_upper = false;
202 
203  // Ensure throttle is within bounds of 0 to 1000
204  int16_t thr_in_min = rel_pwm_to_thr_range(_min_throttle);
205  if (_throttle_control_input <= thr_in_min) {
206  _throttle_control_input = thr_in_min;
207  limit.throttle_lower = true;
208  }
209  if (_throttle_control_input >= _max_throttle) {
210  _throttle_control_input = _max_throttle;
211  limit.throttle_upper = true;
212  }
213 
214  roll_pwm = calc_roll_pwm();
215  pitch_pwm = calc_pitch_pwm();
216  yaw_pwm = calc_yaw_pwm();
217  throttle_radio_output = calc_throttle_radio_output();
218 
219  // calculate roll and pitch for each motor
220  // set rpy_low and rpy_high to the lowest and highest values of the motors
221  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
222  if (motor_enabled[i]) {
223  rpy_out[i] = roll_pwm * _roll_factor[i] * get_compensation_gain() +
224  pitch_pwm * _pitch_factor[i] * get_compensation_gain();
225 
226  // record lowest roll pitch command
227  if (rpy_out[i] < rpy_low) {
228  rpy_low = rpy_out[i];
229  }
230  // record highest roll pich command
231  if (rpy_out[i] > rpy_high) {
232  rpy_high = rpy_out[i];
233  }
234  }
235  }
236 
237  // calculate throttle that gives most possible room for yaw (range 1000 ~ 2000) which is the lower of:
238  // 1. mid throttle - average of highest and lowest motor (this would give the maximum possible room margin above the highest motor and below the lowest)
239  // 2. the higher of:
240  // a) the pilot's throttle input
241  // b) the mid point between the pilot's input throttle and hover-throttle
242  // Situation #2 ensure we never increase the throttle above hover throttle unless the pilot has commanded this.
243  // Situation #2b allows us to raise the throttle above what the pilot commanded but not so far that it would actually cause the copter to rise.
244  // We will choose #1 (the best throttle for yaw control) if that means reducing throttle to the motors (i.e. we favour reducing throttle *because* it provides better yaw control)
245  // We will choose #2 (a mix of pilot and hover throttle) only when the throttle is quite low. We favour reducing throttle instead of better yaw control because the pilot has commanded it
246  int16_t motor_mid = (rpy_low+rpy_high)/2;
247  out_best_thr_pwm = min(out_mid_pwm - motor_mid, max(throttle_radio_output, throttle_radio_output*max(0,1.0f-_throttle_thr_mix)+get_hover_throttle_as_pwm()*_throttle_thr_mix));
248 
249  // calculate amount of yaw we can fit into the throttle range
250  // this is always equal to or less than the requested yaw from the pilot or rate controller
251  yaw_allowed = min(out_max_pwm - out_best_thr_pwm, out_best_thr_pwm - out_min_pwm) - (rpy_high-rpy_low)/2;
252  yaw_allowed = max(yaw_allowed, _yaw_headroom);
253 
254  if (yaw_pwm >= 0) {
255  // if yawing right
256  if (yaw_allowed > yaw_pwm * get_compensation_gain()) {
257  yaw_allowed = yaw_pwm * get_compensation_gain(); // to-do: this is bad form for yaw_allows to change meaning to become the amount that we are going to output
258  }else{
259  limit.yaw = true;
260  }
261  }else{
262  // if yawing left
263  yaw_allowed = -yaw_allowed;
264  if (yaw_allowed < yaw_pwm * get_compensation_gain()) {
265  yaw_allowed = yaw_pwm * get_compensation_gain(); // to-do: this is bad form for yaw_allows to change meaning to become the amount that we are going to output
266  }else{
267  limit.yaw = true;
268  }
269  }
270 
271  // add yaw to intermediate numbers for each motor
272  rpy_low = 0;
273  rpy_high = 0;
274  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
275  if (motor_enabled[i]) {
276  rpy_out[i] = rpy_out[i] +
277  yaw_allowed * _yaw_factor[i];
278 
279  // record lowest roll+pitch+yaw command
280  if( rpy_out[i] < rpy_low ) {
281  rpy_low = rpy_out[i];
282  }
283  // record highest roll+pitch+yaw command
284  if( rpy_out[i] > rpy_high) {
285  rpy_high = rpy_out[i];
286  }
287  }
288  }
289 
290  // check everything fits
291  thr_adj = throttle_radio_output - out_best_thr_pwm;
292 
293  // calculate upper and lower limits of thr_adj
294  int16_t thr_adj_max = max(out_max_pwm-(out_best_thr_pwm+rpy_high),0);
295 
296  // if we are increasing the throttle (situation #2 above)..
297  if (thr_adj > 0) {
298  // increase throttle as close as possible to requested throttle
299  // without going over out_max_pwm
300  if (thr_adj > thr_adj_max){
301  thr_adj = thr_adj_max;
302  // we haven't even been able to apply full throttle command
303  limit.throttle_upper = true;
304  }
305  }else if(thr_adj < 0){
306  // decrease throttle as close as possible to requested throttle
307  // without going under out_min_pwm or over out_max_pwm
308  // earlier code ensures we can't break both boundaries
309  int16_t thr_adj_min = min(out_min_pwm-(out_best_thr_pwm+rpy_low),0);
310  if (thr_adj > thr_adj_max) {
311  thr_adj = thr_adj_max;
312  limit.throttle_upper = true;
313  }
314  if (thr_adj < thr_adj_min) {
315  thr_adj = thr_adj_min;
316  }
317  }
318 
319  // do we need to reduce roll, pitch, yaw command
320  // earlier code does not allow both limit's to be passed simultaneously with abs(_yaw_factor)<1
321  if ((rpy_low+out_best_thr_pwm)+thr_adj < out_min_pwm){
322  // protect against divide by zero
323  if (rpy_low != 0) {
324  rpy_scale = (float)(out_min_pwm-thr_adj-out_best_thr_pwm)/rpy_low;
325  }
326  // we haven't even been able to apply full roll, pitch and minimal yaw without scaling
327  limit.roll_pitch = true;
328  limit.yaw = true;
329  }else if((rpy_high+out_best_thr_pwm)+thr_adj > out_max_pwm){
330  // protect against divide by zero
331  if (rpy_high != 0) {
332  rpy_scale = (float)(out_max_pwm-thr_adj-out_best_thr_pwm)/rpy_high;
333  }
334  // we haven't even been able to apply full roll, pitch and minimal yaw without scaling
335  limit.roll_pitch = true;
336  limit.yaw = true;
337  }
338 
339  // add scaled roll, pitch, constrained yaw and throttle for each motor
340  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
341  if (motor_enabled[i]) {
342  motor_out[i] = out_best_thr_pwm+thr_adj +
343  rpy_scale*rpy_out[i];
344  }
345  }
346 
347  // apply thrust curve and voltage scaling
348  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
349  if (motor_enabled[i]) {
350  motor_out[i] = apply_thrust_curve_and_volt_scaling(motor_out[i], out_min_pwm, out_max_pwm);
351  }
352  }
353 
354  // clip motor output if required (shouldn't be)
355  for (i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++) {
356  if (motor_enabled[i]) {
357  motor_out[i] = constrain_int16(motor_out[i], out_min_pwm, out_max_pwm);
358  }
359  }
360 
361  // send output to each motor
362  hal.rcout->cork();
363  for( i=0; i<AP_MOTORS_MAX_NUM_MOTORS; i++ ) {
364  if( motor_enabled[i] ) {
365  hal.rcout->write(i, motor_out[i]);
366  }
367  }
368  hal.rcout->push();
369 }
