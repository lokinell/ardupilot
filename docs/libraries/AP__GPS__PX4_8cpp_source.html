46 // update internal state if new GPS information is available
47 bool
48 AP_GPS_PX4::read(void)
49 {
50  bool updated = false;
51  orb_check(_gps_sub, &updated);
52 
53  if (updated) {
54  if (OK == orb_copy(ORB_ID(vehicle_gps_position), _gps_sub, &_gps_pos)) {
55  state.last_gps_time_ms = hal.scheduler->millis();
56  state.status = (AP_GPS::GPS_Status) (_gps_pos.fix_type | AP_GPS::NO_FIX);
57  state.num_sats = _gps_pos.satellites_used;
58  state.hdop = uint16_t(_gps_pos.eph*100.0f + .5f);
59 
60  if (_gps_pos.fix_type >= 2) {
61  state.location.lat = _gps_pos.lat;
62  state.location.lng = _gps_pos.lon;
63  state.location.alt = _gps_pos.alt/10;
64 
65  state.ground_speed = _gps_pos.vel_m_s;
66  state.ground_course_cd = wrap_360_cd(degrees(_gps_pos.cog_rad)*100);
67  state.hdop = _gps_pos.eph*100;
68 
69  // convert epoch timestamp back to gps epoch - evil hack until we get the genuine
70  // raw week information (or APM switches to Posix epoch ;-) )
71  uint64_t epoch_ms = uint64_t(_gps_pos.time_utc_usec/1000.+.5);
72  uint64_t gps_ms = epoch_ms - DELTA_POSIX_GPS_EPOCH + LEAP_MS_GPS_2014;
73  state.time_week = uint16_t(gps_ms / uint64_t(MS_PER_WEEK));
74  state.time_week_ms = uint32_t(gps_ms - uint64_t(state.time_week)*MS_PER_WEEK);
75 
76  if (_gps_pos.time_utc_usec == 0) {
77  // This is a work-around for https://github.com/PX4/Firmware/issues/1474
78  // reject position reports with invalid time, as APM adjusts it's clock after the first lock has been aquired
79  state.status = AP_GPS::NO_FIX;
80  }
81  }
82  if (_gps_pos.fix_type >= 3) {
83  state.have_vertical_velocity = _gps_pos.vel_ned_valid;
84  state.velocity.x = _gps_pos.vel_n_m_s;
85  state.velocity.y = _gps_pos.vel_e_m_s;
86  state.velocity.z = _gps_pos.vel_d_m_s;
87  state.speed_accuracy = _gps_pos.s_variance_m_s;
88  state.have_speed_accuracy = true;
89  }
90  else {
91  state.have_vertical_velocity = false;
92  }
93  }
94  }
95 
96  return updated;
97 }
