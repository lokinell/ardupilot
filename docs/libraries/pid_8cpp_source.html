22 float PID::get_pid(float error, float scaler)
23 {
24  uint32_t tnow = hal.scheduler->millis();
25  uint32_t dt = tnow - _last_t;
26  float output = 0;
27  float delta_time;
28 
29  if (_last_t == 0 || dt > 1000) {
30  dt = 0;
31 
32  // if this PID hasn't been used for a full second then zero
33  // the intergator term. This prevents I buildup from a
34  // previous fight mode from causing a massive return before
35  // the integrator gets a chance to correct itself
36  reset_I();
37  }
38  _last_t = tnow;
39 
40  delta_time = (float)dt / 1000.0f;
41 
42  // Compute proportional component
43  output += error * _kp;
44 
45  // Compute derivative component if time has elapsed
46  if ((fabsf(_kd) > 0) && (dt > 0)) {
47  float derivative;
48 
49  if (isnan(_last_derivative)) {
50  // we've just done a reset, suppress the first derivative
51  // term as we don't want a sudden change in input to cause
52  // a large D output change
53  derivative = 0;
54  _last_derivative = 0;
55  } else {
56  derivative = (error - _last_error) / delta_time;
57  }
58 
59  // discrete low pass filter, cuts out the
60  // high frequency noise that can drive the controller crazy
61  float RC = 1/(2*PI*_fCut);
62  derivative = _last_derivative +
63  ((delta_time / (RC + delta_time)) *
64  (derivative - _last_derivative));
65 
66  // update state
67  _last_error = error;
68  _last_derivative = derivative;
69 
70  // add in derivative component
71  output += _kd * derivative;
72  }
73 
74  // scale the P and D components
75  output *= scaler;
76 
77  // Compute integral component if time has elapsed
78  if ((fabsf(_ki) > 0) && (dt > 0)) {
79  _integrator += (error * _ki) * scaler * delta_time;
80  if (_integrator < -_imax) {
81  _integrator = -_imax;
82  } else if (_integrator > _imax) {
83  _integrator = _imax;
84  }
85  output += _integrator;
86  }
87 
88  return output;
89 }
