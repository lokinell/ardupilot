37 /*
38  update the state of the airspeed calibration - needs to be called
39  once a second
40  */
41 float Airspeed_Calibration::update(float airspeed, const Vector3f &vg)
42 {
43  // Perform the covariance prediction
44  // Q is a diagonal matrix so only need to add three terms in
45  // C code implementation
46  // P = P + Q;
47  P.a.x += Q0;
48  P.b.y += Q0;
49  P.c.z += Q1;
50 
51  // Perform the predicted measurement using the current state estimates
52  // No state prediction required because states are assumed to be time
53  // invariant plus process noise
54  // Ignore vertical wind component
55  float TAS_pred = state.z * pythagorous3(vg.x - state.x, vg.y - state.y, vg.z);
56  float TAS_mea = airspeed;
57 
58  // Calculate the observation Jacobian H_TAS
59  float SH1 = sq(vg.y - state.y) + sq(vg.x - state.x);
60  if (SH1 < 0.000001f) {
61  // avoid division by a small number
62  return state.z;
63  }
64  float SH2 = 1/sqrtf(SH1);
65 
66  // observation Jacobian
67  Vector3f H_TAS(
68  -(state.z*SH2*(2*vg.x - 2*state.x))/2,
69  -(state.z*SH2*(2*vg.y - 2*state.y))/2,
70  1/SH2);
71 
72  // Calculate the fusion innovaton covariance assuming a TAS measurement
73  // noise of 1.0 m/s
74  // S = H_TAS*P*H_TAS' + 1.0; % [1 x 3] * [3 x 3] * [3 x 1] + [1 x 1]
75  Vector3f PH = P * H_TAS;
76  float S = H_TAS * PH + 1.0f;
77 
78  // Calculate the Kalman gain
79  // [3 x 3] * [3 x 1] / [1 x 1]
80  Vector3f KG = PH / S;
81 
82  // Update the states
83  state += KG*(TAS_mea - TAS_pred); // [3 x 1] + [3 x 1] * [1 x 1]
84 
85  // Update the covariance matrix
86  Vector3f HP2 = H_TAS * P;
87  P -= KG.mul_rowcol(HP2);
88 
89  // force symmetry on the covariance matrix - necessary due to rounding
90  // errors
91  float P12 = 0.5f * (P.a.y + P.b.x);
92  float P13 = 0.5f * (P.a.z + P.c.x);
93  float P23 = 0.5f * (P.b.z + P.c.y);
94  P.a.y = P.b.x = P12;
95  P.a.z = P.c.x = P13;
96  P.b.z = P.c.y = P23;
97 
98  // Constrain diagonals to be non-negative - protects against rounding errors
99  P.a.x = max(P.a.x, 0.0f);
100  P.b.y = max(P.b.y, 0.0f);
101  P.c.z = max(P.c.z, 0.0f);
102 
103  state.x = constrain_float(state.x, -aparm.airspeed_max, aparm.airspeed_max);
104  state.y = constrain_float(state.y, -aparm.airspeed_max, aparm.airspeed_max);
105  state.z = constrain_float(state.z, 0.5f, 1.0f);
106 
107  return state.z;
108 }

114 void AP_Airspeed::update_calibration(const Vector3f &vground)
115 {
116  if (!_autocal) {
117  // auto-calibration not enabled
118  return;
119  }
120 
121  // set state.z based on current ratio, this allows the operator to
122  // override the current ratio in flight with autocal, which is
123  // very useful both for testing and to force a reasonable value.
124  float ratio = constrain_float(_ratio, 1.0f, 4.0f);
125 
126  _calibration.state.z = 1.0f / sqrtf(ratio);
127 
128  // calculate true airspeed, assuming a airspeed ratio of 1.0
129  float dpress = get_differential_pressure();
130  float true_airspeed = sqrtf(dpress) * _EAS2TAS;
131 
132  float zratio = _calibration.update(true_airspeed, vground);
133 
134  if (isnan(zratio) || isinf(zratio)) {
135  return;
136  }
137 
138  // this constrains the resulting ratio to between 1.0 and 4.0
139  zratio = constrain_float(zratio, 0.5f, 1.0f);
140  _ratio.set(1/sq(zratio));
141  if (_counter > 60) {
142  if (_last_saved_ratio > 1.05f*_ratio ||
143  _last_saved_ratio < 0.95f*_ratio) {
144  _ratio.save();
145  _last_saved_ratio = _ratio;
146  _counter = 0;
147  }
148  } else {
149  _counter++;
150  }
151 }
