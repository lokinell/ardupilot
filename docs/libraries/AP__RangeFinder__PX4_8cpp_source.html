103 void AP_RangeFinder_PX4::update(void)
104 {
105  if (_fd == -1) {
106  set_status(RangeFinder::RangeFinder_NotConnected);
107  return;
108  }
109 
110  struct distance_sensor_s range_report;
111  float sum = 0;
112  uint16_t count = 0;
113 
114  if (_last_max_distance_cm != ranger._max_distance_cm[state.instance] ||
115  _last_min_distance_cm != ranger._min_distance_cm[state.instance]) {
116  float max_distance = ranger._max_distance_cm[state.instance]*0.01f;
117  float min_distance = ranger._min_distance_cm[state.instance]*0.01f;
118  if (ioctl(_fd, RANGEFINDERIOCSETMAXIUMDISTANCE, (unsigned long)&max_distance) == 0 &&
119  ioctl(_fd, RANGEFINDERIOCSETMINIUMDISTANCE, (unsigned long)&min_distance) == 0) {
120  _last_max_distance_cm = ranger._max_distance_cm[state.instance];
121  _last_min_distance_cm = ranger._min_distance_cm[state.instance];
122  }
123  }
124 
125 
126  while (::read(_fd, &range_report, sizeof(range_report)) == sizeof(range_report) &&
127  range_report.timestamp != _last_timestamp) {
128  // take reading
129  sum += range_report.current_distance;
130  count++;
131  _last_timestamp = range_report.timestamp;
132  }
133 
134  // if we have not taken a reading in the last 0.2s set status to No Data
135  if (hal.scheduler->micros64() - _last_timestamp >= 200000) {
136  set_status(RangeFinder::RangeFinder_NoData);
137  }
138 
139  if (count != 0) {
140  state.distance_cm = sum / count * 100.0f;
141  state.distance_cm += ranger._offset[state.instance];
142 
143  // update range_valid state based on distance measured
144  update_status();
145  }
146 }
