60 // update - updates led according to timed_updated. Should be called at 50Hz
61 void ToneAlarm_Linux::update()
62 {
63  // exit immediately if we haven't initialised successfully
64  if (!_initialized) {
65  return;
66  }
67 
68  // check for arming failure
69  if (AP_Notify::events.arming_failed) {
70  play_tune(TONE_ARMING_FAILURE_TUNE);
71  }
72 
73  // check if arming status has changed
74  if (flags.armed != AP_Notify::flags.armed) {
75  flags.armed = AP_Notify::flags.armed;
76  if (flags.armed) {
77  // arming tune
78  play_tune(TONE_ARMING_WARNING_TUNE);
79  }else{
80  // disarming tune
81  play_tune(TONE_NOTIFY_NEUTRAL_TUNE);
82  }
83  }
84 
85  // check if battery status has changed
86  if (flags.failsafe_battery != AP_Notify::flags.failsafe_battery) {
87  flags.failsafe_battery = AP_Notify::flags.failsafe_battery;
88  if (flags.failsafe_battery) {
89  // low battery warning tune
90  play_tune(TONE_BATTERY_WARNING_FAST_TUNE);
91  }
92  }
93 
94  // check parachute release
95  if (flags.parachute_release != AP_Notify::flags.parachute_release) {
96  flags.parachute_release = AP_Notify::flags.parachute_release;
97  if (flags.parachute_release) {
98  // parachute release warning tune
99  play_tune(TONE_PARACHUTE_RELEASE_TUNE);
100  }
101  }
102 }
