57 void AP_InertialSensor_Backend::_notify_new_gyro_raw_sample(uint8_t instance,
58  const Vector3f &gyro)
59 {
60  float dt;
61 
62  if (_imu._gyro_raw_sample_rates[instance] <= 0) {
63  return;
64  }
65 
66  dt = 1.0f / _imu._gyro_raw_sample_rates[instance];
67 
68  // compute delta angle
69  Vector3f delta_angle = (gyro + _imu._last_raw_gyro[instance]) * 0.5f * dt;
70 
71  // compute coning correction
72  // see page 26 of:
73  // Tian et al (2010) Three-loop Integration of GPS and Strapdown INS with Coning and Sculling Compensation
74  // Available: http://www.sage.unsw.edu.au/snap/publications/tian_etal2010b.pdf
75  // see also examples/coning.py
76  Vector3f delta_coning = (_imu._delta_angle_acc[instance] +
77  _imu._last_delta_angle[instance] * (1.0f / 6.0f));
78  delta_coning = delta_coning % delta_angle;
79  delta_coning *= 0.5f;
80 
81  // integrate delta angle accumulator
82  // the angles and coning corrections are accumulated separately in the
83  // referenced paper, but in simulation little difference was found between
84  // integrating together and integrating separately (see examples/coning.py)
85  _imu._delta_angle_acc[instance] += delta_angle + delta_coning;
86 
87  // save previous delta angle for coning correction
88  _imu._last_delta_angle[instance] = delta_angle;
89  _imu._last_raw_gyro[instance] = gyro;
90 }
