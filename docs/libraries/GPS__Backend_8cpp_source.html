64 uint64_t AP_GPS::time_epoch_usec(uint8_t instance)
65 {
66  const GPS_State &istate = state[instance];
67  if (istate.last_gps_time_ms == 0) {
68  return 0;
69  }
70  const uint64_t ms_per_week = 7000ULL*86400ULL;
71  const uint64_t unix_offset = 17000ULL*86400ULL + 52*10*7000ULL*86400ULL - 15000ULL;
72  uint64_t fix_time_ms = unix_offset + istate.time_week*ms_per_week + istate.time_week_ms;
73  // add in the milliseconds since the last fix
74  return (fix_time_ms + (hal.scheduler->millis() - istate.last_gps_time_ms)) * 1000ULL;
75 }

82 void AP_GPS_Backend::make_gps_time(uint32_t bcd_date, uint32_t bcd_milliseconds)
83 {
84  uint8_t year, mon, day, hour, min, sec;
85  uint16_t msec;
86 
87  year = bcd_date % 100;
88  mon = (bcd_date / 100) % 100;
89  day = bcd_date / 10000;
90 
91  uint32_t v = bcd_milliseconds;
92  msec = v % 1000; v /= 1000;
93  sec = v % 100; v /= 100;
94  min = v % 100; v /= 100;
95  hour = v % 100; v /= 100;
96 
97  int8_t rmon = mon - 2;
98  if (0 >= rmon) {
99  rmon += 12;
100  year -= 1;
101  }
102 
103  // get time in seconds since unix epoch
104  uint32_t ret = (year/4) - 15 + 367*rmon/12 + day;
105  ret += year*365 + 10501;
106  ret = ret*24 + hour;
107  ret = ret*60 + min;
108  ret = ret*60 + sec;
109 
110  // convert to time since GPS epoch
111  ret -= 272764785UL;
112 
113  // get GPS week and time
114  state.time_week = ret / (7*86400UL);
115  state.time_week_ms = (ret % (7*86400UL)) * 1000;
116  state.time_week_ms += msec;
117 }
