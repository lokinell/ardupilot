236 bool AP_Baro_MS56XX::_check_crc(void)
237 {
238  int16_t cnt;
239  uint16_t n_rem;
240  uint16_t crc_read;
241  uint8_t n_bit;
242  uint16_t n_prom[8] = { _serial->read_16bits(CMD_MS5611_PROM_Setup),
243  _C1, _C2, _C3, _C4, _C5, _C6,
244  _serial->read_16bits(CMD_MS5611_PROM_CRC) };
245  n_rem = 0x00;
246 
247  /* save the read crc */
248  crc_read = n_prom[7];
249 
250  /* remove CRC byte */
251  n_prom[7] = (0xFF00 & (n_prom[7]));
252 
253  for (cnt = 0; cnt < 16; cnt++) {
254  /* uneven bytes */
255  if (cnt & 1) {
256  n_rem ^= (uint8_t)((n_prom[cnt >> 1]) & 0x00FF);
257 
258  } else {
259  n_rem ^= (uint8_t)(n_prom[cnt >> 1] >> 8);
260  }
261 
262  for (n_bit = 8; n_bit > 0; n_bit--) {
263  if (n_rem & 0x8000) {
264  n_rem = (n_rem << 1) ^ 0x3000;
265 
266  } else {
267  n_rem = (n_rem << 1);
268  }
269  }
270  }
271 
272  /* final 4 bit remainder is CRC value */
273  n_rem = (0x000F & (n_rem >> 12));
274  n_prom[7] = crc_read;
275 
276  /* return true if CRCs match */
277  return (0x000F & crc_read) == (n_rem ^ 0x00);
278 }

281 /*
282  Read the sensor. This is a state machine
283  We read one time Temperature (state=1) and then 4 times Pressure (states 2-5)
284  temperature does not change so quickly...
285 */
286 void AP_Baro_MS56XX::_timer(void)
287 {
288  // Throttle read rate to 100hz maximum.
289  if (hal.scheduler->micros() - _last_timer < 10000) {
290  return;
291  }
292 
293  if (!_serial->sem_take_nonblocking()) {
294  return;
295  }
296 
297  if (_state == 0) {
298  // On state 0 we read temp
299  uint32_t d2 = _serial->read_24bits(0);
300  if (d2 != 0) {
301  _s_D2 += d2;
302  _d2_count++;
303  if (_d2_count == 32) {
304  // we have summed 32 values. This only happens
305  // when we stop reading the barometer for a long time
306  // (more than 1.2 seconds)
307  _s_D2 >>= 1;
308  _d2_count = 16;
309  }
310 
311  if (_serial->write(ADDR_CMD_CONVERT_D1)) { // Command to read pressure
312  _state++;
313  }
314  } else {
315  /* if read fails, re-initiate a temperature read command or we are
316  * stuck */
317  _serial->write(ADDR_CMD_CONVERT_D2);
318  }
319  } else {
320  uint32_t d1 = _serial->read_24bits(0);
321  if (d1 != 0) {
322  // occasional zero values have been seen on the PXF
323  // board. These may be SPI errors, but safest to ignore
324  _s_D1 += d1;
325  _d1_count++;
326  if (_d1_count == 128) {
327  // we have summed 128 values. This only happens
328  // when we stop reading the barometer for a long time
329  // (more than 1.2 seconds)
330  _s_D1 >>= 1;
331  _d1_count = 64;
332  }
333  // Now a new reading exists
334  _updated = true;
335 
336  if (_state == 4) {
337  if (_serial->write(ADDR_CMD_CONVERT_D2)) { // Command to read temperature
338  _state = 0;
339  }
340  } else {
341  if (_serial->write(ADDR_CMD_CONVERT_D1)) { // Command to read pressure
342  _state++;
343  }
344  }
345  } else {
346  /* if read fails, re-initiate a pressure read command or we are
347  * stuck */
348  _serial->write(ADDR_CMD_CONVERT_D1);
349  }
350  }
351 
352  _last_timer = hal.scheduler->micros();
353  _serial->sem_give();
354 }

356 void AP_Baro_MS56XX::update()
357 {
358  if (!_use_timer) {
359  // if we're not using the timer then accumulate one more time
360  // to cope with the calibration loop and minimise lag
361  accumulate();
362  }
363 
364  if (!_updated) {
365  return;
366  }
367  uint32_t sD1, sD2;
368  uint8_t d1count, d2count;
369 
370  // Suspend timer procs because these variables are written to
371  // in "_update".
372  hal.scheduler->suspend_timer_procs();
373  sD1 = _s_D1; _s_D1 = 0;
374  sD2 = _s_D2; _s_D2 = 0;
375  d1count = _d1_count; _d1_count = 0;
376  d2count = _d2_count; _d2_count = 0;
377  _updated = false;
378  hal.scheduler->resume_timer_procs();
379 
380  if (d1count != 0) {
381  _D1 = ((float)sD1) / d1count;
382  }
383  if (d2count != 0) {
384  _D2 = ((float)sD2) / d2count;
385  }
386  _calculate();
387 }

394 // Calculate Temperature and compensated Pressure in real units (Celsius degrees*100, mbar*100).
395 void AP_Baro_MS5611::_calculate()
396 {
397  float dT;
398  float TEMP;
399  float OFF;
400  float SENS;
401 
402  // Formulas from manufacturer datasheet
403  // sub -15c temperature compensation is not included
404 
405  // we do the calculations using floating point allows us to take advantage
406  // of the averaging of D1 and D1 over multiple samples, giving us more
407  // precision
408  dT = _D2-(((uint32_t)_C5)<<8);
409  TEMP = (dT * _C6)/8388608;
410  OFF = _C2 * 65536.0f + (_C4 * dT) / 128;
411  SENS = _C1 * 32768.0f + (_C3 * dT) / 256;
412 
413  if (TEMP < 0) {
414  // second order temperature compensation when under 20 degrees C
415  float T2 = (dT*dT) / 0x80000000;
416  float Aux = TEMP*TEMP;
417  float OFF2 = 2.5f*Aux;
418  float SENS2 = 1.25f*Aux;
419  TEMP = TEMP - T2;
420  OFF = OFF - OFF2;
421  SENS = SENS - SENS2;
422  }
423 
424  float pressure = (_D1*SENS/2097152 - OFF)/32768;
425  float temperature = (TEMP + 2000) * 0.01f;
426  _copy_to_frontend(_instance, pressure, temperature);
427 }

475 void AP_Baro_MS5637::_calculate()
476 {
477  int32_t dT, TEMP;
478  int64_t OFF, SENS;
479  int32_t raw_pressure = _D1;
480  int32_t raw_temperature = _D2;
481 
482  // Formulas from manufacturer datasheet
483  // sub -15c temperature compensation is not included
484 
485  dT = raw_temperature - (((uint32_t)_C5) << 8);
486  TEMP = 2000 + ((int64_t)dT * (int64_t)_C6) / 8388608;
487  OFF = (int64_t)_C2 * (int64_t)131072 + ((int64_t)_C4 * (int64_t)dT) / (int64_t)64;
488  SENS = (int64_t)_C1 * (int64_t)65536 + ((int64_t)_C3 * (int64_t)dT) / (int64_t)128;
489 
490  if (TEMP < 2000) {
491  // second order temperature compensation when under 20 degrees C
492  int32_t T2 = ((int64_t)3 * ((int64_t)dT * (int64_t)dT) / (int64_t)8589934592);
493  int64_t aux = (TEMP - 2000) * (TEMP - 2000);
494  int64_t OFF2 = 61 * aux / 16;
495  int64_t SENS2 = 29 * aux / 16;
496 
497  TEMP = TEMP - T2;
498  OFF = OFF - OFF2;
499  SENS = SENS - SENS2;
500  }
501 
502  int32_t pressure = ((int64_t)raw_pressure * SENS / (int64_t)2097152 - OFF) / (int64_t)32768;
503  float temperature = TEMP * 0.01f;
504  _copy_to_frontend(_instance, (float)pressure, temperature);
