165 size_t Print::printNumber(unsigned long n, uint8_t base) {
166  char buf[8 * sizeof(long) + 1]; // Assumes 8-bit chars plus zero byte.
167  char *str = &buf[sizeof(buf) - 1];
168 
169  *str = '\0';
170 
171  // prevent crash if called with base == 1
172  if (base < 2) base = 10;
173 
174  do {
175  unsigned long m = n;
176  n /= base;
177  char c = m - base * n;
178  *--str = c < 10 ? c + '0' : c + 'A' - 10;
179  } while(n);
180 
181  return write(str);
182 }

184 size_t Print::printFloat(float number, uint8_t digits)
185 {
186  size_t n = 0;
187 
188  // Handle negative numbers
189  if (number < 0.0f)
190  {
191  n += print('-');
192  number = -number;
193  }
194 
195  // Round correctly so that print(1.999, 2) prints as "2.00"
196  float rounding = 0.5;
197  for (uint8_t i=0; i<digits; ++i)
198  rounding /= 10.0f;
199 
200  number += rounding;
201 
202  // Extract the integer part of the number and print it
203  unsigned long int_part = (unsigned long)number;
204  float remainder = number - (float )int_part;
205  n += print(int_part);
206 
207  // Print the decimal point, but only if there are digits beyond
208  if (digits > 0) {
209  n += print(".");
210  }
211 
212  // Extract digits from the remainder one at a time
213  while (digits-- > 0)
214  {
215  remainder *= 10.0f;
216  int toPrint = int(remainder);
217  n += print(toPrint);
218  remainder -= toPrint;
219  }
220 
221  return n;
222 }
