43 void FLYMAPLEUARTDriver::begin(uint32_t b)
44 {
45  // Dont let the ISRs access the ring buffers until we are fully set up:
46  nvic_irq_disable(_hws->c_dev()->irq_num);
47  _hws->begin(b);
48  if (_txBuf)

51  rb_init(_hws->c_dev()->rb, _rxBufSize, _rxBuf); // Get the RX ring buffer size we want

53 }

55 void FLYMAPLEUARTDriver::begin(uint32_t b, uint16_t rxS, uint16_t txS)
56 {
57  // Our private buffers can only grow, never shrink
58  // rxS == 0 or txS == 0 means no change to buffer size
59  uint8_t* oldRxBuf = NULL;
60  uint8_t* oldTxBuf = NULL;
61  // Maybe a new TX buffer?
62  if (txS && (txS > _txBufSize))
63  {
64  oldTxBuf = _txBuf;
65  _txBuf = (uint8_t*)malloc(txS); // Caution: old contents lost
66  _txBufSize = txS;
67  }
68  // Maybe a new RX buffer?
69  if (rxS && (rxS > _rxBufSize))
70  {
71  oldRxBuf = _rxBuf;
72  _rxBuf = (uint8_t*)malloc(rxS); // Caution: old contents lost
73  _rxBufSize = rxS;
74  }
75 
76  // Dont let the IRs access the ring buffers until we are fully set up:
77  nvic_irq_disable(_hws->c_dev()->irq_num);
78  begin(b); // libmaple internal ring buffer reinitialised to defaults here
79  if (_txBuf)
80  rb_init(_hws->c_dev()->tx_rb, _txBufSize, _txBuf); // Get the TX ring buffer size we want
81  if (_rxBuf)
82  rb_init(_hws->c_dev()->rb, _rxBufSize, _rxBuf); // Get the RX ring buffer size we want
83  nvic_irq_enable(_hws->c_dev()->irq_num);
84 
85  // Now its safe to free any old buffers
86  if (oldTxBuf)
87  free(oldTxBuf);
88  if (oldRxBuf)
89  free(oldRxBuf);
90 }
