32 // update mount position - should be called periodically
33 void AP_Mount_SToRM32_serial::update()
34 {
35  // exit immediately if not initialised
36  if (!_initialised) {
37  return;
38  }
39 
40  read_incoming(); // read the incoming messages from the gimbal
41 
42  // flag to trigger sending target angles to gimbal
43  bool resend_now = false;
44 
45  // update based on mount mode
46  switch(get_mode()) {
47  // move mount to a "retracted" position. To-Do: remove support and replace with a relaxed mode?
48  case MAV_MOUNT_MODE_RETRACT:
49  {
50  const Vector3f &target = _state._retract_angles.get();
51  _angle_ef_target_rad.x = ToRad(target.x);
52  _angle_ef_target_rad.y = ToRad(target.y);
53  _angle_ef_target_rad.z = ToRad(target.z);
54  }
55  break;
56 
57  // move mount to a neutral position, typically pointing forward
58  case MAV_MOUNT_MODE_NEUTRAL:
59  {
60  const Vector3f &target = _state._neutral_angles.get();
61  _angle_ef_target_rad.x = ToRad(target.x);
62  _angle_ef_target_rad.y = ToRad(target.y);
63  _angle_ef_target_rad.z = ToRad(target.z);
64  }
65  break;
66 
67  // point to the angles given by a mavlink message
68  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
69  // do nothing because earth-frame angle targets (i.e. _angle_ef_target_rad) should have already been set by a MOUNT_CONTROL message from GCS
70  break;
71 
72  // RC radio manual angle control, but with stabilization from the AHRS
73  case MAV_MOUNT_MODE_RC_TARGETING:
74  // update targets using pilot's rc inputs
75  update_targets_from_rc();
76  resend_now = true;
77  break;
78 
79  // point mount to a GPS point given by the mission planner
80  case MAV_MOUNT_MODE_GPS_POINT:
81  if(_frontend._ahrs.get_gps().status() >= AP_GPS::GPS_OK_FIX_2D) {
82  calc_angle_to_location(_state._roi_target, _angle_ef_target_rad, true, true);
83  resend_now = true;
84  }
85  break;
86 
87  default:
88  // we do not know this mode so do nothing
89  break;
90  }
91 
92  // resend target angles at least once per second
93  resend_now = resend_now || ((hal.scheduler->millis() - _last_send) > AP_MOUNT_STORM32_SERIAL_RESEND_MS);
94 
95  if ((hal.scheduler->millis() - _last_send) > AP_MOUNT_STORM32_SERIAL_RESEND_MS*2) {
96  _reply_type = ReplyType_UNKNOWN;
97  }
98  if (can_send(resend_now)) {
99  if (resend_now) {
100  send_target_angles(ToDeg(_angle_ef_target_rad.y), ToDeg(_angle_ef_target_rad.x), ToDeg(_angle_ef_target_rad.z));
101  get_angles();
102  _reply_type = ReplyType_ACK;
103  _reply_counter = 0;
104  _reply_length = get_reply_size(_reply_type);
105  } else {
106  get_angles();
107  _reply_type = ReplyType_DATA;
108  _reply_counter = 0;
109  _reply_length = get_reply_size(_reply_type);
110  }
111  }

149 // send_target_angles
150 void AP_Mount_SToRM32_serial::send_target_angles(float pitch_deg, float roll_deg, float yaw_deg)
151 {
152 
153  static cmd_set_angles_struct cmd_set_angles_data = {
154  0xFA,
155  0x0E,
156  0x11,
157  0, // pitch
158  0, // roll
159  0, // yaw
160  0, // flags
161  0, // type
162  0, // crc
163  };
164 
165  // exit immediately if not initialised
166  if (!_initialised) {
167  return;
168  }
169 
170  if ((size_t)_port->txspace() < sizeof(cmd_set_angles_data)) {
171  return;
172  }
173 
174  // reverse pitch and yaw control
175  pitch_deg = -pitch_deg;
176  yaw_deg = -yaw_deg;
177 
178  // send CMD_SETANGLE
179  cmd_set_angles_data.pitch = pitch_deg;
180  cmd_set_angles_data.roll = roll_deg;
181  cmd_set_angles_data.yaw = yaw_deg;
182 
183  uint8_t* buf = (uint8_t*)&cmd_set_angles_data;
184 
185  cmd_set_angles_data.crc = crc_calculate(&buf[1], sizeof(cmd_set_angles_data)-3);
186 
187  for (uint8_t i = 0; i != sizeof(cmd_set_angles_data) ; i++) {
188  _port->write(buf[i]);
189  }
190 
191  // store time of send
192  _last_send = hal.scheduler->millis();
193 }

226

228

263 void AP_Mount_SToRM32_serial::parse_reply() {
264  uint16_t crc;
265  bool crc_ok;
266 
267  switch (_reply_type) {
268  case ReplyType_DATA:
269  crc = crc_calculate(_buffer.bytes, sizeof(_buffer.data)-3);
270  crc_ok = crc == _buffer.data.crc;
271  if (!crc_ok) {
272  break;
273  }
274

276  _current_angle.y = _buffer.data.imu1_pitch;
277  _current_angle.z = _buffer.data.imu1_yaw;
278  break;
279  case ReplyType_ACK:
280  crc = crc_calculate(&_buffer.bytes[1], sizeof(SToRM32_reply_ack_struct)-3);
281  crc_ok = crc == _buffer.ack.crc;
282  break;
