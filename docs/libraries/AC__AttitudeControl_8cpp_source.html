442 // frame_conversion_ef_to_bf - converts earth frame vector to body frame vector
443 void AC_AttitudeControl::frame_conversion_ef_to_bf(const Vector3f& ef_vector, Vector3f& bf_vector)
444 {
445  // convert earth frame rates to body frame rates
446  bf_vector.x = ef_vector.x - _ahrs.sin_pitch() * ef_vector.z;
447  bf_vector.y = _ahrs.cos_roll() * ef_vector.y + _ahrs.sin_roll() * _ahrs.cos_pitch() * ef_vector.z;
448  bf_vector.z = -_ahrs.sin_roll() * ef_vector.y + _ahrs.cos_pitch() * _ahrs.cos_roll() * ef_vector.z;
449 }

451 // frame_conversion_bf_to_ef - converts body frame vector to earth frame vector
452 bool AC_AttitudeControl::frame_conversion_bf_to_ef(const Vector3f& bf_vector, Vector3f& ef_vector)
453 {
454  // avoid divide by zero
455  if (is_zero(_ahrs.cos_pitch())) {
456  return false;
457  }
458  // convert earth frame angle or rates to body frame
459  ef_vector.x = bf_vector.x + _ahrs.sin_roll() * (_ahrs.sin_pitch()/_ahrs.cos_pitch()) * bf_vector.y + _ahrs.cos_roll() * (_ahrs.sin_pitch()/_ahrs.cos_pitch()) * bf_vector.z;
460  ef_vector.y = _ahrs.cos_roll() * bf_vector.y - _ahrs.sin_roll() * bf_vector.z;
461  ef_vector.z = (_ahrs.sin_roll() / _ahrs.cos_pitch()) * bf_vector.y + (_ahrs.cos_roll() / _ahrs.cos_pitch()) * bf_vector.z;
462  return true;
463 }

544 // update_rate_bf_targets - converts body-frame angle error to body-frame rate targets for roll, pitch and yaw axis
545 // targets rates in centi-degrees taken from _angle_bf_error
546 // results in centi-degrees/sec put into _rate_bf_target

585 {

588  float rate_error; // simply target_rate - current_rate
589 
590  // get current rate
591  // To-Do: make getting gyro rates more efficient?
592  current_rate = (_ahrs.get_gyro().x * AC_ATTITUDE_CONTROL_DEGX100);
593

598

601

604

608  }
609

612

742 // sqrt_controller - response based on the sqrt of the error instead of the more common linear response
743 float AC_AttitudeControl::sqrt_controller(float error, float p, float second_ord_lim)
744 {
745  if (is_zero(second_ord_lim) || is_zero(p)) {
746  return error*p;
747  }
748 
749  float linear_dist = second_ord_lim/sq(p);
750 
751  if (error > linear_dist) {
752  return safe_sqrt(2.0f*second_ord_lim*(error-(linear_dist/2.0f)));
753  } else if (error < -linear_dist) {
754  return -safe_sqrt(2.0f*second_ord_lim*(-error-(linear_dist/2.0f)));
755  } else {
756  return error*p;
757  }
758 }
