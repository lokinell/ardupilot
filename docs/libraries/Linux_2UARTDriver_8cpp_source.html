393 size_t UARTDriver::write(const uint8_t *buffer, size_t size)
394 {
395  if (!_initialised) {
396  return 0;
397  }
398  if (!_nonblocking_writes) {
399  /*
400  use the per-byte delay loop in write() above for blocking writes
401  */
402  size_t ret = 0;
403  while (size--) {
404  if (write(*buffer++) != 1) break;
405  ret++;
406  }
407  return ret;
408  }
409 
410  uint16_t _head, space;
411  space = BUF_SPACE(_writebuf);
412  if (space == 0) {
413  return 0;
414  }
415  if (size > space) {
416  size = space;
417  }
418  if (_writebuf_tail < _head) {
419  // perform as single memcpy
420  assert(_writebuf_tail+size <= _writebuf_size);
421  memcpy(&_writebuf[_writebuf_tail], buffer, size);
422  BUF_ADVANCETAIL(_writebuf, size);
423  return size;
424  }
425 
426  // perform as two memcpy calls
427  uint16_t n = _writebuf_size - _writebuf_tail;
428  if (n > size) n = size;
429  assert(_writebuf_tail+n <= _writebuf_size);
430  memcpy(&_writebuf[_writebuf_tail], buffer, n);
431  BUF_ADVANCETAIL(_writebuf, n);
432  buffer += n;
433  n = size - n;
434  if (n > 0) {
435  assert(_writebuf_tail+n <= _writebuf_size);
436  memcpy(&_writebuf[_writebuf_tail], buffer, n);
437  BUF_ADVANCETAIL(_writebuf, n);
438  }
439  return size;

487 /*
488  try to push out one lump of pending bytes
489  return true if progress is made
490  */
491 bool UARTDriver::_write_pending_bytes(void)
492 {
493  uint16_t n;
494 
495  // write any pending bytes
496  uint16_t _tail;
497  uint16_t available_bytes = BUF_AVAILABLE(_writebuf);
498  n = available_bytes;
499  if (_packetise && n > 0 && _writebuf[_writebuf_head] == 254) {
500  // this looks like a MAVLink packet - try to write on
501  // packet boundaries when possible
502  if (n < 8) {
503  n = 0;
504  } else {
505  // the length of the packet is the 2nd byte, and mavlink
506  // packets have a 6 byte header plus 2 byte checksum,
507  // giving len+8 bytes
508  uint16_t ofs = (_writebuf_head + 1) % _writebuf_size;
509  uint8_t len = _writebuf[ofs];
510  if (n < len+8) {
511  // we don't have a full packet yet
512  n = 0;
513  } else if (n > len+8) {
514  // send just 1 packet at a time (so MAVLink packets
515  // are aligned on UDP boundaries)
516  n = len+8;
517  }
518  }
519  }
520 
521  if (n > 0) {
522  uint16_t n1 = _writebuf_size - _writebuf_head;
523  if (n1 >= n) {
524  // do as a single write
525  _write_fd(&_writebuf[_writebuf_head], n);
526  } else {
527  // split into two writes
528  if (_packetise) {
529  // keep as a single UDP packet
530  uint8_t tmpbuf[n];
531  memcpy(tmpbuf, &_writebuf[_writebuf_head], n1);
532  if (n > n1) {
533  memcpy(&tmpbuf[n1], &_writebuf[0], n-n1);
534  }
535  _write_fd(tmpbuf, n);
536  } else {
537  int ret = _write_fd(&_writebuf[_writebuf_head], n1);
538  if (ret == n1 && n > n1) {
539  _write_fd(&_writebuf[_writebuf_head], n - n1);
540  }
541  }
542  }
543  }
544 
545  return BUF_AVAILABLE(_writebuf) != available_bytes;
546 }
