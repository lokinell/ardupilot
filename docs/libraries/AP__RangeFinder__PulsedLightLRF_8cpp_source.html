48 // start_reading() - ask sensor to make a range reading
49 bool AP_RangeFinder_PulsedLightLRF::start_reading()
50 {
51  // get pointer to i2c bus semaphore
52  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
53 
54  // exit immediately if we can't take the semaphore
55  if (i2c_sem == NULL || !i2c_sem->take(1)) {
56  return false;
57  }
58 
59  // send command to take reading
60  if (hal.i2c->writeRegister(AP_RANGEFINDER_PULSEDLIGHTLRF_ADDR,
61  AP_RANGEFINDER_PULSEDLIGHTLRF_MEASURE_REG,
62  AP_RANGEFINDER_PULSEDLIGHTLRF_MSRREG_ACQUIRE) != 0) {
63  i2c_sem->give();
64  return false;
65  }
66 
67  // return semaphore
68  i2c_sem->give();
69 
70  return true;
71 }

73 // read - return last value measured by sensor
74 bool AP_RangeFinder_PulsedLightLRF::get_reading(uint16_t &reading_cm)
75 {
76  uint8_t buff[2];
77 
78  // get pointer to i2c bus semaphore
79  AP_HAL::Semaphore* i2c_sem = hal.i2c->get_semaphore();
80 
81  // exit immediately if we can't take the semaphore
82  if (i2c_sem == NULL || !i2c_sem->take(1)) {
83  return false;
84  }
85 
86  // read the high and low byte distance registers
87  if (hal.i2c->readRegisters(AP_RANGEFINDER_PULSEDLIGHTLRF_ADDR,
88  AP_RANGEFINDER_PULSEDLIGHTLRF_DISTHIGH_REG, 2, &buff[0]) != 0) {
89  i2c_sem->give();
90  return false;
91  }
92 
93  // combine results into distance
94  reading_cm = ((uint16_t)buff[0]) << 8 | buff[1];
95 
96  // return semaphore
97  i2c_sem->give();
98 
99  // kick off another reading for next time
100  // To-Do: replace this with continuous mode
101  hal.scheduler->delay_microseconds(200);
102  start_reading();
103 
104  return true;
105 }
