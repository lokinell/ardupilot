92 void AP_RangeFinder_PX4_PWM::update(void)
93 {
94  if (_fd == -1) {
95  set_status(RangeFinder::RangeFinder_NotConnected);
96  return;
97  }
98 
99  struct pwm_input_s pwm;
100  float sum_cm = 0;
101  uint16_t count = 0;
102  const float scaling = ranger._scaling[state.instance];
103  uint32_t now = hal.scheduler->millis();
104 
105  while (::read(_fd, &pwm, sizeof(pwm)) == sizeof(pwm)) {
106  // report the voltage as the pulse width, so we get the raw
107  // pulse widths in the log
108  state.voltage_mv = pwm.pulse_width;
109 
110  _last_pulse_time_ms = now;
111 
112  // setup for scaling in meters per millisecond
113  float distance_cm = pwm.pulse_width * 0.1f * scaling + ranger._offset[state.instance];
114 
115  float distance_delta_cm = fabsf(distance_cm - _last_sample_distance_cm);
116  _last_sample_distance_cm = distance_cm;
117 
118  if (distance_delta_cm > 100) {
119  // varying by more than 1m in a single sample, which means
120  // between 50 and 100m/s vertically - discard
121  _good_sample_count = 0;
122  continue;
123  }
124 
125  if (_good_sample_count > 1) {
126  count++;
127  sum_cm += distance_cm;
128  _last_timestamp = pwm.timestamp;
129  } else {
130  _good_sample_count++;
131  }
132  }
133 
134  // if we haven't received a pulse for 1 second then we may need to
135  // reset the timer
136  int8_t stop_pin = ranger._stop_pin[state.instance];
137  uint16_t settle_time_ms = (uint16_t)ranger._settle_time_ms[state.instance];
138 
139  if (stop_pin != -1 && out_of_range()) {
140  // we are above the power saving range. Disable the sensor
141  hal.gpio->pinMode(stop_pin, HAL_GPIO_OUTPUT);
142  hal.gpio->write(stop_pin, false);
143  set_status(RangeFinder::RangeFinder_NoData);
144  state.distance_cm = 0;
145  state.voltage_mv = 0;
146  return;
147  }
148 
149  // if we have not taken a reading in the last 0.2s set status to No Data
150  if (hal.scheduler->micros64() - _last_timestamp >= 200000) {
151  set_status(RangeFinder::RangeFinder_NoData);
152  }
153 
154  /* if we haven't seen any pulses for 0.5s then the sensor is
155  probably dead. Try resetting it. Tests show the sensor takes
156  about 0.2s to boot, so 500ms offers some safety margin
157  */
158  if (now - _last_pulse_time_ms > 500U && _disable_time_ms == 0) {
159  ioctl(_fd, SENSORIOCRESET, 0);
160  _last_pulse_time_ms = now;
161 
162  // if a stop pin is configured then disable the sensor for the
163  // settle time
164  if (stop_pin != -1) {
165  hal.gpio->pinMode(stop_pin, HAL_GPIO_OUTPUT);
166  hal.gpio->write(stop_pin, false);
167  _disable_time_ms = now;
168  }
169  }
170 
171  /* the user can configure a settle time. This controls how
172  long the sensor is disabled for using the stop pin when it is
173  reset. This is used both to make sure the sensor is properly
174  reset, and also to allow for power management by running a low
175  duty cycle when it has no signal
176  */
177  if (stop_pin != -1 && _disable_time_ms != 0 &&
178  (now - _disable_time_ms > settle_time_ms)) {
179  hal.gpio->write(stop_pin, true);
180  _disable_time_ms = 0;
181  _last_pulse_time_ms = now;
182  }
183 
184  if (count != 0) {
185  state.distance_cm = sum_cm / count;
186 
187  // update range_valid state based on distance measured
188  update_status();
189  }
190 }
