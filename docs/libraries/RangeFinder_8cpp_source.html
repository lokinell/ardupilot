442  around 10Hz by main loop

445 {

447  if (drivers[i] != NULL) {

571  these checks involve the user lifting or rotating the vehicle so that sensor readings between

585 /*
586  set pre-arm checks to passed if the range finder has been exercised through a reasonable range of movement
587  max distance sensed is at least 50cm > min distance sensed
588  max distance < 200cm
589  min distance sensed is within 10cm of ground clearance or sensor's minimum distance
590  */
591 void RangeFinder::update_pre_arm_check(uint8_t instance)
592 {
593  // return immediately if already passed or no sensor data
594  if (state[instance].pre_arm_check || state[instance].status == RangeFinder_NotConnected || state[instance].status == RangeFinder_NoData) {
595  return;
596  }
597 
598  // update min, max captured distances
599  state[instance].pre_arm_distance_min = min(state[instance].distance_cm, state[instance].pre_arm_distance_min);
600  state[instance].pre_arm_distance_max = max(state[instance].distance_cm, state[instance].pre_arm_distance_max);
601 
602  // Check that the range finder has been exercised through a realistic range of movement
603  if (((state[instance].pre_arm_distance_max - state[instance].pre_arm_distance_min) > RANGEFINDER_PREARM_REQUIRED_CHANGE_CM) &&
604  (state[instance].pre_arm_distance_max < RANGEFINDER_PREARM_ALT_MAX_CM) &&
605  ((int16_t)state[instance].pre_arm_distance_min < (max(_ground_clearance_cm[instance],min_distance_cm(instance)) + 10)) &&
606  ((int16_t)state[instance].pre_arm_distance_min > (min(_ground_clearance_cm[instance],min_distance_cm(instance)) - 10))) {
607  state[instance].pre_arm_check = true;
608  }
