89 void HeatPwm::set_imu_temp(float current)
90 {
91  float error, output;
92 
93  if (hal.scheduler->millis() - _last_temp_update < 5) {
94  return;
95  }
96 
97  /* minimal PI algo without dt */
98  error = _target - current;
99  /* Don't accumulate errors if the integrated error is superior
100  * to the max duty cycle(pwm_period)
101  */
102  if ((fabsf(_sum_error) * _Ki < _period_ns)) {
103  _sum_error = _sum_error + error;
104  }
105 
106  output = _Kp*error + _Ki * _sum_error;
107 
108  if (output > _period_ns) {
109  output = _period_ns;
110  } else if (output < 0) {
111  output = 0;
112  }
113 
114  _set_duty(output);
115  _last_temp_update = hal.scheduler->millis();
116 }
