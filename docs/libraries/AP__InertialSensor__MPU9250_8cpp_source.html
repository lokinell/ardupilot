257 bool AP_MPU9250_BusDriver_SPI::read_data_transaction(uint8_t *samples, uint8_t &n_samples)
258 {
259  /* one register address followed by seven 2-byte registers */
260  struct PACKED {
261  uint8_t cmd;
262  uint8_t int_status;
263  uint8_t v[MPU9250_SAMPLE_SIZE];
264  } rx, tx = { cmd : MPUREG_INT_STATUS | 0x80, };
265 
266  _spi->transaction((const uint8_t *)&tx, (uint8_t *)&rx, sizeof(rx));
267 
268  if (!(rx.int_status & BIT_RAW_RDY_INT)) {
269  n_samples = 0;
270 #if MPU9250_DEBUG
271  hal.console->printf("MPU9250: No sample available.\n");
272 #endif
273  return false;
274  }
275 
276  n_samples = 1;
277  memcpy(&samples[0], &rx.v[0], MPU9250_SAMPLE_SIZE);
278 
279  return true;
280 }

505 /*
506  read from the data registers and update filtered data
507  */
508 void AP_InertialSensor_MPU9250::_read_data_transaction()
509 {
510  uint8_t n_samples;
511  uint8_t rx[MPU9250_SAMPLE_SIZE];
512 
513  Vector3f accel, gyro;
514 
515  if (!_bus->read_data_transaction(rx, n_samples)) {
516  return;
517  }
518 
519 #define int16_val(v, idx) ((int16_t)(((uint16_t)v[2*idx] << 8) | v[2*idx+1]))

582 /*
583  initialise the sensor configuration registers
584  */
585 bool AP_InertialSensor_MPU9250::_hardware_init(void)
586 {
587  // we need to suspend timers to prevent other SPI drivers grabbing
588  // the bus while we do the long initialisation
589  hal.scheduler->suspend_timer_procs();
590 
591  if (!_bus_sem->take(100)) {
592  hal.console->printf("MPU9250: Unable to get semaphore");
593  return false;
594  }
595 
596  // initially run the bus at low speed
597  _bus->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
598 
599  uint8_t value = _register_read(MPUREG_WHOAMI);
600  if (value != MPUREG_WHOAMI_MPU9250 && value != MPUREG_WHOAMI_MPU9255) {
601  hal.console->printf("MPU9250: unexpected WHOAMI 0x%x\n", (unsigned)value);
602  goto fail_whoami;
603  }
604 
605  // Chip reset
606  uint8_t tries;
607  for (tries = 0; tries < 5; tries++) {
608  uint8_t user_ctrl = _register_read(MPUREG_USER_CTRL);
609 
610  /* First disable the master I2C to avoid hanging the slaves on the
611  * auxiliary I2C bus */
612  if (user_ctrl & BIT_USER_CTRL_I2C_MST_EN) {
613  _register_write(MPUREG_USER_CTRL, user_ctrl & ~BIT_USER_CTRL_I2C_MST_EN);
614  hal.scheduler->delay(10);
615  }
616 
617  // reset device
618  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
619  hal.scheduler->delay(100);
620 
621  // bus-dependent initialization
622  _bus->init();
623 
624  // Wake up device and select GyroZ clock. Note that the
625  // MPU9250 starts up in sleep mode, and it can take some time
626  // for it to come out of sleep
627  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_ZGYRO);
628  hal.scheduler->delay(5);
629 
630  // check it has woken up
631  if (_register_read(MPUREG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_ZGYRO) {
632  break;
633  }
634 
635  hal.scheduler->delay(10);
636  uint8_t status = _register_read(MPUREG_INT_STATUS);
637  if ((status & BIT_RAW_RDY_INT) != 0) {
638  break;
639  }
640 #if MPU9250_DEBUG
641  _dump_registers();
642 #endif
643  }
644 
645  if (tries == 5) {
646  hal.console->println("Failed to boot MPU9250 5 times");
647  goto fail_tries;
648  }
649 
650  _register_write(MPUREG_PWR_MGMT_2, 0x00); // only used for wake-up in accelerometer only low power mode
651 
652  // used no filter of 256Hz on the sensor, then filter using
653  // the 2-pole software filter
654  _register_write(MPUREG_CONFIG, BITS_DLPF_CFG_256HZ_NOLPF2);
655 
656  // set sample rate to 1kHz, and use the 2 pole filter to give the
657  // desired rate
658  _register_write(MPUREG_SMPLRT_DIV, DEFAULT_SMPLRT_DIV);
659  _register_write(MPUREG_GYRO_CONFIG, BITS_GYRO_FS_2000DPS); // Gyro scale 2000º/s
660 
661  // RM-MPU-9250A-00.pdf, pg. 15, select accel full scale 16g
662  _register_write(MPUREG_ACCEL_CONFIG,3<<3);
663 
664  // configure interrupt to fire when new data arrives
665  _register_write(MPUREG_INT_ENABLE, BIT_RAW_RDY_EN);
666 
667  // clear interrupt on any read, and hold the data ready pin high
668  // until we clear the interrupt
669  value = _register_read(MPUREG_INT_PIN_CFG);
670  value |= BIT_INT_RD_CLEAR | BIT_LATCH_INT_EN;
671  _register_write(MPUREG_INT_PIN_CFG, value);
672 
673  // now that we have initialized, we set the SPI bus speed to high
674  _bus->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_HIGH);
675 
676  _bus_sem->give();
677 
678  hal.scheduler->resume_timer_procs();
679 
680  return true;
681 
682 fail_tries:
683 fail_whoami:
684  _bus_sem->give();
685  hal.scheduler->resume_timer_procs();
686  _bus->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_HIGH);
687  return false;
688 }
