304

308  _flags.do_cmd_all_done = false;

408 bool AP_Mission::read_cmd_from_storage(uint16_t index, Mission_Command& cmd) const
409 {
410  // exit immediately if index is beyond last command but we always let cmd #0 (i.e. home) be read
411  if (index > (unsigned)_cmd_total && index != 0) {
412  return false;
413  }
414 
415  // special handling for command #0 which is home
416  if (index == 0) {
417  cmd.index = 0;
418  cmd.id = MAV_CMD_NAV_WAYPOINT;
419  cmd.p1 = 0;
420  cmd.content.location = _ahrs.get_home();
421  }else{
422  // Find out proper location in memory by using the start_byte position + the index
423  // we can load a command, we don't process it yet
424  // read WP position
425  uint16_t pos_in_storage = 4 + (index * AP_MISSION_EEPROM_COMMAND_SIZE);
426 
427  cmd.id = _storage.read_byte(pos_in_storage);
428  cmd.p1 = _storage.read_uint16(pos_in_storage+1);
429  _storage.read_block(cmd.content.bytes, pos_in_storage+3, 12);
430 
431  // set command's index to it's position in eeprom
432  cmd.index = index;
433  }
434 
435  // return success
436  return true;

476 {

478  bool copy_alt = false;
479  uint8_t num_turns, radius_m; // used by MAV_CMD_NAV_LOITER_TURNS & _TO_ALT
480  uint8_t heading_req; // used by MAV_CMD_NAV_LOITER_TO_ALT
481 
482  // command's position in mission list and mavlink id
483  cmd.index = packet.seq;
484  cmd.id = packet.command;
485  cmd.content.location.options = 0;
486 
487  // command specific conversions from mavlink packet to mission command
488  switch (cmd.id) {
489 
490  case MAV_CMD_NAV_WAYPOINT: // MAV ID: 16
491  copy_location = true;
492  /*
493  the 15 byte limit means we can't fit both delay and radius
494  in the cmd structure. When we expand the mission structure
495  we can do this properly
496  */
497 #if APM_BUILD_TYPE(APM_BUILD_ArduPlane)
498  // acceptance radius in meters
499  cmd.p1 = packet.param2;
500 #else
501  // delay at waypoint in seconds
502  cmd.p1 = packet.param1;
503 #endif
504  break;
505 
506  case MAV_CMD_NAV_LOITER_UNLIM: // MAV ID: 17
507  copy_location = true;
508  cmd.content.location.flags.loiter_ccw = (packet.param3 < 0); // -1 = counter clockwise, +1 = clockwise
509  break;
510 
511  case MAV_CMD_NAV_LOITER_TURNS: // MAV ID: 18
512  copy_location = true;
513  num_turns = packet.param1; // number of times to circle is held in param1
514  radius_m = fabsf(packet.param3); // radius in meters is held in high in param3
515  cmd.p1 = (((uint16_t)radius_m)<<8) | (uint16_t)num_turns; // store radius in high byte of p1, num turns in low byte of p1
516  cmd.content.location.flags.loiter_ccw = (packet.param3 < 0);
517  break;
518 
519  case MAV_CMD_NAV_LOITER_TIME: // MAV ID: 19
520  copy_location = true;
521  cmd.p1 = packet.param1; // loiter time in seconds
522  cmd.content.location.flags.loiter_ccw = (packet.param3 < 0);
523  break;

526  copy_location = true;
527  break;
