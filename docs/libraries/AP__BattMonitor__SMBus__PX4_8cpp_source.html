55 // read - read latest voltage and current
56 void AP_BattMonitor_SMBus_PX4::read()
57 {
58  bool updated = false;
59  struct battery_status_s batt_status;
60 
61  // check if new info has arrived from the orb
62  orb_check(_batt_sub, &updated);
63 
64  // retrieve latest info
65  if (updated) {
66  if (OK == orb_copy(ORB_ID(battery_status), _batt_sub, &batt_status)) {
67  _state.voltage = batt_status.voltage_v;
68  _state.current_amps = batt_status.current_a;
69  _state.last_time_micros = hal.scheduler->micros();
70  _state.current_total_mah = batt_status.discharged_mah;
71  _state.healthy = true;
72 
73  // read capacity
74  if ((_batt_fd >= 0) && !_capacity_updated) {
75  uint16_t tmp;
76  if (ioctl(_batt_fd, BATT_SMBUS_GET_CAPACITY, (unsigned long)&tmp) == OK) {
77  _capacity_updated = true;
78  set_capacity(tmp);
79  }
80  }
81  }
82  } else if (_state.healthy) {
83  // timeout after 5 seconds
84  if ((hal.scheduler->micros() - _state.last_time_micros) > AP_BATTMONITOR_SMBUS_TIMEOUT_MICROS) {
85  _state.healthy = false;
86  }
87  }
88 }
