103  write some bytes from the simulated GPS
104  */
105 void SITL_State::_gps_write(const uint8_t *p, uint16_t size)
106 {
107  while (size--) {
108  if (_sitl->gps_byteloss > 0.0f) {
109  float r = ((((unsigned)random()) % 1000000)) / 1.0e4;
110  if (r < _sitl->gps_byteloss) {
111  // lose the byte
112  p++;
113  continue;
114  }
115  }
116  write(gps_state.gps_fd, p, 1);
117  if (_sitl->gps2_enable) {
118  write(gps2_state.gps_fd, p, 1);

120  p++;
121  }

124 /*
125  send a UBLOX GPS message
126  */
127 void SITL_State::_gps_send_ubx(uint8_t msgid, uint8_t *buf, uint16_t size)
128 {
129  const uint8_t PREAMBLE1 = 0xb5;
130  const uint8_t PREAMBLE2 = 0x62;
131  const uint8_t CLASS_NAV = 0x1;
132  uint8_t hdr[6], chk[2];
133  hdr[0] = PREAMBLE1;
134  hdr[1] = PREAMBLE2;
135  hdr[2] = CLASS_NAV;
136  hdr[3] = msgid;
137  hdr[4] = size & 0xFF;
138  hdr[5] = size >> 8;
139  chk[0] = chk[1] = hdr[2];
140  chk[1] += (chk[0] += hdr[3]);
141  chk[1] += (chk[0] += hdr[4]);
142  chk[1] += (chk[0] += hdr[5]);
143  for (uint8_t i=0; i<size; i++) {
144  chk[1] += (chk[0] += buf[i]);
145  }
146  _gps_write(hdr, sizeof(hdr));
147  _gps_write(buf, size);
148  _gps_write(chk, sizeof(chk));
149 }

154 static void gps_time(uint16_t *time_week, uint32_t *time_week_ms)
155 {
156  struct timeval tv;
157  gettimeofday(&tv, NULL);
158  const uint32_t epoch = 86400*(10*365 + (1980-1969)/4 + 1 + 6 - 2) - 15;
159  uint32_t epoch_seconds = tv.tv_sec - epoch;
160  *time_week = epoch_seconds / (86400*7UL);
161  *time_week_ms = (epoch_seconds % (86400*7UL))*1000 + tv.tv_usec/1000;

315  struct PACKED mtk_msg {

318  uint8_t msg_class;
319  uint8_t msg_id;
320  int32_t latitude;

322  int32_t altitude;

328  uint8_t ck_a;
329  uint8_t ck_b;
330  } p;
331 
332  p.preamble1 = 0xb5;
333  p.preamble2 = 0x62;
334  p.msg_class = 1;
335  p.msg_id = 5;
336  p.latitude = d->latitude * 1.0e6;
337  p.longitude = d->longitude * 1.0e6;
338  p.altitude = d->altitude * 100;
339  p.ground_speed = pythagorous2(d->speedN, d->speedE) * 100;
340  p.ground_course = ToDeg(atan2f(d->speedE, d->speedN)) * 1000000.0f;
341  if (p.ground_course < 0.0f) {
342  p.ground_course += 360.0f * 1000000.0f;
343  }
344  p.satellites = d->have_lock?_sitl->gps_numsats:3;
345  p.fix_type = d->have_lock?3:1;
346 
347  // the spec is not very clear, but the time field seems to be
348  // milliseconds since the start of the day in UTC time,
349  // done in powers of 100.
350  // The date is powers of 100 as well, but in days since 1/1/2000
351  struct tm tm;
352  struct timeval tv;
353 
354  gettimeofday(&tv, NULL);
355  tm = *gmtime(&tv.tv_sec);
356  uint32_t hsec = (tv.tv_usec / (10000*20)) * 20; // always multiple of 20
357 
358  p.utc_time = hsec + tm.tm_sec*100 + tm.tm_min*100*100 + tm.tm_hour*100*100*100;
359 
360  swap_uint32((uint32_t *)&p.latitude, 5);
361  swap_uint32((uint32_t *)&p.utc_time, 1);
362  mtk_checksum(&p.msg_class, sizeof(p)-4, &p.ck_a, &p.ck_b);
363 
364  _gps_write((uint8_t*)&p, sizeof(p));
365 }

577 void SITL_State::_sbp_send_message(uint16_t msg_type, uint16_t sender_id, uint8_t len, uint8_t *payload)
578 {
579  if (len != 0 && payload == 0) {
580  return; //SBP_NULL_ERROR;
581  }
582 
583  uint8_t preamble = 0x55;
584  _gps_write(&preamble, 1);
585  _gps_write((uint8_t*)&msg_type, 2);
586  _gps_write((uint8_t*)&sender_id, 2);
587  _gps_write(&len, 1);
588  if (len > 0) {
589  _gps_write((uint8_t*)payload, len);
590  }
591

601 void SITL_State::_update_gps_sbp(const struct gps_data *d)
602 {
603  struct PACKED sbp_gps_time_t {
604  uint16_t wn; //< GPS week number
605  uint32_t tow; //< GPS Time of Week rounded to the nearest ms
606  int32_t ns; //< Nanosecond remainder of rounded tow
607  uint8_t flags; //< Status flags (reserved)
608  } t;
609  struct PACKED sbp_pos_llh_t {
610  uint32_t tow; //< GPS Time of Week
611  double lat; //< Latitude
612  double lon; //< Longitude
613  double height; //< Height
614  uint16_t h_accuracy; //< Horizontal position accuracy estimate
615  uint16_t v_accuracy; //< Vertical position accuracy estimate
616  uint8_t n_sats; //< Number of satellites used in solution
617  uint8_t flags; //< Status flags
618  } pos;
619  struct PACKED sbp_vel_ned_t {
620  uint32_t tow; //< GPS Time of Week
621  int32_t n; //< Velocity North coordinate
622  int32_t e; //< Velocity East coordinate
623  int32_t d; //< Velocity Down coordinate
624  uint16_t h_accuracy; //< Horizontal velocity accuracy estimate
625  uint16_t v_accuracy; //< Vertical velocity accuracy estimate
626  uint8_t n_sats; //< Number of satellites used in solution
627  uint8_t flags; //< Status flags (reserved)
628  } velned;
629  struct PACKED sbp_dops_t {
630  uint32_t tow; //< GPS Time of Week
631  uint16_t gdop; //< Geometric Dilution of Precision
632  uint16_t pdop; //< Position Dilution of Precision
633  uint16_t tdop; //< Time Dilution of Precision
634  uint16_t hdop; //< Horizontal Dilution of Precision
635  uint16_t vdop; //< Vertical Dilution of Precision
636  } dops;
637 
638  static const uint16_t SBP_HEARTBEAT_MSGTYPE = 0xFFFF;
639  static const uint16_t SBP_GPS_TIME_MSGTYPE = 0x0100;
640  static const uint16_t SBP_DOPS_MSGTYPE = 0x0206;
641  static const uint16_t SBP_POS_LLH_MSGTYPE = 0x0201;
642  static const uint16_t SBP_VEL_NED_MSGTYPE = 0x0205;
643 
644  uint16_t time_week;
645  uint32_t time_week_ms;
646 
647  gps_time(&time_week, &time_week_ms);
648 
649  t.wn = time_week;
650  t.tow = time_week_ms;
651  t.ns = 0;
652  t.flags = 0;
653  _sbp_send_message(SBP_GPS_TIME_MSGTYPE, 0x2222, sizeof(t), (uint8_t*)&t);
654 
655  if (!d->have_lock) {
656  return;
657  }
658 
659  pos.tow = time_week_ms;
660  pos.lon = d->longitude;
661  pos.lat= d->latitude;
662  pos.height = d->altitude;
663  pos.h_accuracy = 5e3;
664  pos.v_accuracy = 10e3;
665  pos.n_sats = _sitl->gps_numsats;
666 
667  // Send single point position solution
668  pos.flags = 0;
669  _sbp_send_message(SBP_POS_LLH_MSGTYPE, 0x2222, sizeof(pos), (uint8_t*)&pos);
670  // Send "pseudo-absolute" RTK position solution
671  pos.flags = 1;
672  _sbp_send_message(SBP_POS_LLH_MSGTYPE, 0x2222, sizeof(pos), (uint8_t*)&pos);
673 
674  velned.tow = time_week_ms;
675  velned.n = 1e3 * d->speedN;
676  velned.e = 1e3 * d->speedE;
677  velned.d = 1e3 * d->speedD;
678  velned.h_accuracy = 5e3;
679  velned.v_accuracy = 5e3;
680  velned.n_sats = _sitl->gps_numsats;
681  velned.flags = 0;
682  _sbp_send_message(SBP_VEL_NED_MSGTYPE, 0x2222, sizeof(velned), (uint8_t*)&velned);
683 
684  static uint32_t do_every_count = 0;
685  if (do_every_count % 5 == 0) {
686 
687  dops.tow = time_week_ms;
688  dops.gdop = 1;
689  dops.pdop = 1;
690  dops.tdop = 1;
691  dops.hdop = 100;
692  dops.vdop = 1;
693  _sbp_send_message(SBP_DOPS_MSGTYPE, 0x2222, sizeof(dops),
694  (uint8_t*)&dops);
695 
696  uint32_t system_flags = 0;
697  _sbp_send_message(SBP_HEARTBEAT_MSGTYPE, 0x2222,
698  sizeof(system_flags),
699  (uint8_t*)&system_flags);
700 
701  }
702  do_every_count++;
703 }

732  */
733 void SITL_State::_update_gps(double latitude, double longitude, float altitude,
734  double speedN, double speedE, double speedD, bool have_lock)
735 {
736  struct gps_data d;
737  char c;
738  Vector3f glitch_offsets = _sitl->gps_glitch;
739 
740  //Capture current position as basestation location for
741  if (!_gps_has_basestation_position) {
742  if (have_lock) {
743  _gps_basestation_data.latitude = latitude;
744  _gps_basestation_data.longitude = longitude;
745  _gps_basestation_data.altitude = altitude;
746  _gps_basestation_data.speedN = speedN;
747  _gps_basestation_data.speedE = speedE;
748  _gps_basestation_data.speedD = speedD;
749  _gps_basestation_data.have_lock = have_lock;
750  _gps_has_basestation_position = true;
751  }
752  }
753 
754  // run at configured GPS rate (default 5Hz)
755  if ((hal.scheduler->millis() - gps_state.last_update) < (uint32_t)(1000/_sitl->gps_hertz)) {
756  return;
757  }
758 
759  // swallow any config bytes
760  if (gps_state.gps_fd != 0) {
761  read(gps_state.gps_fd, &c, 1);
762  }
763  if (gps2_state.gps_fd != 0) {
764  read(gps2_state.gps_fd, &c, 1);
765  }
766 
767  gps_state.last_update = hal.scheduler->millis();
768  gps2_state.last_update = hal.scheduler->millis();
769 
770  d.latitude = latitude + glitch_offsets.x;
771  d.longitude = longitude + glitch_offsets.y;
772  d.altitude = altitude + glitch_offsets.z;
773 
774  if (_sitl->gps_drift_alt > 0) {
775  // slow altitude drift
776  d.altitude += _sitl->gps_drift_alt*sinf(hal.scheduler->millis()*0.001f*0.02f);
777  }
778 
779  d.speedN = speedN;
780  d.speedE = speedE;
781  d.speedD = speedD;
782  d.have_lock = have_lock;
783 
784  // add in some GPS lag
785  _gps_data[next_gps_index++] = d;
786  if (next_gps_index >= gps_delay+1) {
787  next_gps_index = 0;
788  }
789 
790  d = _gps_data[next_gps_index];
791 
792  if (_sitl->gps_delay != gps_delay) {
793  // cope with updates to the delay control
794  gps_delay = _sitl->gps_delay;
795  for (uint8_t i=0; i<gps_delay; i++) {
796  _gps_data[i] = d;
797  }
798  }
799 
800  if (gps_state.gps_fd == 0 && gps2_state.gps_fd == 0) {
801  return;
802  }
803 
804  switch ((SITL::SITL::GPSType)_sitl->gps_type.get()) {
805  case SITL::SITL::GPS_TYPE_NONE:
806  // no GPS attached
807  break;
808 
809  case SITL::SITL::GPS_TYPE_UBLOX:
810  _update_gps_ubx(&d);
811  break;
812 
813  case SITL::SITL::GPS_TYPE_MTK:
814  _update_gps_mtk(&d);
815  break;
816 
817  case SITL::SITL::GPS_TYPE_MTK16:
818  _update_gps_mtk16(&d);
819  break;
820 
821  case SITL::SITL::GPS_TYPE_MTK19:
822  _update_gps_mtk19(&d);
823  break;
824 
825  case SITL::SITL::GPS_TYPE_NMEA:
826  _update_gps_nmea(&d);
827  break;
828 
829  case SITL::SITL::GPS_TYPE_SBP:
830  _update_gps_sbp(&d);
831  break;
832 
833  case SITL::SITL::GPS_TYPE_FILE:
834  _update_gps_file(&d);
835  break;
836 
837  }
838 }
