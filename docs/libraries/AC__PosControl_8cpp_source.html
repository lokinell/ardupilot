777 void AC_PosControl::pos_to_rate_xy(xy_mode mode, float dt, float ekfNavVelGainScaler)
778 {
779  Vector3f curr_pos = _inav.get_position();
780  float linear_distance; // the distance we swap between linear and sqrt velocity response
781  float kP = ekfNavVelGainScaler * _p_pos_xy.kP(); // scale gains to compensate for noisy optical flow measurement in the EKF
782 
783  // avoid divide by zero
784  if (kP <= 0.0f) {
785  _vel_target.x = 0.0f;
786  _vel_target.y = 0.0f;
787  }else{
788  // calculate distance error
789  _pos_error.x = _pos_target.x - curr_pos.x;
790  _pos_error.y = _pos_target.y - curr_pos.y;
791 
792  // constrain target position to within reasonable distance of current location
793  _distance_to_target = pythagorous2(_pos_error.x, _pos_error.y);
794  if (_distance_to_target > _leash && _distance_to_target > 0.0f) {
795  _pos_target.x = curr_pos.x + _leash * _pos_error.x/_distance_to_target;
796  _pos_target.y = curr_pos.y + _leash * _pos_error.y/_distance_to_target;
797  // re-calculate distance error
798  _pos_error.x = _pos_target.x - curr_pos.x;
799  _pos_error.y = _pos_target.y - curr_pos.y;
800  _distance_to_target = _leash;
801  }
802 
803  // calculate the distance at which we swap between linear and sqrt velocity response
804  linear_distance = _accel_cms/(2.0f*kP*kP);
805 
806  if (_distance_to_target > 2.0f*linear_distance) {
807  // velocity response grows with the square root of the distance
808  float vel_sqrt = safe_sqrt(2.0f*_accel_cms*(_distance_to_target-linear_distance));
809  _vel_target.x = vel_sqrt * _pos_error.x/_distance_to_target;
810  _vel_target.y = vel_sqrt * _pos_error.y/_distance_to_target;
811  }else{
812  // velocity response grows linearly with the distance
813  _vel_target.x = _p_pos_xy.kP() * _pos_error.x;
814  _vel_target.y = _p_pos_xy.kP() * _pos_error.y;
815  }
816 
817  if (mode == XY_MODE_POS_LIMITED_AND_VEL_FF) {
818  // this mode is for loiter - rate-limiting the position correction
819  // allows the pilot to always override the position correction in
820  // the event of a disturbance
821 
822  // scale velocity within limit
823  float vel_total = pythagorous2(_vel_target.x, _vel_target.y);
824  if (vel_total > POSCONTROL_VEL_XY_MAX_FROM_POS_ERR) {
825  _vel_target.x = POSCONTROL_VEL_XY_MAX_FROM_POS_ERR * _vel_target.x/vel_total;
826  _vel_target.y = POSCONTROL_VEL_XY_MAX_FROM_POS_ERR * _vel_target.y/vel_total;
827  }
828 
829  // add velocity feed-forward
830  _vel_target.x += _vel_desired.x;
831  _vel_target.y += _vel_desired.y;
832  } else {
833  if (mode == XY_MODE_POS_AND_VEL_FF) {
834  // add velocity feed-forward
835  _vel_target.x += _vel_desired.x;
836  _vel_target.y += _vel_desired.y;
837  }
838 
839  // scale velocity within speed limit
840  float vel_total = pythagorous2(_vel_target.x, _vel_target.y);
841  if (vel_total > _speed_cms) {
842  _vel_target.x = _speed_cms * _vel_target.x/vel_total;
843  _vel_target.y = _speed_cms * _vel_target.y/vel_total;
844  }
845  }
846  }

851 void AC_PosControl::rate_to_accel_xy(float dt, float ekfNavVelGainScaler)
852 {
853  const Vector3f &vel_curr = _inav.get_velocity(); // current velocity in cm/s
854  Vector2f vel_xy_p, vel_xy_i;
855 
856  // reset last velocity target to current target
857  if (_flags.reset_rate_to_accel_xy) {
858  _vel_last.x = _vel_target.x;
859  _vel_last.y = _vel_target.y;
860  _flags.reset_rate_to_accel_xy = false;

864  if (dt > 0.0f) {
865  if (!_flags.freeze_ff_xy) {

867  _accel_feedforward.y = (_vel_target.y - _vel_last.y)/dt;
868  } else {
869  // stop the feed forward being calculated during a known discontinuity
870  _flags.freeze_ff_xy = false;
871  }
872  } else {

874  _accel_feedforward.y = 0.0f;

878  _vel_last.x = _vel_target.x;
879  _vel_last.y = _vel_target.y;

882  _vel_error.x = _vel_target.x - vel_curr.x;
883  _vel_error.y = _vel_target.y - vel_curr.y;
