910 void
911 AP_InertialSensor::_init_gyro()
912 {
913  uint8_t num_gyros = min(get_gyro_count(), INS_MAX_INSTANCES);
914  Vector3f last_average[INS_MAX_INSTANCES], best_avg[INS_MAX_INSTANCES];
915  Vector3f new_gyro_offset[INS_MAX_INSTANCES];
916  float best_diff[INS_MAX_INSTANCES];
917  bool converged[INS_MAX_INSTANCES];
918 
919  // exit immediately if calibration is already in progress
920  if (_calibrating) {
921  return;
922  }
923 
924  // record we are calibrating
925  _calibrating = true;
926 
927  // flash leds to tell user to keep the IMU still
928  AP_Notify::flags.initialising = true;
929 
930  // cold start
931  hal.console->print("Init Gyro");
932 
933  /*
934  we do the gyro calibration with no board rotation. This avoids
935  having to rotate readings during the calibration
936  */
937  enum Rotation saved_orientation = _board_orientation;
938  _board_orientation = ROTATION_NONE;
939 
940  // remove existing gyro offsets
941  for (uint8_t k=0; k<num_gyros; k++) {
942  _gyro_offset[k].set(Vector3f());
943  new_gyro_offset[k].zero();
944  best_diff[k] = 0;
945  last_average[k].zero();
946  converged[k] = false;
947  }
948 
949  for(int8_t c = 0; c < 5; c++) {
950  hal.scheduler->delay(5);
951  update();
952  }
953 
954  // the strategy is to average 50 points over 0.5 seconds, then do it
955  // again and see if the 2nd average is within a small margin of
956  // the first
957 
958  uint8_t num_converged = 0;
959 
960  // we try to get a good calibration estimate for up to 30 seconds
961  // if the gyros are stable, we should get it in 1 second
962  for (int16_t j = 0; j <= 30*4 && num_converged < num_gyros; j++) {
963  Vector3f gyro_sum[INS_MAX_INSTANCES], gyro_avg[INS_MAX_INSTANCES], gyro_diff[INS_MAX_INSTANCES];
964  Vector3f accel_start;
965  float diff_norm[INS_MAX_INSTANCES];
966  uint8_t i;
967 
968  memset(diff_norm, 0, sizeof(diff_norm));
969 
970  hal.console->print("*");
971 
972  for (uint8_t k=0; k<num_gyros; k++) {
973  gyro_sum[k].zero();
974  }
975  accel_start = get_accel(0);
976  for (i=0; i<50; i++) {
977  update();
978  for (uint8_t k=0; k<num_gyros; k++) {
979  gyro_sum[k] += get_gyro(k);
980  }
981  hal.scheduler->delay(5);
982  }
983 
984  Vector3f accel_diff = get_accel(0) - accel_start;
985  if (accel_diff.length() > 0.2f) {
986  // the accelerometers changed during the gyro sum. Skip
987  // this sample. This copes with doing gyro cal on a
988  // steadily moving platform. The value 0.2 corresponds
989  // with around 5 degrees/second of rotation.
990  continue;
991  }
992 
993  for (uint8_t k=0; k<num_gyros; k++) {
994  gyro_avg[k] = gyro_sum[k] / i;
995  gyro_diff[k] = last_average[k] - gyro_avg[k];
996  diff_norm[k] = gyro_diff[k].length();
997  }
998 
999  for (uint8_t k=0; k<num_gyros; k++) {
1000  if (j == 0) {
1001  best_diff[k] = diff_norm[k];
1002  best_avg[k] = gyro_avg[k];
1003  } else if (gyro_diff[k].length() < ToRad(0.1f)) {
1004  // we want the average to be within 0.1 bit, which is 0.04 degrees/s
1005  last_average[k] = (gyro_avg[k] * 0.5f) + (last_average[k] * 0.5f);
1006  if (!converged[k] || last_average[k].length() < new_gyro_offset[k].length()) {
1007  new_gyro_offset[k] = last_average[k];
1008  }
1009  if (!converged[k]) {
1010  converged[k] = true;
1011  num_converged++;
1012  }
1013  } else if (diff_norm[k] < best_diff[k]) {
1014  best_diff[k] = diff_norm[k];
1015  best_avg[k] = (gyro_avg[k] * 0.5f) + (last_average[k] * 0.5f);
1016  }
1017  last_average[k] = gyro_avg[k];
1018  }
1019  }
1020 
1021  // we've kept the user waiting long enough - use the best pair we
1022  // found so far
1023  hal.console->println();
1024  for (uint8_t k=0; k<num_gyros; k++) {
1025  if (!converged[k]) {
1026  hal.console->printf("gyro[%u] did not converge: diff=%f dps\n",
1027  (unsigned)k, (double)ToDeg(best_diff[k]));
1028  _gyro_offset[k] = best_avg[k];
1029  // flag calibration as failed for this gyro
1030  _gyro_cal_ok[k] = false;
1031  } else {
1032  _gyro_cal_ok[k] = true;
1033  _gyro_offset[k] = new_gyro_offset[k];
1034  }
1035  }
1036 
1037  // restore orientation
1038  _board_orientation = saved_orientation;
1039 
1040  // record calibration complete
1041  _calibrating = false;
1042 
1043  // stop flashing leds
1044  AP_Notify::flags.initialising = false;
1045 }

1094 // _calibrate_model - perform low level accel calibration
1095 // accel_sample are accelerometer samples collected in 6 different positions
1096 // accel_offsets are output from the calibration routine
1097 // accel_scale are output from the calibration routine
1098 // returns true if successful
1099 bool AP_InertialSensor::_calibrate_accel(const Vector3f accel_sample[6],
1100  Vector3f& accel_offsets,
1101  Vector3f& accel_scale,
1102  float max_abs_offsets,
1103  enum Rotation rotation)
1104 {
1105  int16_t i;
1106  int16_t num_iterations = 0;
1107  float eps = 0.000000001f;
1108  float change = 100.0f;
1109  float data[3];
1110  float beta[6];
1111  float delta[6];
1112  float ds[6];
1113  float JS[6][6];
1114  bool success = true;
1115 
1116  // reset
1117  beta[0] = beta[1] = beta[2] = 0;
1118  beta[3] = beta[4] = beta[5] = 1.0f/GRAVITY_MSS;
1119 
1120  while( num_iterations < 20 && change > eps ) {
1121  num_iterations++;
1122 
1123  _calibrate_reset_matrices(ds, JS);
1124 
1125  for( i=0; i<6; i++ ) {
1126  data[0] = accel_sample[i].x;
1127  data[1] = accel_sample[i].y;
1128  data[2] = accel_sample[i].z;
1129  _calibrate_update_matrices(ds, JS, beta, data);
1130  }
1131 
1132  _calibrate_find_delta(ds, JS, delta);
1133 
1134  change = delta[0]*delta[0] +
1135  delta[0]*delta[0] +
1136  delta[1]*delta[1] +
1137  delta[2]*delta[2] +
1138  delta[3]*delta[3] / (beta[3]*beta[3]) +
1139  delta[4]*delta[4] / (beta[4]*beta[4]) +
1140  delta[5]*delta[5] / (beta[5]*beta[5]);
1141 
1142  for( i=0; i<6; i++ ) {
1143  beta[i] -= delta[i];
1144  }
1145  }
1146 
1147  // copy results out
1148  accel_scale.x = beta[3] * GRAVITY_MSS;
1149  accel_scale.y = beta[4] * GRAVITY_MSS;
1150  accel_scale.z = beta[5] * GRAVITY_MSS;
1151  accel_offsets.x = beta[0] * accel_scale.x;
1152  accel_offsets.y = beta[1] * accel_scale.y;
1153  accel_offsets.z = beta[2] * accel_scale.z;
1154 
1155  // sanity check scale
1156  if( accel_scale.is_nan() || fabsf(accel_scale.x-1.0f) > 0.1f || fabsf(accel_scale.y-1.0f) > 0.1f || fabsf(accel_scale.z-1.0f) > 0.1f ) {
1157  success = false;
1158  }
1159 
1160  if (accel_offsets.is_nan() ||
1161  fabsf(accel_offsets.x) > max_abs_offsets ||
1162  fabsf(accel_offsets.y) > max_abs_offsets ||
1163  fabsf(accel_offsets.z) > max_abs_offsets) {
1164  success = false;
1165  }
1166 
1167  // return success or failure
1168  return success;
1169 }

1171 void AP_InertialSensor::_calibrate_update_matrices(float dS[6], float JS[6][6],
1172  float beta[6], float data[3])
1173 {
1174  int16_t j, k;
1175  float dx, b;
1176  float residual = 1.0f;
1177  float jacobian[6];
1178 
1179  for( j=0; j<3; j++ ) {
1180  b = beta[3+j];
1181  dx = (float)data[j] - beta[j];
1182  residual -= b*b*dx*dx;
1183  jacobian[j] = 2.0f*b*b*dx;
1184  jacobian[3+j] = -2.0f*b*dx*dx;
1185  }
1186 
1187  for( j=0; j<6; j++ ) {
1188  dS[j] += jacobian[j]*residual;
1189  for( k=0; k<6; k++ ) {
1190  JS[j][k] += jacobian[j]*jacobian[k];
1191  }
1192  }
1193 }
