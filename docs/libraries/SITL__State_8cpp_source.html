239  switch (size) {

241  static uint32_t last_report;
242  static uint32_t count;
243 
244  if (d.fg_pkt.magic != 0x4c56414f) {
245  fprintf(stdout, "Bad FDM packet - magic=0x%08x\n", d.fg_pkt.magic);

257  _synthetic_clock_mode = true;

259

263  double *v = &_sitl->state.latitude;

265  if (isinf(v[i]) || isnan(v[i]) || fabs(v[i]) > 1.0e10) {
266  v[i] = 0;
267  }
268  }
269  }

272  count++;
273  if (hal.scheduler->millis() - last_report > 1000) {
274  //fprintf(stdout, "SIM %u FPS\n", count);
275  count = 0;
276  last_report = hal.scheduler->millis();

282  uint8_t i;

333  This allows simulation of servo lag
334  */
335 void SITL_State::_apply_servo_filter(float deltat)
336 {
337  if (_sitl->servo_rate < 1.0f) {
338  // no limit
339  return;
340  }
341  // 1000 usec == 90 degrees
342  uint16_t max_change = deltat * _sitl->servo_rate * 1000 / 90;
343  if (max_change == 0) {
344  max_change = 1;
345  }
346  for (uint8_t i=0; i<SITL_NUM_CHANNELS; i++) {
347  int16_t change = (int16_t)pwm_output[i] - (int16_t)last_pwm_output[i];
348  if (change > max_change) {
349  pwm_output[i] = last_pwm_output[i] + max_change;
350  } else if (change < -max_change) {
351  pwm_output[i] = last_pwm_output[i] - max_change;
352  }
353  }
354 }

358  create sitl_input structure for sending to FDM

360 void SITL_State::_simulator_servos(Aircraft::sitl_input &input)
361 {
362  static uint32_t last_update_usec;
363 
364  /* this maps the registers used for PWM outputs. The RC
365  * driver updates these whenever it wants the channel output
366  * to change */
367  uint8_t i;
368 
369  if (last_update_usec == 0) {
370  for (i=0; i<SITL_NUM_CHANNELS; i++) {
371  pwm_output[i] = 1000;
372  }
373  if (_vehicle == ArduPlane) {
374  pwm_output[0] = pwm_output[1] = pwm_output[3] = 1500;
375  pwm_output[7] = 1800;
376  }
377  if (_vehicle == APMrover2) {
378  pwm_output[0] = pwm_output[1] = pwm_output[2] = pwm_output[3] = 1500;
379  pwm_output[7] = 1800;
380  }
381  for (i=0; i<SITL_NUM_CHANNELS; i++) {
382  last_pwm_output[i] = pwm_output[i];
383  }
384  }
385 
386  // output at chosen framerate
387  uint32_t now = hal.scheduler->micros();
388  float deltat = (now - last_update_usec) * 1.0e-6f;
389  last_update_usec = now;
390 
391  _apply_servo_filter(deltat);
392 
393  // pass wind into simulators, using a wind gradient below 60m
394  float altitude = _barometer?_barometer->get_altitude():0;
395  float wind_speed = _sitl->wind_speed;
396  if (altitude < 0) {
397  altitude = 0;
398  }
399  if (altitude < 60) {
400  wind_speed *= altitude / 60;
401  }
402  input.wind.speed = wind_speed;
403  input.wind.direction = _sitl->wind_direction;
404  input.wind.turbulence = _sitl->wind_turbulance;
405 
406  for (i=0; i<SITL_NUM_CHANNELS; i++) {
407  if (pwm_output[i] == 0xFFFF) {
408  input.servos[i] = 0;
409  } else {
410  input.servos[i] = pwm_output[i];
411  }
412  last_pwm_output[i] = pwm_output[i];
413  }
414 
415  if (_vehicle == ArduPlane) {
416  // add in engine multiplier
417  if (input.servos[2] > 1000) {
418  input.servos[2] = ((input.servos[2]-1000) * _sitl->engine_mul) + 1000;
419  if (input.servos[2] > 2000) input.servos[2] = 2000;
420  }
421  _motors_on = ((input.servos[2]-1000)/1000.0f) > 0;
422  } else if (_vehicle == APMrover2) {
423  // add in engine multiplier
424  if (input.servos[2] != 1500) {
425  input.servos[2] = ((input.servos[2]-1500) * _sitl->engine_mul) + 1500;
426  if (input.servos[2] > 2000) input.servos[2] = 2000;
427  if (input.servos[2] < 1000) input.servos[2] = 1000;
428  }
429  _motors_on = ((input.servos[2]-1500)/500.0f) != 0;
430  } else {
431  _motors_on = false;
432  // apply engine multiplier to first motor
433  input.servos[0] = ((input.servos[0]-1000) * _sitl->engine_mul) + 1000;
434  // run checks on each motor
435  for (i=0; i<4; i++) {
436  // check motors do not exceed their limits
437  if (input.servos[i] > 2000) input.servos[i] = 2000;
438  if (input.servos[i] < 1000) input.servos[i] = 1000;
439  // update motor_on flag
440  if ((input.servos[i]-1000)/1000.0f > 0) {
441  _motors_on = true;
442  }
443  }
444  }
445 
446  float throttle = _motors_on?(input.servos[2]-1000) / 1000.0f:0;
447  // lose 0.7V at full throttle
448  float voltage = _sitl->batt_voltage - 0.7f*throttle;
449 
450  // assume 50A at full throttle
451  _current = 50.0f * throttle;
452 
453  // assume 3DR power brick
454  voltage_pin_value = ((voltage / 10.1f) / 5.0f) * 1024;
455  current_pin_value = ((_current / 17.0f) / 5.0f) * 1024;
456 }
