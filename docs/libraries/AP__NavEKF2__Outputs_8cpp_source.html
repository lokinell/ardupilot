28  return false;

218  // This is the normal mode of operation where we can use the EKF position states

229  pos.x = tempPosNE.x;

453  status.flags.vert_pos = !hgtTimeout && filterHealthy; // vertical position estimate valid
454  status.flags.terrain_alt = gndOffsetValid && filterHealthy; // terrain height estimate valid
