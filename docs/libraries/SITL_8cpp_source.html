130 /*
131  convert a set of roll rates from earth frame to body frame
132  output values are in radians/second
133 */
134 void SITL::convert_body_frame(double rollDeg, double pitchDeg,
135  double rollRate, double pitchRate, double yawRate,
136  double *p, double *q, double *r)
137 {
138  double phi, theta, phiDot, thetaDot, psiDot;
139 
140  phi = ToRad(rollDeg);
141  theta = ToRad(pitchDeg);
142  phiDot = ToRad(rollRate);
143  thetaDot = ToRad(pitchRate);
144  psiDot = ToRad(yawRate);
145 
146  *p = phiDot - psiDot*sin(theta);
147  *q = cos(phi)*thetaDot + sin(phi)*psiDot*cos(theta);
148  *r = cos(phi)*psiDot*cos(theta) - sin(phi)*thetaDot;
149 }

152 /*
153  convert angular velocities from body frame to
154  earth frame.
155 
156  all inputs and outputs are in radians/s
157 */
158 Vector3f SITL::convert_earth_frame(const Matrix3f &dcm, const Vector3f &gyro)
159 {
160  float p = gyro.x;
161  float q = gyro.y;
162  float r = gyro.z;
163 
164  float phi, theta, psi;
165  dcm.to_euler(&phi, &theta, &psi);
166 
167  float phiDot = p + tanf(theta)*(q*sinf(phi) + r*cosf(phi));
168  float thetaDot = q*cosf(phi) - r*sinf(phi);
169  if (fabsf(cosf(theta)) < 1.0e-20f) {
170  theta += 1.0e-10f;
171  }
172  float psiDot = (q*sinf(phi) + r*cosf(phi))/cosf(theta);
173  return Vector3f(phiDot, thetaDot, psiDot);
174 }
